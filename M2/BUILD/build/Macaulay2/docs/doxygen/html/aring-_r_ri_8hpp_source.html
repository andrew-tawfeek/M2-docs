268 void power(ElementType &result, const ElementType &a, int n) const
269 {
270 if (n >= 2)
271 {
272 if (n%2 == 0)
273 {
274 ElementType b;
275 init(b);
276 power(b,a,n/2);
277 mult(result,b,b);
278 }
279 else
280 {
281 ElementType b;
282 init(b);
283 power(b,a,n-1);
284 mult(result,a,b);
285 }
286 }
287 else if (n == 1)
288 mpfi_set(&result,&a);
289 else if (n == 0)

292 throw 20;
