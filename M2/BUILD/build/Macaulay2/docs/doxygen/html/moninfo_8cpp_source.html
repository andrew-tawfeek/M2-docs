75monomial_word MonomialInfo::monomial_heft(const_packed_monomial m) const
76{
77 ncalls_weight++;
78 const_packed_monomial m1 = m + firstvar;
79 monomial_word sum = mModuleHeftDegrees[get_component(m)];
80 for (int j = 0; j < mHeftDegrees.size(); ++j)
81 {
82 sum += mHeftDegrees[j] * m1[j];
83 }
84 return sum;
