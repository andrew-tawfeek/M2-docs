16class ConcreteModuleMonoSet : public ModuleMonoSet {
17public:
18 typedef PolyRing::Monoid Monoid;
19 struct NoData {};
20 typedef StaticMonoLookup<UseKDTree, NoData, AllowRemovals, UseDivMask>
21 Lookup;
22 typedef typename Lookup::Entry Entry;
23
24 ConcreteModuleMonoSet(const Monoid& monoid, const size_t componentCount):
25 mMonoid(monoid),
26 mComponentCount(componentCount),
27 mLookups(std::allocator<Lookup>().allocate(mComponentCount))
28 {
29 for (size_t component = 0; component < componentCount; ++component)
30 new (&mLookups[component]) Lookup(monoid);
31 }
32
33 virtual ~ConcreteModuleMonoSet() {
34 for(size_t component = 0; component < mComponentCount; ++component)
35 mLookups[component].~Lookup();
36 std::allocator<Lookup>().deallocate(mLookups, mComponentCount);
37 }
38
39 virtual bool insert(ConstMonoRef m) {
40 const auto c = monoid().component(m);
41 MATHICGB_ASSERT(c < componentCount());
42 if (member(m))
43 return false;
44 if (AllowRemovals)
45 mLookups[c].removeMultiples(m);
46 mLookups[c].insert(m, NoData());
47 return true;
48 }
49
50 virtual bool member(ConstMonoRef m) {
51 const auto c = monoid().component(m);
52 MATHICGB_ASSERT(c < componentCount());
53 return mLookups[c].divisor(m) != 0;
54 }
55
56 virtual std::string name() const {
57 return Lookup(monoid()).getName();
58 }
59
60 virtual void display(std::ostream& out) const {
61 std::vector<ConstMonoPtr> monomials;
62 for (size_t c = 0; c < componentCount(); ++c) {
63 const Lookup& lookup = mLookups[c];
64 if (lookup.size() == 0)
65 continue;
66 out << " " << c << ": ";
67 monomials.clear();
68 lookup.forAll([&](const Entry& entry) {
69 monomials.emplace_back(entry.mono().ptr());
70 });
71 const auto& monoid = this->monoid(); // workaround for gcc 4.5.3 issue
72 const auto cmp = [&](ConstMonoPtr a, ConstMonoPtr b) {
73 return monoid.lessThan(*a, *b);
74 };
75 std::sort(monomials.begin(), monomials.end(), cmp);
76 for (auto mono = monomials.cbegin(); mono != monomials.cend(); ++mono) {
77 MathicIO<>().writeMonomial(monoid, false, **mono, out);
78 out << " ";
79 }
80 out << '\n';
81 }
82 }
83
84 virtual void getMonomials(std::vector<ConstMonoPtr>& monomials) const {
85 for (size_t c = 0; c < componentCount(); ++c) {
86 mLookups[c].forAll(
87 [&](const Entry& entry) {
88 monomials.push_back(entry.mono().ptr());
89 }
90 );
91 }
92 }
93
94 virtual void forAllVirtual(EntryOutput& consumer) {
95 for (size_t c = 0; c < componentCount(); ++c) {
96 mLookups[c].forAll(
97 [&](const Entry& entry) {
98 consumer.proceed(entry.mono());
99 }
100 );
101 }
102 }
103
104
105 virtual size_t elementCount() const {
106 size_t count = 0;
107 for (size_t c = 0; c < componentCount(); ++c)
108 count += mLookups[c].size();
109 return count;
110 }
111
112 virtual size_t getMemoryUse() const {
113 size_t count = 0;
114 for (size_t c = 0; c < componentCount(); ++c)
115 count += mLookups[c].getMemoryUse();
116 return count;
117 }

119 const Monoid& monoid() const {return mMonoid;}
120 size_t componentCount() const {return mComponentCount;}
121
122private:
123 const Monoid& mMonoid;
124
125 // We cannot use std::vector as Lookup might not be copyable or movable.
126 const size_t mComponentCount;
127 Lookup* const mLookups;
