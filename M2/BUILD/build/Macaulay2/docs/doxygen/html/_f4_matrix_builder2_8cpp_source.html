38 void initializeRowsToReduce(std::vector<RowTask>& tasks) {
39 // If aF-bG is an S-pair that is added as a bottom row in the matrix, and
40 // we do not already have a reducer for the common leading term of aF and
41 // bG, then we can instead set bG as a top/reducer/pivot row and put aF
42 // as a bottom row. This way, we do not need to add a reducer for that
43 // column if that monomial turns out to be used. We can also use the
44 // faster code path for adding a single multiple like aF to the matrix
45 // instead of the more complicated path that adds the difference between
46 // two multiples aF-bG.
47 //
48 // If bG is already the top/reducer/pivot row for its leading monomial,
49 // then we can simply replace a bottom row aF-bG with aF directly, since
50 // the first computation that will happen will be to calculate aF-bG.
51 // This way we get to use the much faster code in the matrix reducer for
52 // making that reduction/Gaussian-elimination step. More importantly, we
53 // have 1 row less to construct.
54 //
55 // The above two cases turn rows specified like aF-bG into rows specified
56 // more simply as just aF. Note that we can interchange aF and bG in these
57 // arguments with no problems.
58 //
59 // These two cases will apply to all
60 // S-pairs with a given leading monomial if and only if there is a single
61 // bG that appears in all of those S-pairs. The absolute best case is
62 // if that single common bG is at the same time the most preferable
63 // (sparsest or oldest) reducer for that monomial. It is not trivial to
64 // realize this (in fact I should write a paper it), but the particular
65 // way that S-pair elimination works in MathicGB ensures that this best
66 // case is always the case. So the code below will get rid of all the
67 // S-pairs aF-bG and replace them with single bottom rows aF.
68 // However, if the S-pair elimination code is ever changed, that may
69 // no longer be the case, so this code is still supposed to work even
70 // if some S-pairs are not gotten rid of. Still, there is an assert
71 // to flag it if the S-pair elimination is accidentally broken so that it
72 // no longer gives this guarantee.
73
74
75 // Bring S-pairs with the same leading monomial together by ordering
76 // them according in increasing order of monomial. Put the non-S-pairs
77 // together at the front.
78 auto cmp = [&](const RowTask& a, const RowTask& b) {
79 if (a.sPairPoly == nullptr)
80 return b.sPairPoly != nullptr;
81 else
82 return monoid().lessThan(*a.desiredLead, *b.desiredLead);
83 };
84 mtbb::parallel_sort(tasks.begin(), tasks.end(), cmp);
85
86 const auto taskCount = tasks.size();
87 for (size_t i = 0; i < taskCount;) {
88 if (tasks[i].sPairPoly == 0) {
89 ++i;
90 continue;
91 }
92 MATHICGB_ASSERT(tasks[i].desiredLead != nullptr);
93 MATHICGB_ASSERT(ColReader(mMap).find(*tasks[i].desiredLead).first == 0);
94
95 // Create column for the lead term that cancels in the S-pair
96 if (mIsColumnToLeft.size() >= std::numeric_limits<ColIndex>::max())
97 throw std::overflow_error("Too many columns in matrix.");
98 const auto newIndex = static_cast<ColIndex>(mIsColumnToLeft.size());
99 const auto inserted =
100 mMap.insert(std::make_pair(tasks[i].desiredLead, newIndex));
101 mIsColumnToLeft.push_back(true);
102 const auto& mono = inserted.first.second;
103
104 // Schedule the two parts of the S-pair as separate rows. This adds a row
105 // while creating the column in the hash table without adding a reducer
106 // removes a row, effectively making this operation equivalent to taking
107 // half of the S-pair and using it as a reducer.
108 auto desiredLead = monoid().alloc();
109 monoid().copy(*mono, *desiredLead);
110 RowTask newTask = {desiredLead.release(), tasks[i].sPairPoly, nullptr};
111
112 // Now we can strip off any part of an S-pair with the same cancelling lead
113 // term that equals a or b since those rows are in the matrix.
114 const auto* const a = tasks[i].poly;
115 const auto* const b = tasks[i].sPairPoly;
116
117 tasks[i].sPairPoly = 0;
118
119 tasks.push_back(newTask);
120 for (++i; i < taskCount; ++i) {
121 // auto& task = tasks[i];
122 if (tasks[i].sPairPoly == 0)
123 continue;
124 MATHICGB_ASSERT(tasks[i].desiredLead != nullptr);
125 if (!monoid().equal(*tasks[i].desiredLead, *mono))
126 break;
127 if (tasks[i].poly == a || tasks[i].poly == b) {
128 tasks[i].poly = tasks[i].sPairPoly;
129 tasks[i].sPairPoly = 0;
130 } else if (tasks[i].sPairPoly == a || tasks[i].sPairPoly == b) {
131 tasks[i].sPairPoly = 0;
132 } else
133 MATHICGB_ASSERT(false);
134 }
135 }
136 }

138 void buildMatrixAndClear(std::vector<RowTask>& tasks, QuadMatrix& quadMatrix) {
139 MATHICGB_ASSERT(&quadMatrix.ring() == &ring());
140 MATHICGB_LOG_TIME(F4MatrixBuild2) <<
141 "\n***** Constructing matrix *****\n";
142
143 if (tasks.empty()) {
144 quadMatrix.clear();
145 return;
146 }
147
148 initializeRowsToReduce(tasks);

150 // The MonoRef's cannot be Mono's since enumerable_thread_specific
151 // apparently requires the stored data type to be copyable and
152 // Mono is not copyable.
153 struct ThreadData {
154 MonoRef tmp1;
155 MonoRef tmp2;
156 F4ProtoMatrix block;
157 };
158
159 mtbb::enumerable_thread_specific<ThreadData> threadData([&](){
160 // We need to grab a lock since monoid isn't internally synchronized.
161 const mtbb::lock_guard guard(mCreateColumnLock);
162 ThreadData data = {
163 *monoid().alloc().release(),
164 *monoid().alloc().release(),
165 F4ProtoMatrix()
166 };
167 return data;
168 });
169
170 // Construct the matrix as pre-blocks
171 mtbb::parallel_for_each(tasks.begin(), tasks.end(),
172 [&](const RowTask& task, TaskFeeder& feeder)
173 {
174 auto& data = threadData.local();
175 const auto& poly = *task.poly;
176
177 // It is perfectly permissible for task.sPairPoly to be non-null. The
178 // assert is there because of an interaction between S-pair
179 // elimination/choice and the use of halves of S-pairs as reducers. The
180 // current effect of these is that *all* S-pairs have a component split
181 // off so that sPairPoly is always null (this is non-trivial to
182 // realize). So if this assert goes off, you've messed that interaction
183 // up somehow or you are using this class in some new way. So you can
184 // remove the assert if necessary.
185 MATHICGB_ASSERT(task.sPairPoly == 0);
186
187 if (task.sPairPoly != 0) {
188 monoid().colons(
189 poly.leadMono(),
190 task.sPairPoly->leadMono(),
191 data.tmp2,
192 data.tmp1
193 );
194 appendRowSPair
195 (poly, data.tmp1, *task.sPairPoly, data.tmp2, data.block, feeder);
196 return;
197 }
198 if (task.desiredLead == nullptr)
199 monoid().setIdentity(data.tmp1);
200 else
201 monoid().divide(poly.leadMono(), *task.desiredLead, data.tmp1);
202 appendRow(data.tmp1, *task.poly, data.block, feeder);
203 });
204 MATHICGB_ASSERT(!threadData.empty()); // as tasks empty causes early return
205
206 // Free the monomials from all the tasks
207 for (const auto& task : tasks)
208 if (task.desiredLead != nullptr)
209 monoid().freeRaw(*task.desiredLead.castAwayConst());
210 tasks.clear();
211
212 // Move the proto-matrices across all threads into the projection.
213 F4MatrixProjection projection
214 (ring(), static_cast<ColIndex>(mMap.entryCount()));
215 for (auto& data : threadData) {
216 monoid().freeRaw(data.tmp1);
217 monoid().freeRaw(data.tmp2);
218 projection.addProtoMatrix(std::move(data.block));
219 }
220
221 // Sort columns by monomial and tell the projection of the resulting order
222 MonomialMap<ColIndex>::Reader reader(mMap);
223 typedef std::pair<ColIndex, ConstMonoPtr> IndexMono;
224 auto toPtr = [](std::pair<ColIndex, ConstMonoRef> p) {
225 return std::make_pair(p.first, &p.second);
226 };
227 std::vector<IndexMono> columns;
228 std::transform
229 (reader.begin(), reader.end(), std::back_inserter(columns), toPtr);
230
231 const auto cmp = [&](const IndexMono& a, const IndexMono b) {
232 return monoid().lessThan(*b.second, *a.second);
233 };
234 mtbb::parallel_sort(columns.begin(), columns.end(), cmp);
235
236 const auto colEnd = columns.end();
237 for (auto it = columns.begin(); it != colEnd; ++it) {
238 const auto p = *it;
239 projection.addColumn(p.first, *p.second, mIsColumnToLeft[p.first]);
240 }
241
242 quadMatrix = projection.makeAndClear(mMemoryQuantum);
243
244 MATHICGB_LOG(F4MatrixSizes)
245 << "F4["
246 << mathic::ColumnPrinter::commafy(quadMatrix.rowCount())
247 << " by "
248 << mathic::ColumnPrinter::commafy(
249 quadMatrix.computeLeftColCount() + quadMatrix.computeRightColCount())
250 << "]" << std::endl;
251
252#ifdef MATHICGB_DEBUG
253 for (size_t side = 0; side < 2; ++side) {
254 auto& monos = side == 0 ?
255 quadMatrix.leftColumnMonomials : quadMatrix.rightColumnMonomials;
256 for (auto it = monos.begin(); it != monos.end(); ++it) {
257 MATHICGB_ASSERT(!it->isNull());
258 }
259 }
260 for (RowIndex row = 0; row < quadMatrix.topLeft.rowCount(); ++row) {
261 MATHICGB_ASSERT(quadMatrix.topLeft.entryCountInRow(row) > 0);
262 MATHICGB_ASSERT(quadMatrix.topLeft.leadCol(row) == row);
263 }
264 MATHICGB_ASSERT(quadMatrix.debugAssertValid());
265#endif
266 }

303 MATHICGB_NO_INLINE
304 std::pair<ColIndex, ConstMonoRef> createColumn(
305 ConstMonoRef monoA,
306 ConstMonoRef monoB,
307 TaskFeeder& feeder
308 ) {
309 const mtbb::lock_guard guard(mCreateColumnLock);
310 // see if the column exists now after we have synchronized
311 {
312 const auto found(ColReader(mMap).findProduct(monoA, monoB));
313 if (found.first != 0)
314 return std::make_pair(*found.first, *found.second);
315 }
316
317 // The column really does not exist, so we need to create it
318 monoid().multiply(monoA, monoB, mTmp);
319 if (!monoid().hasAmpleCapacity(*mTmp))
320 mathic::reportError("Monomial exponent overflow in F4MatrixBuilder2.");
321
322 // look for a reducer of mTmp
323 const size_t reducerIndex = mBasis.classicReducer(*mTmp);
324 const bool insertLeft = (reducerIndex != static_cast<size_t>(-1));
325
326 // Create the new left or right column
327 if (mIsColumnToLeft.size() >= std::numeric_limits<ColIndex>::max())
328 throw std::overflow_error("Too many columns in QuadMatrix");
329 const auto newIndex = static_cast<ColIndex>(mIsColumnToLeft.size());
330 const auto inserted = mMap.insert(std::make_pair(mTmp.ptr(), newIndex));
331 mIsColumnToLeft.push_back(insertLeft);
332
333 // schedule new task if we found a reducer
334 if (insertLeft) {
335 RowTask task = {};
336 task.poly = &mBasis.poly(reducerIndex);
337 task.desiredLead = inserted.first.second;
338 feeder.add(task);
339 }
340
341 return std::make_pair(*inserted.first.first, *inserted.first.second);
342 }

346 void appendRow(
347 ConstMonoRef multiple,
348 const Poly& poly,
349 F4ProtoMatrix& block,
350 TaskFeeder& feeder
351 ) {
352 const auto begin = poly.begin();
353 const auto end = poly.end();
354 const auto count = poly.termCount();
355 MATHICGB_ASSERT(count < std::numeric_limits<ColIndex>::max());
356 auto indices = block.makeRowWithTheseScalars(poly);
357
358 auto it = begin;
359 if ((count % 2) == 1) {
360 ColReader reader(mMap);
361 const auto col = findOrCreateColumn
362 (it.mono(), multiple, reader, feeder);
363 MATHICGB_ASSERT(it.coef() < std::numeric_limits<Scalar>::max());
364 MATHICGB_ASSERT(!field().isZero(it.coef()));
365 *indices = col.first;
366 ++indices;
367 ++it;
368 }
369 updateReader:
370 ColReader colMap(mMap);
371 while (it != end) {
372 MATHICGB_ASSERT(it.coef() < std::numeric_limits<Scalar>::max());
373 MATHICGB_ASSERT(!field().isZero(it.coef()));
374 // Unused. Remove. const auto scalar1 = static_cast<Scalar>(it.coef());
375 const auto mono1 = it.mono();
376
377 auto it2 = it;
378 ++it2;
379 MATHICGB_ASSERT(it2.coef() < std::numeric_limits<Scalar>::max());
380 MATHICGB_ASSERT(!field().isZero(it2.coef()));
381 // Unused. Remove. const auto scalar2 = static_cast<Scalar>(it2.coef());
382 const auto mono2 = it2.mono();
383
384 const auto colPair = colMap.findTwoProducts(mono1, mono2, multiple);
385 if (colPair.first == 0 || colPair.second == 0) {
386 createColumn(mono1, multiple, feeder);
387 createColumn(mono2, multiple, feeder);
388 goto updateReader;
389 }
390
391 *indices = *colPair.first;
392 ++indices;
393 *indices = *colPair.second;
394 ++indices;
395
396 it = ++it2;
397 }
398 }

402 void appendRowSPair(
403 const Poly& poly,
404 ConstMonoRef multiply,
405 const Poly& sPairPoly,
406 ConstMonoRef sPairMultiply,
407 F4ProtoMatrix& block,
408 TaskFeeder& feeder
409 ) {
410 MATHICGB_ASSERT(!poly.isZero());
411 auto itA = poly.begin();
412 const auto endA = poly.end();
413
414 MATHICGB_ASSERT(!sPairPoly.isZero());
415 auto itB = sPairPoly.begin();
416 const auto endB = sPairPoly.end();
417
418 // skip leading terms since they cancel
419 MATHICGB_ASSERT(itA.coef() == itB.coef());
420 ++itA;
421 ++itB;
422
423 // @todo: handle overflow of termCount addition here
424 MATHICGB_ASSERT(poly.termCount() + sPairPoly.termCount() - 2 <=
425 std::numeric_limits<ColIndex>::max());
426 const auto maxCols =
427 static_cast<ColIndex>(poly.termCount() + sPairPoly.termCount() - 2);
428 auto row = block.makeRow(maxCols);
429 const auto indicesBegin = row.first;
430
431 const ColReader colMap(mMap);
432
433 auto mulA = multiply;
434 auto mulB = sPairMultiply;
435 while (itB != endB && itA != endA) {
436 const auto colA = findOrCreateColumn
437 (itA.mono(), mulA, colMap, feeder);
438 const auto colB = findOrCreateColumn
439 (itB.mono(), mulB, colMap, feeder);
440 const auto cmp = monoid().compare(colA.second, colB.second);
441
442 coefficient coeff = 0;
443 ColIndex col;
444 if (cmp != LT) {
445 coeff = itA.coef();
446 col = colA.first;
447 ++itA;
448 }
449 if (cmp != GT) {
450 coeff = ring().coefficientSubtract(coeff, itB.coef());
451 col = colB.first;
452 ++itB;
453 }
454 MATHICGB_ASSERT(coeff < std::numeric_limits<Scalar>::max());
455 if (coeff != 0) {
456 *row.first++ = col;
457 *row.second++ = static_cast<Scalar>(coeff);
458 }
459 }
460
461 for (; itA != endA; ++itA) {
462 const auto colA = findOrCreateColumn
463 (itA.mono(), mulA, colMap, feeder);
464 *row.first++ = colA.first;
465 *row.second++ = static_cast<Scalar>(itA.coef());
466 }
467
468 for (; itB != endB; ++itB) {
469 const auto colB = findOrCreateColumn
470 (itB.mono(), mulB, colMap, feeder);
471 const auto negative = ring().coefficientNegate(itB.coef());
472 *row.first = colB.first;
473 ++row.first;
474 *row.second = static_cast<Scalar>(negative);
475 ++row.second;
476 }
477
478 const auto toRemove =
479 maxCols - static_cast<ColIndex>(row.first - indicesBegin);
480 block.removeLastEntries(block.rowCount() - 1, toRemove);
481 }

486 MATHICGB_NO_INLINE
487 std::pair<ColIndex, ConstMonoRef> findOrCreateColumn(
488 ConstMonoRef monoA,
489 ConstMonoRef monoB,
490 TaskFeeder& feeder
491 ) {
492 const auto col = ColReader(mMap).findProduct(monoA, monoB);
