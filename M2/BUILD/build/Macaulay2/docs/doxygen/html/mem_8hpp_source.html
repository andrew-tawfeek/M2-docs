116inline void *stash::new_elem()
117// Allocate space for an object from this stash.
118{
119 return newarray_clear(char, element_size);
120 acquireSpinLock(&list_spinlock);
121 n_allocs++;
122 n_inuse++;
123 if (n_inuse > highwater) highwater = n_inuse;
124 if (free_list == nullptr)
125 {
126 if (n_per_slab == 0)
127 {
128 void *result = newarray_clear(char, element_size);
129 // allocated_amount += element_size;
130 releaseSpinLock(&list_spinlock);
131 return result;
132 }
133 chop_slab();
134 }
135 assert(free_list != NULL); // chop_slab should not let this happen.
136 void *result = free_list;
137 free_list = *(reinterpret_cast<void **>(free_list));
138 releaseSpinLock(&list_spinlock);
139 return result;
140}
