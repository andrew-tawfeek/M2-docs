69TEST(MathicIO, ReadWriteMonomial) {
70 typedef PolyRing::Monoid Monoid;
71 typedef Monoid::VarIndex VarIndex;
72 typedef Monoid::Exponent Exponent;
73 static const auto NoComponent = static_cast<Exponent>(-1);
74 static const auto NoExponent = NoComponent;
75 static const auto NoVar = static_cast<VarIndex>(-1);
76
77 Monoid m(28);
78
79 // canonicalStr is str as it should appear after being read and
80 // printed. If canonicalStr is null then this is the same as str.
81 auto check = [&](
82 const char* const str,
83 const Exponent component,
84 const VarIndex var1,
85 const Exponent exp1,
86 const VarIndex var2,
87 const Exponent exp2,
88 const char* const canonicalStr
89 ) {
90 const bool doComponent = component != NoComponent;
91
92 // read monomial from string
93 auto monoRead = m.alloc();
94 Scanner in(str);
95 MathicIO<>().readMonomial(m, doComponent, *monoRead, in);
96
97 // directly make monomial
98 auto monoSet = m.alloc();
99 if(var1!=NoVar)
100 m.setExponent(var1, exp1, *monoSet);
101 if(var2!=NoVar)
102 m.setExponent(var2, exp2, *monoSet);
103 if (doComponent)
104 m.setComponent(component, *monoSet);
105 ASSERT_TRUE(m.equal(*monoRead, *monoSet)) << "Str: " << str;
106
107 // print monomial
108 std::ostringstream out;
109 MathicIO<>().writeMonomial(m, doComponent, *monoRead, out);
110 const auto correctStr = canonicalStr == 0 ? str : canonicalStr;
111 ASSERT_EQ(correctStr, out.str());
112 };
113
114 check("1", NoComponent, NoVar,NoExponent, NoVar,NoExponent, 0);
115 check("1<0>", 0, NoVar,NoExponent, NoVar,NoExponent, 0);
116 check("1<1>", 1, NoVar,NoExponent, NoVar,NoExponent, 0);
117 check("1<999>", 999, NoVar,NoExponent, NoVar,NoExponent, 0);
118
119 check("a1", NoComponent, 0,1, NoVar,NoExponent, "a");
120 check("b10<0>", 0, 1,10, NoVar,NoExponent, 0);
121 check("A11", NoComponent, 26,11, NoVar,NoExponent, 0);
122 check("B99<1>", 1, 27,99, NoVar,NoExponent, 0);
123
124 check("ab", NoComponent, 0,1, 1,1, 0);
125 check("ba", NoComponent, 0,1, 1,1, "ab");
126 check("a0c3b1", NoComponent, 1,1, 2,3, "bc3");
127 check("ab<2>", 2, 0,1, 1,1, 0);
128}
