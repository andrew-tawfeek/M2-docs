31std::unique_ptr<Basis> PolyBasis::initialIdeal() const {
32 std::unique_ptr<Basis> basis(new Basis(mRing));
33 size_t const basisSize = size();
34 for (size_t gen = 0; gen != basisSize; ++gen) {
35 if (!retired(gen) && leadMinimal(gen)) {
36 std::unique_ptr<Poly> p(new Poly(mRing));
37 p->append(1, leadMono(gen));
38 basis->insert(std::move(p));
39 }
40 }
41 basis->sort();
42 return basis;
43}

107size_t PolyBasis::divisor(ConstMonoRef mono) const {
108 size_t index = monoLookup().divisor(mono);
109 MATHICGB_ASSERT((index == static_cast<size_t>(-1)) ==
110 (divisorSlow(mono) == static_cast<size_t>(-1)));
111 MATHICGB_ASSERT(index == static_cast<size_t>(-1) ||
112 monoid().divides(leadMono(index), mono));
113 return index;

125size_t PolyBasis::divisorSlow(ConstMonoRef mono) const {
126 const size_t stop = size();

133bool PolyBasis::leadMinimalSlow(size_t index) const {
134 MATHICGB_ASSERT(index < size());
135 MATHICGB_ASSERT(!retired(index));
136 const auto lead = leadMono(index);
137 EntryCIter const skip = mEntries.begin() + index;
138 EntryCIter const stop = mEntries.end();
139 for (EntryCIter it = mEntries.begin(); it != stop; ++it) {
140 if (it->retired)
141 continue;
142 const auto itLead = it->poly->leadMono();
143 if (monoid().divides(itLead, lead) && it != skip)
144 return false;
145 }
146 return true;
147}
