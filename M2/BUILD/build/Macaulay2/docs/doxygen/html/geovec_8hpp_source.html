91inline void vecHeap::add(vecterm *p)
92{
93 mLead = -1;
94 int len = K->n_nonzero_terms(p);
95 int i = 0;
96 while (len >= heap_size[i]) i++;
97 K->add_vec_to(heap[i], p);
98 len = K->n_nonzero_terms(heap[i]);
99 p = NULL;
100 while (len >= heap_size[i])
101 {
102 i++;
103 K->add_vec_to(heap[i], heap[i - 1]);
104 len = K->n_nonzero_terms(heap[i]);
105 heap[i - 1] = NULL;
106 }
107 if (i > top_of_heap) top_of_heap = i;
108}

118inline const vecterm *vecHeap::get_lead_term()
119{
120 int lead_so_far = -1;
121 for (int i = 0; i <= top_of_heap; i++)
122 {
123 if (heap[i] == NULL) continue;
124 if (lead_so_far < 0)
125 {
126 lead_so_far = i;
127 continue;
128 }
129 int cmp = compare(heap[lead_so_far], heap[i]);
130 if (cmp == GT) continue;
131 if (cmp == LT)
132 {
133 lead_so_far = i;
134 continue;
135 }
136 // At this point we have equality
137 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
138 vecterm *tmp = heap[i];
139 heap[i] = tmp->next;
140 tmp->next = NULL;
141 K->remove_vec(tmp);
142
143 if (K->is_zero(heap[lead_so_far]->coeff))
144 {
145 // Remove, and start over
146 tmp = heap[lead_so_far];
147 heap[lead_so_far] = tmp->next;
148 tmp->next = NULL;
149 K->remove_vec(tmp);
150 lead_so_far = -1;
151 i = -1;
152 }
153 }
154 mLead = lead_so_far;
155 if (lead_so_far < 0) return NULL;
156 vecterm *result = heap[lead_so_far];
157 return result;
158}
159inline vecterm *vecHeap::remove_lead_term()
160{
161 if (mLead < 0) get_lead_term();
162 if (mLead < 0) return NULL;
163 vecterm *result = heap[mLead];
164 heap[mLead] = result->next;
165 result->next = NULL;
166 mLead = -1;
167 return result;
168}
