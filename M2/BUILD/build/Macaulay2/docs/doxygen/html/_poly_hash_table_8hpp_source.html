55 MATHICGB_INLINE
56 std::pair<Node*, bool> insertProduct(ConstMonoRef a, ConstMonoRef b) {
57 auto newNode = new (mNodes.alloc()) Node();
58 monoid().multiply(a, b, newNode->mono());
59 const auto abHash = monoid().hash(newNode->mono());
60 auto& bucket = mBuckets[hashToIndex(abHash)];
61
62 for (auto node = bucket; node != nullptr; node = node->next()) {
63 if (abHash != monoid().hash(node->mono()))
64 continue;
65 if (monoid().equal(newNode->mono(), node->mono())) {
66 mNodes.free(newNode);
67 return std::make_pair(node, false); // found a*b.
68 }
69 }
70
71 mRing.coefficientSet(newNode->value(), 0);
72 newNode->next() = bucket;
73 bucket = newNode;
74 ++mSize;
75 if (mSize >= mMaxSize)
76 rehash(bucketCount() * 2);
77 return std::make_pair(newNode, true); // inserted mono
78 }

98 void remove(Node* nodeToRemove) {
99 MATHICGB_ASSERT(nodeToRemove != 0);
100 MATHICGB_ASSERT(mNodes.fromPool(nodeToRemove));
101 const auto index = hashToIndex(monoid().hash(nodeToRemove->mono()));
102 auto nodePtr = &mBuckets[index];
103 while (*nodePtr != nodeToRemove) {
104 MATHICGB_ASSERT(*nodePtr != nullptr);
105 nodePtr = &(*nodePtr)->next();
106 }
107 *nodePtr = nodeToRemove->next();
108 mNodes.free(nodeToRemove);
109 --mSize;
110 }
