18 void MemoryBlocks::Block::newBlock(size_t capacityInBytes) {
19 if (!isNull()) {
20 // We set aside space for a block at the end of the memory. Use that
21 // space to store the block for the old memory.
22 const size_t blockStructOffset = alignNoOverflow(getBytesInBlock());
23 Block* block = reinterpret_cast<Block*>(begin() + blockStructOffset);
24 block->_previous = this->previousBlock();
25 block->_begin = begin();
26 block->_end = end();
27 block->_position = position();
28 _previous = block;
29 }
30
31 // make space for block information at end of new memory, but do not
32 // use it yet.
33 const size_t aligned = alignThrowOnOverflow(capacityInBytes);
34 const size_t total = aligned + sizeof(Block);
35 if (total < aligned) // check overflow
36 throw std::bad_alloc();
37 _begin = new char[total];
38 _position = _begin;
39 _end = _begin + capacityInBytes;
40
41 MEMT_ASSERT(!isNull());
42 MEMT_ASSERT(empty());
43 MEMT_ASSERT(capacityInBytes == getBytesInBlock());
44 }
