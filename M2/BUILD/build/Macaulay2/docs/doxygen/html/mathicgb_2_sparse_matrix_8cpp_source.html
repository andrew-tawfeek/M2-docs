11void SparseMatrix::takeRowsFrom(SparseMatrix&& matrix) {
12 if (matrix.mRows.empty())
13 return;
14
15 if (mRows.empty()) {
16 *this = std::move(matrix);
17 return;
18 }
19
20 Block* oldestBlock = &matrix.mBlock;
21 while (oldestBlock->mPreviousBlock != 0)
22 oldestBlock = oldestBlock->mPreviousBlock;
23
24 if (mBlock.mHasNoRows) // only put mBlock in chain of blocks if non-empty
25 oldestBlock->mPreviousBlock = mBlock.mPreviousBlock;
26 else
27 oldestBlock->mPreviousBlock = new Block(std::move(mBlock));
28 mBlock = std::move(matrix.mBlock);
29
30 mRows.insert(mRows.end(), matrix.mRows.begin(), matrix.mRows.end());
31 matrix.clear();
32}

38) {

50void SparseMatrix::sortRowsByIncreasingPivots() {
51 SparseMatrix ordered;
52 const auto rowCount = this->rowCount();
53
54 std::vector<RowIndex> rows(rowCount);
55 for (RowIndex row = 0; row < rowCount; ++row)
56 rows[row] = row;
57
58 const auto lexLess = [&](const RowIndex a, const RowIndex b) -> bool {
59 auto aIt = rowBegin(a);
60 auto bIt = rowBegin(b);
61 const auto aEnd = rowEnd(a);
62 const auto bEnd = rowEnd(b);
63 for (; aIt != aEnd && bIt != bEnd; ++aIt, ++bIt) {
64 if (*aIt < *bIt)
65 return true;
66 if (*aIt > *bIt)
67 return false;
68 ++aIt;
69 ++bIt;
70 }
71 return aIt == aEnd && bIt != bEnd;
72 };
73 std::sort(rows.begin(), rows.end(), lexLess);
74
75 // construct ordered with pivot columns in increasing order
76 ordered.clear();
77 for (size_t i = 0; i < rowCount; ++i)
78 ordered.appendRow(*this, rows[i]);
79 *this = std::move(ordered);
80}

152void SparseMatrix::appendRowAndNormalize(
153 const SparseMatrix& matrix,
154 const RowIndex row,
155 const Scalar modulus
156) {
157 MATHICGB_ASSERT(row < matrix.rowCount());
158 auto it = matrix.rowBegin(row);
159 const auto end = matrix.rowEnd(row);
160 if (it != end) {
161 appendEntry(it.index(), 1);
162 const Scalar lead = it.scalar();
163 ++it;
164 if (it != end) {
165 const Scalar inverse = modularInverse(lead, modulus);
166 do {
167 const uint32 prod = static_cast<uint32>(inverse) * it.scalar();
168 const uint16 prodMod = static_cast<uint16>(prod % modulus);
169 appendEntry(it.index(), prodMod);
170 ++it;
171 } while (it != end);
172 }
173 }
174 rowDone();
175}

177void SparseMatrix::appendRow(const SparseMatrix& matrix, const RowIndex row) {
178 MATHICGB_ASSERT(row < matrix.rowCount());
179
180 const auto size = matrix.entryCountInRow(row);
181 while (mBlock.mScalars.capacityToGo() < size)
182 growEntryCapacity();
183 MATHICGB_ASSERT(mBlock.mScalars.capacityToGo() ==
184 mBlock.mColIndices.capacityToGo());
185
186 auto const data = matrix.mRows[row];
187 mBlock.mScalars.memcpy(data.mScalarsBegin, size);
188 mBlock.mColIndices.memcpy(data.mIndicesBegin, size);
189 rowDone();
190}

211bool SparseMatrix::operator==(const SparseMatrix& matrix) const {
212 const auto count = rowCount();
213 if (count != matrix.rowCount())
214 return false;
215 for (RowIndex row = 0; row < count; ++row) {
216 if (entryCountInRow(row) != matrix.entryCountInRow(row))
217 return false;
218 const auto end = rowEnd(row);
219 auto matrixIt = matrix.rowBegin(row);
220 for (auto it = rowBegin(row); it != end; ++it, ++matrixIt)
221 if (*it != *matrixIt)
222 return false;
223 }
224 return true;
225}

267void SparseMatrix::appendRowWithModulusNormalized(
268 std::vector<uint64> const& v,
269 const Scalar modulus
270) {
271 uint16 multiply = 1;
272 bool first = true;
273 const auto count = static_cast<ColIndex>(v.size());
274 for (ColIndex col = 0; col < count; ++col) {
275 Scalar scalar = static_cast<Scalar>(v[col] % modulus);
276 if (scalar == 0)
277 continue;
278 if (first) {
279 multiply = modularInverse(scalar, modulus);
280 scalar = 1;
281 first = false;
282 } else {
283 uint32 prod = static_cast<uint32>(multiply) * scalar;
284 scalar = prod % modulus;
285 }
286 appendEntry(col, scalar);
287 }
288 rowDone();
289}

315void SparseMatrix::reserveFreeEntries(const size_t freeCount) {
316 if (freeCount <= mBlock.mColIndices.capacity() - mBlock.mColIndices.size())
317 return;
318 // We need to copy over the pending entries, so we need space for those
319 // entries on top of freeCount.
320 const size_t count = freeCount + ( // todo: detect overflow for this addition
321 mBlock.mHasNoRows ?
322 mBlock.mColIndices.size() :
323 std::distance(mRows.back().mIndicesEnd, mBlock.mColIndices.end())
324 );
325
326 // @todo: fix memory leaks on exception
327
328 auto oldBlock = new Block(std::move(mBlock));
329 MATHICGB_ASSERT(mBlock.mColIndices.begin() == 0);
330 MATHICGB_ASSERT(mBlock.mScalars.begin() == 0);
331 MATHICGB_ASSERT(mBlock.mHasNoRows);
332 MATHICGB_ASSERT(mBlock.mPreviousBlock == 0);
333
334 {
335 const auto begin = new ColIndex[count];
336 const auto capacityEnd = begin + count;
337 mBlock.mColIndices.releaseAndSetMemory(begin, begin, capacityEnd);
338 }
339
340 {
341 const auto begin = new Scalar[count];
342 const auto capacityEnd = begin + count;
343 mBlock.mScalars.releaseAndSetMemory(begin, begin, capacityEnd);
344 }
345
346 // copy pending entries over
347 if (oldBlock->mHasNoRows) {
348 mBlock.mColIndices.rawAssign
349 (oldBlock->mColIndices.begin(), oldBlock->mColIndices.end());
350 mBlock.mScalars.rawAssign
351 (oldBlock->mScalars.begin(), oldBlock->mScalars.end());
352 delete oldBlock; // no reason to keep it around
353 } else {
354 mBlock.mColIndices.rawAssign
355 (mRows.back().mIndicesEnd, oldBlock->mColIndices.end());
356 mBlock.mScalars.rawAssign
357 (mRows.back().mScalarsEnd, oldBlock->mScalars.end());
358
359 // remove the pending entries from old block so that counting the number
360 // of entries will give the correct result in future.
361 oldBlock->mColIndices.resize
362 (std::distance(oldBlock->mColIndices.begin(), mRows.back().mIndicesEnd));
363 oldBlock->mScalars.resize
364 (std::distance(oldBlock->mScalars.begin(), mRows.back().mScalarsEnd));
365 mBlock.mPreviousBlock = oldBlock;
366 }
367}

369void SparseMatrix::growEntryCapacity() {
370 MATHICGB_ASSERT(mBlock.mColIndices.size() == mBlock.mScalars.size());
371 MATHICGB_ASSERT(mBlock.mColIndices.capacity() == mBlock.mScalars.capacity());
372 MATHICGB_ASSERT(mBlock.mColIndices.size() <= mBlock.mColIndices.capacity());
373
374 // TODO: handle overflow of arithmetic here
375 if (mMemoryQuantum != 0 &&
376 (!mBlock.mHasNoRows || mBlock.mPreviousBlock == 0)
377 )
378 reserveFreeEntries(mMemoryQuantum);
379 else if (mBlock.mColIndices.capacity() == 0)
380 reserveFreeEntries(1 << 14); // minimum block size
381 else {
382 // do this if the quantum is not set or if the quantum is too small
383 // to store a single row being built.
384 reserveFreeEntries(mBlock.mColIndices.capacity() * 2);
385 }
386
387 MATHICGB_ASSERT(mBlock.mColIndices.size() <= mBlock.mColIndices.capacity());
388
389 MATHICGB_ASSERT(mBlock.mColIndices.size() == mBlock.mScalars.size());
390}

467void SparseMatrix::write(const Scalar modulus, FILE* file) const {
468 const auto storedRowCount = rowCount();
469
470 writeOne(static_cast<uint32>(storedRowCount), file);
471 writeOne(static_cast<uint32>(computeColCount()), file);
472 writeOne(static_cast<uint32>(modulus), file);
473 writeOne(static_cast<uint64>(entryCount()), file);
474
475 // write scalars
476 for (SparseMatrix::RowIndex row = 0; row < storedRowCount; ++row) {
477 const auto count = entryCountInRow(row);
478 if (fwrite(&rowBegin(row).scalar(), sizeof(uint16), count, file) != count)
479 mathic::reportError("error while writing to file.");
480 }
481
482 // write indices
483 for (SparseMatrix::RowIndex row = 0; row < storedRowCount; ++row) {
484 const auto count = entryCountInRow(row);
485 if (fwrite(&rowBegin(row).index(), sizeof(uint32), count, file) != count)
486 mathic::reportError("error while writing to file.");
487 }
488
489 std::vector<uint32> entryCounts;
490 for (SparseMatrix::RowIndex row = 0; row < storedRowCount; ++row)
491 entryCounts.push_back(entryCountInRow(row));
492 writeMany<uint32>(entryCounts, file);
493}

495SparseMatrix::Scalar SparseMatrix::read(FILE* file) {
496 MATHICGB_ASSERT(file != 0);
497
498 const auto rowCount = readOne<uint32>(file);
499 [[maybe_unused]] const auto colCount = readOne<uint32>(file);
500 const auto modulus = readOne<uint32>(file);
501 const auto entryCount64 = readOne<uint64>(file);
502 if (entryCount64 > std::numeric_limits<size_t>::max())
503 throw std::bad_alloc();
504 const auto entryCount = static_cast<size_t>(entryCount64);
505
506 // Allocate memory to hold the matrix in one block.
507 clear();
508 reserveFreeEntries(entryCount);
509 mRows.reserve(rowCount);
510 MATHICGB_ASSERT(mBlock.mPreviousBlock == 0); // only one block
511
512 // @todo: we can read directly into the block. Do that.
513
514 // Read scalars.
515 {
516 mBlock.mScalars.resize(entryCount);
517 std::vector<uint16> scalars;
518 readMany(file, entryCount, scalars);
519 std::copy(scalars.begin(), scalars.end(), mBlock.mScalars.begin());
520 }
521
522 // Read column indices.
523 {
524 mBlock.mColIndices.resize(entryCount);
525 std::vector<uint32> indices;
526 readMany(file, entryCount, indices);
527 std::copy(indices.begin(), indices.end(), mBlock.mColIndices.begin());
528 }
529
530 // Read where rows begin and end.
531 {
532 std::vector<uint32> sizes;
533 readMany(file, rowCount, sizes);
534 uint32 runningOffset = 0;
535 for (auto it = sizes.begin(); it != sizes.end(); ++it) {
536 Row row;
537 row.mIndicesBegin = mBlock.mColIndices.begin() + runningOffset;
538 row.mScalarsBegin = mBlock.mScalars.begin() + runningOffset;
539 runningOffset += *it;
540 row.mIndicesEnd = mBlock.mColIndices.begin() + runningOffset;
541 row.mScalarsEnd = mBlock.mScalars.begin() + runningOffset;
542 mRows.push_back(row);
543 }
544 MATHICGB_ASSERT(runningOffset == entryCount);
545 }
546
547 MATHICGB_ASSERT(mBlock.mPreviousBlock == 0); // still only one block
548 return modulus;
549}

551void SparseMatrix::writePBM(FILE* file) {
552 // See http://netpbm.sourceforge.net/doc/pbm.html
553
554 const auto rowCount = this->rowCount();
555 auto colCount = this->computeColCount();
556 colCount += 8-(colCount % 8);
557
558 // Write PBM header
559 {
560 std::stringstream out;
561 out << "P1 " << colCount << ' ' << rowCount << '\n';
562 fputs(out.str().c_str(), file);
563 }
564
565 for (RowIndex row = 0; row < rowCount; ++row) {
566 const auto end = rowEnd(row);
567 auto it = rowBegin(row);
568
569 for (ColIndex col = 0; col < colCount; ++col) {
570 if (it != end && col == it.index()) {
571 fputc('1', file);
572 ++it;
573 } else
574 fputc('0', file);
575 }
576 fputc('\n', file);
577 }
578}
