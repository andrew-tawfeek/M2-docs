177

179
180 void reset();
181 // Clear out the hash table, resetting all values to 0, and
182 // all counting values back to 0 (count,nclashes,max_run_length)
183 // BUT: the size is kept the same.
184

191 // displays on stderr some info about the hash table

193
194 void show() const;
195 // displays the hash table
196 // and then each value is displayed, with hash value.
197 // in form [loc, hash, value]
