271 inline void* Arena::alloc(size_t size) {
272 // It is OK to check capacity before aligning size as capacity is aligned.
273 // This single if checks for three different special circumstances:
274 // * size is 0 (size - 1 will overflow)
275 // * there is not enough capacity (size > capacity)
276 // * aligning size would cause an overflow (capacity is aligned)
277 const size_t capacity = block().getBytesToRight();
278 MEMT_ASSERT(capacity % MemoryAlignment == 0);
279 if (size - 1 >= capacity) {
280 MEMT_ASSERT(size == 0 || size > capacity);
281 if (size == 0) {
282 size = 1;
283 if (capacity > 0)
284 goto capacityOK;
285 }
286 growCapacity(size);
287 }
288 capacityOK:
289 MEMT_ASSERT(0 < size);
290 MEMT_ASSERT(size <= block().getBytesToRight());
291 MEMT_ASSERT(MemoryBlocks::alignNoOverflow(size) <= block().getBytesToRight());
292
293 char* ptr = block().position();
294 block().setPosition(ptr + MemoryBlocks::alignNoOverflow(size));
295
296#ifdef MEMT_DEBUG
297 _debugAllocs.push_back(ptr);
298#endif
299 return ptr;
300 }

332 template<class T>
333 std::pair<T*, T*> Arena::allocArrayNoCon(size_t elementCount) {
334 if (elementCount > static_cast<size_t>(-1) / sizeof(T))
335 throw std::bad_alloc();
336 const size_t size = elementCount * sizeof(T);
337 MEMT_ASSERT(size / sizeof(T) == elementCount);
338 char* buffer = static_cast<char*>(alloc(size));
339 T* array = reinterpret_cast<T*>(buffer);
340 T* arrayEnd = reinterpret_cast<T*>(buffer + size);
341 return std::make_pair(array, arrayEnd);
342 }

346 std::pair<T*, T*> p = allocArrayNoCon<T>(elementCount);
347 T* it = p.first;

350 new (it) T();

353 freeTopArray<T>(p.first, it);

383 inline void* Arena::guardPoint() {
384 // Supporting guard points is significantly more tricky than it
385 // may at first seem. Do not alter guardPoint() and
386 // restoreToGuardPoint() unless you are certain you understand the
387 // code completely and have carefully considered the change.
388 //
389 // We cannot allow the guard point to be inside an empty block
390 // since such blocks can be deallocated and that memory could then
391 // potentially be somewhere inside a later-allocated block. This
392 // problem does not appear for a non-empty block because the guard
393 // point is invalidated if any of the allocations in that block
394 // are later freed.
395
396 if (!block().empty()) {
397 MEMT_ASSERT_NO_ASSUME(!_debugAllocs.empty());
398 MEMT_ASSERT_NO_ASSUME(block().isInBlock(_debugAllocs.back()));
399 MEMT_ASSERT_NO_ASSUME(_debugAllocs.back() < block().position());
400 return block().position(); // no problems in this case
401 }
402
403 if (block().hasPreviousBlock()) {
404 // The previous block is not empty as then it would have been
405 // deallocated, so it is safe to take the guard point from the
406 // previous block.
407 MEMT_ASSERT(!block().previousBlock()->empty());
408 MEMT_ASSERT_NO_ASSUME(!_debugAllocs.empty());
409 MEMT_ASSERT_NO_ASSUME
410 (block().previousBlock()->isInBlock(_debugAllocs.back()));
411 MEMT_ASSERT_NO_ASSUME
412 (_debugAllocs.back() < block().previousBlock()->position());
413 return block().previousBlock()->position();
414 } else {
415 // Here the arena is empty, so there is no non-empty block.
416 MEMT_ASSERT(isEmpty());
417 MEMT_ASSERT_NO_ASSUME(_debugAllocs.empty());
418 // Return null to indicate empty arena.
419 return 0;
420 }
421 }

423 inline void Arena::restoreToGuardPoint(void* ptr) {
424 if (ptr == static_cast<void const*>(0)) { // null indicates empty arena
425 freeAllAllocs();
426 return;
427 }
428
429 Block* b = &block();
430 while (!(b->begin() < ptr && ptr <= b->end())) {
431 b->setPosition(b->begin());
432 b = b->previousBlock();
433
434 // If you get an assert here then most likely the guard point was
435 // invalidated by deallocating memory that was live at the point
436 // the guard point was created.
437 MEMT_ASSERT(b != 0);
438 }
439 MEMT_ASSERT(b->begin() < ptr && ptr <= b->end());
440 b->setPosition(ptr);
441
442#ifdef MEMT_DEBUG // update _debugAllocs
443 // there is always at least one allocation in the block of the
444 // guard ptr, so _debugAllocs will have a pointer in that block.
445 while (!_debugAllocs.empty() && !b->isInBlock(_debugAllocs.back()))
446 _debugAllocs.pop_back();
447 MEMT_ASSERT(!_debugAllocs.empty());
448 while (!_debugAllocs.empty() && _debugAllocs.back() >= ptr) {
449 MEMT_ASSERT(b->isInBlock(_debugAllocs.back()));
450 _debugAllocs.pop_back();
451 }
452 MEMT_ASSERT(!_debugAllocs.empty());
453 MEMT_ASSERT(b->isInBlock(_debugAllocs.back()));
454 MEMT_ASSERT(_debugAllocs.back() < ptr);
455#endif
456 }
