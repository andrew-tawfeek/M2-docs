33// expect(i,j) encodes a matrix with interesting bit patterns that
34// are supposed to be likely to surface errors in how monomials are
35// stored inside a vector.
36uint32 expect(size_t mono, size_t var, size_t varCount) {
37 const auto unique = (static_cast<uint32>(var + varCount * mono + 1) % 127);
38
39 while (true) {
40 // 000
41 if (mono == 0)
42 return 0;
43 --mono;
44
45 // 100
46 // 010
47 // 001
48 if (mono < varCount)
49 return var == mono ? unique : 0;
50 mono -= varCount;
51
52 // 000
53 // 100
54 // 110
55 // 111
56 if (mono < varCount + 1)
57 return var < mono ? unique : 0;
58 mono -= varCount + 1;
59
60 // 111
61 // 011
62 // 001
63 // 000
64 if (mono < varCount + 1)
65 return var >= mono ? unique : 0;
66 mono -= varCount + 1;
67
68 // 101
69 // 010
70 if (mono < 4)
71 return (var % 2) == (mono % 2) ? unique : 0;
72 mono -= 4;
73
74 // 100100
75 // 010010
76 // 001001
77 if (mono < 6)
78 return (var % 3) == (mono % 3) ? unique : 0;
79 mono -= 6;
80
81 // mix the patterns
82 mono += var % 17;
83 }
84};

95template<class Monoid, class MonoVector>
96void testMonoVector() {
97 typedef typename Monoid::VarIndex VarIndex;
98
99 Monoid monoid(13);
100 MonoVector v(monoid);
101 MonoVector v2(monoid);
102 ASSERT_EQ(v2.monoid(), monoid);
103 const auto varCount = monoid.varCount();
104
105 ASSERT_TRUE(v.empty());
106 size_t count = 1000;
107
108
109 // Not a correctness error, but empty vectors should preferably not
110 // use any memory.
111 ASSERT_EQ(0, v.memoryBytesUsed());
112
113 for (size_t i = 0; i < count; ++i) {
114 ASSERT_EQ(i, v.size());
115 v.push_back(); // push_back, no param
116 ASSERT_GT(v.memoryBytesUsed(), 0u);
117 ASSERT_FALSE(v.empty()); // empty
118 ASSERT_EQ(i + 1, v.size()); // size
119
120
121 ASSERT_TRUE(monoid.isIdentity(v.back())); // isIdentity true, back non-const
122 bool allZero = true;
123 for (VarIndex var = 0; var < varCount; ++var) {
124 const auto exponent = expect(i, var, varCount);
125 if (exponent != 0) {
126 allZero = false;
127 monoid.setExponent(var, exponent, v.back());
128 }
129 }
130 ASSERT_EQ(allZero, monoid.isIdentity(v.back())); // isIdentity false
131 v2.push_back(v.back()); // push_back with param
132 ASSERT_TRUE(monoid.equal(v.back(), v2.back()));
133 }
134 auto it = v.begin();
135 ASSERT_EQ(it, v.cbegin());
136 for (size_t i = 0; i < count; ++i, ++it) {
137 typename MonoVector::const_iterator tmp;
138 tmp = it;
139 ASSERT_EQ(tmp, it);
140 ASSERT_TRUE(v.end() != it);
141
142 for (VarIndex var = 0; var < monoid.varCount(); ++var) {
143 ASSERT_EQ(expect(i, var, varCount), monoid.exponent(*it, var));
144 }
145 }
146 ASSERT_EQ(v.end(), it);
147 ASSERT_EQ(v.cend(), it);
148
149 ASSERT_EQ(v, v2); // operator== true
150 monoid.setExponent(0, 1 + monoid.exponent(v2.back(), 0), v2.back());
151 ASSERT_TRUE(v != v2); // operator!=, true, same length
152
153 auto& vc = const_cast<const MonoVector&>(v);
154 ASSERT_TRUE(monoid.equal(v.front(), *v2.begin())); // front, non-const
155 ASSERT_TRUE(monoid.equal(vc.front(), *v2.begin())); // front, const
156 ASSERT_TRUE(monoid.equal(vc.back(), v.back())); // back, non-const
157
158 auto v3(v2); // copy constructor
159 ASSERT_EQ(v3.monoid(), monoid);
160 ASSERT_TRUE(v != v3 && v2 == v3);
161 v2.swap(v); // member swap
162 ASSERT_TRUE(v == v3 && v2 != v3);
163 std::swap(v, v2); // std::swap
164 ASSERT_TRUE(v != v3 && v2 == v3);
165 using std::swap;
166 swap(v, v2); // let compiler decide which swap to use
167 ASSERT_TRUE(v == v3 && v2 != v3);
168 swap(v, v2); // get back to original state
169 ASSERT_TRUE(v != v3 && v2 == v3);
170
171 ASSERT_FALSE(v3 != v2); // operator!=, false, same length
172 v3.push_back();
173 ASSERT_TRUE(v3 != v2); // operator!=, true, different length
174
175
176 ASSERT_FALSE(v3 == v);
177 v3 = v; // copy assignment
178 ASSERT_EQ(v3.monoid(), monoid);
179 ASSERT_EQ(v3, v);
180
181 ASSERT_FALSE(v3.empty());
182 v2 = std::move(v3); // move assignment
183 ASSERT_EQ(v2.monoid(), monoid);
184 ASSERT_EQ(v2, v);
185 ASSERT_TRUE(v3.empty());
186
187 ASSERT_FALSE(v2.empty());
188 auto v4(std::move(v2)); // move constructor
189 ASSERT_EQ(v4.monoid(), monoid);
190 ASSERT_TRUE(v2.empty());
191 ASSERT_EQ(v4, v);
192
193 ASSERT_FALSE(v.empty());
194 v.clear();
195 ASSERT_TRUE(v.empty());
196}

217TYPED_TEST(Monoids, ReadWriteMonoid) {
218 typedef TypeParam Monoid;
219 typedef typename Monoid::VarIndex VarIndex;
220
221 const auto check = [](
222 const char* const inStr,
223 const char* const outStr,
224 const VarIndex varCount,
225 const VarIndex gradingCount
226 ) -> void {
227 for (int i = 0; i < 2; ++i) {
228 const char* str = i == 0 ? inStr : outStr;
229 if (str == 0)
230 continue;
231
232 std::istringstream inStream(str);
233 Scanner in(inStream);
234 const auto order = MathicIO<Monoid>().
235 readOrderWithVarCount(Monoid::HasComponent, in);
236 const Monoid monoid(order);
237 const auto constructedOrder =
238 monoid.makeOrder
239 (order.componentsAscendingDesired(), order.schreyering());
240 std::ostringstream out;
241 MathicIO<Monoid>().writeOrderWithVarCount
242 (constructedOrder, Monoid::HasComponent, out);
243 ASSERT_EQ(outStr, out.str());
244 ASSERT_EQ(varCount, order.varCount());
245 ASSERT_EQ(varCount, monoid.varCount());
246 ASSERT_EQ(varCount, constructedOrder.varCount());
247 ASSERT_EQ(gradingCount, monoid.gradingCount());
248 }
249 };
250 check("0 0\n", "0\nrevlex 0\n", 0, 0);
251 check("1 1\n 2\n", "1\nrevlex 1\n 2\n", 1, 1);
252 check("1 2\n 3\n 4\n", "1\nrevlex 2\n 3\n 4\n", 1, 2);
253 check("2 2\n 3 4\n 5 6\n", "2\nrevlex 2\n 3 4\n 5 6\n", 2, 2);
254 check("4 1\n 1 1 1 1\n", "4\nrevlex 1\n 1 1 1 1\n", 4, 1);
255
256 check("0 lex 0", "0\nlex 0\n", 0, 0);
257 check("1 lex 1 2", "1\nlex 1\n 2\n", 1, 1);
258 check("1 lex 2 3 4", "1\nlex 2\n 3\n 4\n", 1, 2);
259 check("2 lex 2 3 4 5 6", "2\nlex 2\n 3 4\n 5 6\n", 2, 2);
260 check("4 lex 1 1 1 1 1", "4\nlex 1\n 1 1 1 1\n", 4, 1);
261
262 if (Monoid::HasComponent) {
263 check("2 2\n component\n 5 6\n", "2\nrevlex 2\n component\n 5 6\n", 2, 2);
264 check
265 ("2 2\n 3 4\n revcomponent\n","2\nrevlex 2\n 3 4\n revcomponent\n", 2, 2);
266 check("0 lex 1 component", "0\nlex 1\n component\n", 0, 1);
267 check("1 lex 1 revcomponent", "1\nlex 1\n revcomponent\n", 1, 1);
268 check("5 lex 1 revcomponent", "5\nlex 1\n revcomponent\n", 5, 1);
269 }

272TYPED_TEST(Monoids, MonoPool) {
273 typedef TypeParam Monoid;
274 typedef typename Monoid::VarIndex VarIndex;
275 typedef typename Monoid::Mono Mono;
276
277 for (int q = 0; q < 2; ++q) {
278 Monoid monoid(13);
279 typename Monoid::MonoPool pool(monoid);
280 const auto varCount = monoid.varCount();
281
282 const auto count = 1000;
283 std::vector<Mono> monos;
284 for (int i = 0; i < count; ++i) {
285 pool.alloc();
286 pool.free(pool.alloc());
287 auto m1 = pool.alloc();
288 ASSERT_TRUE(monoid.isIdentity(*m1));
289 auto m2 = pool.alloc();
290 ASSERT_TRUE(monoid.isIdentity(*m2));
291 for (VarIndex var = 0; var < varCount; ++var) {
292 monoid.setExponent(var, 1, *m1);
293 monoid.setExponent(var, 1, *m2);
294 }
295 if (i > 10) {
296 using std::swap;
297 swap(m2, monos[i - 10]);
298 }
299 monos.emplace_back(std::move(m1));
300 }
301
302 // This ensures that we get to each entry in monos exactly once.
303 MATHICGB_ASSERT((count % 17) != 0);
304 int i = 0;
305 do {
306 MATHICGB_ASSERT(!monos[i].isNull());
307 ASSERT_FALSE(monoid.isIdentity(*monos[i]));
308 pool.free(std::move(monos[i]));
309 ASSERT_TRUE(monos[i].isNull());
310 pool.free(std::move(monos[i]));
311 ASSERT_TRUE(monos[i].isNull());
312 i = (i + 17) % count;
313 } while (i != 0);
314
315 // If the ordering of monomials inside the pool has anything to do with
316 // allocation and deallocation order, then the monomials inside the
317 // pool are at this point all jumbled around. All the entries were also
318 // non-zero before, so we test that new allocations are the identity.
319
320 for (int i = 0; i < count; ++i) {
321 monos[i] = pool.alloc();
322 ASSERT_TRUE(monoid.isIdentity(*monos[i]));
323 for (VarIndex var = 0; var < varCount; ++var)
324 monoid.setExponent(var, expect(i, var, varCount), *monos[i]);
325 }
326 for (int i = 0; i < count; ++i) {
327 for (VarIndex var = 0; var < varCount; ++var) {
328 ASSERT_EQ(expect(i, var, varCount), monoid.exponent(*monos[i], var));
329 }
330 }
331 // everything should be free'd now. Let's do all that again.
332 }
333}

421TYPED_TEST(Monoids, MultiplyDivide) {
422 typedef TypeParam Monoid;
423 Monoid m(49);
424 typename Monoid::MonoPool pool(m);
425 auto monoOwner = pool.alloc();
426 auto mono = *monoOwner;
427 auto check = [&](const char* const str, const bool component) -> void {
428 if (component && !Monoid::HasComponent)
429 return;
430 auto v = parseVector(m, str, component);
431 MATHICGB_ASSERT(v.size() == 3);
432 const auto& a = v.front();
433 const auto& b = *++v.begin();
434 const auto& c = v.back();
435 ASSERT_EQ(m.hashOfProduct(a, b), m.hash(c));
436 ASSERT_EQ(m.hashOfProduct(a, b), m.hashOfProduct(b, a));
437
438 // isProductOf
439 ASSERT_TRUE(m.isProductOf(a, b, c));
440 ASSERT_TRUE(m.isProductOfHintTrue(a, b, c));
441 ASSERT_TRUE(m.isTwoProductsOfHintTrue(a, a, b, c, c));
442
443
444 // a*b == c using multiply
445 m.multiply(a, b, mono);
446 ASSERT_TRUE(m.equal(c, mono));
447 ASSERT_TRUE(m.compare(c, mono) == Monoid::EqualTo);
448 ASSERT_EQ(m.hash(c), m.hash(mono));
449
450 // c/a == b using divide
451 m.divide(a, c, mono);
452 ASSERT_TRUE(m.equal(b, mono));
453 ASSERT_TRUE(m.compare(b, mono) == Monoid::EqualTo);
454 ASSERT_EQ(m.hash(b), m.hash(mono));
455
456 // c/b == a using divideInPlace
457 m.copy(c, mono);
458 m.divideInPlace(b, mono);
459 ASSERT_TRUE(m.equal(a, mono));
460 ASSERT_TRUE(m.compare(a, mono) == Monoid::EqualTo);
461 ASSERT_EQ(m.hash(a), m.hash(mono));
462
463 // a*b == c using multiplyInPlace
464 m.copy(a, mono);
465 m.multiplyInPlace(b, mono);
466 ASSERT_TRUE(m.equal(c, mono));
467 ASSERT_TRUE(m.compare(c, mono) == Monoid::EqualTo);
468 ASSERT_EQ(m.hash(c), m.hash(mono));
469
470 // divides, check properties that mono=a*b should have
471 ASSERT_TRUE(m.divides(mono, c));
472 ASSERT_TRUE(m.divides(c, mono));
473 ASSERT_TRUE(m.divides(a, mono));
474 ASSERT_TRUE(m.divides(b, mono));
475
476 // divides, general
477 ASSERT_TRUE(m.divides(m, mono, c));
478 ASSERT_TRUE(m.divides(m, c, mono));
479 ASSERT_TRUE(m.divides(m, a, mono));
480 ASSERT_TRUE(m.divides(m, b, mono));
481
482 if (!m.isIdentity(a)) {
483 ASSERT_TRUE(m.lessThan(b, mono));
484 ASSERT_FALSE(m.lessThan(mono, b));
485 ASSERT_TRUE(m.compare(mono, b) == Monoid::GreaterThan);
486 ASSERT_FALSE(m.divides(mono, b));
487 ASSERT_FALSE(m.divides(m, mono, b));
488
489 ASSERT_FALSE(m.isProductOf(a, c, b));
490 ASSERT_FALSE(m.isProductOfHintTrue(a, c, b));
491 ASSERT_FALSE(m.isTwoProductsOfHintTrue(c, c, a, b, b));
492 ASSERT_FALSE(m.isTwoProductsOfHintTrue(b, c, a, c, b));
493 ASSERT_FALSE(m.isTwoProductsOfHintTrue(c, b, a, b, c));
494 } else {
495 ASSERT_TRUE(m.equal(b, mono));
496 ASSERT_TRUE(m.compare(b, mono) == Monoid::EqualTo);
497 ASSERT_TRUE(m.divides(mono, b));
498 ASSERT_TRUE(m.divides(m, mono, b));
499 }
500
501 if (!m.isIdentity(b)) {
502 ASSERT_TRUE(m.lessThan(a, mono));
503 ASSERT_FALSE(m.lessThan(mono, a));
504 ASSERT_TRUE(m.compare(mono, a) == Monoid::GreaterThan);
505 ASSERT_FALSE(m.divides(mono, a));
506 ASSERT_FALSE(m.divides(m, mono, a));
507
508 ASSERT_FALSE(m.isProductOf(c, b, a));
509 ASSERT_FALSE(m.isProductOfHintTrue(b, c, a));
510 ASSERT_FALSE(m.isTwoProductsOfHintTrue(c, c, b, a, a));
511 ASSERT_FALSE(m.isTwoProductsOfHintTrue(a, c, b, c, a));
512 ASSERT_FALSE(m.isTwoProductsOfHintTrue(c, a, b, a, c));
513 } else {
514 ASSERT_TRUE(m.equal(a, mono));
515 ASSERT_TRUE(m.compare(a, mono) == Monoid::EqualTo);
516 ASSERT_TRUE(m.divides(m, mono, a));
517 }
518
519 // Check that aliased parameters work.
520 m.multiply(mono, mono, mono);
521 m.divide(mono, mono, mono);
522 MATHICGB_ASSERT(m.isIdentity(mono));
523
524 // Check that negative exponents work.
525 if (Monoid::HasComponent && m.component(a) != m.component(b))
526 return;
527 m.divideToNegative(a, b, mono);
528 m.multiply(a, mono, mono);
529 ASSERT_TRUE(m.equal(mono, b));
530
531 m.divideToNegative(b, a, mono);
532 m.multiply(b, mono, mono);
533 ASSERT_TRUE(m.equal(mono, a));
534 };
535 check("1 1 1", false);
536 check("a<5> 1<0> a<5>", true);
537 check("1 Vx Vx", false);
538 check("aV bx abxV", false);
539 check("a a2 a3", false);
540 check("V<2> V2<0> V3<2>", true);
541 check("arlgh svug arlg2hsvu", false);
542 check
543 ("abcdefghiV<7> ab2c3d4e5f6g7h8i9V11<0> a2b3c4d5e6f7g8h9i10V12<7>", true);
544}

546TYPED_TEST(Monoids, LcmColon) {
547 typedef TypeParam Monoid;
548 Monoid mNonConst(49);
549 auto& m = mNonConst;
550 typename Monoid::MonoPool pool(m);
551 auto monoOwner = pool.alloc();
552 auto mono = *monoOwner;
553 auto mono2Owner = pool.alloc();
554 auto mono2 = *mono2Owner;
555 auto check = [&](const char* const str, const bool component) -> void {
556 if (component && !Monoid::HasComponent)
557 return;
558 auto v = parseVector(m, str, component);
559 MATHICGB_ASSERT(v.size() == 3);
560 const auto& a = v.front();
561 const auto& b = *++v.begin();
562 const auto& lcm = v.back();
563
564 // isLcm (+general)
565 ASSERT_TRUE(m.isLcm(a, b, lcm));
566 ASSERT_TRUE(m.isLcm(m, a, m, b, lcm));
567 m.copy(lcm, mono);
568 m.setExponent(1, m.exponent(mono, 1) + 1, mono);
569 ASSERT_FALSE(m.isLcm(a, b, mono));
570 ASSERT_FALSE(m.isLcm(m, a, m, b, mono));
571
572 // dividesLcm
573 ASSERT_TRUE(m.dividesLcm(lcm, a, b));
574 ASSERT_FALSE(m.dividesLcm(mono, a, b));
575 ASSERT_TRUE(m.dividesLcm(a, a, a));
576 ASSERT_TRUE(m.dividesLcm(a, a, b));
577 ASSERT_TRUE(m.dividesLcm(b, b, b));
578 ASSERT_TRUE(m.dividesLcm(b, b, a));
579
580 // dividesLcm, general
581 ASSERT_TRUE(m.dividesLcm(m, lcm, m, a, b));
582 ASSERT_FALSE(m.dividesLcm(m, mono, m, a, b));
583 ASSERT_TRUE(m.dividesLcm(m, a, m, a, a));
584 ASSERT_TRUE(m.dividesLcm(m, a, m, a, b));
585 ASSERT_TRUE(m.dividesLcm(m, b, m, b, b));
586 ASSERT_TRUE(m.dividesLcm(m, b, m, b, a));
587
588 // lcm(a, b)
589 m.lcm(a, b, mono);
590 ASSERT_TRUE(m.equal(mono, lcm));
591 ASSERT_TRUE(m.compare(mono, lcm) == Monoid::EqualTo);
592 ASSERT_EQ(m.hash(lcm), m.hash(mono));
593
594 // lcm(b, a), general
595 m.lcm(m, b, m, a, mono);
596 ASSERT_TRUE(m.equal(mono, lcm));
597 ASSERT_TRUE(m.compare(mono, lcm) == Monoid::EqualTo);
598 ASSERT_EQ(m.hash(lcm), m.hash(mono));
599
600 // colons
601 m.colons(a, b, mono, mono2);
602 m.multiply(b, mono, mono);
603 m.multiply(a, mono2, mono2);
604 ASSERT_TRUE(m.equal(mono, lcm));
605 ASSERT_TRUE(m.compare(mono, lcm) == Monoid::EqualTo);
606 ASSERT_TRUE(m.equal(mono2, lcm));
607 ASSERT_TRUE(m.compare(mono2, lcm) == Monoid::EqualTo);
608 };
609 check("1 1 1", false);
610 check("a<2> 1<2> a<2>", true);
611 check("1 Vx Vx", false);
612 check("aV bx abxV", false);
613 check("a a2 a2", false);
614 check("V<3> V2<3> V2<3>", true);
615 check("arlgh svug arlghsvu", false);
616 check("a6b7c8d9efghiV ab2c3d4e5f6g7h8i9V11 a6b7c8d9e5f6g7h8i9V11", false);
