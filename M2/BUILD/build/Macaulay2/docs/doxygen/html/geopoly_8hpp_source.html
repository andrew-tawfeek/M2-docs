82inline void polyheap::add(Nterm *p)
83{
84 int len = F->n_terms(p);
85 int i = 0;
86 while (len >= heap_size[i]) i++;
87
88 ring_elem tmp1 = heap[i];
89 ring_elem tmp2 = p;
90 F->add_to(tmp1, tmp2);
91 heap[i] = tmp1;
92
93 len = F->n_terms(heap[i]);
94 p = NULL;
95 while (len >= heap_size[i])
96 {
97 i++;
98
99 tmp1 = heap[i];
100 tmp2 = heap[i - 1];
101 F->add_to(tmp1, tmp2);
102 heap[i] = tmp1;
103
104 len = F->n_terms(heap[i]);
105 heap[i - 1] = NULL;
106 }
107 if (i > top_of_heap) top_of_heap = i;
108}

110inline Nterm *polyheap::remove_lead_term()
111{
112 int lead_so_far = -1;
113 for (int i = 0; i <= top_of_heap; i++)
114 {
115 if (heap[i] == NULL) continue;
116 if (lead_so_far < 0)
117 {
118 lead_so_far = i;
119 continue;
120 }
121 int cmp = EQ; // F->compare(heap[lead_so_far], heap[i]);
122 if (cmp == GT) continue;
123 if (cmp == LT)
124 {
125 lead_so_far = i;
126 continue;
127 }
128 // At this point we have equality
129 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
130 Nterm *tmp = heap[i];
131 heap[i] = tmp->next;
132 tmp->next = NULL;
133 F->remove(reinterpret_cast<ring_elem &>(tmp));
134
135 if (K->is_zero(heap[lead_so_far]->coeff))
136 {
137 // Remove, and start over
138 tmp = heap[lead_so_far];
139 heap[lead_so_far] = tmp->next;
140 tmp->next = NULL;
141 F->remove(reinterpret_cast<ring_elem &>(tmp));
142 lead_so_far = -1;
143 i = -1;
144 }
145 }
146 if (lead_so_far < 0) return NULL;
147 Nterm *result = heap[lead_so_far];
148 heap[lead_so_far] = result->next;
149 result->next = NULL;
150 return result;
151}

153inline Nterm *polyheap::value()
154{

156 for (int i = 0; i <= top_of_heap; i++)
157 {
158 if (heap[i] == NULL) continue;
159 ring_elem tmp1 = result;
160 ring_elem tmp2 = heap[i];
161 F->add_to(tmp1, tmp2);
162 result = tmp1;
163 heap[i] = NULL;
164 }
165 top_of_heap = -1;
166 return result;
