42 // Construct a hash table with at least requestedBucketCount buckets. There
43 // may be more buckets. Currently the number is rounded up to the next power
44 // of two.
45 FixedSizeMonomialMap(
46 const size_t requestedBucketCount,
47 const PolyRing& ring
48 ):
49 mHashToIndexMask(computeHashMask(requestedBucketCount)),
50 mBuckets(
51 make_unique_array<Atomic<Node*>>(hashMaskToBucketCount(mHashToIndexMask))
52 ),
53 mRing(ring),
54 mNodeAlloc(Node::bytesPerNode(ring.monoid()))
55 {
56 // Calling new int[x] does not zero the array. std::atomic has a trivial
57 // constructor so the same thing is true of new atomic[x]. Calling
58 // new int[x]() is supposed to zero initialize but this apparently
59 // does not work on GCC. So we have to fill the table with nulls
60 // manually. This was wonderful to debug btw.
61 // We can store relaxed as the constructor does not run concurrently.
62 setTableEntriesToNullRelaxed();
63 }

73 FixedSizeMonomialMap(
74 const size_t requestedBucketCount,
75 FixedSizeMonomialMap<T>&& map
76 ):
77 mHashToIndexMask(computeHashMask(requestedBucketCount)),
78 mBuckets(
79 make_unique_array<Atomic<Node*>>(hashMaskToBucketCount(mHashToIndexMask))
80 ),
81 mRing(map.ring()),
82 mNodeAlloc(std::move(map.mNodeAlloc))
83 {
84 // We can store relaxed as the constructor does not run concurrently.
85 const auto relax = std::memory_order_relaxed;
86
87 setTableEntriesToNullRelaxed();
88 const auto tableEnd = map.mBuckets.get() + map.bucketCount();
89 for (auto tableIt = map.mBuckets.get(); tableIt != tableEnd; ++tableIt) {
90 for (Node* node = tableIt->load(); node != 0;) {
91 const size_t index = hashToIndex(monoid().hash(node->mono()));
92 const auto next = node->next(relax);
93 node->setNext(mBuckets[index].load(relax), relax);
94 mBuckets[index].store(node, relax);
95 node = next;
96 }
97 }

126 std::pair<const mapped_type*, ConstMonoPtr>
127 find(ConstMonoRef mono) const {
128 const auto monoHash = monoid().hash(mono);
129 const auto* node = bucketAtIndex(hashToIndex(monoHash));
130 for (; node != 0; node = node->next(std::memory_order_consume)) {
131 // To my surprise, it seems to be faster to comment out this branch.
132 // I guess the hash table has too few collisions to make it worth it.
133 //if (monoHash != mRing.monomialHashValue(node->mono))
134 // continue;
135 if (monoid().equalHintTrue(mono, node->mono()))
136 return std::make_pair(&node->value, node->mono().ptr());
137 }
138 return std::make_pair(nullptr, ConstMonoPtr());
139 }

141 // As find on the product a*b but also returns the monomial that is the
142 // product.
143 MATHICGB_INLINE
144 std::pair<const mapped_type*, ConstMonoPtr> findProduct(
145 ConstMonoRef a,
146 ConstMonoRef b
147 ) const {
148 const HashValue abHash = monoid().hashOfProduct(a, b);
149 const Node* node = bucketAtIndex(hashToIndex(abHash));
150 for (; node != 0; node = node->next(std::memory_order_consume)) {
151 // To my surprise, it seems to be faster to comment out this branch.
152 // I guess the hash table has too few collisions to make it worth it.
153 //if (abHash != mRing.monomialHashValue(node->mono))
154 // continue;
155 if (monoid().isProductOfHintTrue(a, b, node->mono()))
156 return std::make_pair(&node->value, node->mono().ptr());
157 }
158 return std::make_pair(nullptr, nullptr);
159 }

162 MATHICGB_INLINE
163 std::pair<const mapped_type*, const mapped_type*> findTwoProducts(
164 ConstMonoRef a1,
165 ConstMonoRef a2,
166 ConstMonoRef b
167 ) const {
168 const HashValue a1bHash = monoid().hashOfProduct(a1, b);
169 const HashValue a2bHash = monoid().hashOfProduct(a2, b);
170 const Node* const node1 = bucketAtIndex(hashToIndex(a1bHash));
171 const Node* const node2 = bucketAtIndex(hashToIndex(a2bHash));
172
173 if (
174 node1 != 0 &&
175 node2 != 0 &&
176 monoid().isTwoProductsOfHintTrue
177 (a1, a2, b, node1->mono(), node2->mono())
178 )
179 return std::make_pair(&node1->value, &node2->value);
180 else
181 return std::make_pair(findProduct(a1, b).first, findProduct(a2, b).first);
182 }

192 std::pair< std::pair<const mapped_type*, ConstMonoPtr>, bool>
193 insert(const value_type& value) {
194 const mtbb::lock_guard lockGuard(mInsertionMutex);
195
196 // find() loads buckets with memory_order_consume, so it may seem like
197 // we need some extra synchronization to make sure that we have the
198 // most up to date view of the bucket that value.first goes in -
199 // what if a pending insert is hiding in a cache of some other processor
200 // somewhere? We in fact have an up to date view of every bucket in the
201 // the table already because inserts only happen while holding the
202 // insertion mutex and by locking that mutex we have synchronized with
203 // all threads that previously did insertions.
204 {
205 const auto found = find(*value.first);
206 if (found.first != 0) {
207 auto p = std::make_pair(found.first, *found.second);
208 return std::make_pair(p, false); // key already present
209 }
210 }
211
212 const auto node = static_cast<Node*>(mNodeAlloc.alloc());
213 const size_t index = hashToIndex(monoid().hash(*value.first));
214 // the constructor initializes the first field of node->mono, so
215 // it has to be called before copying the monomial.
216 new (node) Node(bucketAtIndex(index), value.second);
217 monoid().copy(*value.first, node->mono());
218
219 // we cannot store with memory_order_relaxed here because unlocking the
220 // lock only synchronizes with threads who later grab the lock - it does
221 // not synchronize with reading threads since they do not grab the lock.
222 mBuckets[index].store(node, std::memory_order_release);
223
224 auto p = std::make_pair(&node->value, node->constMono());
225 return std::make_pair(p, true); // successful insertion
226 }

231 void clearNonConcurrent() {
232#ifdef MATHICGB_DEBUG
233 // requires mutual exclusion from both readers and writers, but we can
234 // only assert on this for the writers.
235 if (mInsertionMutex.try_lock())
236 mInsertionMutex.unlock();
237 else {
238 MATHICGB_ASSERT(false);
239 }
240#endif
241 // we can store relaxed as the client supplies synchronization.
242 setTableEntriesToNullRelaxed();
243
244 // This is the reason that we cannot support this operation concurrently -
245 // we have no way to know when it is safe to deallocate the monomials
246 // since readers do no synchronization.
247 mNodeAlloc.freeAllBuffers();
248 }

252 const auto tableEnd = mBuckets.get() + bucketCount();
253 for (auto tableIt = mBuckets.get(); tableIt != tableEnd; ++tableIt)
254 tableIt->store(0, std::memory_order_relaxed);

282 static HashValue computeHashMask(const size_t requestedBucketCount) {
283 // round request up to nearest power of 2.
284 size_t pow2 = 1;
285 while (pow2 < requestedBucketCount && 2 * pow2 != 0)
286 pow2 *= 2;
287 MATHICGB_ASSERT(pow2 > 0 && (pow2 & (pow2 - 1)) == 0); // power of two
288
289 // If casting to a hash value overflows, then we get the maximum
290 // possible number of buckets based on the range of the hash
291 // value type. Only unsigned overflow is defined, so we need
292 // to assert that the hash type is unsigned.
293 static_assert(!std::numeric_limits<HashValue>::is_signed, "");
294 const auto hashToIndexMask = static_cast<HashValue>(pow2 - 1);
295 MATHICGB_ASSERT(pow2 == hashMaskToBucketCount(hashToIndexMask));
296 return hashToIndexMask;
297 }
298
299 static size_t hashMaskToBucketCount(const HashValue mask) {
300 const auto count = static_cast<size_t>(mask) + 1u; // should be power of 2
301 MATHICGB_ASSERT(count > 0 && (count & (count - 1)) == 0);
302 return count;
303 }
