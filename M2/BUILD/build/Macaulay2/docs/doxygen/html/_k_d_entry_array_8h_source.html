146 template<class C, class EE>
147 template<class Iter>
148 Iter KDEntryArray<C, EE>::split(
149 Iter begin,
150 Iter end,
151 size_t& var,
152 Exponent& exp,
153 const C& conf,
154 const ExtEntry* extEntry
155 ) {
156 MATHIC_ASSERT(begin != end);
157 const size_t varCount = conf.getVarCount();
158 for (size_t i = 0; i < varCount; ++i) {
159 var = (var + 1) % conf.getVarCount();
160
161 typename C::Exponent min;
162 typename C::Exponent max;
163 if (extEntry == 0) {
164 min = conf.getExponent(getEntry(*begin), var);
165 max = conf.getExponent(getEntry(*begin), var);
166 } else {
167 min = conf.getExponent(getEntry(*extEntry), var);
168 max = conf.getExponent(getEntry(*extEntry), var);
169 }
170 for (Iter it = begin; it != end; ++it) {
171 min = std::min(min, conf.getExponent(getEntry(*it), var));
172 max = std::max(max, conf.getExponent(getEntry(*it), var));
173 }
174 if (min == max)
175 continue;
176 // this formula for the average avoids overflow
177 exp = min + (max - min) / 2;
178 SplitEqualOrLess cmp(var, exp, conf);
179 return std::partition(begin, end, cmp);
180 }
181 MATHIC_ASSERT(false);
182 throw std::logic_error("ERROR: Inserted duplicate entry into a KD tree.");
183 }

286 size_t KDEntryArray<C, EE>::removeMultiples
287 (const EM& monomial, MO& out, const C& conf) {
288 MATHIC_ASSERT(C::AllowRemovals);
289 if (C::LeafSize == 1) { // special case for performance
290 if (empty() || !monomial.divides(*begin(), conf))
291 return 0;
292 out.push_back(begin()->get());
293 pop_back();
294 return 1;
295 }
296
297 iterator it = begin();
298 iterator oldEnd = end();
299 for (; it != oldEnd; ++it) {
300 if (monomial.divides(*it, conf)) {
301 out.push_back(it->get());
302 break;
303 }
304 }
305 if (it == oldEnd)
306 return 0;
307 iterator newEnd = it;
308 for (++it; it != oldEnd; ++it) {
309 if (!monomial.divides(*it, conf)) {
310 *newEnd = *it;
311 ++newEnd;
312 } else
313 out.push_back(it->get());
314 }
315 // cannot just set _end directly as the superfluous
316 // entries at the end need to be destructed.
317 const size_t newSize = std::distance(begin(), newEnd);
318 const size_t removedCount = size() - newSize;
319 MATHIC_ASSERT(newSize < size());
320 do {
321 pop_back();
322 } while (newSize < size());
323 MATHIC_ASSERT(size() == newSize);
324 return removedCount;
325 }
