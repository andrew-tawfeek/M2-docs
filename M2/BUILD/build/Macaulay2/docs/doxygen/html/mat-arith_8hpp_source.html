180template <typename MatType>
181void normSquared(SubMatrix<MatType> M,
182 typename MatType::CoeffRing::RealElementType& result)
183{
184 const auto& C = M.matrix.ring();
185 const auto& R = C.real_ring();
186 typename MatType::CoeffRing::RealRingType::Element c(R);
187 R.set_zero(result);
188 for (long rA = M.begin_row; rA < M.end_row; ++rA)
189 for (long cA = M.begin_column; cA < M.end_column; ++cA)
190 {
191 C.abs_squared(c, M.matrix.entry(rA, cA));
192 R.add(result, result, c);
193 }
194}
