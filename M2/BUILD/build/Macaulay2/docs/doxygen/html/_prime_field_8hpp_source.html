109 template<class Integer>
110 Element toElement(Integer&& i) const {
111 typedef typename std::remove_reference<Integer>::type NoRefInteger;
112 static_assert(std::numeric_limits<NoRefInteger>::is_integer, "");
113
114 // We need to take the modulus of i to put it into the range [0;charac()).
115 // That is more tricky to get right than it might seem.
116 //
117 // The sign of a % b is implementation defined in C++ if either of a or b
118 // are negative. We need the positive remainder so the operands have to be
119 // positive. Another reason for this is that we could not allow b to be
120 // converted to a signed integer since it might not be representable that
121 // way.
122 //
123 // If Integer is signed and i is std::numeric_limits<Integer>::min() then
124 // it is undefined behavior to evaluate the expression -i since -i is not
125 // representable, leading to a signed overflow. So we have to cast to
126 // unsigned before doing the minus.
127 typedef typename std::make_unsigned<NoRefInteger>::type Unsigned;
128 if (i < 0) {
129 // Negate i to get a positive number, then negate again to cancel out
130 // the first negation. The first cast to unsigned is to avoid
131 // undefined behavior from -i. The second is there because apparently,
132 // at least on GCC, the unary - re-introduces signed and we need to
133 // be unsigned to avoid sign extension. We need zero extension.
134 const auto unsignedNegative =
135 static_cast<Unsigned>(-static_cast<Unsigned>(i));
136 return negative(Element(unsignedNegative % charac()));
137 } else
138 return Element(static_cast<Unsigned>(i) % charac());
139 }

229template<class T>
230auto PrimeField<T>::inverse(const Element elementA) const -> Element {
231 // We do two turns of the extended Euclidian algorithm per
232 // loop. Usually the sign of x changes each time through the loop,
233 // but we avoid that by representing every other x as its negative,
234 // which is the value minusLastX. This way no negative values ever
235 // appear and we do not need any extra bits.
236
237 auto a = elementA.value();
238 MATHICGB_ASSERT(0 < a);
239 MATHICGB_ASSERT(a < charac());
240 auto b = charac();
241 auto minusLastX = static_cast<T>(0);
242 auto x = static_cast<T>(1);
243 while (true) {
244 MATHICGB_ASSERT(x <= charac());
245 MATHICGB_ASSERT(minusLastX <= charac());
246
247 // first turn
248 if (a == 1)
249 break;
250 const auto firstQuotient = b / a;
251 b -= firstQuotient * a;
252 minusLastX += firstQuotient * x;
253
254 // second turn
255 if (b == 1) {
256 MATHICGB_ASSERT(!isZero(minusLastX));
257 MATHICGB_ASSERT(minusLastX < charac());
258 x = charac() - minusLastX;
259 break;
260 }
261 const auto secondQuotient = a / b;
262 a -= secondQuotient * b;
263 x += secondQuotient * minusLastX;
264 }
265 MATHICGB_ASSERT(x >= 1);
266 MATHICGB_ASSERT(x < charac());
267
268 const Element inverseElement(x);
269 MATHICGB_ASSERT(isOne(product(elementA, inverseElement)));
270 return inverseElement;
271}
