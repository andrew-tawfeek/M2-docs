330 template<class Iter>
331 void BinaryKDTree<C>::reset(Iter insertBegin, Iter insertEnd, const DivMaskCalculator& calc) {
332 clear();
333 if (insertBegin == insertEnd)
334 return;
335
336 typedef InsertTodo<Iter> Task;
337 typedef std::vector<Task> TaskCont;
338 TaskCont todo;
339
340 Interior* parent = 0;
341 bool isEqualOrLessChild = false;
342 while (true) {
343 Node* node = 0;
344 const size_t insertCount = std::distance(insertBegin, insertEnd);
345 const bool isLeaf = (insertCount <= C::LeafSize);
346 if (isLeaf)
347 node = new (_arena.allocObjectNoCon<Leaf>())
348 Leaf(insertBegin, insertEnd, _arena, calc, _conf);
349 else {
350 size_t var =
351 (parent == 0 ? static_cast<size_t>(-1) : parent->getVar());
352
353 Exponent exp;
354 Iter middle = KDEntryArray<C, ExtEntry>::
355 split(insertBegin, insertEnd, var, exp, _conf);
356 Interior* interior = new (_arena.allocObjectNoCon<Interior>())
357 Interior(var, exp);
358
359 MATHIC_ASSERT(middle != insertBegin && middle != insertEnd);
360 // push strictly-greater on todo
361 Task task;
362 task.begin = middle;
363 task.end = insertEnd;
364 task.parent = interior;
365 todo.push_back(task);
366 // set up equal-or-less
367 insertEnd = middle;
368 node = interior;
369 }
370
371 if (parent == 0) {
372 MATHIC_ASSERT(_root == 0);
373 _root = node;
374 } else if (isEqualOrLessChild)
375 parent->setEqualOrLess(node);
376 else
377 parent->setStrictlyGreater(node);
378
379 if (isLeaf) {
380 // grab next item from todo
381 if (todo.empty())
382 break;
383 Task task = todo.back();
384 todo.pop_back();
385 insertBegin = task.begin;
386 insertEnd = task.end;
387 parent = task.parent;
388 // only strictlyGreater goes on todo
389 isEqualOrLessChild = false;
390 } else {
391 isEqualOrLessChild = true;
392 parent = &node->asInterior();
393 // continue with equal-or-less as next item
394 }
395 }
396 MATHIC_ASSERT(_root != 0);
397
398 if (C::UseTreeDivMask) {
399 // record nodes in tree using breadth first search
400 typedef std::vector<Interior*> NodeCont;
401 NodeCont nodes;
402 if (_root->isInterior())
403 nodes.push_back(&_root->asInterior());
404 for (size_t i = 0; i < nodes.size(); ++i) {
405 Interior* node = nodes[i];
406 if (node->getEqualOrLess().isInterior())
407 nodes.push_back(&node->getEqualOrLess().asInterior());
408 if (node->getStrictlyGreater().isInterior())
409 nodes.push_back(&node->getStrictlyGreater().asInterior());
410 }
411 // compute div masks in reverse order of breath first search
412 typename NodeCont::reverse_iterator it = nodes.rbegin();
413 typename NodeCont::reverse_iterator end = nodes.rend();
414 for (; it != end; ++it) {
415 Interior* node = *it;
416 node->updateToLowerBound(node->getEqualOrLess());
417 node->updateToLowerBound(node->getStrictlyGreater());
418 }
419 }
420 MATHIC_ASSERT(debugIsValid());
421 }

423 template<class C>
424 typename BinaryKDTree<C>::Entry* BinaryKDTree<C>::findDivisor
425 (const ExtMonoRef& extMonomial) {
426
427 MATHIC_ASSERT(debugIsValid());
428 MATHIC_ASSERT(_tmp.empty());
429 if (_root == 0)
430 return 0;
431 Node* node = _root;
432 while (true) {
433 while (node->isInterior()) {
434 Interior& interior = node->asInterior();
435 if (C::UseTreeDivMask &&
436 !interior.getDivMask().canDivide(extMonomial.getDivMask()))
437 goto next;
438
439 if (interior.getExponent() <
440 _conf.getExponent(extMonomial.get(), interior.getVar()))
441 _tmp.push_back(&interior.getStrictlyGreater());
442 node = &interior.getEqualOrLess();
443 }
444
445 {
446 MATHIC_ASSERT(node->isLeaf());
447 Leaf& leaf = node->asLeaf();
448 LeafIt leafIt = leaf.entries().findDivisor(extMonomial, _conf);
449 if (leafIt != leaf.entries().end()) {
450 MATHIC_ASSERT(_conf.divides(leafIt->get(), extMonomial.get()));
451 _tmp.clear();
452 return &leafIt->get();
453 }
454 }
455 next:
456 if (_tmp.empty())
457 break;
458 node = _tmp.back();
459 _tmp.pop_back();
460 }
461 MATHIC_ASSERT(_tmp.empty());
462 return 0;

622 bool BinaryKDTree<C>::debugIsValid() const {
623#ifndef MATHIC_DEBUG
624 return true;
625#else
626 MATHIC_ASSERT(_tmp.empty());
627 MATHIC_ASSERT(!_conf.getDoAutomaticRebuilds() || _conf.getRebuildRatio() > 0);
628
629 if (_root == 0)
630 return true;
631
632 // record all nodes
633 std::vector<Node*> nodes;
634 nodes.push_back(_root);
635 size_t sizeSum = 0;
636 for (size_t i = 0; i < nodes.size(); ++i) {
637 Node* node = nodes[i];
638 if (node->isInterior()) {
639 MATHIC_ASSERT(node->asInterior().getVar() < _conf.getVarCount());
640 nodes.push_back(&node->asInterior().getStrictlyGreater());
641 nodes.push_back(&node->asInterior().getEqualOrLess());
642 } else
643 sizeSum += node->asLeaf().entries().size();
644 }
645
646 // check the recorded nodes
647 MATHIC_ASSERT(_tmp.empty());
648 for (size_t i = 0; i < nodes.size(); ++i) {
649 Node* nodei = nodes[i];
650 if (nodei->isLeaf()) {
651 MATHIC_ASSERT(nodei->asLeaf().entries().debugIsValid());
652 continue;
653 }
654 Interior& interior = nodei->asInterior();
655 size_t var = interior.getVar();
656 Exponent exp = interior.getExponent();
657
658 MATHIC_ASSERT(_tmp.empty());
659 // check equal or less than sub tree
660 _tmp.push_back(&interior.getEqualOrLess());
661 while (!_tmp.empty()) {
662 Node* node = _tmp.back();
663 _tmp.pop_back();

669 }
670 if (node->isInterior()) {
671 _tmp.push_back(&node->asInterior().getStrictlyGreater());
672 _tmp.push_back(&node->asInterior().getEqualOrLess());
673 } else {
674 MATHIC_ASSERT(node->asLeaf().entries().allLessThanOrEqualTo(var, exp, _conf));
675 }
676 }
677
678 // check strictly greater
679 _tmp.push_back(&interior.getStrictlyGreater());
680 while (!_tmp.empty()) {
681 Node* node = _tmp.back();
682 _tmp.pop_back();
683 if (C::UseTreeDivMask) {
684 if (node->isInterior())
685 MATHIC_ASSERT(interior.getDivMask().canDivide(node->asInterior().getDivMask()));
686 else
687 MATHIC_ASSERT(interior.getDivMask().canDivide(node->asLeaf().entries().getDivMask()));
688 }
689 if (node->isInterior()) {
690 _tmp.push_back(&node->asInterior().getStrictlyGreater());
691 _tmp.push_back(&node->asInterior().getEqualOrLess());
692 } else {

694 }

698#endif

744 MATHIC_ASSERT(entries().size() > 0);
745 size_t var = (parent == 0 ? static_cast<size_t>(-1) : parent->getVar());

749 (entries().begin(), entries().end(), var, exp, conf, &extEntry);

756 Interior& interior = *new (arena.allocObjectNoCon<Interior>())
757 Interior(*this, other, var, exp);
