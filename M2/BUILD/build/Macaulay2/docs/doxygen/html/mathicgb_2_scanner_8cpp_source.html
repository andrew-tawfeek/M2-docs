77bool Scanner::match(const char* const str) {
78 eatWhite();
79 MATHICGB_ASSERT(str != 0);
80 const auto size = std::strlen(str);
81 if (!ensureBuffer(size))
82 return false;
83 if (size == 0)
84 return true;
85 if (peek() != *str)
86 return false;
87 if (std::strncmp(&*mBufferPos, str + 1, size - 1) != 0)
88 return false;
89 ignore(size);
90 return true;
91}

98void Scanner::expect(const char* str) {
99 MATHICGB_ASSERT(str != 0);
100 eatWhite();
101
102 const char* it = str;
103 while (*it != '\0') {
104 int character = get();
105 if (*it == character) {
106 ++it;
107 continue;
108 }
109
110 // Read the rest of what is there to improve error message.
111 // TODO: read at least one char in total even if not alnum.
112 std::ostringstream got;
113 if (character == EOF && it == str)
114 got << "no more input";
115 else {
116 got << '\"' << std::string(str, it);
117 if (isalnum(character))
118 got << static_cast<char>(character);
119 while (isalnum(peek()))
120 got << static_cast<char>(get());
121 got << '\"';
122 }
123
124 reportErrorUnexpectedToken(str, got.str());
125 }
126}

169bool Scanner::readBuffer(size_t minRead) {
170 auto saveCount = std::distance(mBufferPos, mBuffer.end());
171 if (mBufferPos != mBuffer.begin() && mBufferPos != mBuffer.end())
172 std::copy(mBufferPos, mBuffer.end(), mBuffer.begin());
173 mBuffer.resize(std::max(saveCount + minRead, mBuffer.capacity()));
174 auto readInto = reinterpret_cast<char*>(mBuffer.data() + saveCount);
175 auto readCount = mBuffer.size() - saveCount;
176
177 size_t didReadCount = 0;
178 if (mFile != 0) {
179 didReadCount = fread(readInto, 1, readCount, mFile);
180 } else if (mStream != 0) {
181 mStream->read(readInto, readCount);
182 const std::streamsize maxSizeT =
183 std::numeric_limits<std::make_signed<size_t>::type>::max();
184 if (mStream->gcount() > maxSizeT)
185 throw std::bad_alloc();
186 didReadCount = static_cast<size_t>(mStream->gcount());
187 }
188 mBuffer.resize(saveCount + didReadCount);
189 mBufferPos = mBuffer.begin();
190
191 return didReadCount >= minRead;
192}
