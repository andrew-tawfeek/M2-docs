48 for (size_t gen = 0; gen < basis.size(); ++gen)

53 void setSchreyerMultipliers(MonoVector&& moduleAdjustments) {
54 MATHICGB_ASSERT(moduleAdjustments.monoid() == monoid());
55 MATHICGB_ASSERT(mSchreyerMultipliersMemory.empty() ||
56 mSchreyerMultipliersMemory.size() == componentCount());
57 mSchreyerMultipliersMemory = std::move(moduleAdjustments);
58
59 mSchreyerMultipliers.clear();
60 for (
61 auto it = mSchreyerMultipliersMemory.begin();
62 it != mSchreyerMultipliersMemory.end();
63 ++it
64 ) {
65 // in the absence of a separate monoid for (non-module) monomials,
66 // at least we can check that the component is zero.
67 MATHICGB_ASSERT(this->monoid().component(*it) == 0);
68
69 // todo: there should be a better way of indexing into a
70 // MonoVector.
71 mSchreyerMultipliers.emplace_back((*it).ptr());
72 }
