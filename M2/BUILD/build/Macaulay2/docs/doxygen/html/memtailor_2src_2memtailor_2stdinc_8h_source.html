7#ifdef _MSC_VER // For Microsoft Compiler in Visual Studio C++.
8// Sometimes you know that a function will be called very rarely so you want to
9// tell the compiler not to inline it even if it could be inlined at only a
10// modest increase in code size. That is what MEMT_NO_INLINE does.
11#define MEMT_NO_INLINE __declspec(noinline)
12
13// Sometimes the compiler just will not inline functions that should
14// be inlined. Use sparingly --- preferably only if a profiler says
15// that a tiny often-called function consumes a significant amount of time.
16#define MEMT_INLINE __forceinline

18// Tells the compiler to always assume that the expression X is true.
19#define MEMT_ASSUME(X) __assume(X)
20
21// As MEMT_ASSUME, but might actually evaluate X at run-time if it has
22// side-effects. The point is that this can be used on compilers with no other
23// support for assuming things. So there is no difference on MS VC++.
24#define MEMT_ASSUME_AND_MAY_EVALUATE(X) __assume(X)
25
26// Tells the compiler that this function returns a pointer that is not an alias
27// for any other point that is currently valid in the program - like malloc.
28#define MEMT_RETURN_NO_ALIAS __declspec(restrict)
29
30// Tells the compiler that this function will never throw an exception.
31#define MEMT_NOTHROW __declspec(nothrow)
32
33// Tells the compiler that this function has no effects except the return value
34// and the return value depends only on the arguments and first-level
35// indirections of the arguments. (this is the common denominator of GCC
36// and MS VC++ capabilities)
37#define MEMT_PURE __declspec(noalias)
38
39// Tells the compiler that the return value of this function must be looked
40// at by the caller. For example this is appropriate for realloc.
41#define MEMT_MUST_CHECK_RETURN_VALUE
42
43// Tells the compiler that the current line of code cannot be reached.
44#define MEMT_UNREACHABLE __assume(false)
45
46// Tells the compiler that a variable that is a pointer (not a reference)
47// does not alias any other pointer that is used in the current scope.
48#define MEMT_RESTRICT __restrict
