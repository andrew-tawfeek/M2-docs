154 ElementArray& coefficientInserter() { return coeffs; }
155 void setPoly(ResPolynomial& result)
156 {
157 ncalls++;
158 result.len = static_cast<int>(mRing.vectorArithmetic().size(coeffs));
159 std::swap(result.coeffs, coeffs);
160 result.monoms.resize(result.len * mRing.monoid().max_monomial_size());
161
162 // copy monoms: not pointers, actual monoms
163 res_monomial_word* monomptr = result.monoms.data();
164 for (int i = 0; i < result.len; i++)
165 {
166 mRing.monoid().copy(monoms[i], monomptr);
167 monomptr += mRing.monoid().monomial_size(monoms[i]);
168 }
169 }
170
171 static void setPolyFromArrays(ResPolynomial& result,
172 int len,
173 ElementArray& coeffs,
174 std::vector<res_monomial_word>& monoms)
175 {
176 ncalls_fromarray++;
177 result.len = len;
178 result.coeffs.swap(coeffs);
179 std::swap(result.monoms, monoms);
180 }
