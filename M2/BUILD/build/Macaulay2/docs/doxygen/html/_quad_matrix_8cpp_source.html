119void QuadMatrix::printStatistics(std::ostream& out) const {
120 typedef mathic::ColumnPrinter ColPr;
121
122 ColPr pr;
123 pr.addColumn(false, " ", "");
124 pr.addColumn(false, "", "");
125 pr.addColumn(false, "", "");
126 pr.addColumn(false, "", "");
127 pr.addColumn(true, "", "");
128
129 const auto totalMemory = memoryUse();
130
131 auto printDataCol = [&](
132 std::ostream& out,
133 const SparseMatrix& top,
134 const SparseMatrix& bottom,
135 const SparseMatrix::ColIndex colCount
136 ) {
137 auto printDataCell = [&](const SparseMatrix& matrix) {
138 const auto entryCount = matrix.entryCount();
139 const uint64 area =
140 static_cast<uint64>(matrix.rowCount()) * static_cast<uint64>(colCount);
141 const auto memory = matrix.memoryUse();
142
143 out << ColPr::withSIPrefix(entryCount) << " -"
144 << ColPr::percentIntegerFixed(entryCount, area) << " \n"
145 << ColPr::bytesInUnit(memory) << " -"
146 << ColPr::percentIntegerFixed(matrix.memoryUseTrimmed(), memory)
147 << " \n"
148 << ColPr::percentInteger(memory, totalMemory) << " \n";
149 };
150
151 out << ColPr::commafy(colCount) << " \n";
152 const char* const line = "------------------\n";
153 out << line;
154 printDataCell(top);
155 out << line;
156 printDataCell(bottom);
157 out << line;
158 };
159
160 pr[0] << "\n/\n" << ColPr::commafy(topLeft.rowCount())
161 << " |\nrows |\n|\n|\n"
162 << ColPr::commafy(bottomLeft.rowCount())
163 << " |\nrows |\n|\n\\\n";
164 printDataCol(pr[1], topLeft, bottomLeft, computeLeftColCount());
165 pr[2] << " \n|\n|\n|\n|\n|\n|\n|\n|\n|\n";
166 printDataCol(pr[3], topRight, bottomRight, computeRightColCount());
167
168 const char* const legend =
169 "| non-zero (density)\n| memory (used)\n| of total memory\n";
170 pr[4] << " columns\n\\\n" << legend << "|\n" << legend << "/\n";
171
172 out << '\n' << pr
173 << " Total memory: " << ColPr::bytesInUnit(memoryUse()) << " ("
174 << ColPr::percentInteger(memoryUseTrimmed(), totalMemory)
175 << " used)\n\n";
176}

178QuadMatrix QuadMatrix::toCanonical() const {
179 class RowComparer {
180 public:
181 RowComparer(const SparseMatrix& matrix): mMatrix(matrix) {}
182 bool operator()(SparseMatrix::RowIndex a, SparseMatrix::RowIndex b) const {
183 auto itA = mMatrix.rowBegin(a);
184 const auto endA = mMatrix.rowEnd(a);
185 auto itB = mMatrix.rowBegin(b);
186 const auto endB = mMatrix.rowEnd(b);
187 for (; itA != endA; ++itA, ++itB) {
188 if (itB == endB)
189 return true;
190
191 if (itA.index() > itB.index())
192 return true;
193 if (itA.index() < itB.index())
194 return false;
195
196 if (itA.scalar() > itB.scalar())
197 return false;
198 if (itA.scalar() < itB.scalar())
199 return true;
200 }
201 return false;
202 }
203
204 private:
205 const SparseMatrix& mMatrix;
206 };
207
208 // todo: eliminate left/right code duplication here
209 QuadMatrix matrix(ring());
210 { // left side
211 std::vector<SparseMatrix::RowIndex> rows;
212 for (SparseMatrix::RowIndex row = 0; row < topLeft.rowCount(); ++row)
213 rows.push_back(row);
214 {
215 RowComparer comparer(topLeft);
216 std::sort(rows.begin(), rows.end(), comparer);
217 }
218
219 matrix.topLeft.clear();
220 matrix.topRight.clear();
221 for (SparseMatrix::RowIndex i = 0; i < rows.size(); ++i) {
222 matrix.topLeft.appendRow(topLeft, rows[i]);
223 matrix.topRight.appendRow(topRight, rows[i]);
224 }
225 }
226 { // right side
227 std::vector<SparseMatrix::RowIndex> rows;
228 for (SparseMatrix::RowIndex row = 0; row < bottomLeft.rowCount(); ++row)
229 rows.push_back(row);
230 {
231 RowComparer comparer(bottomLeft);
232 std::sort(rows.begin(), rows.end(), comparer);
233 }
234
235 matrix.bottomLeft.clear();
236 matrix.bottomRight.clear();
237 for (SparseMatrix::RowIndex i = 0; i < rows.size(); ++i) {
238 matrix.bottomLeft.appendRow(bottomLeft, rows[i]);
239 matrix.bottomRight.appendRow(bottomRight, rows[i]);
240 }
241 }
242
243 matrix.leftColumnMonomials = leftColumnMonomials;
244 matrix.rightColumnMonomials = rightColumnMonomials;
245
246 return matrix;
247}

269 std::vector<SparseMatrix::ColIndex> sortColumnMonomialsAndMakePermutation(
270 QuadMatrix::Monomials& monomials,
271 const QuadMatrix::Monoid& monoid
272 ) {
273 typedef SparseMatrix::ColIndex ColIndex;
274 MATHICGB_ASSERT(monomials.size() <= std::numeric_limits<ColIndex>::max());
275 const ColIndex colCount = static_cast<ColIndex>(monomials.size());
276 // Monomial needs to be non-const as we are going to put these
277 // monomials back into the vector of monomials which is not const.
278 std::vector<std::pair<QuadMatrix::ConstMonoPtr, ColIndex>> columns;
279 columns.reserve(colCount);
280 for (ColIndex col = 0; col < colCount; ++col)
281 columns.push_back(std::make_pair(monomials[col], col));
282 std::sort(
283 columns.begin(),
284 columns.end(),
285 ColumnComparer<QuadMatrix::Monoid>(monoid)
286 );
287
288 // Apply sorting permutation to monomials. This is why it is necessary to
289 // copy the values in monomial out of there: in-place application of a
290 // permutation is messy.
291 MATHICGB_ASSERT(columns.size() == colCount);
292 MATHICGB_ASSERT(monomials.size() == colCount);
293 for (size_t col = 0; col < colCount; ++col) {
294 MATHICGB_ASSERT(
295 col == 0 ||
296 monoid.lessThan(*columns[col].first, *columns[col - 1].first)
297 );
298 monomials[col] = columns[col].first;
299 }
300
301 // Construct permutation of indices to match permutation of monomials
302 std::vector<ColIndex> permutation(colCount);
303 for (ColIndex col = 0; col < colCount; ++col) {
304 // The monomial for column columns[col].second is now the
305 // monomial for col, so we need the inverse map for indices.
306 permutation[columns[col].second] = col;
307 }
308
309 return permutation;
310 }
311}

313void QuadMatrix::sortColumnsLeftRightParallel() {
314 typedef SparseMatrix::ColIndex ColIndex;
315 std::vector<ColIndex> leftPermutation;
316 std::vector<ColIndex> rightPermutation;
317
318 mtbb::parallel_for(0, 2, 1, [&](int i) {
319 if (i == 0)
320 leftPermutation =
321 sortColumnMonomialsAndMakePermutation(leftColumnMonomials, monoid());
322 else
323 rightPermutation =
324 sortColumnMonomialsAndMakePermutation(rightColumnMonomials, monoid());
325 });
326
327 mtbb::parallel_for(0, 4, 1, [&](int i) {
328 if (i == 0)
329 topRight.applyColumnMap(rightPermutation);
330 else if (i == 1)
331 bottomRight.applyColumnMap(rightPermutation);
332 else if (i == 2)
333 topLeft.applyColumnMap(leftPermutation);
334 else {
335 MATHICGB_ASSERT(i == 3);
336 bottomLeft.applyColumnMap(leftPermutation);
337 }
338 });
339}
