48 // Represents a term multiple of a polynomial,
49 // together with a current term of the multiple.
50 struct MultipleWithPos {
51 MultipleWithPos(const Poly& poly, NewConstTerm multiple);
52
53 Poly::ConstTermIterator pos;
54 const Poly::ConstTermIterator end;
55 NewTerm multiple;
56
57 // invariant: current is the monomial product of multiple.monom
58 // and pos.mono().
59 MonoPtr current;
60
61 // Ensures the invariant, so sets current to the product of
62 // multiple.monom and pos.mono().
63 void computeCurrent(const PolyRing& ring);
64 void currentCoefficient(const PolyRing& ring, Coefficient& coeff);
65 void addCurrentCoefficient(const PolyRing& ring, Coefficient& coeff);
66 void destroy(const PolyRing& ring);
67
68 // Points to a circular list of entries that have the same current
69 // monomial. If no other such entries have been discovered, then
70 // chain points to this object itself. We use a circular linked list
71 // as those allow merging in O(1) time.
72 MultipleWithPos* chain;
73 void mergeChains(MultipleWithPos& entry) {
74 // This only works if *this and entry are not already in the
75 // same chain!
76 std::swap(chain, entry.chain);
77 }
78 };

80 class Configuration : public ReducerHelper::DedupConfiguration {
81 public:

83 Configuration(const PolyRing& ring): DedupConfiguration(ring) {}
84 CompareResult compare(const Entry& a, const Entry& b) const {
85 return ring().monoid().compare(*a->current, *b->current);
86 }
87 Entry deduplicate(Entry a, Entry b) const {
88 a->mergeChains(*b);
89 return a;
90 }
91 };

118template<template<typename> class Q>
119void ReducerPackDedup<Q>::insertTail(NewConstTerm multiple, const Poly& poly) {
120 if (poly.termCount() <= 1)

182template<template<typename> class Q>
183void ReducerPackDedup<Q>::MultipleWithPos::destroy(const PolyRing& ring) {
184 MultipleWithPos* entry = this;
185 do {
186 ring.monoid().freeRaw(*entry->current);
187 ring.monoid().freeRaw(*entry->multiple.mono);
188 MultipleWithPos* next = entry->chain;
189 MATHICGB_ASSERT(next != 0);
190
191 // Call the destructor to destruct the iterators into std::vector.
192 // In debug mode MSVC puts those in a linked list and the destructor
193 // has to be called since it takes an iterator off the list. There were
194 // memory corruption problems in debug mode on MSVC before doing this.
195 entry->~MultipleWithPos();
196
197 entry = next;
198 } while (entry != this);
199}

201template<template<typename> class Q>
202bool ReducerPackDedup<Q>::leadTerm(NewConstTerm& result) {
203 if (!mLeadTermKnown) {
204 do {
205 if (mQueue.empty())
206 return false;
207 auto entry = mQueue.top();
208 entry->currentCoefficient(mRing, mLeadTerm.coef);
209 while (true) {
210 // store the chained elements
211 const auto chainBegin = entry->chain;
212 const auto chainEnd = entry; // the list is circular
213 entry->chain = entry; // detach any chained elements
214
215 // handle the entry itself
216 std::swap(mLeadTerm.mono, entry->current);
217 ++entry->pos;
218 if (entry->pos == entry->end) {
219 mQueue.pop();
220 entry->destroy(mRing);
221 mPool.free(entry);
222 } else {
223 entry->computeCurrent(mRing);
224 // Inserted spans must be in descending order
225 MATHICGB_ASSERT(mQueue.getConfiguration().ring().
226 monoid().lessThan(*entry->current, *mLeadTerm.mono));
227 mQueue.decreaseTop(entry);
228 }
229
230 // handle any chained elements
231 auto chain = chainBegin;
232 while (chain != chainEnd) {
233 MATHICGB_ASSERT(chain != 0);
234 MATHICGB_ASSERT(mRing.monoid().equal(*chain->current, *mLeadTerm.mono));
235
236 const auto next = chain->chain;
237 chain->chain = chain; // detach from remaining chained elements
238
239 chain->addCurrentCoefficient(mRing, mLeadTerm.coef);
240 ++chain->pos;
241 if (chain->pos == chain->end) {
242 chain->destroy(mRing);
243 mPool.free(chain);
244 } else {
245 chain->computeCurrent(mRing);
246 // Inserted spans must be in descending order
247 MATHICGB_ASSERT(mQueue.getConfiguration().ring().
248 monoid().lessThan(*chain->current, *mLeadTerm.mono));
249 mQueue.push(chain);
250 }
251 chain = next;
252 }
253
254 if (mQueue.empty())
255 break;
256
257 entry = mQueue.top();
258 if (!mRing.monoid().equal(*entry->current, *mLeadTerm.mono))
259 break;
260 entry->addCurrentCoefficient(mRing, mLeadTerm.coef);
261 }
262 } while (mRing.coefficientIsZero(mLeadTerm.coef));
263 mLeadTermKnown = true;
264 }
265
266 result = mLeadTerm;
267 return true;
268}

281 class MonomialFree {
