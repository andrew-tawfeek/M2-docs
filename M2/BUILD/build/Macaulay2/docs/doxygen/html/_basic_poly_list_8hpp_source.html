107template<typename S>
108void toStream(const BasicPolyList& Fs, S &str)
109{
110 str.idealBegin(Fs.size());
111 for (auto& F : Fs)
112 {
113 str.appendPolynomialBegin(F.mCoefficients.size());
114 int monomStart = 0;
115 for (auto i=0; i<F.mCoefficients.size(); ++i)
116 {
117 auto monomEnd = monomStart + F.mMonomials[monomStart];
118 if (F.mComponents.empty())
119 str.appendTermBegin(0);
120 else
121 str.appendTermBegin(F.mComponents[i]);
122 for (auto j=monomStart+1; j<monomEnd; j += 2)
123 str.appendExponent(F.mMonomials[j], F.mMonomials[j+1]);
124 str.appendTermDone(F.mCoefficients[i]);
125 monomStart = monomEnd;
126 }
127 str.appendPolynomialDone();
128 }
129 str.idealDone();
130}
