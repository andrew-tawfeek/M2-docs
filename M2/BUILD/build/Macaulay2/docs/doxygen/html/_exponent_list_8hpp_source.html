138 static int compare(ConstExponents a, ConstExponents b)
139 {
140 Exponent alen = length(a++) - 1;
141 Exponent blen = length(b++) - 1;
142 // TODO: try using std::lexicographical_compare?
143 for (size_t i = 0; i < std::min(alen, blen); i++)
144 {
145 Exponent c = *a++ - *b++;
146 if (c == 0) continue;
147 return (c > 0 ? GT : LT);
148 }
149 if (alen == blen) return EQ;
150 return (alen > blen ? GT : LT);
