278typename SMat<CoeffRing>::sparsevec *SMat<CoeffRing>::vec_copy(
279 const sparsevec *v) const
280{

282 sparsevec *result = &head;
283 for (const sparsevec *p = v; p != 0; p = p->next)
284 {
285 sparsevec *w = vec_new();
286 result->next = w;
287 result = w;

319template <typename CoeffRing>
320void SMat<CoeffRing>::vec_set_entry(sparsevec *&v,
321 size_t r,
322 const elem &a) const
323{
324 sparsevec *p;
325 bool iszero = ring().is_zero(a);
326 sparsevec head;
327 head.next = v;
328 for (p = &head; p->next != 0; p = p->next)
329 if (p->next->row <= r) break;
330
331 if (p->next == 0 || p->next->row < r)
332 {
333 if (iszero) return;
334 sparsevec *w = vec_new();
335 w->next = p->next;
336 w->row = r;
337 ring().init_set(w->coeff, a);
338 p->next = w;
339 }
340 else if (p->next->row == r)
341 {
342 if (iszero)
343 {
344 // delete node
345 sparsevec *tmp = p->next;
346 p->next = tmp->next;
347 vec_remove_node(tmp);
348 }
349 else
350 ring().set(p->next->coeff, a);
351 }
352 v = head.next;
353}

355template <typename CoeffRing>
356void SMat<CoeffRing>::vec_interchange_rows(sparsevec *&v,
357 size_t i,
358 size_t j) const
359{
360 sparsevec *p;
361 if (i == j) return;
362 if (v == 0) return;
363 if (i < j)
364 {
365 size_t tmp = i;
366 i = j;
367 j = tmp;
368 }
369 // So now i > j.
370 sparsevec head;
371 head.next = v;
372 sparsevec *vec1;
373 sparsevec *vec2;
374 for (p = &head; p->next != 0; p = p->next)
375 if (p->next->row <= i) break;
376 vec1 = p;
377 for (; p->next != 0; p = p->next)
378 if (p->next->row <= j) break;
379 vec2 = p;
380 if (vec1->next != 0 && vec1->next->row == i)
381 {
382 if (vec2->next != 0 && vec2->next->row == j)
383 {
384 std::swap(vec1->next->coeff, vec2->next->coeff);
385 return;
386 }
387 }
388 else if (vec2->next != 0 && vec2->next->row == j)
389 {
390 sparsevec *tmp = vec1;
391 vec1 = vec2;
392 vec2 = tmp;
393 j = i; // Used below.
394 }
395 else
396 return;
397
398 sparsevec *tmp = vec1->next;
399 if (vec2 != tmp)
400 {
401 vec1->next = tmp->next;
402 tmp->next = vec2->next;
403 vec2->next = tmp;
404 }
405 tmp->row = j;
406 v = head.next;
407}

514template <typename CoeffRing>
515void SMat<CoeffRing>::vec_add_to(sparsevec *&v, sparsevec *&w) const
516// v := v+w, w := 0
517{
518 if (w == 0) return;
519 if (v == 0)
520 {
521 v = w;
522 w = 0;
523 return;
524 }
525 sparsevec head;
526 sparsevec *result = &head;
527 while (true)
528 if (v->row < w->row)
529 {
530 result->next = w;
531 result = result->next;
532 w = w->next;
533 if (w == 0)
534 {
535 result->next = v;
536 v = head.next;
537 return;
538 }
539 }
540 else if (v->row > w->row)
541 {
542 result->next = v;
543 result = result->next;
544 v = v->next;
545 if (v == 0)
546 {
547 result->next = w;
548 v = head.next;
549 w = 0;
550 return;
551 }
552 }
553 else
554 {
555 sparsevec *tmv = v;
556 sparsevec *tmw = w;
557 v = v->next;
558 w = w->next;
559 ring().add(tmv->coeff, tmv->coeff, tmw->coeff);
560 if (ring().is_zero(tmv->coeff))
561 {
562 vec_remove_node(tmv);
563 }
564 else
565 {
566 result->next = tmv;
567 result = result->next;
568 }
569 vec_remove_node(tmw);
570 if (w == 0)
571 {
572 result->next = v;
573 v = head.next;
574 return;
575 }
576 if (v == 0)
577 {
578 result->next = w;
579 v = head.next;
580 w = 0;
581 return;
582 }
583 }

586template <typename CoeffRing>
587void SMat<CoeffRing>::vec_row_op(sparsevec *&v,
588 size_t r1,
589 const elem &a,
590 size_t r2) const
591// row(r1 in v) := row(r1 in v) + a * row(r2 in v)
592{
593 sparsevec *p;
594 sparsevec *vec2 = 0;
595 for (p = v; p != 0; p = p->next)
596 if (p->row == r2)
597 {
598 vec2 = p;
599 break;
600 }
601 if (vec2 == 0) return;
602 typename CoeffRing::Element c(ring());
603 ring().set_zero(c);
604 ring().mult(c, vec2->coeff, a);
605 if (ring().is_zero(c))
606 {
607 return; // nothing to change
608 }
609 // Now add c to the r1'th row of v
610 sparsevec head;
611 head.next = v;
612 for (p = &head; p->next != 0; p = p->next)
613 if (p->next->row <= r1) break;
614 if (p->next == 0 || p->next->row < r1)
615 {
616 // Make a new node
617 sparsevec *w = vec_new();
618 w->next = p->next;
619 w->row = r1;
620 ring().init_set(w->coeff, c);
621 p->next = w;
622 }
623 else
624 {
625 ring().add(p->next->coeff, p->next->coeff, c);
626 if (ring().is_zero(p->next->coeff))
627 {
628 sparsevec *tmp = p->next;
629 p->next = tmp->next;
630 vec_remove_node(tmp);
631 }
632 }
633 v = head.next;
634}

636template <typename CoeffRing>
637void SMat<CoeffRing>::vec_row_op2(sparsevec *&v,
638 size_t r1,
639 size_t r2,
640 const elem &a1,
641 const elem &a2,
642 const elem &b1,
643 const elem &b2) const
644// row(r1 in v) := a1 * row(r1 in v) + a2 * row(r2 in v)
645// row(r2 in v) := b1 * row(r1 in v) + b2 * row(r2 in v) (RHS refers to previous
646// values)
647{
648 // v[row r1] = a1 * v[r1] + a2 * v[r2]
649 // v[row r2] = b1 * v[r1] + b2 * v[r2]
650 Element e1(ring()), e2(ring()), c1(ring()), c2(ring()), c3(ring()),
651 c4(ring());
652
653 ring().set_zero(c1);
654 ring().set_zero(c2);
655 ring().set_zero(c3);
656 ring().set_zero(c4);
657 ring().set_zero(e1);
658 ring().set_zero(e2);
659 bool r1_nonzero = vec_get_entry(v, r1, e1);
660 bool r2_nonzero = vec_get_entry(v, r2, e2);
661 if (!r1_nonzero && !r2_nonzero)
662 {
663 return;
664 }
665 if (r1_nonzero)
666 {
667 ring().mult(c1, a1, e1);
668 ring().mult(c3, b1, e1);
669 }
670 if (r2_nonzero)
671 {
672 ring().mult(c2, a2, e2);
673 ring().mult(c4, b2, e2);
674 }
675
676 ring().add(c1, c1, c2);
677 ring().add(c3, c3, c4);
678 vec_set_entry(v, r1, c1);
679 vec_set_entry(v, r2, c3);
680}

693template <typename CoeffRing>
694void SMat<CoeffRing>::vec_dot_product(sparsevec *v,
695 sparsevec *w,
696 elem &result) const
697{
698 Element a(ring());
699 ring().set_zero(a);
700 ring().set_zero(result);
701 while (true)
702 {
703 if (v == 0 || w == 0) break;
704 if (v->row > w->row)
705 v = v->next;
706 else if (v->row < w->row)
707 w = w->next;
708 else
709 {
710 ring().mult(a, v->coeff, w->coeff);
711 ring().add(result, result, a);
712 v = v->next;
713 w = w->next;
714 }
715 }
716}

718template <typename CoeffRing>
719void SMat<CoeffRing>::vec_sort(sparsevec *&f) const
720{
721 if (f == 0 || f->next == 0) return;
722 sparsevec *f1 = 0;
723 sparsevec *f2 = 0;
724 while (f != 0)
725 {
726 sparsevec *t = f;
727 f = f->next;
728 t->next = f1;
729 f1 = t;
730
731 if (f == 0) break;
732 t = f;
733 f = f->next;
734 t->next = f2;
735 f2 = t;
736 }
737
738 vec_sort(f1);
739 vec_sort(f2);
740 vec_add_to(f1, f2);
741 f = f1;
742}

744template <typename CoeffRing>

748{
749 size_t *perminv = newarray_atomic(size_t, perm->len);
750 for (size_t i = 0; i < perm->len; i++) perminv[perm->array[i]] = i;
751
752 size_t end_row = start_row + perm->len;
753
754 for (sparsevec *w = v; w != 0; w = w->next)
755 if (w->row >= start_row && w->row < end_row)
756 w->row = start_row + perminv[w->row - start_row];
757 vec_sort(v);
758 freemem(perminv);
759}

769template <typename CoeffRing>
770void SMat<CoeffRing>::vec_delete_rows(sparsevec *&v, size_t i, size_t j) const
771{
772 size_t n_to_delete = j - i + 1;
773 sparsevec head;
774 sparsevec *w = &head;
775 w->next = v;
776 while (w->next != 0 && w->next->row >= i)
777 if (w->next->row <= j)
778 {

780 sparsevec *tmp = w->next;
781 w->next = tmp->next;
782 vec_remove_node(tmp);
783 }
784 else
785 {
786 w = w->next;
787 w->row -= n_to_delete;
788 }
789 v = head.next;
790}

982template <typename CoeffRing>
983bool SMat<CoeffRing>::row_permute(size_t start_row, M2_arrayint perm)
984{
985 // We copy one row to another location for each cycle in 'perm' of length > 1.
986 size_t nrows_to_permute = perm->len;
987 bool *done = newarray_atomic(bool, nrows_to_permute);
988 for (size_t i = 0; i < nrows_to_permute; i++) done[i] = true;
989 for (size_t i = 0; i < nrows_to_permute; i++)
990 {
991 size_t j = perm->array[i];
992 if (!done[j])
993 {
994 ERROR("expected permutation");
995 freemem(done);
996 return false;
997 }
998 done[j] = false;
999 }
1000 for (size_t c = 0; c < ncols_; c++) vec_permute(columns_[c], start_row, perm);
1001 return true;
1002}

1004template <typename CoeffRing>
1005bool SMat<CoeffRing>::column_permute(size_t start_col, M2_arrayint perm)
1006{
1007 size_t ncols_to_permute = perm->len;
1008 bool *done = newarray_atomic(bool, ncols_to_permute);
1009 sparsevec **tmpvecs = newarray(sparsevec *, ncols_to_permute);
1010 for (size_t i = 0; i < ncols_to_permute; i++) done[i] = true;
1011 for (size_t i = 0; i < ncols_to_permute; i++)
1012 {
1013 size_t j = perm->array[i];
1014 if (!done[j])
1015 {
1016 ERROR("expected permutation");
1017 freemem(done);
1018 return false;
1019 }
1020 done[j] = false;
1021 }
1022 for (size_t i = 0; i < ncols_to_permute; i++)
1023 tmpvecs[i] = columns_[start_col + perm->array[i]];
1024 for (size_t i = 0; i < ncols_to_permute; i++)
1025 columns_[start_col + i] = tmpvecs[i];
1026 freemem(tmpvecs);
1027 freemem(done);
1028 return true;
1029}
