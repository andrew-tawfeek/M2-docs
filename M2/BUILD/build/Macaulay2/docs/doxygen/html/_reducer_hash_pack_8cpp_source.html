139template<template<typename> class Q>
140bool ReducerHashPack<Q>::leadTerm(NewConstTerm& result) {
141 while (!mQueue.empty()) {
142 auto entry = mQueue.top();
143 MATHICGB_ASSERT(entry != nullptr);
144
145 if (!mRing.coefficientIsZero(entry->node->value())) {
146 result.coef = entry->node->value();
147 result.mono = entry->node->mono().ptr();
148 return true;
149 }
150 removeLeadTerm();
151 }
152 return false;
153}

155template<template<typename> class Q>
156void ReducerHashPack<Q>::removeLeadTerm() {
157 MATHICGB_ASSERT(!mQueue.empty());
158
159 auto entry = mQueue.top();
160 MATHICGB_ASSERT(entry != 0);
161
162 // remove node from hash table first since we are going to be changing
163 // the monomial after this, and if we do that before the hash value will
164 // change. That might prompt an assert inside the hash table.
165 mHashTable.remove(entry->node);
166
167 MATHICGB_ASSERT(entry->pos != entry->end);
168 while (true) {
169 ++entry->pos;
170 if (entry->pos == entry->end) {
171 mQueue.pop();
172 entry->destroy(mRing);
173 mPool.free(entry);
174 break;
175 }
176
177 const auto p = mHashTable.insertProduct(entry->multiple, *entry->pos);
178 if (p.second) {
179 entry->node = p.first;
180 mQueue.decreaseTop(entry);
181 break;
182 }
183 }
184}

186template<template<typename> class Q>
187void ReducerHashPack<Q>::insertEntry(MultipleWithPos* entry) {
188 MATHICGB_ASSERT(entry != 0);
189 for (; entry->pos != entry->end; ++entry->pos) {
190 const auto p = mHashTable.insertProduct(entry->multiple, *entry->pos);
191 if (p.second) {
192 entry->node = p.first;
193 mQueue.push(entry);
194 return;
195 }
196 }
197 entry->destroy(mRing);
198 mPool.free(entry);
199}

201template<template<typename> class Q>
202void ReducerHashPack<Q>::resetReducer() {
203 class MonomialFree {
204 public:
205 MonomialFree(const PolyRing& ring): mRing(ring) {}
206
207 bool proceed(MultipleWithPos* entry) {
208 entry->destroy(mRing);
209 return true;
210 }
211 private:
212 const PolyRing& mRing;
213 };
214
215 MonomialFree freeer(mRing);
216 mQueue.forAll(freeer);
217 mQueue.clear();
218 mHashTable.clear();
219}
