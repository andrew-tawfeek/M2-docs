99void PolyRing::mysteriousSPairMonomialRoutine(ConstMonomial newSig,
100 ConstMonomial newLead,
101 ConstMonomial baseDivSig,
102 ConstMonomial baseDivLead,
103 Monomial result) const
104{
105 result[0] = 0; // set component
106 for (size_t i = 1; i <= varCount(); ++i) {
107 exponent x = newSig[i] - baseDivSig[i] + baseDivLead[i];
108 if (newLead[i] <= x)
109 x = std::numeric_limits<exponent>::max();
110 result[i] = std::max(baseDivLead[i], x);
111 }
112}

114// The following two read/write monomials in the form:
115// letter number letter number ... letter number < number >
116// allowed letters: a-zA-Z
117// also allowed instead of letter: [ number ], [0] refers to first var, etc.
118// What about errors??
119void PolyRing::monomialParse(std::istream &i,
120 Monomial& result) const
121{
122// monoid().parseM2(i, result);
123 // first initialize result:
124 for (size_t j=0; j< maxMonomialSize(); j++) result[j] = 0;
125
126 uint64 e;
127 int v, x;
128 // now look at the next char
129 for (;;)
130 {
131 char next = i.peek();
132 if (isalpha(next))
133 {
134 // determine variable
135 v = next - 'a';
136 if (v > 25 || v < 0)
137 v = next - 'A' + 26;
138 if (v < 0 || v > 51)
139 std::cerr << "variable out of range" << std::endl;
140 i.get(); // move past the letter
141 next = i.peek();
142 e = 1;
143 if (isdigit(next))
144 i >> e;
145 result[v+1] = static_cast<exponent>(e);
146 }
147 else if (next == '<')
148 {
149 // get component
150 i.get();
151 i >> x;
152 *result = x; // place component
153 if (i.peek() == '>') i.get();
154 }
155 else
156 {
157 break; // We assume that we are at the end of the monomial
158 }
159 }
160 setWeightsAndHash(result);
161}

163void PolyRing::monomialDisplay(std::ostream &o,
164 ConstMonomial a,
165 bool print_comp,
166 bool print_one) const
167{
168 // We display a monomial in the form that can be used with monomialParse
169 // print_one: only is consulted in print_comp is false.
170
171 bool printed_any = false;
172 for (size_t i=0; i< varCount(); i++) {
173 if (a[i+1] != 0) {
174 printed_any = true;
175 if (i <= 25)
176 o << static_cast<unsigned char>('a' + i);
177 else
178 o << static_cast<unsigned char>('A' + (i - 26));
179 if (a[i+1] != 1)
180 o << static_cast<int64>(a[i+1]);
181 }
182 }
183 if (print_comp)
184 o << '<' << static_cast<int64>(*a.mValue) << '>';
185 else if (!printed_any && print_one)
186 o << "1";
187}

189void PolyRing::monomialDisplay(FILE* file,
190 ConstMonomial mono,
191 bool printComponent,
192 bool printOne) const
193{
194 const unsigned int letterCount = 26;
195 MATHICGB_ASSERT(getNumVars() <= 2 * letterCount);
196 bool printedAny = false;
197 for (size_t var = 0; var < varCount(); ++var) {
198 exponent e = monomialExponent(mono, var);
199 if (e == 0)
200 continue;
201 printedAny = true;
202 char varChar;
203 if (var < letterCount)
204 varChar = static_cast<char>('a' + var);
205 else
206 varChar = static_cast<char>('A' + (var - letterCount));
207 fputc(varChar, file);
208 if (e != 1)
209 fprintf(file, "%i", e);
210 }
211 if (printComponent)
212 fprintf(file, "<%i>", mono.component());
213 else if (!printedAny && printOne)
214 fputc('1', file);
215}
