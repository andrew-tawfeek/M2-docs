25bool matchInclude(Scanner& in, bool& system) {
26 if (!in.match("#include "))
27 return false;
28
29 char delim = '\"';
30 system = false;
31 if (!in.match('\"')) {
32 if (in.match('<')) {
33 system = true;
34 delim = '>';
35 } else
36 in.reportError("Expected < or \" after #include.");
37 }
38
39 in.readUntil(delim);
40 in.expect(delim);
41 return true;
42}

80void checkInclusionGuard(Scanner& in, const std::string& filename) {
81 auto normalize = [](const std::string& str) {
82 std::string norm;
83 for (size_t i = 0; i < str.size(); ++i)
84 if (str[i] != '_')
85 norm += std::toupper(str[i]);
86 return norm;
87 };
88
89 in.expect("#ifndef ");
90 const auto macroName = in.readString();
91 const auto fileNorm = normalize
92 ("MATHICGB_" + stripEnding(filename) + "_GUARD");
93
94 if (fileNorm != normalize(macroName)) {
95 std::ostringstream err;
96 err << "Inclusion guard name does not match file name.\n";
97 err << "Normalization removes _ and converts to upper case.\n";
98 err << " Filename normalizes to: " << fileNorm << '\n';
99 err << "macro name normalizes to: " << normalize(macroName) << '\n';
100 in.reportError(err.str());
101 }
102
103 in.expect("#define ");
104 std::string macro2;
105 macro2 = in.readString();
106 if (macroName != macro2) {
107 std::ostringstream out;
108 out << "Inclusion guard #ifndef and #define macro name mismatch.\n";
109 out << "#ifndef macro name: " << macroName << '\n';
110 out << "#define macro name: " << macro2 << '\n';
111 in.reportError(out.str());
112 }

115void checkOther(const std::string& filename) {
116 std::ifstream file(filename.c_str(), std::ios_base::binary);
117 file.peek();
118 if (!file)
119 error("could not open file");
120 Scanner in(file);
121 bool mgbNamespace = false;
122 while (!in.matchEOF()) {
123 if (in.peek() == '\r')
124 in.reportError("File contains dos/windows line ending character \\r.");
125 if (in.peek() == '\t')

134 mgbNamespace = true;

136 in.get();
137 }
138 if (!mgbNamespace)
139 in.reportError("MATHICGB_NAMESPACE_BEGIN does not appear in file");

142void checkFile(std::string filename) {
143 try {
144 std::cout << "Checking file " << filename << std::endl;
