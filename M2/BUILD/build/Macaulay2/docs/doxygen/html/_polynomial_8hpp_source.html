43struct Monom
44// Format for monomials:
45 // A monomial is an array of ints, the first of which is the length of that array (including length field).
46 // e.g. [2 0] (is currently the empty monomial, but this class knows nothing about specific monomials.
47 // each monomial is of the form:
48 // <length: n+2> <degree: (currently) n> <var1> <var2> ... <varn>
49 // e.g. xyxy: 6 4 0 1 0 1
50 // xy23x: 27 25 0 1 1 ... 1 0
51 // 2 monomials: xzx, xy (dot is only there for readability)
52 // 5 3 0 2 0 . 4 2 0 1
53 // TODO: There are now weights in the monomial (which only the monoid knows about),
54 // so this needs to be updated.

56 Monom() : mValue(nullptr) {}

63 int size() const { return *mValue; }
64
65 const int* begin() const { return mValue; }
66 const int* end() const { return mValue + *mValue; }

116 static int compare(const ModuleMonom& m1, const ModuleMonom& m2)
117 {
118 if (m1[2] > m2[2]) return GT;
119 if (m1[2] < m2[2]) return LT;
120 if (m1[3] > m2[3]) return GT;
121 if (m1[3] < m2[3]) return LT;

154inline ModuleMonom monomToModuleMonom(const Monom& a, int comp, std::pair<int*, int*> allocated_result)
155{
156 assert(allocated_result.second-allocated_result.first >= ModuleMonom::sizeOfCorrespondingModuleMonom(a));
157 auto begin = allocated_result.first;
158 begin[0] = ModuleMonom::sizeOfCorrespondingModuleMonom(a);
159 begin[1] = 0; // index
160 begin[2] = 0; // hashval
161 begin[3] = comp;
162 std::copy(a.begin()+1, a.end(), begin+4);
163 return ModuleMonom(begin);
164}
