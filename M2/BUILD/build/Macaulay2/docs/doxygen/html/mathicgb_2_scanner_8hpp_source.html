207template<class T>
208T Scanner::readInteger(const bool negate) {
209 static_assert(std::numeric_limits<T>::is_integer, "");
210
211 eatWhite();
212 const bool minus = !match('+') && match('-');
213 const bool positive = minus == negate;
214 if (!peekDigit())
215 reportErrorUnexpectedToken("an integer", "");
216 // Skip leading zeroes and return if the number is zero.
217 if (peek() == '0') {
218 while (peek() == '0')
219 get();
220 if (!peekDigit())
221 return static_cast<T>(0);
222 }
223
224 MATHICGB_ASSERT(peekDigit());
225 MATHICGB_ASSERT(peek() != 0);
226
227 // Checking this here allows us to recognize -0 as non-negative.
228 if (!positive && !std::numeric_limits<T>::is_signed)
229 reportErrorUnexpectedToken("a positive integer", "");
230
231 const auto min = std::numeric_limits<T>::min();
232 const auto max = std::numeric_limits<T>::max();
233 auto t = static_cast<T>(0);
234 while (peekDigit()) {
235 const auto c = static_cast<char>(get());
236 const auto d = positive ? c - '0' : -(c - '0');
237 if (positive ? t > (max - d) / 10 : t < (min - d) / 10) {
238 std::ostringstream err;
239 err << "an integer in the range [" << unchar(min)
240 << ", " << unchar(max) << ']';
241 reportErrorUnexpectedToken(err.str(), "");
242 }
243 t = t * 10 + d;
244 }
245 MATHICGB_ASSERT(t != static_cast<T>(0)); // We already handled zero above.
246 return t;
247}

250typename PrimeField<T>::Element Scanner::readModular(
251 const PrimeField<T>& field,
252 const bool negate
253) {
254 static_assert(std::numeric_limits<T>::is_integer, "");
255
256 // Otherwise we need to consider that the most negative value's
257 // negative cannot be represented as a positive number when reading.
258 // todo: reinstate this assert once we get rid of signed coefficients.
259 //static_assert(!std::is_signed<T>::value, "");

261 eatWhite();
262 const bool minus = !match('+') && match('-');
263 const bool positive = minus == negate;
264 if (!peekDigit())
265 reportErrorUnexpectedToken("an integer", "");
266 const auto e = field.toElement(readInteger<T>(false));
