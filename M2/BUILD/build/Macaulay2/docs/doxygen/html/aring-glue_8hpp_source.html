516 bool lower_associate_divisor(ring_elem &f, const ring_elem g) const
517 {
518 mpq_srcptr a = MPQ_VAL(f);
519 mpq_srcptr b = MPQ_VAL(g);
520 int sa = mpq_sgn(a);
521 int sb = mpq_sgn(b);
522 int s = (sa == 0 ? sb : sa);
523 mpq_ptr result = getmemstructtype(mpq_ptr);
524 mpq_init(result);
525
526 mpz_gcd(mpq_numref(result), mpq_numref(a), mpq_numref(b));
527 mpz_lcm(mpq_denref(result), mpq_denref(a), mpq_denref(b));
528 if (s != mpq_sgn(result)) mpq_neg(result, result);
529 mpz_reallocate_limbs(mpq_numref(result));
530 mpz_reallocate_limbs(mpq_denref(result));
531 f = ring_elem(result);
532 return true; // the answer could become lower, if a newer g has a larger
533 // denom
534 }

537 {
538 if (is_zero(c))
539 {
540 c = g;
541 return;
542 }
543 mpq_srcptr a = MPQ_VAL(c);
544 mpq_srcptr b = MPQ_VAL(g);
545 int sa = mpq_sgn(a);
546 mpq_ptr result = getmemstructtype(mpq_ptr);
547 mpq_init(result);
548

550 mpz_lcm(mpq_denref(result), mpq_denref(a), mpq_denref(b));
551 if (sa != mpq_sgn(result)) mpq_neg(result, result);
552 mpz_reallocate_limbs(mpq_numref(result));
553 mpz_reallocate_limbs(mpq_denref(result));
554 c = ring_elem(result);

589namespace RingPromoter {
590 template <typename SourceRing, typename TargetRing>
591 bool promoter(const Ring *R,
592 const Ring *S,
593 const ring_elem fR,
594 ring_elem &resultS)
595 {
596 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
597 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
598 const SourceRing &R1 =
599 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
600 const TargetRing &S1 =
601 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
602
603 typename SourceRing::Element fR1(R1);
604 typename TargetRing::Element gS1(S1);
605
606 R1.from_ring_elem(fR1, fR);
607 bool retval = mypromote(R1, S1, fR1, gS1);
608 if (retval) S1.to_ring_elem(resultS, gS1);
609 return retval;
610 }
611
612 template <typename SourceRing, typename TargetRing>
613 bool lifter(const Ring *R,
614 const Ring *S,
615 ring_elem &result_gR,
616 const ring_elem gS)
617 {
618 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
619 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
620 const SourceRing &R1 =
621 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
622 const TargetRing &S1 =
623 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
624
625 typename SourceRing::Element fR1(R1);
626 typename TargetRing::Element gS1(S1);
627
628 S1.from_ring_elem(gS1, gS);
629 bool retval = mylift(R1, S1, fR1, gS1); // sets fR1.
630 if (retval) R1.to_ring_elem(result_gR, fR1);
631 return retval;
632 }

635template <typename RingType>
636bool ConcreteRing<RingType>::promote(const Ring *R,
637 const ring_elem fR,
638 ring_elem &resultS) const
639{
640 const Ring *S = this;
641 // fprintf(stderr, "calling promote\n");
642 namespace RP = RingPromoter;
643 if (R == globalZZ)
644 {
645 resultS = S->from_int(fR.get_mpz());
646 return true;
647 }
648 if (R == S)
649 {
650 resultS = copy(fR);
651 return true;
652 }
653 switch (R->ringID())
654 {
655 case M2::ring_ZZp:
656 switch (S->ringID())
657 {
658 case M2::ring_ZZp:
659 return false;
660 case M2::ring_ZZpFfpack:
661 return RP::promoter<ARingZZp, ARingZZpFFPACK>(R, S, fR, resultS);
662 default:
663 return false;
664 }
665 break;
666 case M2::ring_ZZpFfpack:
667 switch (S->ringID())
668 {
669 case M2::ring_ZZp:
670 return RP::promoter<ARingZZpFFPACK, ARingZZp>(R, S, fR, resultS);
671 case M2::ring_ZZpFfpack:
672 return RP::promoter<ARingZZpFFPACK, ARingZZpFFPACK>(
673 R, S, fR, resultS);
674 default:
675 return false;
676 }
677 case M2::ring_QQ:
678 switch (S->ringID())
679 {
680 case M2::ring_RR:
681 return RP::promoter<ARingQQ, ARingRR>(R, S, fR, resultS);
682 case M2::ring_RRR:
683 return RP::promoter<ARingQQ, ARingRRR>(R, S, fR, resultS);
684 case M2::ring_RRi:
685 return RP::promoter<ARingQQ, ARingRRi>(R, S, fR, resultS);
686 case M2::ring_CC:
687 return RP::promoter<ARingQQ, ARingCC>(R, S, fR, resultS);
688 case M2::ring_CCC:
689 return RP::promoter<ARingQQ, ARingCCC>(R, S, fR, resultS);
690 case M2::ring_CCi:
691 return RP::promoter<ARingQQ, ARingCCi>(R, S, fR, resultS);
692 default:
693 return false;
694 }
695 case M2::ring_RR:
696 switch (S->ringID())
697 {
698 case M2::ring_RR:
699 return RP::promoter<ARingRR, ARingRR>(R, S, fR, resultS);
700 case M2::ring_RRR:
701 return RP::promoter<ARingRR, ARingRRR>(R, S, fR, resultS);
702 case M2::ring_RRi:
703 return RP::promoter<ARingRR, ARingRRi>(R, S, fR, resultS);
704 case M2::ring_CC:
705 return RP::promoter<ARingRR, ARingCC>(R, S, fR, resultS);
706 case M2::ring_CCC:
707 return RP::promoter<ARingRR, ARingCCC>(R, S, fR, resultS);
708 case M2::ring_CCi:
709 return RP::promoter<ARingRR, ARingCCi>(R, S, fR, resultS);
710 default:
711 return false;
712 }
713 case M2::ring_RRR:
714 switch (S->ringID())
715 {
716 case M2::ring_RR:
717 return RP::promoter<ARingRRR, ARingRR>(R, S, fR, resultS);
718 case M2::ring_RRR:
719 return RP::promoter<ARingRRR, ARingRRR>(R, S, fR, resultS);
720 case M2::ring_RRi:
721 return RP::promoter<ARingRRR, ARingRRi>(R, S, fR, resultS);
722 case M2::ring_CC:
723 return RP::promoter<ARingRRR, ARingCC>(R, S, fR, resultS);
724 case M2::ring_CCC:
725 return RP::promoter<ARingRRR, ARingCCC>(R, S, fR, resultS);
726 case M2::ring_CCi:
727 return RP::promoter<ARingRRR, ARingCCi>(R, S, fR, resultS);
728 default:
729 return false;
730 }
731 case M2::ring_RRi:
732 switch (S->ringID())
733 {
734 case M2::ring_RR:
735 return RP::promoter<ARingRRi, ARingRR>(R, S, fR, resultS);
736 case M2::ring_RRR:

783// given a natural map: R --> S = this,
784// 'lift' sets result_gR with an element which maps to gS, if possible.
785// true is returned iff this is possible.
786template <typename RingType>
787bool ConcreteRing<RingType>::lift(const Ring *R,
788 const ring_elem gS,
789 ring_elem &result_gR) const
790{
791 const Ring *S = this;
792
793 namespace RP = RingPromoter;
794 if (R == S)
795 {
796 result_gR = gS;
797 return true;
798 }
799 if (R == globalZZ)
800 {
801 printf("error!! lift called with no ZZ lifting method\n");
802 // MES:TODO!! WRITE ME
803 return true;
804 }
805 switch (R->ringID())
806 {
807 case M2::ring_ZZp:
808 switch (S->ringID())
809 {
810 case M2::ring_ZZp:
811 return false;
812 case M2::ring_ZZpFfpack:
813 return RP::lifter<ARingZZp, ARingZZpFFPACK>(R, S, result_gR, gS);
814 default:
815 return false;
816 }
817 break;
818 case M2::ring_ZZpFfpack:
819 switch (S->ringID())
820 {
821 case M2::ring_ZZp:
822 return RP::lifter<ARingZZpFFPACK, ARingZZp>(R, S, result_gR, gS);
823 case M2::ring_ZZpFfpack:
824 return RP::lifter<ARingZZpFFPACK, ARingZZpFFPACK>(
825 R, S, result_gR, gS);
826 default:
827 return false;
828 }
829 case M2::ring_RR:
830 switch (S->ringID())
831 {
832 case M2::ring_RR:
833 return RP::lifter<ARingRR, ARingRR>(R, S, result_gR, gS);
834 case M2::ring_RRR:
835 return RP::lifter<ARingRR, ARingRRR>(R, S, result_gR, gS);
836 case M2::ring_RRi:
837 return RP::lifter<ARingRR, ARingRRi>(R, S, result_gR, gS);
838 case M2::ring_CC:
839 return RP::lifter<ARingRR, ARingCC>(R, S, result_gR, gS);
840 case M2::ring_CCC:
841 return RP::lifter<ARingRR, ARingCCC>(R, S, result_gR, gS);
842 default:
843 return false;
844 }
845 case M2::ring_RRR:
846 switch (S->ringID())
847 {
848 case M2::ring_RR:
849 return RP::lifter<ARingRRR, ARingRR>(R, S, result_gR, gS);
850 case M2::ring_RRR:
851 return RP::lifter<ARingRRR, ARingRRR>(R, S, result_gR, gS);
852 case M2::ring_RRi:
853 return RP::lifter<ARingRRR, ARingRRi>(R, S, result_gR, gS);
854 case M2::ring_CC:
855 return RP::lifter<ARingRRR, ARingCC>(R, S, result_gR, gS);
856 case M2::ring_CCC:
857 return RP::lifter<ARingRRR, ARingCCC>(R, S, result_gR, gS);
858 default:
859 return false;
860 }
861 case M2::ring_CC:
862 switch (S->ringID())
863 {
864 case M2::ring_CC:
865 return RP::lifter<ARingCC, ARingCC>(R, S, result_gR, gS);
866 case M2::ring_CCC:
867 return RP::lifter<ARingCCC, ARingCC>(R, S, result_gR, gS);
868 default:
869 return false;
870 }
871 case M2::ring_CCC:
872 switch (S->ringID())
873 {
874 case M2::ring_CC:
875 return RP::lifter<ARingCC, ARingCCC>(R, S, result_gR, gS);
876 case M2::ring_CCC:
877 return RP::lifter<ARingCCC, ARingCCC>(R, S, result_gR, gS);
878 default:
879 return false;
880 }
881 default:
882#ifndef NDEBUG
883 fprintf(stderr,
884 "oh no: rings not in list\n, R->ringID()=%d S->ringID()=%d\n",
885 R->ringID(),
886 S->ringID());
887#endif
888 break;
889 };
890 return false;
891}
