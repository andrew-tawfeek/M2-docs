539 bool lower_associate_divisor(ring_elem &f, const ring_elem g) const
540 {
541 mpq_srcptr a = MPQ_VAL(f);
542 mpq_srcptr b = MPQ_VAL(g);
543 int sa = mpq_sgn(a);
544 int sb = mpq_sgn(b);
545 int s = (sa == 0 ? sb : sa);
546 mpq_ptr result = getmemstructtype(mpq_ptr);
547 mpq_init(result);
548
549 mpz_gcd(mpq_numref(result), mpq_numref(a), mpq_numref(b));
550 mpz_lcm(mpq_denref(result), mpq_denref(a), mpq_denref(b));
551 if (s != mpq_sgn(result)) mpq_neg(result, result);
552 mpz_reallocate_limbs(mpq_numref(result));
553 mpz_reallocate_limbs(mpq_denref(result));
554 f = ring_elem(result);
555 return true; // the answer could become lower, if a newer g has a larger
556 // denom
557 }

563 c = g;

565 }

567 mpq_srcptr b = MPQ_VAL(g);

612namespace RingPromoter {
613 template <typename SourceRing, typename TargetRing>
614 bool promoter(const Ring *R,
615 const Ring *S,
616 const ring_elem fR,
617 ring_elem &resultS)
618 {
619 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
620 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
621 const SourceRing &R1 =
622 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
623 const TargetRing &S1 =
624 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
625
626 typename SourceRing::Element fR1(R1);
627 typename TargetRing::Element gS1(S1);
628
629 R1.from_ring_elem(fR1, fR);
630 bool retval = mypromote(R1, S1, fR1, gS1);
631 if (retval) S1.to_ring_elem(resultS, gS1);
632 return retval;
633 }
634
635 template <typename SourceRing, typename TargetRing>
636 bool lifter(const Ring *R,
637 const Ring *S,
638 ring_elem &result_gR,
639 const ring_elem gS)
640 {
641 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
642 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
643 const SourceRing &R1 =
644 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
645 const TargetRing &S1 =
646 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
647
648 typename SourceRing::Element fR1(R1);
649 typename TargetRing::Element gS1(S1);
650
651 S1.from_ring_elem(gS1, gS);
652 bool retval = mylift(R1, S1, fR1, gS1); // sets fR1.
653 if (retval) R1.to_ring_elem(result_gR, fR1);
654 return retval;
655 }

658template <typename RingType>
659bool ConcreteRing<RingType>::promote(const Ring *R,
660 const ring_elem fR,
661 ring_elem &resultS) const
662{
663 const Ring *S = this;
664 // fprintf(stderr, "calling promote\n");
665 namespace RP = RingPromoter;
666 if (R == globalZZ)
667 {
668 resultS = S->from_int(fR.get_mpz());
669 return true;
670 }
671 if (R == S)
672 {
673 resultS = copy(fR);
674 return true;
675 }
676 switch (R->ringID())
677 {
678 case M2::ring_ZZp:
679 switch (S->ringID())
680 {
681 case M2::ring_ZZp:
682 return false;
683 case M2::ring_ZZpFfpack:
684 return RP::promoter<ARingZZp, ARingZZpFFPACK>(R, S, fR, resultS);
685 default:
686 return false;
687 }
688 break;
689 case M2::ring_ZZpFfpack:
690 switch (S->ringID())
691 {
692 case M2::ring_ZZp:
693 return RP::promoter<ARingZZpFFPACK, ARingZZp>(R, S, fR, resultS);
694 case M2::ring_ZZpFfpack:
695 return RP::promoter<ARingZZpFFPACK, ARingZZpFFPACK>(
696 R, S, fR, resultS);
697 default:
698 return false;
699 }
700 case M2::ring_QQ:
701 switch (S->ringID())
702 {
703 case M2::ring_RR:
704 return RP::promoter<ARingQQ, ARingRR>(R, S, fR, resultS);
705 case M2::ring_RRR:
706 return RP::promoter<ARingQQ, ARingRRR>(R, S, fR, resultS);
707 case M2::ring_RRi:
708 return RP::promoter<ARingQQ, ARingRRi>(R, S, fR, resultS);
709 case M2::ring_CC:
710 return RP::promoter<ARingQQ, ARingCC>(R, S, fR, resultS);
711 case M2::ring_CCC:
712 return RP::promoter<ARingQQ, ARingCCC>(R, S, fR, resultS);
713 case M2::ring_CCi:
714 return RP::promoter<ARingQQ, ARingCCi>(R, S, fR, resultS);
715 default:
716 return false;
717 }
718 case M2::ring_RR:
719 switch (S->ringID())
720 {
721 case M2::ring_RR:
722 return RP::promoter<ARingRR, ARingRR>(R, S, fR, resultS);
723 case M2::ring_RRR:
724 return RP::promoter<ARingRR, ARingRRR>(R, S, fR, resultS);
725 case M2::ring_RRi:
726 return RP::promoter<ARingRR, ARingRRi>(R, S, fR, resultS);
727 case M2::ring_CC:
728 return RP::promoter<ARingRR, ARingCC>(R, S, fR, resultS);
729 case M2::ring_CCC:
730 return RP::promoter<ARingRR, ARingCCC>(R, S, fR, resultS);
731 case M2::ring_CCi:
732 return RP::promoter<ARingRR, ARingCCi>(R, S, fR, resultS);
733 default:
734 return false;
735 }
736 case M2::ring_RRR:
737 switch (S->ringID())
738 {
739 case M2::ring_RR:
740 return RP::promoter<ARingRRR, ARingRR>(R, S, fR, resultS);
741 case M2::ring_RRR:
742 return RP::promoter<ARingRRR, ARingRRR>(R, S, fR, resultS);
743 case M2::ring_RRi:
744 return RP::promoter<ARingRRR, ARingRRi>(R, S, fR, resultS);
745 case M2::ring_CC:
746 return RP::promoter<ARingRRR, ARingCC>(R, S, fR, resultS);
747 case M2::ring_CCC:
748 return RP::promoter<ARingRRR, ARingCCC>(R, S, fR, resultS);
749 case M2::ring_CCi:
750 return RP::promoter<ARingRRR, ARingCCi>(R, S, fR, resultS);
751 default:
752 return false;
753 }
754 case M2::ring_RRi:
755 switch (S->ringID())
756 {
757 case M2::ring_RR:
758 return RP::promoter<ARingRRi, ARingRR>(R, S, fR, resultS);
759 case M2::ring_RRR:

806// given a natural map: R --> S = this,
807// 'lift' sets result_gR with an element which maps to gS, if possible.
808// true is returned iff this is possible.
809template <typename RingType>
810bool ConcreteRing<RingType>::lift(const Ring *R,
811 const ring_elem gS,
812 ring_elem &result_gR) const
813{
814 const Ring *S = this;
815
816 namespace RP = RingPromoter;
817 if (R == S)
818 {
819 result_gR = gS;
820 return true;
821 }
822 if (R == globalZZ)
823 {
824 printf("error!! lift called with no ZZ lifting method\n");
825 // MES:TODO!! WRITE ME
826 return true;
827 }
828 switch (R->ringID())
829 {
830 case M2::ring_ZZp:
831 switch (S->ringID())
832 {
833 case M2::ring_ZZp:
834 return false;
835 case M2::ring_ZZpFfpack:
836 return RP::lifter<ARingZZp, ARingZZpFFPACK>(R, S, result_gR, gS);
837 default:
838 return false;
839 }
840 break;
841 case M2::ring_ZZpFfpack:
842 switch (S->ringID())
843 {
844 case M2::ring_ZZp:
845 return RP::lifter<ARingZZpFFPACK, ARingZZp>(R, S, result_gR, gS);
846 case M2::ring_ZZpFfpack:
847 return RP::lifter<ARingZZpFFPACK, ARingZZpFFPACK>(
848 R, S, result_gR, gS);
849 default:
850 return false;
851 }
852 case M2::ring_RR:
853 switch (S->ringID())
854 {
855 case M2::ring_RR:
856 return RP::lifter<ARingRR, ARingRR>(R, S, result_gR, gS);
857 case M2::ring_RRR:
858 return RP::lifter<ARingRR, ARingRRR>(R, S, result_gR, gS);
859 case M2::ring_RRi:
860 return RP::lifter<ARingRR, ARingRRi>(R, S, result_gR, gS);
861 case M2::ring_CC:
862 return RP::lifter<ARingRR, ARingCC>(R, S, result_gR, gS);
863 case M2::ring_CCC:
864 return RP::lifter<ARingRR, ARingCCC>(R, S, result_gR, gS);
865 default:
866 return false;
867 }
868 case M2::ring_RRR:
869 switch (S->ringID())
870 {
871 case M2::ring_RR:
872 return RP::lifter<ARingRRR, ARingRR>(R, S, result_gR, gS);
873 case M2::ring_RRR:
874 return RP::lifter<ARingRRR, ARingRRR>(R, S, result_gR, gS);
875 case M2::ring_RRi:
876 return RP::lifter<ARingRRR, ARingRRi>(R, S, result_gR, gS);
877 case M2::ring_CC:
878 return RP::lifter<ARingRRR, ARingCC>(R, S, result_gR, gS);
879 case M2::ring_CCC:
880 return RP::lifter<ARingRRR, ARingCCC>(R, S, result_gR, gS);
881 default:
882 return false;
883 }
884 case M2::ring_CC:
885 switch (S->ringID())
886 {
887 case M2::ring_CC:
888 return RP::lifter<ARingCC, ARingCC>(R, S, result_gR, gS);
889 case M2::ring_CCC:
890 return RP::lifter<ARingCCC, ARingCC>(R, S, result_gR, gS);
891 default:
892 return false;
893 }
894 case M2::ring_CCC:
895 switch (S->ringID())
896 {
897 case M2::ring_CC:
898 return RP::lifter<ARingCC, ARingCCC>(R, S, result_gR, gS);
899 case M2::ring_CCC:
900 return RP::lifter<ARingCCC, ARingCCC>(R, S, result_gR, gS);
901 default:
902 return false;
903 }
904 default:
905#ifndef NDEBUG
906 fprintf(stderr,
907 "oh no: rings not in list\n, R->ringID()=%d S->ringID()=%d\n",
908 R->ringID(),
909 S->ringID());
910#endif
911 break;
912 };
913 return false;
914}
