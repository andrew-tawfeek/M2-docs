22// A priority queue on S-pairs where the priority is based on a
23// signature as in signature Grobner basis algorithms. The class is
24// not responsible for eliminating S-pairs or doing anything beyond
25// order the S-pairs.

27public:

39 //typedef std::pair<size_t, monomial> IndexSig;

42
43 // Takes the minimal signature in the queue and adds all S-pairs of
44 // that signature to pairs. Clears pairs first. Returns null and
45 // leaves pairs empty if the queue is empty.
46 //
47 // This class does not have an empty() method on purpose - you are
48 // supposed to call this method until it returns null.
49 virtual Mono popSignature(Pairs& pairs) = 0;
50
51 // If (x, sig) is an element of pairsConsumed then (pairWith, x) is
52 // added to the queue. sig must be the signature of the S-pair
53 // (pairWith, x).
54 //
55 // ATTENTION: the calls to pushPairs must have pairWith in the
56 // sequence 0, 1, 2, 3 and so on. It follows from this that the
57 // queue can figure out what pairWith is without being told. Thus
58 // the purpose of pairWith is to make it possible to make an
59 // assertion saying that the caller and the queue agree on what
60 // pairWith is.
61 //
62 // ATTENTION: pairsConsumed will be cleared and the signatures in it
63 // will be freed on the ring. This is because the queue will need to
64 // alter pairsConsumed in various ways and clearing it after that is
65 // cleaner than exposing what's being done to
66 // pairsConsumed. Especially since after you did pushPairs there
67 // would not be a reason to care about what its content was.
68 virtual void pushPairs(size_t pairWith, IndexSigs& pairsConsumed) = 0;
69
70 // Returns a string that describes the queue.
71 virtual std::string name() const = 0;
72
73 // Returns the number of pairs currently in the queue.
74 virtual size_t pairCount() const = 0;
75
76 // Returns number of bytes of memory used.
77 virtual size_t memoryUse() const = 0;
78
79 static std::unique_ptr<SigSPairQueue> create(SigPolyBasis const& basis);
80};
