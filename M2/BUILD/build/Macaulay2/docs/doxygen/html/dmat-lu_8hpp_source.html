337template <class RingType>
338bool DMatLinAlg<RingType>::solve(const Mat& B, Mat& X)
339{
340 // printf("in dmat-lu: solve\n");
341 const Mat& LU = mLUObject.LUinPlace();
342 // printf("in dmat-lu: after LU solve\n");
343
344 // For each column of B, we solve it separately.
345
346 // Step 1: Take a column, and permute it via mPerm, into b.
347 // Step 2: Solve the lower triangular system Ly=b
348 // If there is no solution, then return false.
349 // Step 3: Solve the upper triangular system Ux = y.
350
351 // Sizes of objects here:
352 // A is m x n
353 // L is m x r
354 // U is r x n. Here, r <= m, and A has rank r.
355 // b is a vector 0..m-1
356 // y is a vector 0..r-1
357 // x is a vector 0..n-1
358
359 // printf("entering DMatLinAlg::solve\n");
360 const std::vector<size_t>& pivotColumns = mLUObject.pivotColumns();
361 const std::vector<size_t>& perm = mLUObject.permutation();
362 size_t rk = pivotColumns.size();
363
364 typename RingType::Element tmp(ring()), tmp2(ring());
365
366 typedef typename RingType::ElementArray ElementArray;
367 ElementArray b(ring(), LU.numRows());
368 ElementArray y(ring(), rk);
369 ElementArray x(ring(), LU.numColumns());
370
371 X.resize(LU.numColumns(), B.numColumns());
372
373 for (size_t col = 0; col < B.numColumns(); col++)
374 {
375 // Step 0: erase x (b will be set below, y is also set when needed).
376
377 for (size_t i = 0; i < LU.numColumns(); i++) ring().set_zero(x[i]);
378
379 // Step 1: set b to be the permuted i-th column of B.
380 for (size_t r = 0; r < B.numRows(); r++)
381 ring().set(b[r], B.entry(perm[r], col));
382
385
386 // Step 2: Solve Ly=b
387 for (size_t i = 0; i < rk; i++)
388 {
389 ring().set(y[i], b[i]);
390 for (size_t j = 0; j < i; j++)
391 {
392 ring().mult(tmp, LU.entry(i, j), y[j]);
393 ring().subtract(y[i], y[i], tmp);
394 }
395 }
396
399
400 // Step 2B: see if the solution is consistent
401 for (size_t i = rk; i < LU.numRows(); i++)
402 {
403 ring().set(tmp, b[i]);
404 for (size_t j = 0; j < rk; j++)
405 {
406 ring().mult(tmp2, LU.entry(i, j), y[j]);
407 ring().subtract(tmp, tmp, tmp2);
408 }
409 if (!ring().is_zero(tmp))
410 {
411 // printf("returning false\n");
412 return false;
413 }
414 }
415
417
418 // Step 3: Solve Ux=y
419 // and place x back into X as col-th column
420 for (long i = rk - 1; i >= 0; --i)
421 {
422 ring().set(x[i], y[i]);
423 for (size_t j = i + 1; j <= rk - 1; j++)
424 {
425 ring().mult(tmp, LU.entry(i, pivotColumns[j]), x[j]);
426 ring().subtract(x[i], x[i], tmp);
427 }
428 ring().divide(x[i], x[i], LU.entry(i, pivotColumns[i]));
429 ring().set(X.entry(pivotColumns[i], col), x[i]);
430
435 }
438
443 }
444
445 return true; // The system seems to have been consistent
446}

577template <class RingType>
578size_t DMatLinAlg<RingType>::kernel(Mat& X)
579{
580 const Mat& LU = mLUObject.LUinPlace();
581 const std::vector<size_t>& pivotColumns = mLUObject.pivotColumns();
582
583 typename RingType::Element tmp(ring()), tmp2(ring());
584
585 // First, let's set X to be the correct size:
586 X.resize(LU.numColumns(), LU.numColumns() - pivotColumns.size());
587
588 size_t col = 0;
589 size_t nextpivotidx = 0;
590 size_t nextpivotcol =
591 (pivotColumns.size() > 0 ? pivotColumns[0] : LU.numColumns());
592 size_t colX = 0;
593 while (colX < X.numColumns())
594 {
595 if (col == nextpivotcol)
596 {
597 col++;
598 nextpivotidx++;
599 nextpivotcol =
600 (nextpivotidx < pivotColumns.size() ? pivotColumns[nextpivotidx]
601 : LU.numColumns());
602 continue;
603 }
604 // At this point, we are ready to create a column of X.
605 ring().set_from_long(X.entry(col, colX), -1);
606 // Now we loop through and set the elements in the rows of X = pivot
607 // columns.
608 for (long p = nextpivotidx - 1; p >= 0; p--)
609 {
610 // set X.entry(pivotColumns[p], colX)
611 ring().set(tmp, LU.entry(p, col));
612 for (size_t i = nextpivotidx - 1; i >= p + 1; i--)
613 {
614 ring().mult(tmp2,
615 LU.entry(p, pivotColumns[i]),
616 X.entry(pivotColumns[i], colX));
617 ring().subtract(tmp, tmp, tmp2);
618 }
619 ring().divide(tmp, tmp, LU.entry(p, pivotColumns[p]));
620 ring().set(X.entry(pivotColumns[p], colX), tmp);
621 }
622 colX++;
623 col++;
624 }
625 return X.numColumns();
626}
