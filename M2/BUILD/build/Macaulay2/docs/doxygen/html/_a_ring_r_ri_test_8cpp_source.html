45bool almostEqual(const M2::ARingRRi& R,
46 int nbits,
47 const M2::ARingRRi::ElementType& a,
48 const M2::ARingRRi::ElementType& b)
49{
50 mpfr_t epsilon;
51 mpfr_init2(epsilon, R.get_precision());
52 mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN);
53
54 mpfr_t c,d;
55 mpfr_init2(c, R.get_precision());
56 mpfr_init2(d, R.get_precision());
57
58 mpfr_sub(c,&(a.left),&(b.left),MPFR_RNDN);
59 mpfr_sub(d,&(a.right),&(b.right),MPFR_RNDN);
60
61 bool retL = mpfr_cmpabs(c, epsilon) < 0,
62 retR = mpfr_cmpabs(d, epsilon) < 0;
63
64 mpfr_clear(d);
65 mpfr_clear(c);
66 mpfr_clear(epsilon);
67 return retL and retR;
68}

222 {

226 // Test commutativity
227 // test: a*b = b*a
228 // test: a+b == b+a

236 // Test associativity

238 // test: a*(b*c) == (a*b)*c

243 EXPECT_TRUE(almostEqual(R,-94,d,e));
244
245 R.mult(e, b, c);

251
252 // Test distributivity
