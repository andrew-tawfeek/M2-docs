17 void BufferPool::growCapacity() {
18 // ** Calcuate size of block (doubles capacity)
19 size_t size = block().getBytesInBlock();
20 if (size == 0) {
21 // start out at 10 buffers
22 MEMT_ASSERT(block().isNull());
23 if (_bufferSize > std::numeric_limits<size_t>::max() / 10)
24 throw std::bad_alloc(); // _bufferSize * 10 overflows
25 size = _bufferSize * 10;
26 } else {
27 // double the size
28 if (size > std::numeric_limits<size_t>::max() / 2)
29 throw std::bad_alloc(); // size * 2 overflows
30 size *= 2;
31 }
32
33 // ** Allocate next block
34 MEMT_ASSERT(MemoryBlocks::alignNoOverflow(size) == size);
35 MEMT_ASSERT(size > block().getBytesInBlock());
36 _blocks.allocBlock(size);
37 }
