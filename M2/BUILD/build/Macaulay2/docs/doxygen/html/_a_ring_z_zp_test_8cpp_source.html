71template <typename RT>
72void testCoerceToLongInteger(const RT& R)
73{
74 long top1 = (R.characteristic() > 10000 ? 10000 : R.characteristic());
75 long bottom2 = (R.characteristic() > 10000 ? R.characteristic() - 10000 : 0);
76 for (long i = 0; i < top1; i++)
77 {
78 typename RT::ElementType a;
79 R.init(a);
80 R.set_from_long(a, i);
81 long b = R.coerceToLongInteger(a);
82 if (b < 0) b += R.characteristic();
83 EXPECT_EQ(b, i);
84 }
85 for (long i = bottom2; i < R.characteristic(); i++)
86 {
87 typename RT::ElementType a;
88 R.init(a);
89 R.set_from_long(a, i);
90 long b = R.coerceToLongInteger(a);
91 if (b < 0) b += R.characteristic();
92 EXPECT_EQ(b, i);
93 }
94}

156{
157 std::istringstream i("+1234 +345 -235*a");
158 M2::ARingZZp R(32003);
159 M2::ARingZZp::ElementType a;
160 R.init(a);
161 while (true)
162 {
163 while (isspace(i.peek())) i.get();
164
165 if (!isdigit(i.peek()) && i.peek() != '+' && i.peek() != '-') break;
166
167 fromStream(i, R, a);
168
169 buffer o;
170 R.elem_text_out(o, a);
171 std::cout << o.str() << " peek: " << i.peek() << std::endl;
172 }
173 R.clear(a);
