65template<template<typename> class Q>
66void ReducerHash<Q>::insertTail(NewConstTerm multiplier, const Poly& f) {
67 if (f.termCount() <= 1)
68 return;
69
70 mNodesTmp.clear();
71 auto it = f.begin();
72 const auto end = f.end();
73 for (++it; it != end; ++it) {
74 auto p = mHashTable.insertProduct(*it, multiplier);
75 if (p.second)
76 mNodesTmp.emplace_back(p.first);
77 }
78 if (!mNodesTmp.empty())
79 mQueue.push(mNodesTmp.begin(), mNodesTmp.end());
80}

82template<template<typename> class Q>
83void ReducerHash<Q>::insert(ConstMonoRef multiplier, const Poly& f) {
84 mNodesTmp.clear();
85 const auto end = f.end();
86 for (auto it = f.begin(); it != end; ++it) {
87 auto p = mHashTable.insertProduct(it.mono(), multiplier, it.coef());
88 if (p.second)
89 mNodesTmp.emplace_back(p.first);
90 }
91 if (!mNodesTmp.empty())
92 mQueue.push(mNodesTmp.begin(), mNodesTmp.end());
93}

95template<template<typename> class Q>
96bool ReducerHash<Q>::leadTerm(NewConstTerm& result) {
97 while (!mQueue.empty()) {
98 const auto top = mQueue.top();
99 if (!mRing.coefficientIsZero(top->value())) {
100 result.coef = top->value();
101 result.mono = top->mono().ptr();
102 return true;
103 }
104 mQueue.pop();
105 mHashTable.remove(top);
106 }
107 return false;
108}
