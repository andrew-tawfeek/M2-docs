66template<class Iterator1, class Iterator2>
67class Zip {
68public:
69 typedef decltype(*std::declval<Iterator1>()) ValueType1;
70 typedef decltype(*std::declval<Iterator2>()) ValueType2;
71
76 typedef std::forward_iterator_tag iterator_category;
77 typedef std::pair<ValueType1, ValueType2> value_type;
78 typedef ptrdiff_t difference_type;
79 typedef value_type* pointer;
80 typedef value_type& reference;
81
82 Zip() {}
83 Zip(Iterator1 it1, Iterator2 it2): mIt1(it1), mIt2(it2) {}
84
85 Zip& operator++() {
86 ++mIt1;
87 ++mIt2;
88 return *this;
89 }
90
91 value_type operator*() const {
92 // We cannot use std::make_pair here. If .first or .second is a
93 // reference, then std::make_pair will remove the reference and then the
94 // conversion to value_type will add the reference back in - but now the
95 // reference is not what was originally referenced, instead it is a
96 // reference to the temporary object returned by std::make_pair, which
97 // then promptly goes out of scope, leading to undefined behavior.
98 return value_type(*mIt1, *mIt2);
99 }
100
102 bool operator!=(const Zip& it) const {return mIt1 != it.mIt1;}
103
105 bool operator==(const Zip& it) const {return mIt1 == it.mIt1;}
106
107private:
108 Iterator1 mIt1;
109 Iterator2 mIt2;
110};

360template<class Iterator>
361auto adjPairRange(
362 const Iterator& begin,
363 const Iterator& end
364) -> decltype(oppositePairRange(zip(end, end, end, end))) {
365 if (begin == end)
366 return oppositePairRange(zip(end, end, end, end));
367 // This is a bit tricky. What we really want is
368 // zip(begin, end - 1, begin + 1, end)
369 // The -1 is bad because we want this to work for forward iterators
370 // that do not have operator--(). We cannot just leave out the -1 because
371 // zip defines the length of the range according to the first component
372 // range. However, if we swap the ranges around to
373 // zip(begin + 1, end, begin, end - 1)
374 // then we can replace this by
375 // zip(begin + 1, end, begin, end)
376 // because zip does not actually care about the length of the second range.
377 // The problem now is that the pairs of elements will be the wrong way
378 // around, but oppositePairRange solves that nicely.
379 auto pastBegin = begin;
380 ++pastBegin;
381 return oppositePairRange(zip(pastBegin, end, begin, end));
382}
