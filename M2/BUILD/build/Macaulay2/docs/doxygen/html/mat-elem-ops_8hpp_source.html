32 static size_t lead_row(const Mat& mat, size_t col)
33 /* returns the largest index row which has a non-zero value in column 'col'.
34 returns -1 if the column has no non-zero entries */
35 {
36 size_t row = mat.numRows();
37 while (row != 0)
38 {
39 --row;
40 if (!mat.ring().is_zero(mat.entry(row, col))) return row;
41 }
42 return static_cast<size_t>(-1);
43 }

45 static size_t lead_row(const Mat& mat, size_t col, ElementType& result)
46 /* returns the largest index row which has a non-zero value in column 'col'.
47 Also sets result to be the entry at this index.
48 returns -1 if the column is 0, or if col is out of range
49 No error is flagged. */
50 {
51 size_t row = mat.numRows();
52 while (row != 0)
53 {
54 --row;
55 if (!mat.ring().is_zero(mat.entry(row, col)))
56 {
57 mat.ring().set(result, mat.entry(row, col));
58 return row;
59 }
60 }
61 return static_cast<size_t>(-1);

168 const ElementType& b1,

257 static bool row_permute(Mat& mat, size_t start_row, M2_arrayint perm)
258 {
259 // We copy one row to another location for each cycle in 'perm' of length > 1
260 size_t nrows_to_permute = perm->len;
261 std::unique_ptr<bool[]> done(new bool[nrows_to_permute]);
262 for (size_t i = 0; i < nrows_to_permute; i++) done[i] = true;
263 // We check that this is a permutation
264 for (size_t i = 0; i < nrows_to_permute; i++)
265 {
266 size_t j = perm->array[i];
267 if (!done[j])
268 {
269 ERROR("expected permutation");
270 return false;
271 }
272 done[j] = false;
273 }
274 typename RT::ElementArray tmp(mat.ring(), mat.numColumns());
275 size_t next = 0;
276
277 while (next < nrows_to_permute)
278 {
279 if (done[next] || perm->array[next] == next)
280 {
281 next++;
282 }
283 else
284 {
285 // store row 'next' into tmp
286 for (int i=0; i<mat.numColumns(); ++i) std::swap(tmp[i], mat.entry(start_row + next, i));
287
288 size_t r = next;
289 while (perm->array[r] != next)
290 {
291 // copy row perm[r] to row r
292 for (int i = 0; i < mat.numColumns(); ++i)
293 std::swap(mat.entry(start_row + perm->array[r], i),
294 mat.entry(start_row + r, i));
295 done[r] = true;
296 r = perm->array[r];
297 }
298 // we swap tmp and r
299 for (int i=0; i<mat.numColumns(); ++i) std::swap(tmp[i], mat.entry(start_row + r, i));
300 done[r] = true;
301 }
302 }
303 return true;
304 }

306 static bool column_permute(Mat& mat, size_t start_col, M2_arrayint perm)
307 {
308 // We copy one col to another location for each cycle in 'perm' of length > 1
309 size_t ncols_to_permute = perm->len;
310 std::unique_ptr<bool[]> done(new bool[ncols_to_permute]);
311 for (size_t i = 0; i < ncols_to_permute; i++) done[i] = true;
312 // We check that this is a permutation
313 for (size_t i = 0; i < ncols_to_permute; i++)
314 {
315 size_t j = perm->array[i];
316 if (!done[j])
317 {
318 ERROR("expected permutation");
319 return false;
320 }
321 done[j] = false;
322 }
323 typename RT::ElementArray tmp(mat.ring(), mat.numRows());
324 size_t next = 0;
325
326 while (next < ncols_to_permute)
327 {
328 if (done[next] || perm->array[next] == next)
329 {
330 next++;
331 }
332 else
333 {
334 // store col 'next' into tmp
335 for (int i=0; i<mat.numRows(); ++i) std::swap(tmp[i], mat.entry(i, start_col + next));
336
337 size_t c = next;
338 while (perm->array[c] != next)
339 {
340 // swap col perm[c] to col c
341 for (int i = 0; i < mat.numRows(); ++i)
342 std::swap(mat.entry(i, start_col + perm->array[c]),
343 mat.entry(i, start_col + c));
344 done[c] = true;
345 c = perm->array[c];
346 }
347 // we swap tmp and c
348 for (int i=0; i<mat.numRows(); ++i) std::swap(tmp[i], mat.entry(i, start_col + c));
349 done[c] = true;
350 }
351 }
352 return true;

431 // An internal function for reduceby_pivots
432 static void perform_reduction(Mat& M,
433 size_t r,
434 size_t c,
435 size_t nr,
436 size_t nc,
437 int pivot_type)
438 // Subroutine of reduce_pivots()
439 // pivot_type: 1 means pivot is 1, -1 means pivot is -1, 0 means pivot is unit
440 {
441 // Flip rows r, nr
442 // Flip cols c, nc
443 // Use (nr,nc) location to remove all terms in columns 0..nc-1
444 // and in row nr.
445 // Replace column nc with all zeros, except 1 in nr row.
446
447 Element pivot(M.ring()), coef(M.ring()), f(M.ring()), zero(M.ring()),
448 one(M.ring());
449 M.ring().set_from_long(zero, 0);
450 M.ring().set_from_long(one, 1);
451
452 interchange_columns(M, c, nc);
453 interchange_rows(M, r, nr);
454 long pivotrow = lead_row(M, nc, pivot);
455 if (pivot_type == -1) // pivot is -1
456 scale_column(M, nc, pivot);
457 else if (pivot_type == 0)
458 divide_column(M, nc, pivot);
459 for (int i = 0; i < nc; i++)
460 {
461 pivotrow = lead_row(M, i, coef);
462 if (pivotrow < 0) continue;
463 if (pivotrow == nr)
464 {
465 // Do the reduction
466 M.ring().negate(f, coef);
467 column_op(M, i, f, nc);
468 }
469 }
470
471 scale_column(M, nc, zero);
472 setEntry(M, nr, nc, one);
473 }

476 static void reduce_by_pivots(Mat& M)
477 {
478 if (M.numRows() == 0 or M.numColumns() == 0) return;
479 size_t nr = M.numRows() - 1;
480 size_t nc = M.numColumns() - 1;
481
482 Element one(M.ring()), minus_one(M.ring());
483 M.ring().set_from_long(one, 1);
484 M.ring().set_from_long(minus_one, -1);
485
486 // After using the pivot element, it is moved to [nrows-1,ncols-1]
487 // and nrows and ncols are decremented.
488
489 for (size_t i = 0; i <= nc; i++)
490 {
491 for (size_t j = 0; j < M.numRows(); ++j)
492 {
493 auto& p = M.entry(j,i);
494 if (M.ring().is_zero(p)) continue;
495 int pivot_type = 0;
496 if (M.ring().is_equal(one, p))
497 pivot_type = 1;
498 else if (M.ring().is_equal(minus_one, p))
499 pivot_type = -1;
500 if (pivot_type != 0)
501 {
502 // printf("before reduction: j=%lu i=%lu:\n",j,i);
503 // displayMat(M);
504 perform_reduction(M, j, i, nr--, nc--, pivot_type);
505 // printf("after reduction: j=%lu i=%lu:\n",j,i);
506 // displayMat(M);
507 if (nr == static_cast<size_t>(-1) or
508 nc == static_cast<size_t>(-1))
509 return;
510 // restart loop with the (new) column i
511 i = -1;
512 break;
513 }
514 }
515 }
516
517 // Now search for other possible pivots
518 for (size_t i = 0; i <= nc; i++)
519 {
520 for (size_t j = 0; j < M.numRows(); ++j)
521 {
522 auto& p = M.entry(j,i);
523 if (M.ring().is_zero(p)) continue;
524 if (!M.ring().is_unit(p)) continue;
525 int pivot_type = 0;
526 if (M.ring().is_equal(one, p))
527 pivot_type = 1;
528 else if (M.ring().is_equal(minus_one, p))
529 pivot_type = -1;
530
531 // printf("before general reduction: j=%lu i=%lu:\n",j,i);
532 // displayMat(M);
533 perform_reduction(M, j, i, nr--, nc--, pivot_type);
534 // printf("after general reduction: j=%lu i=%lu:\n",j,i);
535 // displayMat(M);
536 if (nr == static_cast<size_t>(-1) or nc == static_cast<size_t>(-1))
537 return;
538 // restart loop with the (new) column i
539 i = -1;
540 break;
541 }
542 }

671 i = -1;
672 break;
673 }
674 }
675 }
676
677 // Now search for other possible pivots
678 for (int i=0; i<=nc; i++)
679 {
680 p->set(i);
681 for (; p->valid(); p->next())
682 {
683 ring_elem coef;
684 p->copy_ring_elem(coef);
685 if (!K->is_unit(coef)) continue;
686 int pivot_type = 0;
687 if (K->is_equal(one, coef))
688 pivot_type = 1;
689 else if (K->is_equal(minus_one, coef))
690 pivot_type = -1;
691
692 perform_reduction(M, static_cast<int>(p->row()), i, nr, nc, pivot_type);
693 if (nr == 0 or nc == 0) return;
694 --nr;
695 --nc;
696 // restart loop with the (new) column i
697 i = -1;
698 break;
699 }
700 }
701}
702#endif
703};
