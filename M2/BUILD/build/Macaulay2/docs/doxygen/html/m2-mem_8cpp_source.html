528void M2_debug_free(void *p) {
529 struct FRONT *f;
530 struct REAR *r;
531 int INTS_BODY, i, _trapcount;
532 size_t size;
533 if (p == NULL) return;

537 r = p + sizeof(int)*INTS_BODY;
538 _trapcount = f->trapcount;
