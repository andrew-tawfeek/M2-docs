66void LogDomain<true>::recordTime(TimeInterval interval) {

68 return;

71

92void LogDomain<true>::Timer::stop() {

94 return;
95 mTimerRunning = false;
96 if (!mLogger.enabled())
97 return;
98 TimeInterval interval;
99 interval.realSeconds = (mtbb::tick_count::now() - mRealTicks).seconds();
100 mLogger.recordTime(interval);
101 return;
