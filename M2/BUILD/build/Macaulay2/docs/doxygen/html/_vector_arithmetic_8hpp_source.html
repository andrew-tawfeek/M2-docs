213 void denseCancelFromSparse(ElementArray& dense,
214 const ElementArray& sparse,
215 const Range<const int>& comps) const
216 {
217 // ASSUMPTION: svec[0] == 1.
218 auto& dvec = * elementArray(dense);
219 auto& svec = * elementArray(sparse);
220

222 for (int i=1; i < comps.size(); ++i)
223 mRing->subtract_multiple(dvec[comps[i]], a, svec[i]);

227 void denseCancelFromSparse(ElementArray& dense,
228 const ElementArray& sparse,
229 const Range<const int>& comps,
230 ElementArray& result_multiplier) const
231 {
232 // ASSUMPTION: svec[0] == 1, comps.size() >= 1
233 auto& dvec = * elementArray(dense);
234 auto& svec = * elementArray(sparse);
235 auto& mults = * elementArray(result_multiplier);
236
237 FieldElement b;
238 mRing->init(b);
239 mRing->set(b, dvec[comps[0]]);
240
241 FieldElement one;
242 mRing->init(one);
243 mRing->set_from_long(one, 1);
244 if (not mRing->is_equal(svec[0], one)) // should be minus_one
245 mRing->negate(b, b);
246
247 for (int i=1; i < comps.size(); ++i)
248 {
249 mRing->subtract_multiple(dvec[comps[i]], b, svec[i]);
250 mStats.incrementNumAdditions();
251 }
252 mRing->set_zero(dvec[comps[0]]);
253
254 mRing->negate(b, b);
255 mults.push_back(b); // this transfers ownership of b.
256 }

294 ElementArray& sparse, // output value: sets this value
295 Range<int>& comps, // output value: sets comps

300 {
