44bool almostEqual(const M2::ARingRR& R,
45 unsigned long nbits,
46 const M2::ARingRR::ElementType& a,
47 const M2::ARingRR::ElementType& b)
48{
49 M2::ARingRR::ElementType epsilon = pow(2, static_cast<double>(-nbits));
50 M2::ARingRR::ElementType c;
51 R.subtract(c, a, b);
52 // std::cout << "a = " << a << ", b = " << b << ", c = " << c << ", a-b = " <<
53 // a-b;
54 R.abs(c, c);
55 // std::cout << ", |c| = " << c << ", epsilon = " << epsilon << std::endl;
56 return R.compare_elems(c, epsilon) < 0;
57}

201TEST(ARingRR, axioms)
202{
203 M2::ARingRR R;
204 auto nbits = R.get_precision();
205 ARingElementGenerator<M2::ARingRR> gen(R);
206 M2::ARingRR::ElementType a, b, c, d, e;
207 R.init(a);
208 R.init(b);
209 R.init(c);
210 R.init(d);
211 R.init(e);
212 for (int i = 0; i < ntrials; i++)
213 {
214 gen.nextElement(a);
215 gen.nextElement(b);
216 gen.nextElement(c);
217 // Test commutativity
218 // test: a*b = b*a
219 // test: a+b == b+a
220 R.add(d, a, b);
221 R.add(e, b, a);
222 EXPECT_TRUE(almostEqual(R, nbits - 2, d, e));
223 R.mult(d, a, b);
224 R.mult(e, b, a);
225 EXPECT_TRUE(almostEqual(R, nbits - 2, d, e));
226
227 // Test associativity
228 // test: a+(b+c) == (a+b)+c
229 // test: a*(b*c) == (a*b)*c
230 R.add(e, b, c);
231 R.add(d, a, e); // a+(b+c)
232 R.add(e, a, b);
233 R.add(e, e, c); // (a+b)+c
234 EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
235 R.mult(e, b, c);
236 R.mult(d, a, e); // a*(b*c)
237 R.mult(e, a, b);
238 R.mult(e, e, c); // (a*b)*c
239 EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
240
241 // Test distributivity
242 // test: a*(b+c) == a*b + a*c
243 R.add(e, b, c);
244 R.mult(d, a, e); // a*(b+c)
245 R.mult(b, a, b);
246 R.mult(c, a, c);
247 R.add(e, b, c); // a*b + a*c
248 EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
249 }
250 R.clear(e);
251 R.clear(d);
252 R.clear(c);
253 R.clear(b);
254 R.clear(a);
255}
