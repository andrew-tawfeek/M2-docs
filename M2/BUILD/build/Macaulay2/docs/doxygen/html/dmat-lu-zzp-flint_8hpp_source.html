91 bool solve(const Mat& B, Mat& X)
92 {
93 // printf("in dmat lu zzp flint solve\n");
94 long nrows = mMatrix.numRows();
95 long ncols = mMatrix.numColumns();
96 std::vector<size_t> profile;
97 Mat AB(mMatrix.ring(), nrows, ncols + B.numColumns());
98 concatenateMatrices<Mat>(mMatrix, B, AB);
99 nmod_mat_rref(AB.nmod_mat());
100 LUUtil<RingType>::computePivotColumns(AB, profile);
101 if (profile.size() >= 1 and profile[profile.size() - 1] >= ncols)
102 return false; // system is inconsistent
103 // At this point, we know the solutions. Should we go through FlintQQMat,
104 // or go directly to Mat?
105 X.resize(ncols, B.numColumns());
106 for (long c = 0; c < B.numColumns(); c++)
107 {
108 // Fill in this column
109 for (long r = 0; r < profile.size(); r++)
110 {
111 mMatrix.ring().set(X.entry(profile[r], c), AB.entry(r, ncols + c));
112 }
113 }
114 return true;
115 }
