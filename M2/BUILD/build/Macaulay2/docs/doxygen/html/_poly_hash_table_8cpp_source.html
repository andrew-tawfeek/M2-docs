25void PolyHashTable::rehash(const size_t requestedBucketCount) {
26 const auto newHashToIndexMask = computeHashMask(requestedBucketCount);
27 const auto newBucketCount = hashMaskToBucketCount(newHashToIndexMask);
28 auto newBuckets = make_unique_array<Node*>(newBucketCount);
29 std::fill_n(newBuckets.get(), newBucketCount, nullptr);
30
31 const auto bucketsEnd = mBuckets.get() + bucketCount();
32 for (auto bucket = mBuckets.get(); bucket != bucketsEnd; ++bucket) {
33 for (auto node = *bucket; node != 0;) {
34 const auto hash = monoid().hash(node->mono());
35 const auto newIndex = hashToIndex(hash, newHashToIndexMask);
36 const auto next = node->next();
37 node->next() = newBuckets[newIndex];
38 newBuckets[newIndex] = node;
39 node = next;
40 }
41 }
42
43 mHashToIndexMask = newHashToIndexMask;
44 mBuckets = std::move(newBuckets);
45 mMaxSize = static_cast<size_t>(bucketCount() * maxLoadFactor());
46}

48HashValue PolyHashTable::computeHashMask(const size_t requestedBucketCount) {
49 // round request up to nearest power of 2.
50 size_t pow2 = 1;
51 while (pow2 < requestedBucketCount && 2 * pow2 != 0)
52 pow2 *= 2;
53 MATHICGB_ASSERT(pow2 > 0 && (pow2 & (pow2 - 1)) == 0); // power of two
54
55 // If casting to a hash value overflows, then we get the maximum
56 // possible number of buckets based on the range of the hash
57 // value type. Only unsigned overflow is defined, so we need
58 // to assert that the hash type is unsigned.
59 static_assert(!std::numeric_limits<HashValue>::is_signed, "");
60 const auto hashToIndexMask = static_cast<HashValue>(pow2 - 1);
61 MATHICGB_ASSERT(pow2 == hashMaskToBucketCount(hashToIndexMask));
62 return hashToIndexMask;
63}
