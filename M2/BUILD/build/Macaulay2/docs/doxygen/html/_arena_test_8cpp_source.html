87TEST(Arena, BigAndOverflow) {
88 memt::Arena arena;
89 // aligning size causes overflow
90 ASSERT_THROW(arena.alloc(static_cast<size_t>(-1)), std::bad_alloc);
91
92 // 2x size is an overflow
93 ASSERT_THROW(arena.alloc(static_cast<size_t>(-1)/2 + 1), std::bad_alloc);
94
95 // causes attempt at allocating almost the entire virtual memory space
96 // which cannot succeed
97 ASSERT_THROW(arena.alloc(static_cast<size_t>(-1)/2 - 100), std::bad_alloc);
98
99 // sizeof(long) * x overflows to a smaller value (0).
100 const size_t smallerOverflow = 1ul << (8*sizeof(long) - 1);
101 MEMT_ASSERT(smallerOverflow > 0);
102 //ASSERT(smallerOverflow * sizeof(long) == 0); compiler warning
103 ASSERT_THROW(arena.allocArray<long>(smallerOverflow), std::bad_alloc);
104
105 // sizeof(int) * x overflows to a greater value
106 const size_t greaterOverflow = (~(0ul)) >> 1;
107 MEMT_ASSERT(sizeof(long) >= 4);
108 //ASSERT(greaterOverflow * sizeof(long) > greaterOverflow); compiler warning
109 //ASSERT(greaterOverflow != (greaterOverflow * sizeof(long)) / sizeof(long));
110 ASSERT_THROW(arena.allocArray<long>(greaterOverflow), std::bad_alloc);
111
112 ASSERT_TRUE(arena.isEmpty());
113}

115// Make a class that keeps track of constructions and destructions in
116// a global string. MAKE_HELPER will make a separate instantiation for
117// every test so the tests are still independent in spite of the use
118// of global variables. This is also thread safe as long as two
119// threads are not running the same test at the same time.
120namespace {
121 template<class T, size_t ThrowAt>
122 class _frobby_Helper {
123 public:
124 _frobby_Helper() {
125 _id = ++_count;
126 if (_id == ThrowAt) {
127 _log << 'T' << _id;
128 throw _id;
129 } else
130 _log << '+' << _id;
131 }
132
133 ~_frobby_Helper() {
134 _log << '-' << _id;
135 }
136
137 void setId(size_t id) {_id = id;}
138
139 static std::string resetLog() {
140 std::string log = _log.str();
141 _log.clear();
142 _log.str("");
143 _count = 0;
144 return log;
145 }
146
147 private:
148 size_t _id;
149 static size_t _count;
150 static std::ostringstream _log;
151 };
152
153 template<class T, size_t ThrowAt>
154 size_t _frobby_Helper<T, ThrowAt>::_count = 0;
155
156 template<class T, size_t ThrowAt>
157 std::ostringstream _frobby_Helper<T, ThrowAt>::_log;
158}
159#define MAKE_HELPER(NAME, THROW_AT) \
160 namespace { \
161 struct _frobby_##NAME##HelperTag {}; \
162 typedef _frobby_Helper<_frobby_##NAME##HelperTag, THROW_AT> NAME##Helper; \
163 }
