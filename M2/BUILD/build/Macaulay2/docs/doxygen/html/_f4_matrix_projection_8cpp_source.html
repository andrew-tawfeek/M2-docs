18void F4MatrixProjection::addColumn(
19 const ColIndex projectFrom,
20 ConstMonoRef mono,
21 const bool isLeft
22) {
23 MATHICGB_ASSERT(projectFrom < mColProjectTo.size());
24 MATHICGB_ASSERT
25 (mLeftMonomials.size() + mRightMonomials.size() < mColProjectTo.size());
26
27 auto monoCopy = mRing.monoid().alloc();
28 mRing.monoid().copy(mono, *monoCopy);
29
30 auto& projected = mColProjectTo[projectFrom];
31 if (isLeft) {
32 projected.isLeft = true;
33 projected.index = static_cast<ColIndex>(mLeftMonomials.size());
34 mLeftMonomials.push_back(monoCopy.ptr());
35 } else {
36 projected.isLeft = false;
37 projected.index = static_cast<ColIndex>(mRightMonomials.size());
38 mRightMonomials.push_back(monoCopy.ptr());
39 }
40 monoCopy.release();
41}

52class F4MatrixProjection::TopBottom {
53public:
54 typedef std::pair<Row, Scalar> RowMultiple;
55 typedef std::vector<RowMultiple> RowVector;
56
57 TopBottom(const size_t leftColCount, const PolyRing& ring):
58 mModulus(static_cast<Scalar>(ring.charac())),
59 mTopRows(leftColCount)
60 {
61 MATHICGB_ASSERT(ring.charac() <= std::numeric_limits<Scalar>::max());
62 MATHICGB_ASSERT(leftColCount <= std::numeric_limits<ColIndex>::max());
63 }
64
65 void addRow(const Row& row, ColIndex leadIndex, Scalar leadScalar) {
66 if (row.entryCount == 0)
67 return; // Skip zero rows.
68 if (leadIndex == std::numeric_limits<ColIndex>::max()) {
69 // this row has no left entries, so it cannot be a top row.
70 mBottomRows.push_back(RowMultiple(row, 1));
71 return;
72 }
73
74 const auto currentTop = mTopRows[leadIndex].first;
75 if (
76 currentTop.entryCount != 0 && // already have a reducer and...
77 currentTop.entryCount < row.entryCount // ...it is sparser/better
78 ) {
79 mBottomRows.push_back(RowMultiple(row, 1));
80 } else {
81 if (currentTop.entryCount != 0)
82 mBottomRows.push_back(std::make_pair(currentTop, 1));
83 MATHICGB_ASSERT(leadScalar != 0);
84 const auto inverse = leadScalar == 1 ? // 1 is a common case
85 1 : modularInverse(leadScalar, mModulus);
86 mTopRows[leadIndex] = RowMultiple(row, inverse);
87 }
88 }
89
90 bool debugAssertValid() {
91#ifdef MATHICGB_DEBUG
92 auto check = [](RowMultiple r) {
93 MATHICGB_ASSERT(r.first.entryCount > 0);
94 MATHICGB_ASSERT(r.second != 0);
95 };
96 std::for_each(mTopRows.begin(), mTopRows.end(), check);
97 std::for_each(mBottomRows.begin(), mBottomRows.end(), check);
98#endif
99 return true;
100 }
101
102 Scalar modulus() const {return mModulus;}
103 const RowVector& top() const {return mTopRows;}
104 const RowVector& bottom() const {return mBottomRows;}
105
106 RowVector moveTop() {return mTopRows;}
107 RowVector moveBottom() {return mBottomRows;}
108
109private:
110 const Scalar mModulus;
111 RowVector mTopRows;
112 RowVector mBottomRows;
113};

143 const auto end = preBlocks.end();
144 for (auto it = preBlocks.begin(); it != end; ++it) {
145 auto& block = **it;
146 const auto rowCount = block.rowCount();
147 for (SparseMatrix::RowIndex r = 0; r < rowCount; ++r) {

155 void appendRow(const Row& row) {
156 MATHICGB_ASSERT(row.entryCount > 0); // could be OK, but unexpected
157
158 const auto indicesEnd = row.indices + row.entryCount;
159 if (row.scalars == nullptr)
160 appendRow(row.indices, indicesEnd, row.externalScalars);
161 else
162 appendRow(row.indices, indicesEnd, row.scalars);
163 }

166 MATHICGB_ASSERT(multiplyBy != 0);
167 appendRow(row);

169 const auto rowIndex = mLeft.rowCount() - 1;
170 mLeft.multiplyRow(rowIndex, multiplyBy, mModulus);

229QuadMatrix F4MatrixProjection::makeAndClearOneStep(const size_t quantum) {
230 // Construct top/bottom row permutation
231 TopBottom<F4ProtoMatrix::Row> tb(mLeftMonomials.size(), ring());
232 const auto end = mMatrices.end();
233 for (auto it = mMatrices.begin(); it != end; ++it) {
234 const auto& matrix = **it;
235 const auto rowCount = matrix.rowCount();
236 for (RowIndex r = 0; r < rowCount; ++r) {
237 const auto& row = matrix.row(r); // const ref keeps temporary alive
238 if (row.entryCount == 0)
239 continue; // ignore zero rows
240
241 // *** Look for leading left entry
242 ColIndex lead = 0;
243 for (; lead < row.entryCount; ++lead) {
244 MATHICGB_ASSERT(row.indices[lead] < mColProjectTo.size());
245 auto const projected = mColProjectTo[row.indices[lead]];
246 if (projected.isLeft) {
247 const auto leadScalar = row.scalars != nullptr ?
248 row.scalars[lead] :
249 static_cast<Scalar>(row.externalScalars[lead]);
250 tb.addRow(row, projected.index, leadScalar);
251 goto done;
252 }
253 }
254 // Did not find any left entry.
255 tb.addRow(row, std::numeric_limits<ColIndex>::max(), 0);
256done:;
257 }
258 }
259 MATHICGB_ASSERT(tb.debugAssertValid());
260
261 // Split left/right and top/bottom simultaneously
262 LeftRight top(mColProjectTo, ring(), quantum);
263 top.appendRowsPermuted(tb.moveTop());
264
265 LeftRight bottom(mColProjectTo, ring(), 0);
266 bottom.appendRowsPermuted(tb.moveBottom());
267
268 // Move the data into place
269 QuadMatrix qm(ring());
270 qm.leftColumnMonomials = std::move(mLeftMonomials);
271 qm.rightColumnMonomials = std::move(mRightMonomials);
272
273 qm.topLeft = top.moveLeft();
274 qm.topRight = top.moveRight();
275 qm.bottomLeft = bottom.moveLeft();
276 qm.bottomRight = bottom.moveRight();
277
278 return qm;
279}

283 template<class TopBottom>
284 std::pair<SparseMatrix, SparseMatrix> projectRows(
285 const TopBottom& tb,
286 size_t quantum,
287 SparseMatrix&& in
288 ) {
289 const auto modulus = tb.modulus();
290
291 SparseMatrix top(quantum);
292 const auto topRows = tb.top();
293 const auto rowCountTop =
294 static_cast<SparseMatrix::RowIndex>(topRows.size());
295 for (SparseMatrix::RowIndex toRow = 0; toRow < rowCountTop; ++toRow) {
296 top.appendRow(in, topRows[toRow].first.index);
297 if (topRows[toRow].second != 1)
298 top.multiplyRow(toRow, topRows[toRow].second, modulus);
299 }
300
301 SparseMatrix bottom(quantum);
302 const auto bottomRows = tb.bottom();
303 const auto rowCountBottom =
304 static_cast<SparseMatrix::RowIndex>(bottomRows.size());
305 for (SparseMatrix::RowIndex toRow = 0; toRow < rowCountBottom; ++toRow) {
306 bottom.appendRow(in, bottomRows[toRow].first.index);
307 if (bottomRows[toRow].second != 1)
308 bottom.multiplyRow(toRow, bottomRows[toRow].second, modulus);
309 }
310
311 in.clear();
312 return std::make_pair(std::move(top), std::move(bottom));
313 }

316QuadMatrix F4MatrixProjection::makeAndClearTwoStep(const size_t quantum) {
317 // Split whole matrix into left/right
318 LeftRight lr(mColProjectTo, ring(), quantum);
319 lr.appendRows(mMatrices);
320
321 // Construct top/bottom matrix permutation
322 struct Row {
323 RowIndex index;
324 ColIndex entryCount;
325 };
326 TopBottom<Row> tb(mLeftMonomials.size(), ring());
327 const auto rowCount = lr.left().rowCount();
328 for (SparseMatrix::RowIndex row = 0; row < rowCount; ++row) {
329 const auto leftEntryCount = lr.left().entryCountInRow(row);
330 const auto entryCount = leftEntryCount + lr.right().entryCountInRow(row);
331 MATHICGB_ASSERT(entryCount >= leftEntryCount); // no overflow
332 if (entryCount == 0)
333 continue; // ignore zero rows
334
335 const Row r = {row, entryCount};
336 if (leftEntryCount == 0)
337 tb.addRow(r, std::numeric_limits<ColIndex>::max(), 0);
338 else {
339 const auto entry = lr.left().rowBegin(row);
340 tb.addRow(r, entry.index(), entry.scalar());
341 }
342 }
343 MATHICGB_ASSERT(tb.debugAssertValid());
344
345 QuadMatrix qm(ring());
346 auto left = projectRows(tb, quantum, lr.moveLeft());
347 auto right = projectRows(tb, quantum, lr.moveRight());
348
349 qm.topLeft = std::move(left.first);
350 qm.bottomLeft = std::move(left.second);
351 qm.topRight = std::move(right.first);
352 qm.bottomRight = std::move(right.second);
353 qm.leftColumnMonomials = std::move(mLeftMonomials);
354 qm.rightColumnMonomials = std::move(mRightMonomials);
355 return qm;
356}
