80template<template<typename> class Q>
81void ReducerDedup<Q>::insertTail(NewConstTerm multiple, const Poly& poly) {
82 if (poly.termCount() <= 1)
83 return;
84 mLeadTermKnown = false;
85
86 auto it = poly.begin();
87 const auto end = poly.end();
88 for (++it; it != end; ++it) {
89 NewTerm t;
90 t.mono = mRing.monoid().alloc().release();
91 mRing.monoid().multiply(*multiple.mono, it.mono(), *t.mono);
92 mRing.coefficientMult(multiple.coef, it.coef(), t.coef);
93 mQueue.push(t);
94 }
95}

97template<template<typename> class Q>
98void ReducerDedup<Q>::insert(ConstMonoRef multiple, const Poly& poly) {
99 if (poly.isZero())
100 return;
101 mLeadTermKnown = false;
102
103 const auto end = poly.end();
104 for (auto it = poly.begin(); it != end; ++it) {
105 NewTerm t = {it.coef(), mRing.monoid().alloc().release()};
106 mRing.monoid().multiply(multiple, it.mono(), *t.mono);
107 mQueue.push(t);
108 }
109}

111template<template<typename> class Q>
112bool ReducerDedup<Q>::leadTerm(NewConstTerm& result) {
113 if (!mLeadTermKnown) {
114 do {
115 if (mQueue.empty())
116 return false;
117 mLeadTerm = mQueue.top();
118 mQueue.pop();
119
120 while (!mQueue.empty()) {
121 auto entry = mQueue.top();
122 if (!mRing.monoid().equal(*entry.mono, *mLeadTerm.mono))
123 break;
124 mRing.coefficientAddTo(mLeadTerm.coef, entry.coef);
125 mRing.monoid().freeRaw(*entry.mono);
126 mQueue.pop();
127 }
128 } while (mRing.coefficientIsZero(mLeadTerm.coef));
129 mLeadTermKnown = true;
130 }
131
132 result = mLeadTerm;
133 return true;
134}
