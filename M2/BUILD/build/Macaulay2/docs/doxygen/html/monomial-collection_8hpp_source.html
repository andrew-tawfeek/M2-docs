108// Hash table for monomials (or ...)
109// Functions:
110// insert(monomial,comp): returns index if (monomial,comp) is in table. Otherwise sets value to next size.
111// lookup(monomial,comp): returns index.
112// A function elsewhere will loop through the monomials in a matrix, creating the hash table of these monomials.
113// actually: there might be translation from the monomials.
114// basically: (1) loop through monomials, add (transformed valued) one by one, when needed.
115// (2) loop through monomials, add monomial at row given by 'value' in hash table.
116// Structure itself:
117// hash table (probably from stl)
118// memtailor memory area for monomials.
119// number of elements contained in here
120// Memory management:
121// reserve space for a monomial (and/or copy it in).
122// jettison last one if not needed.

248 // insert (m,comp) into the set. Returns true if it is inserted, i.e. it isn't already in the set.
249 bool insert(Monom m, int comp)
250 {
251 size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
252 std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
253 ModuleMonom a = monomToModuleMonom(m, comp, mon);
254 auto result = mHash.insert(a);
255 bool new_elem = result.second;
256 if (new_elem)
257 {
258 a.setIndex(mElements.size());
259 mElements.push_back(a);
260 }
261 else
262 {
263 mArena.freeTop(a + 0);
264 }
265 return new_elem;
266 }

268 // if (m,comp) is found (and is a monomial with index idx), return {idx, true}
269 // if it is not found, return {-1, false}
270 std::pair<int,bool> find(Monom m, int comp)
271 {
272 size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
273 std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
274 ModuleMonom a = monomToModuleMonom(m, comp, mon);
275 auto result = mHash.find(a); // returns iterator pointing to value, or mHash.end()
276 bool found = (result != mHash.end());
277 int idx = (found ? result->index() : -1);
278 mArena.freeTop(mon.first);
279 return {idx, found};
280 }

304 // hash table
305 // monomial arena
306 // list of pointers to these 'monomials' (so we can sort them?)
307 //
308 // data members:
309 // 1. arena
310 // 2. std::vector of int*'s into arena, unique monomials
311 // 3. unordered_set
312 // each monomial has a spot for its value. When inserted into the hash table, this value is set.
313 //
314 // Conf:
315 // hash function
316 // equality function
317 // comparison function on monomials (needs to be a different class, I think)
318 // Monomial ops:
319 // (monomial,comp) --> moduleMonomial (and back), put in space for int:
320 // --> [len, value, comp, v1, ..., vn] (len includes value)
321 // --> or [value, comp, len, deg, v1, ..., vr] (hash, length, equality, compare) all work on these.
322 //
323 // ops:
324 // H[mon,comp]: first copy monomial to arena, insert it. If there, pop monom, otherwise set 'value' field
325 // lookup: first copy monomial,comp to arena, lookup, then pop copied monomial.
326 // sortMonomials: sorts mons, sets 'value' field in each monomial.
327 // value(H[mon]) = 0-th int.
328private:
329 Conf mConf;
330 memt::Arena mArena; // memory area for monomials
331 Set mHash; // will need to put in hash function, compare fcn from Conf.
332 std::vector<ModuleMonom> mElements; // each monomial points into mArena.
333public:
334 auto begin() const -> decltype(mElements.begin()) { return mElements.begin(); }
335 auto end() const -> decltype(mElements.end()) { return mElements.end(); }
336};
