15bool Tower::initialize(long charac0,
16 M2_ArrayString names0,
17 const VECTOR(ring_elem) & extensions)
18{
19 initialize_ring(charac0);
20 declare_field();
21
22 names = names0;
23 nvars = names->len;
24 level = nvars - 1;
25
26 // Translate extensions to poly's
27 if (extensions.size() == 0)
28 {
29 D = DRing::create(charac0, nvars, nullptr);
30 }
31 else
32 {
33 TowerPolynomial* exts = new TowerPolynomial[extensions.size()];
34 for (int i = 0; i < extensions.size(); i++)
35 exts[i] = reinterpret_cast<TowerPolynomial>(extensions[i].poly_val);
36 D = DRing::create(charac0, nvars, exts);
37 delete[] exts;
38 }
39
40 zeroV = from_long(0);
41 oneV = from_long(1);
42 minus_oneV = from_long(-1);
43
44 return true;
45}

91M2_arrayint Tower::support(const ring_elem a) const
92{
93 TowerPolynomial f1 = TOWER_POLYNOMIAL_VAL(a);
94 std::vector<int> max_degs;
95 D->degrees_of_vars(f1, max_degs);
96 int nelems = 0;
97 for (size_t i = 0; i < max_degs.size(); i++)
98 if (max_degs[i] > 0) nelems++;
99 M2_arrayint result = M2_makearrayint(nelems);
100 int next = 0;
101 for (size_t i = 0; i < max_degs.size(); i++)
102 if (max_degs[i] > 0) result->array[next++] = static_cast<int>(i);
103 return result;
104}

263class TowerEvaluator : public DPolyTraverser
264{
265 const RingMap *map;
266 const Ring *K;
267 SumCollector *H;
268
269 TowerPolynomial f;
270 int first_var;
271 const Ring *target;
272 gc_vector<int> vp;
273 int nvars;
274
275 virtual bool viewTerm(long coeff, const_exponents exp)
276 {
277 // translate exp to varpower
278 // map->eval_term
279 // either:
280 // H->add, or target->add_to
281 vp.resize(0);
282 varpower::from_expvector(nvars, exp, vp);
283 ring_elem c = K->from_long(coeff);
284 ring_elem a = map->eval_term(K, c, vp.data(), first_var, nvars);
285 H->add(a);
286 return true;
287 }
288
289 public:
290 TowerEvaluator(const Tower *T,
291 const RingMap *map0,
292 const ring_elem f0,
293 int first_var0)
294 : DPolyTraverser(T->D),
295 map(map0),
296 f(TOWER_POLYNOMIAL_VAL(f0)),
297 first_var(first_var0),
298 nvars(T->n_vars())
299 {
300 target = map->get_ring();
301 H = target->make_SumCollector();
302 const PolynomialRing *P = target->cast_to_PolynomialRing();
303 K = (P == nullptr ? target : P->getCoefficients());
304 }
305
306 virtual ~TowerEvaluator() { delete H; }
307 ring_elem getValue()
308 {
309 traverse(f);
310 return H->getValue();
311 }
312};

458ring_elem Tower::translate(const PolynomialRing *R, ring_elem fR) const
459{
460 // create a TowerPolynomial in the Tower T, return it.
461
462 const Monoid *M = R->getMonoid();
463 const Ring *K = R->getCoefficients();
464 int nvars = R->n_vars();
465 TowerPolynomial result = nullptr;
466 exponents_t exp = new int[nvars];
467 for (Nterm& t : fR)
468 {
469 M->to_expvector(t.monom, exp);
470 std::pair<bool, long> res = K->coerceToLongInteger(t.coeff);
471 assert(res.first);
472 int c1 = static_cast<int>(res.second);
473
474 D->add_term(result, c1, exp);
475 }
476 delete[] exp;
477 return TOWER_RINGELEM(result);
478}
