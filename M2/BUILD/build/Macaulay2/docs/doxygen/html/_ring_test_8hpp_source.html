153template <typename T>
154void testRingAxioms(const T* R, int ntrials)
155{
156 RingElementGenerator<T> gen(*R);
157 for (int i = 0; i < ntrials; i++)
158 {
159 ring_elem a = gen.nextElement();
160 ring_elem b = gen.nextElement();
161 ring_elem c = gen.nextElement();
162
163 // Test commutativity
164 // test: a*b = b*a
165 // test: a+b == b+a
166 ring_elem d = R->add(a, b);
167 ring_elem e = R->add(b, a);
168 EXPECT_TRUE(R->is_equal(d, e));
169 d = R->mult(a, b);
170 e = R->mult(b, a);
171 EXPECT_TRUE(R->is_equal(d, e));
172
173 // Test associativity
174 // test: a+(b+c) == (a+b)+c
175 // test: a*(b*c) == (a*b)*c
176 d = R->add(a, R->add(b, c));
177 e = R->add(R->add(a, b), c);
178 EXPECT_TRUE(R->is_equal(d, e));
179 d = R->mult(a, R->mult(b, c));
180 e = R->mult(R->mult(a, b), c);
181 EXPECT_TRUE(R->is_equal(d, e));
182
183 // Test distributivity
184 // test: a*(b+c) == a*b + a*c
185 d = R->mult(a, R->add(b, c));
186 e = R->add(R->mult(a, b), R->mult(a, c));
187 EXPECT_TRUE(R->is_equal(d, e));
188 }
