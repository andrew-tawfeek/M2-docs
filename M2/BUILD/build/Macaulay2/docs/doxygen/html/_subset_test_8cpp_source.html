145TEST(Subsets, concatenateSubsets)
146{
147 const int n = 7;
148 const int p = 3;
149 const int q = 2;
150 const int n_choose_p = 35;
151 Subsets C(n, std::max(p, q));
152
153 Subset a(p, 0);
154 Subset b(q, 0);
155 Subset c(p + q, 0);
156 Subset d(p + q, 0);
157
158 int sign;
159 if ((p % 2 == 1) && (q % 2 == 1))
160 sign = -1;
161 else
162 sign = 1;
163 for (size_t i = 0; i < n_choose_p; i++)
164 {
165 C.decode(i, a);
166 EXPECT_TRUE(C.isValid(a));
167 for (size_t j = 0; j < n_choose_p; j++)
168 {
169 C.decode(j, b);
170 EXPECT_TRUE(C.isValid(b));
171 int ret1 = Subsets::concatenateSubsets(a, b, c);
172 int ret2 = Subsets::concatenateSubsets(b, a, d);
173 if (ret1 == 0 || ret2 == 0)
174 {
175 EXPECT_EQ(ret1, ret2);
176 break;
177 }
178 EXPECT_EQ(ret1, sign * ret2);
179 EXPECT_TRUE(sameSubset(c, d));
180 }
181 }
182}
