146template <typename T>
147MutableMatrix /* or null */* MutableMat<T>::mult(const MutableMatrix* B) const
148{
149 // First, make sure B has the same ring/type as 'this'.
150 const T* B1 = B->coerce_const<T>();
151 if (B1 == 0)
152 {
153 ERROR(
154 "mutable matrix/ring type for (mutable) matrix multiplication "
155 "required to be the same");
156 return 0;
157 }
158 // Second, make sure the sizes are correct.
159 if (mat.numColumns() != B1->numRows())
160 {
161 ERROR("matrix sizes do not match in matrix multiplication");
162 return 0;
163 }
164 // create the result matrix
165 MutableMat<T>* result = makeZeroMatrix(n_rows(), B1->numColumns());
166 MatrixOps::mult(mat, *B1, result->mat);
167
168 return result;
169}

175 // First: make sure that A, B have the right ring/matrix type
176 const T* A1 = A->coerce_const<T>();

183 throw exc::engine_error(

185 MatrixOps::addMultipleTo(mat, *A1, *B1);
