191 inline size_t MemoryBlocks::alignNoOverflow(const size_t value) {
192 // this function might look big, but the total compiled code size is
193 // one addition and one bitwise and.
194 const size_t decAlign = MemoryAlignment - 1; // compile time constant
195
196 MEMT_ASSERT((MemoryAlignment & (decAlign)) == 0); // power of 2
197 // This works because MemoryAlignment is a power of 2.
198 const size_t aligned = (value + decAlign) & (~decAlign);
199
200 MEMT_ASSERT(aligned % MemoryAlignment == 0); // alignment
201 MEMT_ASSERT(aligned >= value); // no overflow
202 MEMT_ASSERT(aligned - value < MemoryAlignment); // adjustment minimal
203 return aligned;
204 }

206 inline size_t MemoryBlocks::alignThrowOnOverflow(size_t value) {
207 // this function might look big, but the total compiled code size is
208 // one addition, one branch using a comparison and one bitwise and.
209 const size_t decAlign = MemoryAlignment - 1; // compile time constant
210
211 MEMT_ASSERT((MemoryAlignment & (decAlign)) == 0); // power of 2
212 // This sum overflows if and only if rounding up overflows because
213 // MemoryAlignment is a power of 2.
214 const size_t sum = value + decAlign;
215 if (sum < value)
216 throw std::bad_alloc(); // overflow
217 const size_t aligned = sum & (~decAlign);
218
219 MEMT_ASSERT(aligned % MemoryAlignment == 0); // alignment
220 MEMT_ASSERT(aligned >= value); // no overflow
221 MEMT_ASSERT(aligned - value < MemoryAlignment); // adjustment minimal
222 return aligned;
223 }

225 inline bool MemoryBlocks::Block::isInBlock(const void* ptr) const {
226 // We use a trick to check this using one branch and two subtractions
227 // instead of two branches.
228 const char* p = static_cast<const char*>(ptr);
229 const size_t offset = static_cast<size_t>(p - begin());
230 // if _blockBegin > ptr then offset overflows to a large integer
231 MEMT_ASSERT((offset < getBytesInBlock()) == (begin() <= p && p < end()));
232 return offset < getBytesInBlock();
233 }
