224 void invert(ElementType& result, const ElementType& a) const
225 // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
226 {
227 mpfr_t p, denom;
228 mpfr_init2(p, get_precision());
229 mpfr_init2(denom, get_precision());
230
231 if (mpfr_cmpabs(&a.re, &a.im) >= 0)
232 {
233 // double p = &a.im/&a.re;
234 // double denom = &a.re + p * &a.im;
235 // &result.re = 1.0/denom;
236 // &result.im = - p/denom;
237
238 mpfr_div(p, &a.im, &a.re, MPFR_RNDN);
239 mpfr_mul(denom, p, &a.im, MPFR_RNDN);
240 mpfr_add(denom, denom, &a.re, MPFR_RNDN);
241 mpfr_si_div(&result.re, 1, denom, MPFR_RNDN);
242 mpfr_div(&result.im, p, denom, MPFR_RNDN);
243 mpfr_neg(&result.im, &result.im, MPFR_RNDN);
244 }
245 else
246 {
247 // double p = &a.re/&a.im;
248 // double denom = &a.im + p * &a.re;
249 // &result.re = p/denom;
250 // &result.im = -1.0/denom;
251
252 mpfr_div(p, &a.re, &a.im, MPFR_RNDN);
253 mpfr_mul(denom, p, &a.re, MPFR_RNDN);
254 mpfr_add(denom, denom, &a.im, MPFR_RNDN);
255 mpfr_si_div(&result.im, 1, denom, MPFR_RNDN);
256 mpfr_neg(&result.im, &result.im, MPFR_RNDN);
257 mpfr_div(&result.re, p, denom, MPFR_RNDN);
258 }
259
260 mpfr_clear(p);
261 mpfr_clear(denom);
262 }

327 void mult(ElementType& res, const ElementType& a, const ElementType& b) const
328 {
329 mpfr_t tmp;
330 Element result(*this);
331 mpfr_init2(tmp, get_precision());
332
333 // &result.re = &a.re*&b.re - &a.im*&b.im;
334 mpfr_mul(tmp, &a.re, &b.re, MPFR_RNDN);
335 mpfr_set(&result.value().re, tmp, MPFR_RNDN);
336 mpfr_mul(tmp, &a.im, &b.im, MPFR_RNDN);
337 mpfr_sub(&result.value().re, &result.value().re, tmp, MPFR_RNDN);
338
339 // &result.im = &a.re*&b.im + &a.im*&b.re;
340 mpfr_mul(tmp, &a.re, &b.im, MPFR_RNDN);
341 mpfr_set(&result.value().im, tmp, MPFR_RNDN);
342 mpfr_mul(tmp, &a.im, &b.re, MPFR_RNDN);
343 mpfr_add(&result.value().im, &result.value().im, tmp, MPFR_RNDN);
344
345 set(res, result);
346 mpfr_clear(tmp);

367 void divide(ElementType& res,
368 const ElementType& a,
369 const ElementType& b) const
370 {
371 mpfr_t p, denom;
372 mpfr_init2(p, get_precision());
373 mpfr_init2(denom, get_precision());
374 Element result(*this);
375
376 if (mpfr_cmpabs(&b.re, &b.im) >= 0)
377 {
378 // for v = c + d*i,
379 // p = d/c
380 // c+di = c(1+p*i), so denom is c(1+p^2)
381 // which is c + d*p
382
383 // double p = v.im/v.re;
384 // double denom = v.re + p * v.im;
385 // result.re = (u.re + p*u.im)/denom;
386 // result.im = (u.im - p*u.re)/denom;
387
388 mpfr_div(p, &b.im, &b.re, MPFR_RNDN);
389 mpfr_mul(denom, p, &b.im, MPFR_RNDN);
390 mpfr_add(denom, denom, &b.re, MPFR_RNDN);
391
392 mpfr_mul(&result.value().re, p, &a.im, MPFR_RNDN);
393 mpfr_add(&result.value().re, &result.value().re, &a.re, MPFR_RNDN);
394 mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
395
396 mpfr_mul(&result.value().im, p, &a.re, MPFR_RNDN);
397 mpfr_neg(&result.value().im, &result.value().im, MPFR_RNDN);
398 mpfr_add(&result.value().im, &result.value().im, &a.im, MPFR_RNDN);
399 mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
400 }
401 else
402 {
403 // double p = v.re/v.im;
404 // double denom = v.im + p * v.re;
405 // result.re = (u.re * p + u.im)/denom;
406 // result.im = (-u.re + p * u.im)/denom;
407
408 mpfr_div(p, &b.re, &b.im, MPFR_RNDN);
409 mpfr_mul(denom, p, &b.re, MPFR_RNDN);
410 mpfr_add(denom, denom, &b.im, MPFR_RNDN);
411
412 mpfr_mul(&result.value().re, p, &a.re, MPFR_RNDN);
413 mpfr_add(&result.value().re, &result.value().re, &a.im, MPFR_RNDN);
414 mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
415
416 mpfr_mul(&result.value().im, p, &a.im, MPFR_RNDN);
417 mpfr_sub(&result.value().im, &result.value().im, &a.re, MPFR_RNDN);
418 mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
419 }
420 mpfr_clear(p);
421 mpfr_clear(denom);
422 set(res, result);
423 }

439 void power(ElementType& result, const ElementType& a, int n) const
440 {
441 ElementType curr_pow;
442 init(curr_pow);
443 set_from_long(result, 1);
444 if (n == 0)
445 {
446 }
447 else if (n < 0)
448 {
449 n = -n;
450 invert(curr_pow, a);
451 }
452 else
453 {
454 set(curr_pow, a);
455 }
456 while (n > 0)
457 {
458 if (n % 2)
459 {
460 mult(result, result, curr_pow);
461 }
462 n = n / 2;
463 mult(curr_pow, curr_pow, curr_pow);
464 }
465 clear(curr_pow);
466 }
