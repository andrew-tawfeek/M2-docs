233 void invert(ElementType& result, const ElementType& a) const
234 // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
235 {
236 mpfr_t p, denom;
237 mpfr_init2(p, get_precision());
238 mpfr_init2(denom, get_precision());
239
240 if (mpfr_cmpabs(&a.re, &a.im) >= 0)
241 {
242 // double p = &a.im/&a.re;
243 // double denom = &a.re + p * &a.im;
244 // &result.re = 1.0/denom;
245 // &result.im = - p/denom;
246
247 mpfr_div(p, &a.im, &a.re, MPFR_RNDN);
248 mpfr_mul(denom, p, &a.im, MPFR_RNDN);
249 mpfr_add(denom, denom, &a.re, MPFR_RNDN);
250 mpfr_si_div(&result.re, 1, denom, MPFR_RNDN);
251 mpfr_div(&result.im, p, denom, MPFR_RNDN);
252 mpfr_neg(&result.im, &result.im, MPFR_RNDN);
253 }
254 else
255 {
256 // double p = &a.re/&a.im;
257 // double denom = &a.im + p * &a.re;
258 // &result.re = p/denom;
259 // &result.im = -1.0/denom;
260
261 mpfr_div(p, &a.re, &a.im, MPFR_RNDN);
262 mpfr_mul(denom, p, &a.re, MPFR_RNDN);
263 mpfr_add(denom, denom, &a.im, MPFR_RNDN);
264 mpfr_si_div(&result.im, 1, denom, MPFR_RNDN);
265 mpfr_neg(&result.im, &result.im, MPFR_RNDN);
266 mpfr_div(&result.re, p, denom, MPFR_RNDN);
267 }
268
269 mpfr_clear(p);
270 mpfr_clear(denom);
271 }

336 void mult(ElementType& res, const ElementType& a, const ElementType& b) const
337 {
338 mpfr_t tmp;
339 Element result(*this);
340 mpfr_init2(tmp, get_precision());
341
342 // &result.re = &a.re*&b.re - &a.im*&b.im;
343 mpfr_mul(tmp, &a.re, &b.re, MPFR_RNDN);
344 mpfr_set(&result.value().re, tmp, MPFR_RNDN);
345 mpfr_mul(tmp, &a.im, &b.im, MPFR_RNDN);
346 mpfr_sub(&result.value().re, &result.value().re, tmp, MPFR_RNDN);
347
348 // &result.im = &a.re*&b.im + &a.im*&b.re;
349 mpfr_mul(tmp, &a.re, &b.im, MPFR_RNDN);
350 mpfr_set(&result.value().im, tmp, MPFR_RNDN);
351 mpfr_mul(tmp, &a.im, &b.re, MPFR_RNDN);
352 mpfr_add(&result.value().im, &result.value().im, tmp, MPFR_RNDN);
353
354 set(res, result);
355 mpfr_clear(tmp);

376 void divide(ElementType& res,
377 const ElementType& a,
378 const ElementType& b) const
379 {
380 mpfr_t p, denom;
381 mpfr_init2(p, get_precision());
382 mpfr_init2(denom, get_precision());
383 Element result(*this);
384
385 if (mpfr_cmpabs(&b.re, &b.im) >= 0)
386 {
387 // for v = c + d*i,
388 // p = d/c
389 // c+di = c(1+p*i), so denom is c(1+p^2)
390 // which is c + d*p
391
392 // double p = v.im/v.re;
393 // double denom = v.re + p * v.im;
394 // result.re = (u.re + p*u.im)/denom;
395 // result.im = (u.im - p*u.re)/denom;
396
397 mpfr_div(p, &b.im, &b.re, MPFR_RNDN);
398 mpfr_mul(denom, p, &b.im, MPFR_RNDN);
399 mpfr_add(denom, denom, &b.re, MPFR_RNDN);
400
401 mpfr_mul(&result.value().re, p, &a.im, MPFR_RNDN);
402 mpfr_add(&result.value().re, &result.value().re, &a.re, MPFR_RNDN);
403 mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
404
405 mpfr_mul(&result.value().im, p, &a.re, MPFR_RNDN);
406 mpfr_neg(&result.value().im, &result.value().im, MPFR_RNDN);
407 mpfr_add(&result.value().im, &result.value().im, &a.im, MPFR_RNDN);
408 mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
409 }
410 else
411 {
412 // double p = v.re/v.im;
413 // double denom = v.im + p * v.re;
414 // result.re = (u.re * p + u.im)/denom;
415 // result.im = (-u.re + p * u.im)/denom;
416
417 mpfr_div(p, &b.re, &b.im, MPFR_RNDN);
418 mpfr_mul(denom, p, &b.re, MPFR_RNDN);
419 mpfr_add(denom, denom, &b.im, MPFR_RNDN);
420
421 mpfr_mul(&result.value().re, p, &a.re, MPFR_RNDN);
422 mpfr_add(&result.value().re, &result.value().re, &a.im, MPFR_RNDN);
423 mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
424
425 mpfr_mul(&result.value().im, p, &a.im, MPFR_RNDN);
426 mpfr_sub(&result.value().im, &result.value().im, &a.re, MPFR_RNDN);
427 mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
428 }
429 mpfr_clear(p);
430 mpfr_clear(denom);
431 set(res, result);
432 }

448 void power(ElementType& result, const ElementType& a, int n) const
449 {
450 ElementType curr_pow;
451 init(curr_pow);
452 set_from_long(result, 1);
453 if (n == 0)
454 {
455 }
456 else if (n < 0)
457 {
458 n = -n;
459 invert(curr_pow, a);
460 }
461 else
462 {
463 set(curr_pow, a);
464 }
465 while (n > 0)
466 {
467 if (n % 2)
468 {
469 mult(result, result, curr_pow);
470 }
471 n = n / 2;
472 mult(curr_pow, curr_pow, curr_pow);
473 }
474 clear(curr_pow);
475 }
