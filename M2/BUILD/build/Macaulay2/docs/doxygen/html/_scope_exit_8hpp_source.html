8// Guard holds an action to call and calls it unless it has been released.
9template<class T>
10class Guard {
11public:
12 ~Guard() {
13 if (mOwning && mActive)
14 mAction();
15 }
16
17private:
18 friend struct GuardMaker;
19 Guard(T&& action, const bool& active):
20 mOwning(true), mActive(active), mAction(std::move(action)) {}
21
22 // Most compilers should elide the call to this construtor, but it must be
23 // here anyway and we should support even a crazy compiler that decides to
24 // call it.
25 Guard(Guard<T>&& guard):
26 mOwning(true), mActive(guard.mActive), mAction(std::move(guard.mAction))
27 {
28 assert(guard.mActive);
29 guard.mOwning = false; // to avoid calling mAction twice
30 }
31
32 bool mOwning;
33 const bool& mActive;
34 const T mAction;
35};

71// Example, with need to dismiss:
72// v.push_back(5);
73// MATHICGB_SCOPE_EXIT(name) {v.pop_back();};
74// // ...
75// if (error)
76// return; // the pop_back is done
77// name.dismiss();
78// return; // the pop_back is not done
79//
80// The middle line is a no-op if the name parameter expands to nothing.
81// When NAME expands to nothing, we need the static_cast to prevent
82// the compiler from parsing the middle line as a redeclaration of
83// myBool. In the final line we use that a const reference keeps
84// temporary objects alive until the end of the scope. It would be correct
85// to copy, but this way we can keep the copy constructor private
86// and help out any compiler that has issue with eliding that copy.
87#define MATHICGB_SCOPE_EXIT(NAME) \
88 bool MATHICGB_UNIQUE(active) = true; \
89 ::mgb::Dismisser NAME(static_cast<bool&>(MATHICGB_UNIQUE(active))); \
90 const auto& MATHICGB_UNIQUE(guard) = \
91 ::mgb::GuardMaker(MATHICGB_UNIQUE(active)) + [&]
