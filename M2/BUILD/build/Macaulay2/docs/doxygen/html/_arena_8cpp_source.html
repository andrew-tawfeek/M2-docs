39 void Arena::growCapacity(const size_t needed) {
40 // ** Calcuate size of block (doubles capacity)
41 size_t size = std::max(needed, block().getBytesInBlock());
42 if (size > std::numeric_limits<size_t>::max() / 2)
43 throw std::bad_alloc(); // size * 2 overflows
44 size *= 2;
45 const size_t minimumAlloc = 16 * 1024 - sizeof(Block) - 16;
46 size = std::max(size, minimumAlloc); // avoid many small blocks
47 size = MemoryBlocks::alignThrowOnOverflow(size);
48
49 MEMT_ASSERT(size >= needed);
50 MEMT_ASSERT(size % MemoryAlignment == 0);
51 _blocks.allocBlock(size);
52 }
