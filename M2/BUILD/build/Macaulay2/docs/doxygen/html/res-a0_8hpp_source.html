45// States of the resolution computation
46enum {
47 RES_SKELETON, // Either at beginning of skeleton, or in the middle
48 // if next_pair is null, then at the beginning
49 RES_MONORDER, // Need to set the monomial order, which is always
50 // first use total degree, then compare_num of lead component.
51 RES_MONIDEAL, // At the beginning of, or in th emiddle of
52 // computing the resolution of the monomial ideal
53 RES_REDUCTIONS, // Done with the skeleton, doing computing reductions
54 // next_pair: next pair to (possibly) reduce
55 // if null: at the beginning of reductions.
56 RES_DONE, // If done, given length and degree limit
57 RES_COMPLETE // Completely done with resolution
58};
