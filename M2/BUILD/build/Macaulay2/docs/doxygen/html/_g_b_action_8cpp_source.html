65void GBAction::performAction() {
66 mParams.perform();
67 mGBParams.perform();
68 const std::string projectName = mParams.inputFileNameStem(0);
69
70 // read input
71 const std::string inputBasisFile = projectName + ".ideal";
72 std::ifstream inputFile(inputBasisFile.c_str());
73 if (inputFile.fail())
74 mic::reportError("Could not read input file \"" + inputBasisFile + '\n');
75
76 Scanner in(inputFile);
77 auto p = MathicIO<>().readRing(true, in);
78 auto& ring = *p.first;
79 auto basis = MathicIO<>().readBasis(ring, mModule.value(), in);
80
81 // run algorithm
82 const auto reducerType = Reducer::reducerType(mGBParams.mReducer.value());
83 std::unique_ptr<Reducer> reducer;
84 if (
85 reducerType != Reducer::Reducer_F4_Old &&
86 reducerType != Reducer::Reducer_F4_New
87 ) {
88 reducer = Reducer::makeReducer(reducerType, ring);
89 } else {
90 auto f4Reducer = makeF4Reducer(
91 ring,
92 reducerType == Reducer::Reducer_F4_Old,
93 mMinMatrixToStore.value() > 0 ? projectName : "",
94 mMinMatrixToStore
95 );
96 reducer = std::move(f4Reducer);
97 }
98
99 ClassicGBAlgParams params;
100 params.reducer = reducer.get();
101 params.monoLookupType = mGBParams.mMonoLookup.value();
102 params.preferSparseReducers = mGBParams.mPreferSparseReducers.value();
103 params.sPairQueueType = mGBParams.mSPairQueue.value();
104 params.breakAfter = mGBParams.mBreakAfter.value();
105 params.printInterval = mGBParams.mPrintInterval.value();
106 params.sPairGroupSize = mSPairGroupSize.value();
107 params.reducerMemoryQuantum = mGBParams.mMemoryQuantum.value();
108 params.useAutoTopReduction = mAutoTopReduce.value();
109 params.useAutoTailReduction = mAutoTailReduce.value();
110 params.callback = nullptr;
111
112 const auto gb = mParams.mTaskArena->execute([&basis,this,&params]{
113 return (mModule.value() ?
114 computeModuleGBClassicAlg(std::move(basis), params) :
115 computeGBClassicAlg(std::move(basis), params));
116 });
117
118 if (mGBParams.mOutputResult.value()) {
119 std::ofstream out(projectName + ".gb");
120 MathicIO<>().writeBasis(gb, mModule.value(), out);
121 }
122}
