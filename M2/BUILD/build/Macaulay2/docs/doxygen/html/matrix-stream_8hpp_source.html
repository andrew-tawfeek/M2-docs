122template <typename T>
123void matrixToStream(const Matrix* M, T& stream)
124{
125 const Ring* R = M->get_ring();
126 const PolyRing* P = R->cast_to_PolyRing();
127 assert(P != 0);
128 const Ring* KK = P->getCoefficientRing();
129 size_t nvars = P->n_vars();
130 size_t ncols = M->n_cols();
131 int charac = static_cast<int>(P->characteristic());
132 assert(charac > 0);
133 exponents_t exp =
134 ALLOCATE_EXPONENTS(EXPONENT_BYTE_SIZE(nvars)); // allocated on stack
135 stream.idealBegin(ncols);
136 Matrix::iterator i(M);
137 for (int c = 0; c < ncols; c++)
138 {
139 i.set(c);
140 // We need the length of this column, in number of monomials
141 size_t nterms = 0;
142 for (; i.valid(); i.next())
143 {
144 Nterm* t = i.entry();
145 for ([[maybe_unused]] Nterm& s : t) nterms++;
146 }
147 stream.appendPolynomialBegin(nterms);
148
149 i.set(c);
150 // Now we process each column, sending it to the stream
151 for (; i.valid(); i.next())
152 {
153 Nterm* t = i.entry();
154 for (Nterm& s : t)
155 {
156 P->getMonoid()->to_expvector(s.monom, exp);
157 stream.appendTermBegin(i.row());
158 for (size_t j = 0; j < nvars; j++)
159 if (exp[j] != 0) stream.appendExponent(j, exp[j]);
160 std::pair<bool, long> b = KK->coerceToLongInteger(s.coeff);
161 assert(b.first);
162 int a = static_cast<int>(
163 b.second); // This will fit, as the charac fits into an int
164 if (a < 0) a += charac;
165 stream.appendTermDone(a);
166 }
167 }
168 stream.appendPolynomialDone();
169 }
170 stream.idealDone();
171}
