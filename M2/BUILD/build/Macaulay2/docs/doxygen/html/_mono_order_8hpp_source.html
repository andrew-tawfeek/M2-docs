139 bool isMonomialOrder() const {
140 for (VarIndex var = 0; var < varCount(); ++var) {
141 // Check that x_var > 1.
142 for (size_t grading = 0; ; ++grading) {
143 if (grading == gradingCount()) {
144 // The column was entirely zero, so x_var > 1 if and only if the
145 // base ordering is lex.
146 if (!hasLexBaseOrder())
147 return false;
148 break;
149 }
150 const auto index = grading * varCount() + var;
151 MATHICGB_ASSERT(index < mGradings.size());
152 if (mGradings[index] != 0) {
153 // We have found the first non-zero weight in this column,
154 // so x_var > 1 if and only if this weight is positive.
155 if (mGradings[index] < 0)
156 return false;
157 break;
158 }
159 }
160 }
161 return true;
162 }
