231 void power(ElementType &result, ElementType a, long n) const
232 {
233 if (a != 0)
234 {
235 result = n_powmod2_preinv(a, n, mModulus.n, mModulus.ninv);
236 }
237 else
238 {
239 // case a == 0
240 if (n < 0) throw exc::division_by_zero_error();
241 if (n == 0) return set_from_long(result, 1);
242 if (n > 0) return set_zero(result);
243 }
244 }

246 void power_mpz(ElementType &result, ElementType a, mpz_srcptr n) const
247 {
248 if (a != 0)
249 {
250 unsigned long nbar = mpz_fdiv_ui(n, mCharac - 1);
251 result = n_powmod2_ui_preinv(a, nbar, mModulus.n, mModulus.ninv);
252 }
253 else
254 {
255 // case a == 0
256 if (mpz_sgn(n) == 0)
257 set_from_long(result, 1);
258 else if (mpz_sgn(n) > 0)
259 set_zero(result);
260 else
261 throw exc::division_by_zero_error();
262 }
