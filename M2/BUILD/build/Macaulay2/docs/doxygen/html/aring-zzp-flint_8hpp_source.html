221 void power(ElementType &result, ElementType a, long n) const
222 {
223 if (a != 0)
224 {
225 result = n_powmod2_preinv(a, n, mModulus.n, mModulus.ninv);
226 }
227 else
228 {
229 // case a == 0
230 if (n < 0) throw exc::division_by_zero_error();
231 if (n == 0) return set_from_long(result, 1);
232 if (n > 0) return set_zero(result);
233 }
234 }
235
236 void power_mpz(ElementType &result, ElementType a, mpz_srcptr n) const
237 {
238 if (a != 0)
239 {
240 unsigned long nbar = mpz_fdiv_ui(n, mCharac - 1);
241 result = n_powmod2_ui_preinv(a, nbar, mModulus.n, mModulus.ninv);
242 }
243 else
244 {
245 // case a == 0
246 if (mpz_sgn(n) == 0)
247 set_from_long(result, 1);
248 else if (mpz_sgn(n) > 0)
249 set_zero(result);
250 else
251 throw exc::division_by_zero_error();
252 }
