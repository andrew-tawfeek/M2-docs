225 void power(ElementType& result, const ElementType& a, long n) const
226 {
227 bool n_is_negative = false;
228 if (n < 0)
229 {
230 if (is_zero(a)) throw exc::division_by_zero_error();
231 n_is_negative = true;
232 n = -n;
233 }
234 mpz_pow_ui(mpq_numref(&result), mpq_numref(&a), n);
235 mpz_pow_ui(mpq_denref(&result), mpq_denref(&a), n);
236 if (n_is_negative)
237 mpq_inv(&result, &result);
238 }
