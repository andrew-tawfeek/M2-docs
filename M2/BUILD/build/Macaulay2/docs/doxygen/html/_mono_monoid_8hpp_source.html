91 if (order.componentBefore() != Order::ComponentAfterBaseOrder)

104 static void reverseGradings(const VarIndex varCount, Gradings& gradings) {
105 if (varCount == 0)
106 return;
107 MATHICGB_ASSERT(gradings.size() % varCount == 0);
108 const auto gradingCount = gradings.size() / varCount;
109
110 for (VarIndex grading = 0; grading < gradingCount / 2; ++grading) {
111 for (VarIndex var = 0; var < varCount; ++var) {
112 const auto index = gradingsIndex(grading, var, varCount);
113 const auto oppositeIndex = gradingsOppositeRowIndex
114 (grading, gradingCount, var, varCount);
115 std::swap(gradings[index], gradings[oppositeIndex]);
116 }
117 }

127 static void insertZeroRow(
128 const VarIndex varCount,
129 const size_t insertBeforeRow,
130 Gradings& gradings
131 ) {
132 if (varCount == 0)
133 return;
134 MATHICGB_ASSERT(gradings.size() % varCount == 0);
135 MATHICGB_ASSERT(insertBeforeRow <= gradings.size() / varCount);
136 gradings.resize(gradings.size() + varCount);
137 const auto rowBegin = gradings.begin() + insertBeforeRow * varCount;
138 std::copy_backward(rowBegin, gradings.end() - varCount, gradings.end());
139 std::fill_n(rowBegin, varCount, 0);
140 }

152 std::copy(rowBegin + varCount, gradings.end(), rowBegin);
153 gradings.resize(gradings.size() - varCount);

166 const auto rowEnd = rowBegin + varCount;
167 std::reverse(rowBegin, rowEnd);

531 bool equalHintTrue(ConstMonoRef a, ConstMonoRef b) const {
532 // if a[i] != b[i] then a[i] ^ b[i] != 0, so the or of all xors is zero
533 // if and only if a equals b. This way we avoid having a branch to check
534 // equality for every iteration of the loop, which is a win in the case
535 // that none of the early-exit branches are taken - that is, when a equals
536 // b.
537 Exponent orOfXor = 0;
538 for (VarIndex i = lastExponentIndex(); i != beforeEntriesIndexBegin(); --i)
539 orOfXor |= access(a, i) ^ access(b, i);
540 MATHICGB_ASSERT((orOfXor == 0) == equal(a, b));
541 return orOfXor == 0;
542 }

554
555 bool isProductOfHintTrue(
556 ConstMonoRef a,
557 ConstMonoRef b,
558 ConstMonoRef ab
559 ) const {
560 // We compare more than one exponent at a time using 64 bit integers. This
561 // might go one 32 bit value at the end too far, but since that space is
562 // either a degree or a hash value that is fine --- those values will also
563 // match if the monomials are equal. This does not work for negative
564 // exponents since the overflowing bit will go into the next word.
565 // It is OK that the degree field can be negative (a field we might go
566 // into without caring about it because it shares a 64 bit field with
567 // the last exponent), because it is at the end so the overflowing
568 // bit will not interfere. For this reason we need to have a degree
569 // or a hash value stored there - otherwise two equal monomials could
570 // have different things stored next to them which would confuse this code.
571
572 // todo: ensure 8 byte alignment. Though there seem to be no ill effects
573 // for unaligned access. Performance seems to be no worse than for using
574 // 32 bit integers directly.
575
576 if (sizeof(Exponent) != 4 || (!StoreHash && !StoreOrder) ||
577 __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__)
578 return isProductOf(a, b, ab);
579
580 uint64 orOfXor = 0;
581 for (VarIndex i = varCount() / 2; i != beforeEntriesIndexBegin(); --i) {
582 MATHICGB_ASSERT(access(a, i*2) >= 0);
583 MATHICGB_ASSERT(i == varCount() / 2 || access(a, i*2+1) >= 0);
584
585 uint64 A, B, AB;
586 // We have to use std::memcpy here because just casting to a int64 breaks
587 // the strict aliasing rule which implies undefined behavior. Both MSVC and
588 // gcc don't actually call memcpy here. MSVC is a tiny bit slower for this
589 // code than for casting while GCC seems to be exactly the same speed.
590 std::memcpy(&A, ptr(a, i*2), 8);
591 std::memcpy(&B, ptr(b, i*2), 8);
592 std::memcpy(&AB, ptr(ab, i*2), 8);
593 orOfXor |= AB ^ (A + B);
594 }
595 MATHICGB_ASSERT((orOfXor == 0) == isProductOf(a, b, ab));
596 return orOfXor == 0;
597 }

599 MATHICGB_INLINE bool isTwoProductsOfHintTrue(
600 ConstMonoRef a1,
601 ConstMonoRef a2,

796 CompareResult compare(ConstMonoRef a, ConstMonoRef b) const {
797 MATHICGB_ASSERT(debugOrderValid(a));
798 MATHICGB_ASSERT(debugOrderValid(b));
799
800 VarIndex index;
801
802 if (StoreOrder)
803 index = orderIndexEnd();
804 else {
805 // Check the degrees seperately since they are not stored.
806 auto grading = gradingCount();
807 while (grading != 0) {
808 --grading;
809 const auto cmp = degree(a, grading) - degree(b, grading);
810 if (cmp < 0) return isLexBaseOrder() ? LessThan : GreaterThan;
811 if (cmp > 0) return isLexBaseOrder() ? GreaterThan : LessThan;
812 }
813 index = exponentsIndexEnd();
814 }
815
816 // If StoreOrder is true then this first checks the degrees.
817 // Then the exponents are checked.
818 // Finally, if HasComponent is true, the component is checked.
819 while (index != entriesIndexBegin()) {
820 --index;
821 const auto cmp = access(a, index) - access(b, index);
822 if (cmp < 0) return isLexBaseOrder() ? LessThan : GreaterThan;
823 if (cmp > 0) return isLexBaseOrder() ? GreaterThan : LessThan;
824 }

831 CompareResult compare(ConstMonoRef a, ConstMonoRef b1, ConstMonoRef b2) const {
832 MATHICGB_ASSERT(debugOrderValid(a));
833 MATHICGB_ASSERT(debugOrderValid(b1));
834 MATHICGB_ASSERT(debugOrderValid(b2));
835
836 VarIndex index;
837
838 if (StoreOrder)
839 index = orderIndexEnd();
840 else {
841 // Check the degrees seperately since they are not stored.
842 auto grading = gradingCount();
843 while (grading != 0) {
844 --grading;
845 const auto cmp =
846 degree(a, grading) - (degree(b1, grading) + degree(b2, grading));
847 if (cmp < 0) return isLexBaseOrder() ? LessThan : GreaterThan;
848 if (cmp > 0) return isLexBaseOrder() ? GreaterThan : LessThan;
849 }
850 index = exponentsIndexEnd();
851 }
852
853 // If StoreOrder is true then this first checks the degrees.
854 // Then the exponents are checked.
855 // Finally, if HasComponent is true, the component is checked.
856 while (index != entriesIndexBegin()) {
857 --index;
858 const auto cmp =
859 access(a, index) - (access(b1, index) + access(b2, index));
860 if (cmp < 0) return isLexBaseOrder() ? LessThan : GreaterThan;
861 if (cmp > 0) return isLexBaseOrder() ? GreaterThan : LessThan;
862 }
863 return EqualTo;
864 }

867 return compare(a, b) == LessThan;

878 // If this method returns true for monomials a and b then it is
879 // guaranteed that multiplying a and b together will not overflow
880 // the integers in the representation.
881 bool hasAmpleCapacity(ConstMonoRef mono) const {
882 const auto halfMin = std::numeric_limits<Exponent>::min() / 2;
883 const auto halfMax = std::numeric_limits<Exponent>::max() / 2;
884 MATHICGB_ASSERT(halfMin <= 0);
885 const auto limit = std::min(-halfMin, halfMax);
886 const auto inRange = [&](Exponent value)
887 {return -limit <= value && value <= limit;};
888
889 for (VarIndex i = exponentsIndexBegin(); i != exponentsIndexEnd(); ++i)
890 if (!inRange(access(mono, i)))
891 return false;
892 for (VarIndex grading = 0; grading < gradingCount(); ++grading)
893 if (!inRange(degree(mono, grading)))
894 return false;
895 return true;
896 }
