485bool GroebnerConfiguration::setMonomialOrderInternal(
486 MonomialOrderData order
487) {
488 MATHICGB_ASSERT(Pimpl::baseOrderValid(order.baseOrder));
489 MATHICGB_ASSERT(varCount() != 0 || order.gradingsSize == 0);
490 MATHICGB_ASSERT(varCount() == 0 || order.gradingsSize % varCount() == 0);
491
492 // Check if order is global.
493 if (varCount() != 0) {
494 const size_t rowCount = order.gradingsSize / varCount();
495 for (VarIndex var = 0; var < varCount(); ++var) {
496 // check if 1 < x_var.
497 for (size_t row = 0; row < rowCount; ++row) {
498 const Exponent e = order.gradings[row * varCount() + var];
499 if (e < 0)
500 return false; // order not global
501 if (e > 0)
502 goto orderGlobalForVar;
503 }
504 if (
505 order.baseOrder == RevLexAscendingBaseOrder ||
506 order.baseOrder == RevLexDescendingBaseOrder
507 )
508 return false; // order not global
509 orderGlobalForVar:;
510 }
511 }
512
513 mPimpl->mBaseOrder = order.baseOrder;
514 mPimpl->mGradings.assign
515 (order.gradings, order.gradings + order.gradingsSize);
516 return true;
517}

848 MATHICGB_ASSERT(mPimpl->polyIndex < mPimpl->basis->size());
849
850 const auto& monoid = mPimpl->basis->ring().monoid();
851 const auto& p = *mPimpl->basis->getPoly(mPimpl->polyIndex);
852 MATHICGB_ASSERT(p.ring().monoid() == monoid);
853
854 const auto& from = *mPimpl->mTermIt;
855 const auto com = monoid.component(*from.mono);
856 auto to = mPimpl->tmpTerm.get();
857 for (VarIndex var = 0; var < monoid.varCount(); ++var)
858 to[var] = monoid.externalExponent(*from.mono, var);
859
860 ++(mPimpl->mTermIt);
861 if (mPimpl->mTermIt == p.end()) {
862 ++mPimpl->polyIndex;
863 while (
864 mPimpl->polyIndex < mPimpl->basis->size() &&
865 mPimpl->basis->getPoly(mPimpl->polyIndex)->isZero()
866 )
867 ++(mPimpl->polyIndex);
868
869 if (mPimpl->polyIndex < mPimpl->basis->size())
870 mPimpl->mTermIt = mPimpl->basis->getPoly(mPimpl->polyIndex)->begin();
871 }
872
873 ConstTerm term;
874 term.coef = from.coef;
875 term.exponents = to;
876 term.com = com;
877 return term;
878 }
