78// PointArray
79class PointArray
80{
81 public:
82 using RealVector = std::vector<double>;
83 using Weight = double;
84 PointArray(Weight epsilon, const RealVector& weights)
85 : mEpsilon(epsilon), mWeights(weights)
86 {
87 }
88 PointArray(Weight epsilon, int n) : mEpsilon(epsilon)
89 {
90 double s = 0;
91 for (int i = 0; i < n; i++)
92 {
93 double r = rand();
94 s += r;
95 mWeights.push_back(r);
96 }
97 for (int i = 0; i < n; i++) mWeights[i] /= s;
98 }
99 virtual ~PointArray() {
100 // std::cerr << "entering ~PointArray()" << std::endl;
101 }
102 int lookup_or_append(const RealVector& a)
103 {
104 Weight w = weight(a);
105 int p = lookup(a);
106 if (p != -1) return p;
107 if (mMap.find(w) != mMap.end()) return -1;
108 int ret = static_cast<int>(mPoints.size());
109 mPoints.push_back(a);
110 mMap.insert(std::pair<Weight, int>(w, ret));
111 return ret;
112 }
113 int lookup(const RealVector& a) const
114 {
115 Weight w = weight(a);
116 auto le = left(w);
117 auto ri = right(w);
118 for (auto i = le; i != ri; ++i)
119 if (are_same(mPoints[i->second], a)) return i->second;
120 return -1;
121 }
122 Weight weight(const RealVector& a) const
123 {
124 Weight w = 0;
125 auto n = mWeights.size();
126 assert(n == a.size());
127 for (int i = 0; i < n; ++i) w += mWeights[i] * a[i];
128 return w;
129 }
130 bool are_same(const RealVector& a, const RealVector& b) const
131 {
132 auto n = a.size();
133 assert(n == b.size());
134 for (int i = 0; i < n; ++i)
135 if (fabs(a[i] - b[i]) >= mEpsilon) return false;
136 return true;
137 }
138 void text_out(buffer& o) const
139 {
140 o << "PointArray( " << mPoints.size() << " points: ";
141 for (auto i = mMap.begin(); i != mMap.end(); ++i) o << i->second << " ";
142 o << ")" << newline;
143 }
144
145 private:
146 std::map<Weight, int> mMap;
147 std::vector<RealVector> mPoints;
148 Weight mEpsilon; // tolerance
149 RealVector mWeights; // random positive numbers summing up to 1
150
151 decltype(mMap)::const_iterator left(Weight key) const
152 {
153 return mMap.lower_bound(key - mEpsilon);
154 }
155 decltype(mMap)::const_iterator right(Weight key) const
156 {
157 return mMap.upper_bound(key + mEpsilon);
158 }
159};

279inline complex complex::operator*(complex c)
280{
281 complex tmp;
282 tmp.real = (real * c.real) - (imag * c.imag);
283 tmp.imag = (real * c.imag) + (imag * c.real);
284 return tmp;
285}

287inline complex complex::operator/(complex c)
288{
289 double div = (c.real * c.real) + (c.imag * c.imag);
290 complex tmp;
291 tmp.real = (real * c.real) + (imag * c.imag);
292 tmp.real /= div;
293 tmp.imag = (imag * c.real) - (real * c.imag);
294 tmp.imag /= div;
295 return tmp;
296}

298inline complex complex::getconjugate() const
299{
300 complex tmp;
301 tmp.real = this->real;
302 tmp.imag = this->imag * -1;
303 return tmp;
304}

306inline complex complex::getreciprocal()
307{
308 complex t;
309 t.real = real;
310 t.imag = imag * -1;
311 double div;
312 div = (real * real) + (imag * imag);
313 t.real /= div;
314 t.imag /= div;
315 return t;
316}
317
318inline double complex::getmodulus()
319{
320 double z;
321 z = (real * real) + (imag * imag);
322 z = sqrt(z);
323 return z;
324}

595 // inline functions needed by track
596 void evaluate_slpHxt(int n, const complex* x0t0, complex* Hxt)
597 {
598 if (is_projective)
599 {
600 complex* SxS = newarray_atomic(complex, (n + 1) * (n - 1));
601 complex* TxT = newarray_atomic(complex, (n + 1) * (n - 1));
602 const complex *x0 = x0t0, *t0 = x0t0 + n;
603 const double t = (*(double*)t0) * bigT; // t0 should be real
604 slpSxS->evaluate(n, x0, SxS);
605 slpTxT->evaluate(n, x0, TxT);
606 double c = cos(t), s = sin(t),
607 sqrt_one_minus_productST_2 = sqrt(1 - productST * productST);
608 double mS = c - productST * s / sqrt_one_minus_productST_2;
609 double mT = s / sqrt_one_minus_productST_2;
610 int j, i;
611 for (j = 0; j < n - 1; j++)
612 for (i = 0; i < n; i++)
613 *(Hxt + i * n + j) =
614 *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
615 j = n - 1; // last column
616 for (i = 0; i < n; i++) *(Hxt + i * n + j) = (x0 + i)->getconjugate();
617 i = n; // last row = H_t
618 mS = -s - productST * c / sqrt_one_minus_productST_2;
619 mT = c / sqrt_one_minus_productST_2;
620 for (j = 0; j < n - 1; j++)
621 {
622 complex tt =
623 *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
624 *(Hxt + i * n + j) = tt;
625 }
626 *(Hxt + n * n + n - 1) = complex(0.); // last row and column
627 freemem(SxS);
628 freemem(TxT);
629 }
630 else
631 slpHxt->evaluate(n + 1, x0t0, Hxt);
632 }
