19// Calling these dummy methods from the various reducer file headers ensures
20// that those translation units are registered as necessary. Without this,
21// they will not be linked in, because apparently the linker does not consider
22// the presense of global object constructors as a reason to include a
23// translation unit into the library. It will simply note that no other
24// translation unit has a dependency on anything in that translation unit
25// and then not include it in the library. This was wonderful to diagnose.
26void dummyLinkerFix() {
27 reducerPackDependency();
28 reducerNoDedupDependency();
29 reducerPackDedupDependency();
30 reducerDedupDependency();
31 reducerHashDependency();
32 reducerHashPackDependency();
33 f4ReducerDependency();
34}

61std::unique_ptr<Reducer> Reducer::makeReducer(
62 ReducerType type,
63 PolyRing const& ring
64) {
65 std::unique_ptr<Reducer> reducer =
66 makeReducerNullOnUnknown(type, ring);
67 if (reducer.get() == 0) {
68 std::ostringstream error;
69 error << "Unknown or unimplemented reducer type " << type << ".\n";
70 throw std::runtime_error(error.str());
71 }
72 return reducer;
73}

75std::unique_ptr<Reducer> Reducer::makeReducerNullOnUnknown(
76 ReducerType type,
77 PolyRing const& ring
78) {
79 for (const auto& r : reducerTypes()) {
80 if (type == r->mId)
81 return r->mCreate(ring);
82 }
83 return nullptr;
84}

117void Reducer::displayReducerTypes(std::ostream& out)
118{
119 mathic::ColumnPrinter pr;
120 auto& id = pr.addColumn(false, " ");
121 auto& desc = pr.addColumn(true, " ");
122 auto types = reducerTypes();
123 auto cmp = [](const Registration* a, const Registration* b) {
124 return a->mId < b->mId;
125 };
126 std::sort(types.begin(), types.end(), cmp);
127 for (const auto& r : types) {
128 id << r->mId << '\n';
129 desc << r->mName << '\n';
130 }
131 out << "Reducer types:\n" << pr;
132}
