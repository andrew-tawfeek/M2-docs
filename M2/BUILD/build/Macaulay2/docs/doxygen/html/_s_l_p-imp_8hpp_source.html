156template <typename RT>
157SLEvaluator* SLEvaluatorConcrete<RT>::specialize(
158 const MutableMat<DMat<RT> >* parameters) const
159{
160 size_t nNewParams = parameters->n_rows();
161 if (parameters->n_cols() != 1 || nNewParams > nInputs)
162 throw exc::engine_error("1-column matrix expected; or #parameters > #vars");
163 SLEvaluatorConcrete<RT>* e = new SLEvaluatorConcrete<RT>(*this);
164 if (isCompiled) {
165 e->nParams = nParams + nNewParams;
166 e->nInputs -= nNewParams;
167 delete[] e->parametersAndInputs;
168 e->parametersAndInputs = new ElementType[e->nParams+e->nInputs];
169 std::copy(parametersAndInputs, parametersAndInputs+nParams, e->parametersAndInputs); // old parameters values
170 for (int i = 0; i < nNewParams; ++i)
171 ring().set(e->parametersAndInputs[nParams+i], parameters->getMat().entry(i, 0));
172 }
173 else {
174 for (int i = 0; i < nNewParams; ++i)
175 ring().set(e->values[varsPos[i]], parameters->getMat().entry(i, 0));
176 e->varsPos.erase(e->varsPos.begin(), e->varsPos.begin() + nNewParams);
177 e->nInputs = e->varsPos.size();
178 e->nParams = nParams + nNewParams;
179 }
180 return e;

183template <typename RT>
184void SLEvaluatorConcrete<RT>::computeNextNode()
185{
186 ElementType& v = *vIt;
187 switch (*nIt++)
188 {
189 case SLProgram::MProduct:
190 ring().set_from_long(v, 1);
191 for (int i = 0; i < *numInputsIt; i++)
192 ring().mult(v, v, *(vIt + (*inputPositionsIt++)));
193 numInputsIt++;
194 break;
195 case SLProgram::MSum:
196 ring().set_zero(v);
197 for (int i = 0; i < *numInputsIt; i++)
198 ring().add(v, v, *(vIt + (*inputPositionsIt++)));
199 numInputsIt++;
200 break;
201 case SLProgram::Det:
202 {
203 int n = static_cast<int>(sqrt(*numInputsIt++));
204 DMat<RT> mat(ring(), n, n);
205 for (int i = 0; i < n; i++)
206 for (int j = 0; j < n; j++)
207 ring().set(mat.entry(i, j), *(vIt + (*inputPositionsIt++)));
208 DMatLinAlg<RT>(mat).determinant(v);
209 }
210 break;
211 case SLProgram::Divide:
212 ring().set(v, *(vIt + (*inputPositionsIt++)));
213 ring().divide(v, v, *(vIt + (*inputPositionsIt++)));
214 break;
215 default:
216 std::cerr << "unknown node type\n";
217 abort();
218 }
219}

221template <typename RT>
222bool SLEvaluatorConcrete<RT>::evaluate(const MutableMatrix* inputs,
223 MutableMatrix* outputs)
224{
225 auto inp = dynamic_cast<const MutableMat<DMat<RT> >*>(inputs);
226 auto out = dynamic_cast<MutableMat<DMat<RT> >*>(outputs);
227 if (inp == nullptr)
228 {
229 ERROR("inputs: expected a dense mutable matrix");
230 return false;
231 }
232 if (out == nullptr)
233 {
234 ERROR("outputs: expected a dense mutable matrix");
235 return false;
236 }
237 if (&ring() != &inp->getMat().ring())
238 {
239 ERROR("inputs are in a different ring");
240 return false;
241 }
242 if (&ring() != &out->getMat().ring())
243 {
244 ERROR("outputs are in a different ring");
245 return false;
246 }
247 return evaluate(inp->getMat(), out->getMat());
248}

250template <typename RT>
251bool SLEvaluatorConcrete<RT>::evaluate(const DMat<RT>& inputs,
252 DMat<RT>& outputs)
253{
254 if (nInputs != inputs.numRows() * inputs.numColumns()) {
255 ERROR(
256 "inputs: the number of inputs does not match the number of entries "
257 "in the inputs matrix");
258 std::cout << nInputs
259 << " != " << inputs.numRows() * inputs.numColumns()
260 << std::endl;
261 return false;
262 }
263 if (nOutputs != outputs.numRows() * outputs.numColumns()) {
264 ERROR(
265 "outputs: the number of outputs does not match the number of entries "
266 "in the outputs matrix");
267 std::cout << nOutputs << " != " << outputs.numRows()
268 << " * " << outputs.numColumns() << std::endl;
269 return false;
270 }
271
272 if (isCompiled) {
273 if(parametersAndInputs==nullptr) {
274 (*compiled_fn)(inputs.unsafeArray(), outputs.unsafeArray());
275 } else {
276 std::copy(inputs.unsafeArray(),inputs.unsafeArray()+nInputs,parametersAndInputs+nParams);
277 (*compiled_fn)(parametersAndInputs, outputs.unsafeArray());
278 }
279 return true;
280 } else {
281 size_t i = 0;
282 for (size_t r = 0; r < inputs.numRows(); r++)
283 for (size_t c = 0; c < inputs.numColumns(); c++)
284 ring().set(values[varsPos[i++]], inputs.entry(r, c));
285 nIt = slp->mNodes.begin();
286 numInputsIt = slp->mNumInputs.begin();
287 inputPositionsIt = slp->mInputPositions.begin();
288 for (vIt = values.begin() + slp->inputCounter; vIt != values.end(); ++vIt)
289 computeNextNode();
290 i = 0;
291 for (size_t r = 0; r < outputs.numRows(); r++)
292 for (size_t c = 0; c < outputs.numColumns(); c++)
293 ring().set(outputs.entry(r, c), values[ap(slp->mOutputPositions[i++])]);
294 return true;
295 }

332template <typename RT>
333inline void norm2(const DMat<RT>& M,
334 size_t n,
335 typename RT::RealElementType& result)
336{
337 const auto& C = M.ring();
338 const auto& R = C.real_ring();
339 typename RT::RealRingType::Element c(R);
340 R.set_zero(result);
341 for (size_t i = 0; i < n; i++)
342 {
343 C.abs_squared(c, M.entry(0, i));
344 R.add(result, result, c);
345 }
346}

377template <typename RT>
378bool HomotopyConcrete<RT, FixedPrecisionHomotopyAlgorithm>::track(
379 const MutableMatrix* inputs,
380 MutableMatrix* outputs,
381 MutableMatrix* output_extras,
382 gmp_RR init_dt,
383 gmp_RR min_dt,
384 gmp_RR epsilon, // o.CorrectorTolerance,
385 int max_corr_steps,
386 gmp_RR infinity_threshold,
387 bool checkPrecision)
388{
389 std::chrono::steady_clock::time_point start =
390 std::chrono::steady_clock::now();
391 size_t solveLinearTime = 0, solveLinearCount = 0, evaluateTime = 0;
392
393 // std::cout << "inside
394 // HomotopyConcrete<RT,FixedPrecisionHomotopyAlgorithm>::track" << std::endl;
395 // double the_smallest_number = 1e-13;
396 const Ring* matRing = inputs->get_ring();
397 if (outputs->get_ring() != matRing)
398 {
399 ERROR("outputs and inputs are in different rings");
400 return false;
401 }
402 auto inp = dynamic_cast<const MutableMat<DMat<RT> >*>(inputs);
403 auto out = dynamic_cast<MutableMat<DMat<RT> >*>(outputs);
404 auto out_extras =
405 dynamic_cast<MutableMat<DMat<M2::ARingZZGMP> >*>(output_extras);
406 if (inp == nullptr)
407 {
408 ERROR("inputs: expected a dense mutable matrix");
409 return false;
410 }
411 if (out == nullptr)
412 {
413 ERROR("outputs: expected a dense mutable matrix");
414 return false;
415 }
416 if (out_extras == nullptr)
417 {
418 ERROR("output_extras: expected a dense mutable matrix");
419 return false;
420 }
421
422 auto& in = inp->getMat();
423 auto& ou = out->getMat();
424 auto& oe = out_extras->getMat();
425 size_t n_sols = in.numColumns();
426 size_t n = in.numRows() - 1; // number of x vars
427
428 if (ou.numColumns() != n_sols or ou.numRows() != n + 2)
429 {
430 ERROR("output: wrong shape");
431 return false;
432 }
433 if (oe.numColumns() != n_sols or oe.numRows() != 2)
434 {
435 ERROR("output_extras: wrong shape");
436 return false;
437 }
438
439 const RT& C = in.ring();
440 typename RT::RealRingType R = C.real_ring();
441
442 typedef typename RT::ElementType ElementType;
443 typedef typename RT::Element Element;
444 typedef typename RT::RealRingType::Element RealElement;
445 typedef typename RT::RealRingType::ElementType RealElementType;
446 typedef MatElementaryOps<DMat<RT> > MatOps;
447
448 RealElement t_step(R), min_step2(R), epsilon2(R), infinity_threshold2(R);
449 R.set_from_BigReal(t_step, init_dt); // initial step
450 R.set_from_BigReal(min_step2, min_dt);
451 R.mult(min_step2, min_step2, min_step2); // min_step^2
452 R.set_from_BigReal(epsilon2, epsilon);
453 int tolerance_bits = int(log2(fabs(R.coerceToDouble(epsilon2))));
454 R.mult(epsilon2, epsilon2, epsilon2); // epsilon^2
455 R.set_from_BigReal(infinity_threshold2, infinity_threshold);
456 R.mult(infinity_threshold2, infinity_threshold2, infinity_threshold2);
457 int num_successes_before_increase = 3;
458
459 RealElement t0(R), dt(R), one_minus_t0(R), dx_norm2(R), x_norm2(R), abs2dc(R);
460
461 // constants
462 RealElement one(R), two(R), four(R), six(R), one_half(R), one_sixth(R);
463 RealElementType& dt_factor = one_half;
464 R.set_from_long(one, 1);
465 R.set_from_long(two, 2);
466 R.set_from_long(four, 4);
467 R.set_from_long(six, 6);
468 R.divide(one_half, one, two);
469 R.divide(one_sixth, one, six);
470
471 Element c_init(C), c_end(C), dc(C), one_half_dc(C);
472
473 // think: x_0..x_(n-1), c
474 // c = the homotopy continuation parameter "t" upstair, varies on a (staight
475 // line) segment of complex plane (from c_init to c_end)
476 // t = a real running in the interval [0,1]
477
478 DMat<RT> x0c0(C, n + 1, 1);
479 DMat<RT> x1c1(C, n + 1, 1);
480 DMat<RT> xc(C, n + 1, 1);
481 DMat<RT> HxH(C, n, n + 1);
482 DMat<RT>& Hxt = HxH; // the matrix has the same shape: reuse memory
483 DMat<RT> LHSmat(C, n, n);
484 auto LHS = submatrix(LHSmat);
485 DMat<RT> RHSmat(C, n, 1);
486 auto RHS = submatrix(RHSmat);
487 DMat<RT> dx(C, n, 1);
488 DMat<RT> dx1(C, n, 1);
489 DMat<RT> dx2(C, n, 1);
490 DMat<RT> dx3(C, n, 1);
491 DMat<RT> dx4(C, n, 1);
492 DMat<RT> Jinv_times_random(C, n, 1);
493
494 ElementType& c0 = x0c0.entry(n, 0);
495 ElementType& c1 = x1c1.entry(n, 0);
496 ElementType& c = xc.entry(n, 0);
497 RealElementType& tol2 = epsilon2; // current tolerance squared
498 bool linearSolve_success;
499 for (size_t s = 0; s < n_sols; s++)
500 {
501 SolutionStatus status = PROCESSING;
502 // set initial solution and initial value of the continuation parameter
503 // for(size_t i=0; i<=n; i++)
504 // C.set(x0c0.entry(i,0), in.entry(i,s));
505 submatrix(x0c0) = submatrix(const_cast<DMat<RT>&>(in), 0, s, n + 1, 1);
506 C.set(c_init, c0);
507 C.set(c_end, ou.entry(n, s));
508
509 R.set_zero(t0);
510 bool t0equals1 = false;
511
512 // t_step is actually the initial (absolute) length of step on the
513 // interval [c_init,c_end]
514 // dt is an increment for t on the interval [0,1]
515 R.set(dt, t_step);
516 C.subtract(dc, c_end, c_init);
517 C.abs(abs2dc, dc); // don't want to create new temporary elts: reusing dc
518 // and abs2dc
519 R.divide(dt, dt, abs2dc);
520
521 int predictor_successes = 0;
522 int count = 0; // number of steps
523 // track the real segment (1-t)*c0 + t*c1, a\in [0,1]
524 while (status == PROCESSING and not t0equals1)
525 {
526 if (M2_numericalAlgebraicGeometryTrace > 3)
527 {
528 buffer o;
529 R.elem_text_out(o, t0, true, false, false);
530 std::cout << "t0 = " << o.str();
531 o.reset();
532 C.elem_text_out(o, c0, true, false, false);
533 std::cout << ", c0 = " << o.str() << std::endl;
534 }
535 R.subtract(one_minus_t0, one, t0);
536 if (R.compare_elems(dt, one_minus_t0) > 0)
537 {
538 R.set(dt, one_minus_t0);
539 t0equals1 = true;
540 C.subtract(dc, c_end, c0);
541 C.set(c1, c_end);
542 }
543 else
544 {
545 C.subtract(dc, c_end, c0);
546 C.mult(dc, dc, dt);
547 C.divide(dc, dc, one_minus_t0);
548 C.add(c1, c0, dc);
549 }
550
551 // PREDICTOR in: x0c0,dt
552 // out: dx
553 /* top-level code for Runge-Kutta-4
554 dx1 := solveHxTimesDXequalsMinusHt(x0,t0);
555 dx2 := solveHxTimesDXequalsMinusHt(x0+(1/2)*dx1*dt,t0+(1/2)*dt);
556 dx3 := solveHxTimesDXequalsMinusHt(x0+(1/2)*dx2*dt,t0+(1/2)*dt);
557 dx4 := solveHxTimesDXequalsMinusHt(x0+dx3*dt,t0+dt);
558 (1/6)*dt*(dx1+2*dx2+2*dx3+dx4)
559 */
560
561 C.mult(one_half_dc, dc, one_half);
562
563 // dx1
564 submatrix(xc) = submatrix(x0c0);
565 TIME(evaluateTime, mHxt.evaluate(xc, Hxt))
566
567 LHS = submatrix(Hxt, 0, 0, n, n);
568 RHS = submatrix(Hxt, 0, n, n, 1);
569 MatrixOps::negateInPlace(RHSmat);
570
571 TIME(solveLinearTime,
572 linearSolve_success =
573 MatrixOps::solveLinear(LHSmat, RHSmat, dx1));
574 solveLinearCount++;
575
576 // dx2
577 if (linearSolve_success)
578 {
579 submatrix(dx1) *= one_half_dc; // "dx1" := (1/2)*dx1*dt
580 submatrix(xc, 0, 0, n, 1) += submatrix(dx1);
581 C.add(c, c, one_half_dc);
582
583 TIME(evaluateTime, mHxt.evaluate(xc, Hxt))
584
585 LHS = submatrix(Hxt, 0, 0, n, n);
586 RHS = submatrix(Hxt, 0, n, n, 1);
587 MatrixOps::negateInPlace(RHSmat);
588
589 TIME(solveLinearTime,
590 linearSolve_success =
591 MatrixOps::solveLinear(LHSmat, RHSmat, dx2);)
592 solveLinearCount++;
593 }
594
595 // dx3
596 if (linearSolve_success)
597 {
598 submatrix(dx2) *= one_half_dc; // "dx2" := (1/2)*dx2*dt
599 submatrix(xc, 0, 0, n, 1) = submatrix(x0c0, 0, 0, n, 1);
600 submatrix(xc, 0, 0, n, 1) += submatrix(dx2);
601 // C.add(c,c,one_half_dc); // c should not change here??? or copy
602 // c two lines above???
603
604 TIME(evaluateTime, mHxt.evaluate(xc, Hxt));
605
606 LHS = submatrix(Hxt, 0, 0, n, n);
607 RHS = submatrix(Hxt, 0, n, n, 1);
608 MatrixOps::negateInPlace(RHSmat);
609
610 TIME(solveLinearTime,
611 linearSolve_success =
612 MatrixOps::solveLinear(LHSmat, RHSmat, dx3););
613 solveLinearCount++;
614 }
615
616 // dx4
617 if (linearSolve_success)
618 {
619 submatrix(dx3) *= dc; // "dx3" := dx3*dt
620 submatrix(xc) = submatrix(
621 x0c0); // sets c=c0 as well (not needed for dx2???,dx3)
622 submatrix(xc, 0, 0, n, 1) += submatrix(dx3);
623 C.add(c, c, dc);
624
625 TIME(evaluateTime, mHxt.evaluate(xc, Hxt));
626
627 LHS = submatrix(Hxt, 0, 0, n, n);
628 RHS = submatrix(Hxt, 0, n, n, 1);
629 MatrixOps::negateInPlace(RHSmat);
630
631 TIME(solveLinearTime,
632 linearSolve_success =
633 MatrixOps::solveLinear(LHSmat, RHSmat, dx4););
634 solveLinearCount++;
635 }
636
637 // "dx1" = .5*dx1*dt, "dx2" = .5*dx2*dt, "dx3" = dx3*dt
638 if (linearSolve_success)
639 {
640 submatrix(dx4) *= dc; // "dx4" = dx4*dt
641 submatrix(dx) =
642 submatrix(dx4); // dx = (1/6)*dt*(dx1+2*dx2+2*dx3+dx4)
643 // = (1/6)*(2*"dx1"+4*"dx2"+2*"dx3"+"dx4")
644 submatrix(dx1) *= two;
645 submatrix(dx) += dx1;
646 // submatrix(dx).addMultipleTo(two,dx1);
647 submatrix(dx2) *= four;
648 submatrix(dx) += dx2;
649 submatrix(dx3) *= two;
650 submatrix(dx) += dx3;
651 submatrix(dx) *= one_sixth;
652
653 // update x0c0
654 submatrix(x1c1) = submatrix(x0c0);
655 submatrix(x1c1, 0, 0, n, 1) += dx;
656 C.add(c1, c0, dc);
657 }
658
659 // CORRECTOR
660 bool is_successful;
661 int n_corr_steps = 0;
662 if (linearSolve_success)
663 {
664 do
665 {
666 n_corr_steps++;
667
668 TIME(evaluateTime, mHxH.evaluate(x1c1, HxH));
669
670 LHS = submatrix(HxH, 0, 0, n, n);
671 RHS = submatrix(HxH, 0, n, n, 1);
672 MatrixOps::negateInPlace(RHSmat);
673
674 TIME(solveLinearTime,
675 linearSolve_success =
676 MatrixOps::solveLinear(LHSmat, RHSmat, dx););
677 solveLinearCount++;
678
679 // x1 += dx
680 submatrix(x1c1, 0, 0, n, 1) += dx;
681
682 normSquared(submatrix(dx), dx_norm2);
683 normSquared(submatrix(x1c1, 0, 0, n, 1), x_norm2);
684
685 R.mult(x_norm2, x_norm2, tol2);
686 is_successful = R.compare_elems(dx_norm2, x_norm2) < 0;
687 }
688 while (not is_successful and n_corr_steps < max_corr_steps);
689 }
690 // std::cout << "past corrector loop...\n";
691 if (not linearSolve_success or not is_successful)
692 {
693 // predictor failure
694 predictor_successes = 0;
695 R.mult(dt, dt, dt_factor);
696 t0equals1 = false;
697 C.abs_squared(abs2dc, dc);
698 if (R.compare_elems(abs2dc, min_step2) < 0)
699 status = MIN_STEP_FAILED;
700 }
701 else
702 {
703 // predictor success
704 predictor_successes = predictor_successes + 1;
705 MatOps::setFromSubmatrix(x1c1, 0, n, 0, 0, x0c0); // x1=x0
706 R.add(t0, t0, dt); // increment t: so far only s was incremented
707 count++;
708 if (predictor_successes >= num_successes_before_increase)
709 {
710 predictor_successes = 0;
711 R.divide(dt, dt, dt_factor);
712 }
713 }
714
715 normSquared(submatrix(x0c0, 0, 0, n, 1),
716 x_norm2); // x_norm2 = ||x0||^2
717
718 if (not linearSolve_success)
719 status = SINGULAR;
720 else if (checkPrecision and not t0equals1)
721 { // precision check
722 mHxH.evaluate(x0c0, HxH);
723 MatOps::setFromSubmatrix(HxH, 0, n - 1, 0, n - 1, LHSmat); // Hx
724 // setRandomUnitVector(RHSmat,n);
725 for (int i = 0; i < n; i++) C.random(RHSmat.entry(i, 0));
726
727 TIME(solveLinearTime,
728 linearSolve_success = MatrixOps::solveLinear(
729 LHSmat, RHSmat, Jinv_times_random););
730 solveLinearCount++;
731
732 norm2(Jinv_times_random,
733 n,
734 dx_norm2); // this stands for ||J^{-1}||
736// ||J^{-1}|| should be multiplied by a factor
737// reflecting an estimate on the error of evaluation of J
738#define PRECISION_SAFETY_BITS 10
739 int more_bits = int(log2(fabs(R.coerceToDouble(dx_norm2)))) / 2;
740 int precision_needed = PRECISION_SAFETY_BITS + tolerance_bits + more_bits;
741 if (precision_needed<53) precision_needed = 53;
742 if (M2_numericalAlgebraicGeometryTrace > 3)
743 std::cout << "precision needed = " << precision_needed << " = "
744 << PRECISION_SAFETY_BITS << "(safety) + "
745 << tolerance_bits << "(tolerance) + "
746 << more_bits << "(additional)\n"
747 << "current precision = " << R.get_precision() << std::endl;
748 if (R.get_precision() < precision_needed)
749 status = INCREASE_PRECISION;
750 else if (R.get_precision() != 53 and
751 R.get_precision() > 2 * precision_needed)
752 status = DECREASE_PRECISION;
753 if (M2_numericalAlgebraicGeometryTrace > 3)
754 std::cout << "status = " << status << std::endl;
755 };
756
757 // infinity/origin checks
758 if (status == PROCESSING)
759 {
760 if (R.compare_elems(infinity_threshold2, x_norm2) < 0)
761 status = INFINITY_FAILED;
762 else
763 {
764 if (R.is_zero(x_norm2))
765 status = ORIGIN_FAILED;
766 else
767 {
768 R.divide(x_norm2, one, x_norm2); // 1/||x||^2
769 if (R.compare_elems(infinity_threshold2, x_norm2) < 0)
770 status = ORIGIN_FAILED;
771 }
772 }
773 }
774 }
775 // record the solution
776 // set initial solution and initial value of the continuation parameter
777 for (size_t i = 0; i <= n; i++) C.set(ou.entry(i, s), x0c0.entry(i, 0));
778 C.set(ou.entry(n + 1, s), dc); // store last increment attempted
779 if (status == PROCESSING) status = REGULAR;
780 oe.ring().set_from_long(oe.entry(0, s), status);
781 oe.ring().set_from_long(oe.entry(1, s), count);
782 }
783
784 std::chrono::steady_clock::time_point end = std::chrono::steady_clock::now();
785 if (M2_numericalAlgebraicGeometryTrace > 1)
786 {
787 std::cout << "-- track took "
788 << std::chrono::duration_cast<std::chrono::milliseconds>(end -
789 start)
790 .count()
791 << "ms.\n";
792 std::cout << "-- # of solveLinear calls = " << solveLinearCount
793 << std::endl;
794 std::cout << "-- time of solveLinear calls = " << solveLinearTime
795 << " ns." << std::endl;
796 std::cout << "-- time of evaluate calls = " << evaluateTime << " ns."
797 << std::endl;
798 }
799 return true;
800}
