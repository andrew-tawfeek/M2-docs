105 ClassicGBAlg::ClassicGBAlg(
106 const Basis& basis,
107 Reducer& reducer,
108 int monoLookupType,
109 bool preferSparseReducers,
110 size_t queueType
111 ):
112 mCallback(nullptr),
113 mBreakAfter(0),
114 mPrintInterval(0),
115 mSPairGroupSize(reducer.preferredSetSize()),

144 void ClassicGBAlg::insertPolys(
145 std::vector<std::unique_ptr<Poly> >& polynomials
146 ) {
147 if (!mUseAutoTopReduction) {
148 for (auto it = polynomials.begin(); it != polynomials.end(); ++it) {
149 assert(it->get() != 0);
150 if ((*it)->isZero())
151 continue;
152 if (mBasis.divisor((*it)->leadMono()) != static_cast<size_t>(-1)) {
153 *it = mReducer.classicReduce(**it, mBasis);
154 if ((*it)->isZero())
155 continue;
156 }
157
158 mBasis.insert(std::move(*it));
159 mSPairs.addPairs(mBasis.size() - 1);
160 }
161 polynomials.clear();
162 return;
163 }
164
165 std::vector<size_t> toRetire;
166 std::vector<std::unique_ptr<Poly>> toReduce;
167 std::vector<std::unique_ptr<Poly>> toInsert;
168 std::swap(toInsert, polynomials);
169
170 while (!toInsert.empty()) {
171 // todo: sort by lead term to avoid insert followed by immediate
172 // removal.
173
174 // insert polynomials from toInsert with minimal lead term and
175 // extract those from the basis that become non-minimal.
176 for (auto it = toInsert.begin(); it != toInsert.end(); ++it) {
177 assert(it->get() != 0);
178 if ((*it)->isZero())
179 continue;
180
181 // We check for a divisor from mBasis because a new reducer
182 // might have been added since we did the reduction or perhaps a
183 // non-reduced polynomial was passed in.
184 if (mBasis.divisor((*it)->leadMono()) != static_cast<size_t>(-1))
185 toReduce.push_back(std::move(*it));
186 else {
187 mBasis.insert(std::move(*it));
188 assert(toRetire.empty());
189 mSPairs.addPairsAssumeAutoReduce(mBasis.size() - 1, toRetire);
190 for (auto r = toRetire.begin(); r != toRetire.end(); ++r)
191 toReduce.push_back(mBasis.retire(*r));
192 toRetire.clear();
193 }
194 }
195 toInsert.clear();
196 assert(toRetire.empty());
197
198 // reduce everything in toReduce
199 if (!toReduce.empty()) {
200 mReducer.classicReducePolySet(toReduce, mBasis, toInsert);
201 toReduce.clear();
202 }
203 }
204
205 assert(toRetire.empty());
206 assert(toInsert.empty());
207 assert(toReduce.empty());
208 }

210 void ClassicGBAlg::insertReducedPoly(
211 std::unique_ptr<Poly> polyToInsert
212 ) {
213 assert(polyToInsert.get() != 0);
214 if (polyToInsert->isZero())
215 return;
216 assert(mBasis.divisor(polyToInsert->leadMono()) ==
217 static_cast<size_t>(-1));
218
219 if (tracingLevel > 20) {
220 std::cerr << "inserting basis element " << mBasis.size() << ": ";
221 if (tracingLevel > 100) {
222 MathicIO<>().writePoly(*polyToInsert, false, std::cerr);
223 std::cerr << std::endl;
224 } else {
225 mRing.printMonomialFrobbyM2Format
226 (std::cerr, Monoid::toOld(polyToInsert->leadMono()));
227 if (polyToInsert->termCount() > 1)
228 std::cerr << " + [...]";
229 std::cerr << std::endl;
230 }
231 }
232
233 if (!mUseAutoTopReduction) {
234 size_t const newGen = mBasis.size();
235 mBasis.insert(std::move(polyToInsert));
236 mSPairs.addPairs(newGen);
237 return;
238 }
239
240 std::vector<size_t> toRetireAndReduce;
241 std::vector<Poly*> toReduce;
242
243 try {
244 do {
245 // reduce polynomial and insert into basis
246 {
247 std::unique_ptr<Poly> reduced;
248 if (toReduce.empty()) // if first iteration
249 reduced = std::move(polyToInsert);
250 else {
251 reduced = mReducer.classicReduce(*toReduce.back(), mBasis);
252 if (tracingLevel > 20) {
253 if (reduced->isZero()) {
254 std::cerr << "auto-top-reduce cascade: "
255 "basis element reduced to zero."
256 << std::endl;
257 } else {
258 std::cerr << "auto-top-reduce cascade: "
259 "inserting reduced poly with lead term "
260 << std::endl;
261 mRing.printMonomialFrobbyM2Format
262 (std::cerr, Monoid::toOld(reduced->leadMono()));
263 std::cerr << '\n';
264 }
265 }
266 delete toReduce.back();
267 toReduce.pop_back();
268 }
269 if (reduced->isZero())
270 continue;
271 reduced->makeMonic();
272 mBasis.insert(std::move(reduced));
273 }
274
275 // form S-pairs and retire basis elements that become top reducible.
276 const size_t newGen = mBasis.size() - 1;
277 assert(toRetireAndReduce.empty());
278 mSPairs.addPairsAssumeAutoReduce(newGen, toRetireAndReduce);
279 for (std::vector<size_t>::const_iterator it = toRetireAndReduce.begin();
280 it != toRetireAndReduce.end(); ++it) {
281 toReduce.push_back(0); // allocate space in vector before .release()
282 toReduce.back() = mBasis.retire(*it).release();
283 }
284 toRetireAndReduce.clear();
285 } while (!toReduce.empty());
286 } catch (...) {
287 for (std::vector<Poly*>::iterator it = toReduce.begin();
288 it != toReduce.end(); ++it)
289 delete *it;
290 throw;
291 }
292 assert(toReduce.empty());
293 assert(toRetireAndReduce.empty());
294 }

328 void ClassicGBAlg::step() {
329 assert(!mSPairs.empty());
330 if (tracingLevel > 30)
331 std::cerr << "Determining next S-pair" << std::endl;
332
333 assert(mSPairGroupSize >= 1);
334 std::vector<std::pair<size_t, size_t> > spairGroup;
335 exponent w = 0;
336 for (unsigned int i = 0; i < mSPairGroupSize; ++i) {
337 auto p = mSPairs.pop(w);
338 if (p.first == static_cast<size_t>(-1)) {
339 assert(p.second == static_cast<size_t>(-1));
340 break; // no more S-pairs
341 }
342 assert(p.first != static_cast<size_t>(-1));
343 assert(p.second != static_cast<size_t>(-1));
344 assert(!mBasis.retired(p.first));
345 assert(!mBasis.retired(p.second));
346
347 spairGroup.push_back(p);
348 }
349 if (spairGroup.empty())
350 return; // no more s-pairs
351 std::vector<std::unique_ptr<Poly>> reduced;
352
353 // w is the negative of the degree of the lcm's of the chosen spairs
354 MATHICGB_LOG(SPairDegree) <<
355 spairGroup.size() << " pairs in degree " << -w << std::endl;
356
357 mReducer.classicReduceSPolySet(spairGroup, mBasis, reduced);
358
359 // sort the elements to get deterministic behavior. The order will change
360 // arbitrarily when running multithreaded. Also, if preferring older
361 // reducers, it is of benefit to break ties by preferring the sparser
362 // reducer. Age does not have ties, since each element has a distinct
363 // index, but if they all come from the same matrix then there is
364 // really nothing to distinguish them - the relative age is arbitrarily
365 // chosen. If we order sparsest-first, we will effectively make the
366 // arbitrary choice among reducers from the same matrix in favor of sparser
367 // reducers.
368 auto order = [&](
369 const std::unique_ptr<Poly>& a,
370 const std::unique_ptr<Poly>& b
371 ) {
372 const auto aTermCount = a->termCount();
373 const auto bTermCount = b->termCount();
374 if (aTermCount < bTermCount)
375 return true;
376 if (aTermCount > bTermCount)
377 return false;
378 auto bIt = b->begin();
379 const auto aEnd = a->end();
380 for (auto aIt = a->begin(); aIt != aEnd; ++aIt, ++bIt) {
381 const auto monoCmp =
382 mRing.monoid().compare(aIt.mono(), bIt.mono());
383 if (monoCmp == LT)
384 return true;
385 if (monoCmp == GT)
386 return false;
387 if (aIt.coef() < bIt.coef())
388 return true;
389 if (aIt.coef() > bIt.coef())
390 return false;
391 }
392 return false;
393 };
394 std::sort(reduced.begin(), reduced.end(), order);
395
396 insertPolys(reduced);
397 if (mUseAutoTailReduction)
398 autoTailReduce();
399 }

401 void ClassicGBAlg::autoTailReduce() {
402 assert(mUseAutoTailReduction);
403
404 for (size_t i = 0; i < mBasis.size(); ++i) {
405 if (mBasis.retired(i))
406 continue;
407 if (mBasis.usedAsReducerCount(i) < 1000)
408 continue;
409 mBasis.replaceSameLeadTerm
410 (i, mReducer.classicTailReduce(mBasis.poly(i), mBasis));
411 }
