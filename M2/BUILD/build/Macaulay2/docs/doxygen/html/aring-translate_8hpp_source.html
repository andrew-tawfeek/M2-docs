230// Promote an element of one ring to another.
231// Given a "natural and canonical" map R --> S (depends on the context)
232// and an element fR of R, set result_fS to be the image of fR under this map.
233// Return true if this was done, else return false.
234template <typename RingR, typename RingS>
235bool mypromote(const RingR& R,
236 const RingS& S,
237 const typename RingR::ElementType& fR,
238 typename RingS::ElementType& result_fS)
239{
240 (void) R;
241 (void) S;
242 (void) fR;
243 (void) result_fS;
244 return false;
245}

611inline bool mylift(const ARingRR& R,
612 const ARingRRi& S,
613 ARingRR::ElementType& result_gR,
614 const ARingRRi::ElementType& gS)
615{
616 ARingRRR T(S.get_precision());
617 ARingRRR::Element gT(T);
618 auto gS1 = const_cast<ARingRRi::ElementType&>(gS);
619 S.midpoint(gT,gS1);
620 bool liftstep = mylift(R,T,result_gR,gT);
621 S.diameter(gT,gS1);
622 return liftstep && T.is_zero(gT);
623}

653// ZZ/p --> ZZ/p. 9 versions NONE OF THESE.
654// instead:
655// (1) lift to ZZ (int version?)
656// (2) promote to the version of ZZ/p.
657
658// ZZ/p --> GF(p^n)
659
660// GF(p^m) --> GF(p^n), where m|n, also switch GF type.
661
662// The following are all essentially from_BigRational
663// QQ --> RR
664// QQ --> RRR
665// QQ --> CC
666// QQ --> CCC
667
668// really, there are:
669// RR --> RRR (from double)
670// RR --> CC (imag part = 0)
671// CC --> CCC (from doubles)
672// RRR --> RRR (change precision)
673// RRR --> CCC (imag part = 0)
674// CCC --> CCC (change precision)
675
676// RRR --> RR Use lift.
677// CCC --> CC Use lift.
678
679// RRR --> RR (truncate)
680// RR --> RRR (make new precision)
681// RRR --> RRR (change precision)
682// RR --> CC (imag part = 0)
683// RRR --> CC (RRR --> RR, and imag part = 0)
684// CCC --> CC (truncate)
685// RR --> CCC
686// RRR --> CCC
687// CC --> CCC
688// CCC --> CCC
