573 // This method is made inline to avoid the overhead from calling a function
574 // for every exponent. This is also why mExponents is not inside the pimpl -
575 // otherwise we couldn't access it from here. That then explains why
576 // mExponents is a raw pointer instead of a std::vector - the compiler for the
577 // caller and the library must agree on the memory layout of the object and
578 // that is less likely to introduce problems for a raw pointer than for a
579 // std::vector. In particular, doing it this way allows the library and
580 // the caller to use different implementations of the STL.

582 const VarIndex index,
583 const Exponent exponent
584 ) {
585#ifdef MATHICGB_DEBUG
586 assert(index < varCount());
587#endif
588 mExponents[index] = exponent;

597 inline bool GroebnerConfiguration::setMonomialOrder(
598 const BaseOrder baseOrder,
599 const std::vector<Exponent>& gradings
600 ) {
601 // We cannot do gradings.data() since we may be compiling without C++11
602 // support. We also cannot do &*gradings.begin() if gradings is empty
603 // since then we are dereferencing an invalid iterator - the debug build
604 // of MSVC's STL will correctly flag this as an error.
605 const MonomialOrderData data = {
606 baseOrder,
607 gradings.empty() ? static_cast<Exponent*>(0) : &*gradings.begin(),
608 gradings.size()
609 };
610 return setMonomialOrderInternal(data);
611 }

909 template<class OutputStream>
910 void computeGroebnerBasis(
911 GroebnerInputIdealStream& inputWhichWillBeCleared,
912 OutputStream& output
913 ) {
914 typedef mgbi::IdealAdapter::ConstTerm ConstTerm;
915 mgbi::IdealAdapter ideal;
916 const bool doOutput =
917 mgbi::internalComputeGroebnerBasis(inputWhichWillBeCleared, ideal);
918 if (!doOutput)
919 return;
920
921 ideal.toFirstTerm();
922 const size_t varCount = ideal.varCount();
923 const size_t polyCount = ideal.polyCount();
924 output.idealBegin(polyCount);
925 for (size_t polyIndex = 0; polyIndex < polyCount; ++polyIndex) {
926 const size_t termCount = ideal.termCount(polyIndex);
927 output.appendPolynomialBegin(termCount);
928 for (size_t termIndex = 0; termIndex < termCount; ++termIndex) {
929 const ConstTerm term = ideal.nextTerm();
930 output.appendTermBegin(term.com);
931 for (size_t var = 0; var < varCount; ++var)
932 output.appendExponent(var, term.exponents[var]);
933 output.appendTermDone(term.coef);
934 }
935 output.appendPolynomialDone();
936 }
937 output.idealDone();
938 }
