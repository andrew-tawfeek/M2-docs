239 void add(elem &result, elem a, elem b) const
240 {
241 if (a == 0)
242 result = b;
243 else if (b == 0)
244 result = a;
245 else
246 {
247 int n = a - b;
248 if (n > 0)
249 {
250 if (n == mGF.minusOne())
251 result = 0;
252 else
253 result = modulus_add(b, mGF.oneTable(n), mGF.orderMinusOne());
254 }
255 else if (n < 0)
256 {
257 if (-n == mGF.minusOne())
258 result = 0;
259 else
260 result = modulus_add(a, mGF.oneTable(-n), mGF.orderMinusOne());
261 }
262 else
263 {
264 if (mGF.characteristic() == 2)
265 result = 0;
266 else
267 result =
268 modulus_add(a, mGF.oneTable(mGF.one()), mGF.orderMinusOne());
269 }
270 }
271 }

278 add(result, a, c);

300 void divide(elem &result, elem a, elem b) const
301 {
302 if (b == 0)
303 throw exc::division_by_zero_error();
304 if (a != 0)
305 {
306 int c = a - b;
307 if (c <= 0) c += mGF.orderMinusOne();
308 result = c;
309 }
310 else
311 result = 0;
312 }

314 void power(elem &result, elem a, long n) const
315 {
316 if (a != 0)
317 {
318 long order1 = static_cast<long>(mGF.orderMinusOne());
319 result = static_cast<elem>((a * n) % order1);
320 if (result <= 0) result += mGF.orderMinusOne();
321 }
322 else
323 {
324 // a is the zero element
325 if (n > 0)
326 result = 0;
327 else if (n == 0)
328 result = mGF.one();
329 else
330 throw exc::division_by_zero_error();
331 }
332 }

334 void power_mpz(elem &result, elem a, mpz_srcptr n) const
335 {
336 if (a != 0)
337 {

339 power(result, a, n1);
340 }
341 else
342 {
343 // a is the zero element
344 if (mpz_sgn(n) > 0)
345 result = 0;
346 else if (mpz_sgn(n) == 0)
347 result = mGF.one();
