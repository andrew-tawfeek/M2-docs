205 template<class Iter>
206 void DivMask::Calculator<C, true>::
207 rebuild(Iter begin, Iter end, const C& conf) {
208 const size_t size = std::distance(begin, end);
209 if (size == 0) {
210 rebuildDefault(conf);
211 return;
212 }
213
214 _bits.clear();
215 const size_t varCount = conf.getVarCount();
216 const size_t TotalBits = sizeof(MaskType) * BitsPerByte;
217
218 // ** Determine information about each variable
219 std::vector<VarData> datas;
220 for (size_t var = 0; var < varCount; ++var) {
221 Exponent min = conf.getExponent(*begin, 0);
222 Exponent max = min;
223 for (Iter it = begin; it != end; ++it) {
224 Exponent exp = conf.getExponent(*it, var);
225 if (max < exp)
226 max = exp;
227 if (exp < min)
228 min = exp;
229 }
230 Exponent average = min + (max - min) / 2; // this formula avoids overflow
231 size_t split = 0;
232 for (Iter it = begin; it != end; ++it)
233 if (conf.getExponent(*it, var) < average)
234 ++split;
235 if (split > size / 2)
236 split = size - split;
237 VarData data;
238 data.var = var;
239 data.split = split;
240 datas.push_back(data);
241 }
242 std::sort(datas.begin(), datas.end());
243 MATHIC_ASSERT(datas.size() == varCount);
244
245 // distribute bits to variables according to the data collected
246 std::vector<size_t> bitsForVars(varCount);
247 for (size_t i = 0; i < varCount; ++i) {
248 const size_t var = datas[i].var;
249 bitsForVars[var] = TotalBits / varCount;
250 if (i < TotalBits % varCount)
251 ++bitsForVars[var];
252 }
253
254 // calculate the meaning of each bit and put it in _bits
255 for (size_t var = 0; var < varCount; ++var) {
256 const size_t bitsForVar = bitsForVars[var];
257 if (bitsForVar == 0)
258 continue;
259
260 const bool useRank = false;
261 if (useRank) {
262 std::sort(begin, end, ExponentComparer<C>(var, conf));
263 size_t size = std::distance(begin, end);
264 size_t offset = size / (bitsForVar + 1);
265 if (offset== 0)
266 offset = 1;
267 Exponent lastExp;
268 size_t lastJ = 0;
269 for (size_t i = 1; i <= bitsForVar; ++i) {
270 size_t j = i * offset;
271 if (i > 1 && j < lastJ)
272 j = lastJ + 1;
273 if (j >= size)
274 j = size - 1;
275 if (i > 1) {
276 while (j < size && conf.getExponent(begin[j], var) == lastExp)
277 ++j;
278 if (j == size)
279 break;
280 }
281 _bits.push_back(
282 std::make_pair(var, conf.getExponent(begin[j], var)));
283 lastJ = j;
284 lastExp = conf.getExponent(begin[j], var);
285 if (j == size - 1)
286 break;
287 }
288 } else {
289 Exponent min = conf.getExponent(*begin, 0);
290 Exponent max = min;
291 for (Iter it = begin; it != end; ++it) {
292 Exponent exp = conf.getExponent(*it, var);
293 if (max < exp)
294 max = exp;
295 if (exp < min)
296 min = exp;
297 }
298
299 // divide the range [a,b] into bitsForVar equal pieces
300 // and use the left end points of those ranges
301 // as the points for the bits.
302 Exponent increment = (max - min) / static_cast<Exponent>(bitsForVar); // todo: can avoid cast?
303 if (increment == 0)
304 increment = 1;
305 for (size_t i = 0; i < bitsForVar; ++i)
306 _bits.push_back(std::make_pair(var, min + increment * static_cast<Exponent>(i)));
307 // todo: can avoid cast?
308 }
309 }
310 }

314 rebuildDefault(const C& conf) {
315 _bits.clear();
316 const size_t varCount = conf.getVarCount();
317 const size_t TotalBits = sizeof(MaskType) * BitsPerByte;
318 for (size_t var = 0; var < varCount; ++var) {
319 const size_t bitsForVar =
320 TotalBits / varCount + (var < TotalBits % varCount);
321 Exponent exp = 0;
322 for (size_t i = 0; i < bitsForVar; ++i) {
323 _bits.push_back(std::make_pair(var, exp));
324 exp = (i == 0 ? 1 : exp * 2);
325 }

331 DivMask::MaskType DivMask::Calculator<C, true>::
332 compute(const T& t, const C& conf) const {
333#ifdef MATHIC_TRACK_DIV_MASK_HIT_RATIO
334 ++DivMaskStats::maskComputes;
335#endif
336 typedef typename BitContainer::const_iterator const_iterator;
337 const const_iterator end = _bits.end();
338 DivMask::MaskType mask = 0;
339 for (const_iterator it = _bits.begin(); it != end; ++it)
340 mask = (mask << 1) | (conf.getExponent(t, it->first) > it->second);
