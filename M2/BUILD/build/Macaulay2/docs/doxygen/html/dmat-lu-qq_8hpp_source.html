81 static void findColumnRankProfileFromLU(const fmpq_mat_t B,
82 std::vector<size_t>& profile)
83 {
84 profile.clear();
85 long nrows = fmpq_mat_nrows(B);
86 long ncols = fmpq_mat_ncols(B);
87 long thiscol = 0;
88 long thisrow = 0;
89 while (thisrow < nrows and thiscol < ncols)
90 {
91 if (not fmpq_is_zero(fmpq_mat_entry(B, thisrow, thiscol)))
92 {
93 profile.push_back(thiscol);
94 thisrow++;
95 }
96 thiscol++;
97 }
98 }

199 bool solve(const Mat& B1, Mat& X1)
200 {
201 // This is the version where A = this isn't nec square or full rank...
202 // Plan for now (we will see how efficient this version is): put A|B into a
203 // single matrix.
204 // rref
205 // get column rank profile
206 // from that, either return false (inconsistent), or copy the correct values
207 // into X
208
209 // Too bad flint doesn't do this automatically...
210 // 1. copy A, B to flint objects
211 // 2. concatenate them
212 // 3. rref
213 // 4. pivot columns
214 // 5. if pivot column is in B columns, return false.
215 // 6. otherwise:
216 // resize X
217 // populate X with the solution (only non-zero entries are in pivot
218 // columns
219 // 6. copy
220 // TODO
221
222 long nrows = mInputMatrix.numRows();
223 long ncols = mInputMatrix.numColumns();
224 std::vector<size_t> profile;
225 Mat AB1(mInputMatrix.ring(), nrows, ncols + B1.numColumns());
226 concatenateMatrices<Mat>(mInputMatrix, B1, AB1);
227 FlintQQMat AB(AB1);
228 fmpq_mat_rref(AB.value(), AB.value());
229 findColumnRankProfileFromLU(AB.value(), profile);
230 if (profile[profile.size() - 1] >= ncols)
231 return false; // system is inconsistent
232 // At this point, we know the solutions. Should we go through FlintQQMat,
233 // or go directly to Mat?
234 FlintQQMat X(ncols, B1.numColumns());
235 for (long c = 0; c < B1.numColumns(); c++)
236 {
237 // Fill in this column
238 for (long r = 0; r < profile.size(); r++)
239 {
240 fmpq_set(fmpq_mat_entry(X.value(), profile[r], c),
241 fmpq_mat_entry(AB.value(), r, ncols + c));
242 }
243 }
244 X.toDMat(X1);
245 return true;
246 }
