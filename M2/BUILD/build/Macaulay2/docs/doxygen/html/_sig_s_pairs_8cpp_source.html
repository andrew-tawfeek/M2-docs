65void SigSPairs::setupBaseDivisors(
66 BaseDivisor& divisor1,
67 BaseDivisor& divisor2,
68 size_t& highDivisorCmp,
69 size_t newGenerator
70) {
71 BaseDivContainer divisors;
72 const size_t MaxBaseDivisors = 2;
73 if (mUseBaseDivisors || mUseHighBaseDivisors)
74 divisors.reserve(MaxBaseDivisors + 1);
75
76 MATHICGB_ASSERT(mUseBaseDivisors || mUseHighBaseDivisors);
77 MATHICGB_ASSERT(mKnownSyzygyTri.columnCount() == newGenerator);
78 mKnownSyzygyTri.addColumn();
79
80 if (mUseHighBaseDivisors) {
81 size_t highDivisor = GB->highBaseDivisor(newGenerator);
82 if (highDivisor != static_cast<size_t>(-1)) {
83 // To use a high divisor, the ratio of the other generator has to be
84 // greater than both the ratio of newGenerator and of the high ratio
85 // divisor. We can check both at once by letting highDivisorCmp
86 // be the one out of newGenerator and highDivisor that has the
87 // highest ratio.
88 if (GB->ratioCompare(newGenerator, highDivisor) == GT)
89 highDivisorCmp = newGenerator;
90 else
91 highDivisorCmp = highDivisor;
92 }
93 } else
94 highDivisorCmp = static_cast<size_t>(-1);
95 if (!mUseBaseDivisors)
96 return;
97
98 std::vector<size_t> divs;
99 GB->lowBaseDivisors(divs, MaxBaseDivisors, newGenerator);
100 MATHICGB_ASSERT(divs.size() <= MaxBaseDivisors);
101
102 divisors.resize(divs.size());
103 for (size_t i = 0; i < divisors.size(); ++i) {
104 BaseDivisor& bd = divisors[i];
105 bd.baseDivisor = divs[i];
106
107 // Only use the base divisor technique for generators with ratio
108 // less than both N and baseDivisor. baseDivisorCmp is the
109 // smallest one of these, so it can be used for this comparison.
110 if (GB->ratioCompare(newGenerator, bd.baseDivisor) == LT)
111 bd.ratioLessThan = newGenerator;
112 else
113 bd.ratioLessThan = bd.baseDivisor;
114
115 // Construct a monomial in makeSPair_t2 that can be used
116 // to eliminate s-pairs quickly based on the s-pairs already
117 // eliminated for baseDivisor.
118 auto newSig = GB->signature(newGenerator);
119 auto newLead = GB->leadMono(newGenerator);
120 auto baseDivSig = GB->signature(bd.baseDivisor);
121 auto baseDivLead = GB->leadMono(bd.baseDivisor);
122 bd.baseMonomial = R->allocMonomial();
123 R->mysteriousSPairMonomialRoutine(
124 Monoid::toOld(newSig),
125 Monoid::toOld(newLead),
126 Monoid::toOld(baseDivSig),
127 Monoid::toOld(baseDivLead),
128 bd.baseMonomial
129 );
130 }
131
132 divisor1.baseDivisor = static_cast<size_t>(-1);
133 divisor2.baseDivisor = static_cast<size_t>(-1);
134 if (divisors.size() >= 1)
135 divisor1 = divisors.front();
136 if (divisors.size() == 2) {
137 divisor2 = divisors.back();
138 MATHICGB_ASSERT(GB->ratioCompare
139 (divisor1.ratioLessThan, divisor2.ratioLessThan) != LT);
140 }
141}

143void SigSPairs::makePreSPairs(size_t newGen)
144{
145 MATHICGB_ASSERT(mIndexSigs.empty());
146 MATHICGB_ASSERT(newGen < GB->size());
147 mStats.spairsConstructed += newGen;
148
149 monomial baseDivisorMonomial = 0;
150
151 BaseDivisor divisor1;
152 BaseDivisor divisor2;
153 divisor1.baseDivisor = static_cast<size_t>(-1);
154 divisor2.baseDivisor = static_cast<size_t>(-1);
155 size_t highDivisorCmp = static_cast<size_t>(-1);
156 if (mUseBaseDivisors || mUseHighBaseDivisors)
157 setupBaseDivisors(divisor1, divisor2, highDivisorCmp, newGen);
158
159 monomial hsyz = 0;
160 if (!mPostponeKoszuls)
161 hsyz = R->allocMonomial();
162
163 auto newSig = GB->signature(newGen);
164 auto newLead = GB->leadMono(newGen);
165 auto pairSig = R->allocMonomial();
166
167 if (mUseHighBaseDivisors && divisor1.baseDivisor != static_cast<size_t>(-1))
168 ++mStats.hasLowBaseDivisor;
169 if (mUseHighBaseDivisors && highDivisorCmp != static_cast<size_t>(-1))
170 ++mStats.hasHighBaseDivisor;
171
172 PreSPair result;
173 for (size_t oldGen = 0; oldGen < newGen; oldGen++) {
174 auto oldSig = GB->signature(oldGen);
175 auto oldLead = GB->leadMono(oldGen);
176
177 // Check whether this is a non-regular spair.
178 // 'cmp' is used below too.
179 const int cmp = GB->ratioCompare(newGen, oldGen);
180 if (cmp == EQ) {
181 ++mStats.nonregularSPairs;
182 continue;
183 }
184
185 // check high ratio divisor
186 if (mUseHighBaseDivisors &&
187 highDivisorCmp != static_cast<size_t>(-1) &&
188 GB->ratioCompare(oldGen, highDivisorCmp) == GT &&
189 mKnownSyzygyTri.bitUnordered(oldGen, highDivisorCmp)) {
190 MATHICGB_ASSERT(oldGen != highDivisorCmp); // otherwise ratios should be equal
191 mKnownSyzygyTri.setBit(newGen, oldGen, true);
192 ++mStats.highBaseDivisorHits;
193 // if MATHICGB_DEBUG defined, get to the ASSERT below stating
194 // that this is really a syzygy
195#ifndef MATHICGB_DEBUG
196 continue;
197#endif
198 }
199
200 // check low ratio divisors
201 if (mUseBaseDivisors &&
202 divisor1.baseDivisor != static_cast<size_t>(-1) &&
203 GB->ratioCompare(oldGen, divisor1.ratioLessThan) == LT) {
204 // if no divisor1, also no divisor 2 and also
205 // divisor1 has larger ratio, so skip both checks if divisor1 fails due
206 // to the ratio being too small or because there is no divisor1.
207
208 if (
209 (divisor1.baseDivisor != oldGen && // if divisor1 is a hit
210 mKnownSyzygyTri.bitUnordered(divisor1.baseDivisor, oldGen) &&
211 monoid().divides(oldLead, divisor1.baseMonomial))
212 || // or if divisor2 is a hit
213 (divisor2.baseDivisor != static_cast<size_t>(-1) &&
214 GB->ratioCompare(oldGen, divisor2.ratioLessThan) == LT &&
215 divisor2.baseDivisor != oldGen &&
216 mKnownSyzygyTri.bitUnordered(divisor2.baseDivisor, oldGen) &&
217 monoid().divides(oldLead, divisor2.baseMonomial))
218 ) {
219 mKnownSyzygyTri.setBit(newGen, oldGen, true);
220 ++mStats.lowBaseDivisorHits;
221 // if MATHICGB_DEBUG defined, get to the ASSERT below stating
222 // that this really is a syzygy.
223#ifndef MATHICGB_DEBUG
224 continue;
225#endif
226 }
227 }
228
229 if (cmp == GT)
230 monoid().colonMultiply(newLead, oldLead, newSig, pairSig);
231 else {
232 MATHICGB_ASSERT(cmp == LT);
233 monoid().colonMultiply(oldLead, newLead, oldSig, pairSig);
234 }
235
236 if (Hsyz->member(pairSig)) {
237 ++mStats.syzygyModuleHits;
238#ifdef MATHICGB_DEBUG
239 // Check if actually already elim. by low/high base divisor.
240 // Only check in MATHICGB_DEBUG mode as otherwise we would have taken an
241 // early exit before getting here.
242 if ((mUseBaseDivisors || mUseHighBaseDivisors) &&
243 mKnownSyzygyTri.bit(newGen, oldGen))
244 --mStats.syzygyModuleHits;
245#endif
246 if (mUseBaseDivisors || mUseHighBaseDivisors)
247 mKnownSyzygyTri.setBit(newGen, oldGen, true);
248 continue;
249 }
250 MATHICGB_ASSERT((!mUseBaseDivisors && !mUseHighBaseDivisors)
251 || !mKnownSyzygyTri.bit(newGen, oldGen));
252
253 if (!mPostponeKoszuls) {
254 // add koszul syzygy to Hsyz.
255 MATHICGB_ASSERT(cmp == GT || cmp == LT);
256 if (cmp == GT)
257 monoid().multiply(newSig, oldLead, hsyz);
258 else
259 monoid().multiply(oldSig, newLead, hsyz);
260 if (Hsyz->insert(hsyz))
261 hsyz = R->allocMonomial();
262 if (monoid().relativelyPrime(newLead, oldLead)) {
263 ++mStats.earlyRelativelyPrimePairs;
264 continue;
265 }
266 }
267
268 if (mUseSingularCriterionEarly) {
269 MATHICGB_ASSERT(cmp == GT || cmp == LT);
270 size_t const givesSig = (cmp == GT ? newGen : oldGen);
271 if (
272 GB->ratioCompare(GB->minimalLeadInSig(pairSig), givesSig) == GT &&
273 !monoid().relativelyPrime(newLead, oldLead)
274 ) {
275 ++mStats.earlySingularCriterionPairs;
276 continue;
277 }
278 }
279
280 // construct the PreSPair
281 result.signature = pairSig;
282 pairSig = R->allocMonomial();
283 result.i = static_cast<BigIndex>(oldGen);
284 mIndexSigs.push_back(result);
285 ++mStats.queuedPairs;
286 }
287 R->freeMonomial(pairSig);
288 if (mUseBaseDivisors && ! baseDivisorMonomial.isNull())
289 R->freeMonomial(baseDivisorMonomial);
290 if (!mPostponeKoszuls)
291 R->freeMonomial(hsyz);
292}
