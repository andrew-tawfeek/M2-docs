43TEST(PrimeField, toElement) {
44 const auto max32BitUnsignedPrime = 4294967291u;
45 const PrimeField<unsigned int> pf(max32BitUnsignedPrime);
46
47 // Same number of bits (32)
48 ASSERT_EQ("0", toString(pf.toElement(0)));
49 ASSERT_EQ("1", toString(pf.toElement(1)));
50 ASSERT_EQ("4294967290", toString(pf.toElement(-1)));
51
52 ASSERT_EQ("0", toString(pf.toElement(max32BitUnsignedPrime)));
53 ASSERT_EQ("1", toString(pf.toElement(max32BitUnsignedPrime + 1u)));
54 ASSERT_EQ("4294967290", toString(pf.toElement(max32BitUnsignedPrime - 1u)));
55
56 ASSERT_EQ("4",
57 toString(pf.toElement(std::numeric_limits<unsigned int>::max())));
58 ASSERT_EQ("2147483643",
59 toString(pf.toElement(std::numeric_limits<int>::min())));
60
61 // Fewer number of bits (8)
62 ASSERT_EQ("127", toString(pf.toElement(std::numeric_limits<signed char>::max())));
63 ASSERT_EQ("4294967163",
64 toString(pf.toElement(std::numeric_limits<signed char>::min())));
65
66 ASSERT_EQ("255",
67 toString(pf.toElement(std::numeric_limits<unsigned char>::max())));
68 ASSERT_EQ("0",
69 toString(pf.toElement(std::numeric_limits<unsigned char>::min())));
70
71 // More bits (64)
72 ASSERT_EQ("24", toString(pf.toElement(std::numeric_limits<uint64>::max())));
73 ASSERT_EQ("2147483657",
74 toString(pf.toElement(std::numeric_limits<int64>::max())));
75 ASSERT_EQ("2147483633",
76 toString(pf.toElement(std::numeric_limits<int64>::min())));
77}
