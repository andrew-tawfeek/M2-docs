60 virtual Mono popSignature(Pairs& pairs) {
61 pairs.clear();
62 if (mPairQueue.empty())
63 return 0;
64 auto sig = monoid().alloc();
65 monoid().copy(*mPairQueue.topPairData(), *sig);
66 do {
67 pairs.push_back(mPairQueue.topPair());
68 mPairQueue.pop();
69 } while (
70 !mPairQueue.empty() &&
71 monoid().equal(*mPairQueue.topPairData(), *sig)
72 );
73 return sig;
74 }
75
76 virtual void pushPairs(size_t pairWith, IndexSigs& pairs) {
77 (void)pairWith;
78#ifdef MATHICGB_DEBUG
79 {
80 auto tmp = monoid().alloc();
81 for (size_t i = 0; i < pairs.size(); ++i) {
82 MATHICGB_ASSERT(pairs[i].i < columnCount());
83 mPairQueue.configuration().computePairData
84 (columnCount(), pairs[i].i, tmp.ptr());
85 MATHICGB_ASSERT(monoid().equal(*tmp, *pairs[i].signature));
86 }
87 }
88#endif
89
90 if (columnCount() >= std::numeric_limits<BigIndex>::max())
91 throw std::overflow_error
92 ("Too large basis element index in constructing S-pairs.");
93
94 // sort and insert new column
95 Comparer cmp(ring().monoid());
96 std::sort(pairs.begin(), pairs.end(), cmp);
97 typedef IndexIterator<std::vector<PreSPair>::const_iterator> Iter;
98 mPairQueue.addColumnDescending(Iter(pairs.begin()), Iter(pairs.end()));
99
100 // free signatures
101 for (auto& spair : pairs)
102 monoid().freeRaw(spair.signature);
103 pairs.clear();
104 }

123 typedef MonoPtr PairData;
124 void computePairData(size_t col, size_t row, PairData sig) {
125 MATHICGB_ASSERT(mBasis.ratioCompare(col, row) != EQ);
126 // ensure that ratio(col) > ratio(row)
127 if (mBasis.ratioCompare(col, row) == LT)
128 std::swap(col, row);
129 monoid().colonMultiply(
130 mBasis.leadMono(col),
131 mBasis.leadMono(row),
132 mBasis.signature(col),
133 *sig
134 );
135 }
