214inline int SigPolyBasis::ratioCompare(size_t a, size_t b) const {
215 if (mUseRatioRank) {
216#ifdef MATHICGB_DEBUG
217 int const value =
218 monoid().compare(sigLeadRatio(a), sigLeadRatio(b));
219#endif
220 if (mRatioRanks[a] < mRatioRanks[b]) {
221 MATHICGB_ASSERT_NO_ASSUME(value == LT);
222 return LT;
223 } else if (mRatioRanks[a] > mRatioRanks[b]) {
224 MATHICGB_ASSERT_NO_ASSUME(value == GT);
225 return GT;
226 } else {
227 MATHICGB_ASSERT_NO_ASSUME(value == EQ);
228 return EQ;
229 }
230 } else {
231 // A/a < B/b <=> A < (B/b)a
232
233 monoid().divideToNegative(signature(b), leadMono(b), mTmp);
234 monoid().multiplyInPlace(leadMono(a), mTmp);
235 const auto value = monoid().compare(signature(a), mTmp);
236 MATHICGB_ASSERT
237 (value == monoid().compare(sigLeadRatio(a), sigLeadRatio(b)));
238 return value;
239 }
240}

242inline int SigPolyBasis::StoredRatioCmp::compare(size_t be) const {
243 if (SigPolyBasis::mUseStoredRatioRank) {
244#ifdef MATHICGB_DEBUG
245 const auto value =
246 mBasis.monoid().compare(*mRatio, mBasis.sigLeadRatio(be));
247#endif
248 SigPolyBasis::Rank otherRank = mBasis.ratioRank(be);
249 if (mRatioRank < otherRank) {
250 MATHICGB_ASSERT_NO_ASSUME(value == LT);
251 return LT;
252 } else if (mRatioRank > otherRank) {
253 MATHICGB_ASSERT_NO_ASSUME(value == GT);
254 return GT;
255 } else {
256 MATHICGB_ASSERT_NO_ASSUME(value == EQ);
257 return EQ;
258 }
259 } else {
260 mBasis.monoid().multiply(*mRatio, mBasis.leadMono(be), *mTmp);
261 const auto value = mBasis.monoid().compare(*mTmp, mBasis.signature(be));
262 MATHICGB_ASSERT
263 (value == mBasis.monoid().compare(*mRatio, mBasis.sigLeadRatio(be)));
264 return value;
265 }
