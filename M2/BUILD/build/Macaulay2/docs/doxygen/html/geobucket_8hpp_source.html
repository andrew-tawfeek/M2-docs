96template <class FREEMODULETYPE, class VECTYPE>
97inline void geobucket<FREEMODULETYPE, VECTYPE>::add(VECTYPE p)
98{
99 mLead = -1;
100 int len = F->n_terms(p);
101 int i= 0;
102 while (len >= heap_size[i]) i++;
103 F->add_to(heap[i], p);
104 len = F->n_terms(heap[i]);
105 p = NULL;
106 while (len >= heap_size[i])
107 {
108 i++;
109 F->add_to(heap[i], heap[i-1]);
110 len = F->n_terms(heap[i]);
111 heap[i-1] = NULL;
112 }
113 if (i > top_of_heap)
114 top_of_heap = i;
115}

117template <class FREEMODULETYPE, class VECTYPE>
118inline const VECTYPE geobucket<FREEMODULETYPE, VECTYPE>::get_lead_term()
119{
120 int lead_so_far = -1;
121 for (int i=0; i <= top_of_heap; i++)
122 {
123 if (heap[i] == NULL) continue;
124 if (lead_so_far < 0)
125 {
126 lead_so_far = i;
127 continue;
128 }
129 int cmp = F->compare(heap[lead_so_far], heap[i]);
130 if (cmp == GT) continue;
131 if (cmp == LT)
132 {
133 lead_so_far = i;
134 continue;
135 }
136 // At this point we have equality
137 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
138 VECTYPE tmp = heap[i];
139 heap[i] = tmp->next;
140 tmp->next = NULL;
141 F->remove(tmp);
142
143 if (K->is_zero(heap[lead_so_far]->coeff))
144 {
145 // Remove, and start over
146 tmp = heap[lead_so_far];
147 heap[lead_so_far] = tmp->next;
148 tmp->next = NULL;
149 F->remove(tmp);
150 lead_so_far = -1;
151 i = -1;
152 }
153 }
154 mLead = lead_so_far;
155 if (lead_so_far < 0) return NULL;
156 VECTYPE result = heap[lead_so_far];
157 return result;
158}

160template <class FREEMODULETYPE, class VECTYPE>
161inline VECTYPE geobucket<FREEMODULETYPE, VECTYPE>::remove_lead_term()
162{
163 if (mLead < 0) get_lead_term();
164 if (mLead < 0) return NULL;
165 VECTYPE result = heap[mLead];
166 heap[mLead] = result->next;
167 result->next = NULL;
168 mLead = -1;
169 return result;
170}
