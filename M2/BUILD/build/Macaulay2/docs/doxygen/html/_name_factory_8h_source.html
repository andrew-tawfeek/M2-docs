171 (const NameFactory<AbstractProduct>& factory, const std::string& prefix) {
172 std::vector<std::string> names;
173 factory.namesWithPrefix(prefix, names);
174
175 // if exact string found, then use that one even if there are other
176 // prefix matches.
177 if (std::find(names.begin(), names.end(), prefix) != names.end()) {
178 names.clear();
179 names.push_back(prefix);
180 }
181
182 if (names.empty()) {
183 throwError<UnknownNameException>
184 ("No " + factory.abstractProductName() +
185 " has the prefix \"" + prefix + "\".");
186 }
187
188 if (names.size() >= 2) {
189 std::string errorMsg = "More than one " + factory.abstractProductName() +
190 " has prefix \"" + prefix + "\":\n ";
191 for (size_t name = 0; name < names.size(); ++name)
192 errorMsg += ' ' + names[name];
193 throwError<AmbiguousNameException>(errorMsg);
194 }
195
196 MATHIC_ASSERT(names.size() == 1);
197 return names.back();
198 }
