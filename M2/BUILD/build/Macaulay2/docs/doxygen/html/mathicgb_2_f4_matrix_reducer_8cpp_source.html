37 class DenseRow {
38 public:
39 typedef uint16 Scalar;
40 typedef uint32 ScalarProduct;
41 typedef uint64 ScalarProductSum;
42
43 static ScalarProduct product(const Scalar a, const Scalar b) {
44 return static_cast<ScalarProduct>(a) * b;
45 }
46
47 static void multiplyAdd(
48 const Scalar a,
49 const Scalar b,
50 ScalarProductSum& x
51 ) {
52 x += product(a, b);
53 }
54
55 static void add(const Scalar a, ScalarProductSum& sum) {
56 sum += a;
57 }
58
59 Scalar modulusOf(ScalarProductSum x, Scalar modulus) {
60 return static_cast<Scalar>(x % modulus);
61 }
62
63 DenseRow() {}
64 DenseRow(size_t colCount): mEntries(colCount) {}
65
67 bool takeModulus(const SparseMatrix::Scalar modulus) {
68 ScalarProductSum bitwiseOr = 0; // bitwise or of all entries after modulus
69 const auto end = mEntries.end();
70 for (auto it = mEntries.begin(); it != end; ++it) {
71 if (*it >= modulus)
72 *it = modulusOf(*it, modulus);
73 bitwiseOr |= *it;
74 }
75 return bitwiseOr != 0;
76 }
77
78 size_t colCount() const {return mEntries.size();}
79 bool empty() const {return mEntries.empty();}
80
81 void clear(size_t colCount = 0) {
82 mEntries.clear();
83 mEntries.resize(colCount);
84 }
85
86 ScalarProductSum& operator[](size_t col) {
87 MATHICGB_ASSERT(col < colCount());
88 return mEntries[col];
89 }
90
91 ScalarProductSum const& operator[](size_t col) const {
92 MATHICGB_ASSERT(col < colCount());
93 return mEntries[col];
94 }
95
96 void appendTo(SparseMatrix& matrix) {matrix.appendRow(mEntries);}
97
98 void makeUnitary(const SparseMatrix::Scalar modulus, const size_t lead) {
99 MATHICGB_ASSERT(lead < colCount());
100 MATHICGB_ASSERT(mEntries[lead] != 0);
101
102 const auto end = mEntries.end();
103 auto it = mEntries.begin() + lead;
104 const auto toInvert = static_cast<SparseMatrix::Scalar>(*it % modulus);
105 const auto multiply = modularInverse(toInvert, modulus);
106 *it = 1;
107 for (++it; it != end; ++it) {
108 const auto entry = modulusOf(*it, modulus);
109 if (entry != 0)
110 *it = modulusOf(product(entry, multiply), modulus);
111 else
112 *it = 0;
113 }
114 }

116 void addRow(const SparseMatrix& matrix, SparseMatrix::RowIndex row) {
117 MATHICGB_ASSERT(row < matrix.rowCount());
118 const auto end = matrix.rowEnd(row);
119 for (auto it = matrix.rowBegin(row); it != end; ++it) {
120 MATHICGB_ASSERT(it.index() < colCount());
121 add(it.scalar(), mEntries[it.index()]);
122 }
123 }

125 template<class Iter>
126 void addRowMultiple(
127 const SparseMatrix::Scalar multiple,
128 const Iter begin,
129 const Iter end
130 ) {
131 // I have a matrix reduction that goes from 2.8s to 2.4s on MSVC 2012 by
132 // using entries instead of mEntries, even after removing restrict and
133 // const from entries. That does not make sense to me, but it is a fact
134 // none-the-less, so don't replace entries by mEntries unless you think
135 // it's worth a 14% slowdown of matrix reduction (the whole computation,
136 // not just this method).
137 ScalarProductSum* const MATHICGB_RESTRICT entries = mEntries.data();
138
139#ifdef MATHICGB_DEBUG
140 // These asserts are separated out since otherwise they would also need
141 // to be duplicated due to the manual unrolling.
142 for (auto it = begin; it != end; ++it) {
143 MATHICGB_ASSERT(it.index() < colCount());
144 MATHICGB_ASSERT(entries + it.index() == &mEntries[it.index()]);
145 }
146#endif
147 // I have a matrix reduction that goes from 2.601s to 2.480s on MSVC 2012
148 // by unrolling this loop manually. Unrolling more than once was not a
149 // benefit. So don't undo the unrolling unless you think it's worth a 5%
150 // slowdown of matrix reduction (the whole computation, not just this
151 // method).
152
153 auto it = begin;
154 if (std::distance(begin, end) % 2 == 1) {
155 // Replacing this by a goto into the middle of the following loop
156 // (similar to Duff's device) made the code slower on MSVC 2012.
157 multiplyAdd(it.scalar(), multiple, entries[it.index()]);
158 ++it;
159 }
160 while (it != end) {
161 multiplyAdd(it.scalar(), multiple, entries[it.index()]);
162 ++it;
163 multiplyAdd(it.scalar(), multiple, entries[it.index()]);
164 ++it;
165 }
166 }

168 void rowReduceByUnitary(
169 const SparseMatrix::RowIndex pivotRow,
170 const SparseMatrix& matrix,
171 const SparseMatrix::Scalar modulus
172 ) {
173 MATHICGB_ASSERT(matrix.rowBegin(pivotRow).scalar() == 1); // unitary
174 MATHICGB_ASSERT(modulus > 1);
175
176 auto begin = matrix.rowBegin(pivotRow);
177 const auto col = begin.index();
178 const auto entry = modulusOf(mEntries[col], modulus);
179 mEntries[col] = 0;
180 if (entry == 0)
181 return;
182 ++begin; // can skip first entry as we just set it to zero.
183 addRowMultiple(
184 modularNegativeNonZero(entry, modulus),
185 begin,
186 matrix.rowEnd(pivotRow)
187 );
188 }
189
190 private:
191 std::vector<ScalarProductSum> mEntries;
192 };

194 SparseMatrix reduce(
195 const QuadMatrix& qm,
196 SparseMatrix::Scalar modulus
197 ) {
198 const SparseMatrix& toReduceLeft = qm.bottomLeft;
199 const SparseMatrix& toReduceRight = qm.bottomRight;
200 const SparseMatrix& reduceByLeft = qm.topLeft;
201 const SparseMatrix& reduceByRight = qm.topRight;
202
203 const auto leftColCount = qm.computeLeftColCount();
204 const auto rightColCount =
205 static_cast<SparseMatrix::ColIndex>(qm.computeRightColCount());
206 MATHICGB_ASSERT(leftColCount == reduceByLeft.rowCount());
207 const auto pivotCount = leftColCount;
208 const auto rowCount = toReduceLeft.rowCount();
209
210 // ** pre-calculate what rows are pivots for what columns.
211
212 // Store column indexes instead of row indices as the matrix is square
213 // anyway (so all indices fit) and we are going to store this as a column
214 // index later on.
215 std::vector<SparseMatrix::ColIndex> rowThatReducesCol(pivotCount);
216#ifdef MATHICGB_DEBUG
217 // fill in an invalid value that can be recognized by asserts to be invalid.
218 std::fill(rowThatReducesCol.begin(), rowThatReducesCol.end(), pivotCount);
219#endif
220 for (SparseMatrix::ColIndex pivot = 0; pivot < pivotCount; ++pivot) {
221 MATHICGB_ASSERT(!reduceByLeft.emptyRow(pivot));
222 SparseMatrix::ColIndex col = reduceByLeft.leadCol(pivot);
223 MATHICGB_ASSERT(rowThatReducesCol[col] == pivotCount);
224 rowThatReducesCol[col] = pivot;
225 }
226#ifdef MATHICGB_DEBUG
227 for (SparseMatrix::ColIndex col = 0; col < pivotCount; ++col) {
228 MATHICGB_ASSERT(rowThatReducesCol[col] < pivotCount);
229 }
230#endif
231
232 SparseMatrix reduced(qm.topRight.memoryQuantum());
233
234 mtbb::enumerable_thread_specific<DenseRow> denseRowPerThread([&](){
235 return DenseRow();
236 });
237
238 SparseMatrix tmp(qm.topRight.memoryQuantum());
239
240 std::vector<SparseMatrix::RowIndex> rowOrder(rowCount);
241
242 mtbb::mutex lock;
243 mtbb::parallel_for(mtbb::blocked_range<SparseMatrix::RowIndex>(0, rowCount, 2),
244 [&](const mtbb::blocked_range<SparseMatrix::RowIndex>& range)
245 {
246 auto& denseRow = denseRowPerThread.local();
247 for (auto it = range.begin(); it != range.end(); ++it) {
248 const auto row = it;
249 denseRow.clear(leftColCount);
250 denseRow.addRow(toReduceLeft, row);
251
252 MATHICGB_ASSERT(leftColCount == pivotCount);
253 for (size_t pivot = 0; pivot < pivotCount; ++pivot) {
254 if (denseRow[pivot] != 0) {
255 auto entry = denseRow[pivot];
256 entry %= modulus;
257 if (entry == 0) {
258 denseRow[pivot] = 0;
259 } else {
260 entry = modulus - entry;
261 const auto row = rowThatReducesCol[pivot];
262 MATHICGB_ASSERT(row < pivotCount);
263 MATHICGB_ASSERT(!reduceByLeft.emptyRow(row));
264 MATHICGB_ASSERT(reduceByLeft.leadCol(row) == pivot);
265 MATHICGB_ASSERT(entry < std::numeric_limits<SparseMatrix::Scalar>::max());
266 denseRow.addRowMultiple(
267 static_cast<SparseMatrix::Scalar>(entry),
268 ++reduceByLeft.rowBegin(row),
269 reduceByLeft.rowEnd(row)
270 );
271 denseRow[pivot] = entry;
272 }
273 }
274 }
275 mtbb::lock_guard lockGuard(lock);
276 // const std::lock_guard<std::mutex> lockGuard(lock);
277 for (size_t pivot = 0; pivot < pivotCount; ++pivot) {
278 MATHICGB_ASSERT(denseRow[pivot] < std::numeric_limits<SparseMatrix::Scalar>::max());
279 if (denseRow[pivot] != 0)
280 tmp.appendEntry(rowThatReducesCol[pivot], static_cast<SparseMatrix::Scalar>(denseRow[pivot]));
281 }
282 tmp.rowDone();
283 rowOrder[tmp.rowCount() - 1] = row;
284 }
285 });
286
287 mtbb::parallel_for(mtbb::blocked_range<SparseMatrix::RowIndex>(0, rowCount),
288 [&](const mtbb::blocked_range<SparseMatrix::RowIndex>& range)
289 {for (auto iter = range.begin(); iter != range.end(); ++iter)
290 {
291 const auto i = iter;
292 const auto row = rowOrder[i];
293 auto& denseRow = denseRowPerThread.local();
294
295 denseRow.clear(rightColCount);
296 denseRow.addRow(toReduceRight, row);
297 auto it = tmp.rowBegin(i);
298 const auto end = tmp.rowEnd(i);
299 for (; it != end; ++it) {
300 const auto begin = reduceByRight.rowBegin(it.index());
301 const auto end = reduceByRight.rowEnd(it.index());
302
303 denseRow.addRowMultiple(it.scalar(), begin, end);
304 }
305
306 mtbb::lock_guard lockGuard(lock);
307 // const std::lock_guard<std::mutex> lockGuard(lock);
308
309 bool zero = true;
310 for (SparseMatrix::ColIndex col = 0; col < rightColCount; ++col) {
311 const auto entry =
312 static_cast<SparseMatrix::Scalar>(denseRow[col] % modulus);
313 if (entry != 0) {
314 reduced.appendEntry(col, entry);
315 zero = false;
316 }
317 }
318 if (!zero)
319 reduced.rowDone();
320 }});
321 return reduced;
322 }

324 SparseMatrix reduceToEchelonFormSparse(
325 const SparseMatrix& toReduce,
326 const SparseMatrix::Scalar modulus
327 ) {
328 const auto colCount = toReduce.computeColCount();
329
330 const auto noRow = static_cast<SparseMatrix::RowIndex>(-1);
331
332 // pivotRowOfCol[i] is the pivot in column i or noRow
333 // if we have not identified such a pivot so far.
334 std::vector<SparseMatrix::RowIndex> pivotRowOfCol(colCount, noRow);
335
336 DenseRow rowToReduce(colCount);
337
338 // ** Reduce to row echelon form -- every row is a pivot row.
339 SparseMatrix pivots(colCount);
340 for (SparseMatrix::RowIndex row = 0; row < toReduce.rowCount(); ++row) {
341 if (toReduce.emptyRow(row))
342 continue;
343 rowToReduce.clear(colCount);
344 rowToReduce.addRow(toReduce, row);
345
346 SparseMatrix::ColIndex leadingCol = 0;
347 while (true) { // reduce row by previous pivots
348 for (; leadingCol < colCount; ++leadingCol) {
349 auto& entry = rowToReduce[leadingCol];
350 if (entry != 0) {
351 entry %= modulus;
352 if (entry != 0)
353 break;
354 }
355 }
356 if (leadingCol == colCount)
357 break; // The row has been reduced to zero.
358 const auto pivotRow = pivotRowOfCol[leadingCol];
359 if (pivotRow == noRow) { // If the row is a new pivot.
360 rowToReduce.makeUnitary(modulus, leadingCol);
361 pivotRowOfCol[leadingCol] = pivots.rowCount();
362 rowToReduce.appendTo(pivots);
363 break;
364 }
365 rowToReduce.rowReduceByUnitary(pivotRow, pivots, modulus);
366 }
367 }
368
369 // ** Reduce from row echelon form to reduced row echelon form
370
371 SparseMatrix reduced(colCount);
372 auto pivotCol = colCount;
373 // Reduce pivot rows in descending order of leading column. The reduced
374 // pivots go into reduced and we update pivotRowOfCol to refer to the
375 // row indices in reduced as we go along.
376 while (pivotCol != 0) {
377 --pivotCol;
378 const auto row = pivotRowOfCol[pivotCol];
379 if (row == noRow)
380 continue;
381 rowToReduce.clear(colCount);
382 rowToReduce.addRow(pivots, row);
383 MATHICGB_ASSERT(rowToReduce[pivotCol] == 1); // unitary
384 for (auto col = pivotCol + 1; col != colCount; ++col) {
385 auto& entry = rowToReduce[col];
386 if (entry == 0)
387 continue;
388 entry %= modulus;
389 if (entry == 0)
390 continue;
391 const auto pivotRow = pivotRowOfCol[col];
392 if (pivotRow != noRow)
393 rowToReduce.rowReduceByUnitary(pivotRow, reduced, modulus);
394 MATHICGB_ASSERT(entry < modulus);
395 }
396 pivotRowOfCol[pivotCol] = reduced.rowCount();
397 rowToReduce.appendTo(reduced);
398 }
399 return reduced;
400 }

402 SparseMatrix reduceToEchelonForm(
403 const SparseMatrix& toReduce,
404 const SparseMatrix::Scalar modulus
405 ) {
406 const auto colCount = toReduce.computeColCount();
407 const auto rowCount = toReduce.rowCount();
408
409 // convert to dense representation
410 std::vector<DenseRow> dense(rowCount);
411 mtbb::parallel_for(mtbb::blocked_range<SparseMatrix::RowIndex>(0, rowCount),
412 [&](const mtbb::blocked_range<SparseMatrix::RowIndex>& range)
413 {for (auto it = range.begin(); it != range.end(); ++it)
414 {
415 const auto row = it;
416 if (toReduce.emptyRow(row))
417 continue;
418 dense[row].clear(colCount);
419 dense[row].addRow(toReduce, row);
420 }});
421
422 // invariant: all columns in row to the left of leadCols[row] are zero.
423 std::vector<SparseMatrix::ColIndex> leadCols(rowCount);
424
425 // pivot rows get copied here before being used to reduce the matrix.
426 SparseMatrix reduced(toReduce.memoryQuantum());
427
428 // (col,row) in nextReducers, then use row as a pivot in column col
429 // for the next iteration.
430 std::vector< std::pair<SparseMatrix::ColIndex, SparseMatrix::RowIndex> > nextReducers;
431
432 // isPivotRow[row] is true if row is or has been used as a pivot.
433 std::vector<bool> isPivotRow(rowCount);
434
435 // columnHasPivot[col] is true if a pivot row for column col has
436 // been chosen.
437 std::vector<bool> columnHasPivot(colCount);
438
439 bool firstIteration = true;
440 while (firstIteration || reduced.rowCount() > 0) {
441 firstIteration = false;
442 size_t const reducerCount = reduced.rowCount();
443
444 //std::cout << "reducing " << reduced.rowCount() << " out of " << toReduce.rowCount() << std::endl;
445 mtbb::mutex lock;
446 mtbb::parallel_for(mtbb::blocked_range<SparseMatrix::RowIndex>(0, rowCount),
447 [&](const mtbb::blocked_range<SparseMatrix::RowIndex>& range)
448 {for (auto it = range.begin(); it != range.end(); ++it)
449 {
450 const auto row = it;
451 MATHICGB_ASSERT(leadCols[row] <= colCount);
452 DenseRow& denseRow = dense[row];
453 if (denseRow.empty())
454 continue;
455
456 // reduce by each row of reduced.
457 for (SparseMatrix::RowIndex reducerRow = 0; reducerRow < reducerCount; ++reducerRow) {
458 const auto col = reduced.rowBegin(reducerRow).index();
459 if (denseRow[col] == 0 || (isPivotRow[row] && col == leadCols[row]))
460 continue;
461 denseRow.rowReduceByUnitary(reducerRow, reduced, modulus);
462 }
463
464 // update leadCols[row]
465 SparseMatrix::ColIndex col;
466 MATHICGB_ASSERT(leadCols[row] <= colCount);
467 for (col = leadCols[row]; col < colCount; ++col) {
468 denseRow[col] %= modulus;
469 if (denseRow[col] != 0)
470 break;
471 }
472 leadCols[row] = col;
473 MATHICGB_ASSERT(leadCols[row] <= colCount);
474
475 // note if we have found a new pivot row
476 if (col == colCount)
477 denseRow.clear();
478 else {
479 MATHICGB_ASSERT(col < colCount);
480 bool isNewReducer = false;
481 {
482 mtbb::lock_guard lockGuard(lock);
483 // const std::lock_guard<std::mutex> lockGuard(lock);
484
485 if (!columnHasPivot[col]) {
486 columnHasPivot[col] = true;
487 isNewReducer = true;
488 nextReducers.push_back(std::make_pair(col, row));
489 }
490 }
491 if (isNewReducer)
492 denseRow.makeUnitary(modulus, col);
493 }
494 }});
495
496 reduced.clear();
497 std::sort(nextReducers.begin(), nextReducers.end());
498 for (size_t i = 0; i < nextReducers.size(); ++i) {
499 size_t const row = nextReducers[i].second;
500
501 MATHICGB_ASSERT(static_cast<bool>
502 (columnHasPivot[nextReducers[i].first]));
503 MATHICGB_ASSERT(dense[row].colCount() == colCount);
504 MATHICGB_ASSERT(dense[row][nextReducers[i].first] == 1);
505 MATHICGB_ASSERT(reduced.rowCount() == i);
506 MATHICGB_ASSERT(!isPivotRow[row]);
507
508 dense[row].appendTo(reduced); // already unitary
509 isPivotRow[row] = true;
510 }
511 nextReducers.clear();
512 }
513
514 mtbb::parallel_for(mtbb::blocked_range<size_t>(0, rowCount),
515 [&](const mtbb::blocked_range<size_t>& range)
516 {for (auto it = range.begin(); it != range.end(); ++it)
517 {
518 const size_t row = it;
519 dense[row].takeModulus(modulus);
520 }});
521
522#ifdef MATHICGB_DEBUG
523 std::vector<char> sawPivot(colCount);
524 for (SparseMatrix::RowIndex row = 0; row < rowCount; ++row) {
525 if (dense[row].empty()) {
526 MATHICGB_ASSERT(!isPivotRow[row]);
527 MATHICGB_ASSERT(leadCols[row] == colCount);
528 } else {
529 MATHICGB_ASSERT(isPivotRow[row]);
530
531 const auto leadCol = leadCols[row];
532 MATHICGB_ASSERT(leadCol < colCount);
533 MATHICGB_ASSERT(columnHasPivot[leadCol]);
534 MATHICGB_ASSERT(dense[row][leadCol] == 1);
535 MATHICGB_ASSERT(!sawPivot[leadCol]);
536 sawPivot[leadCol] = true;
537 for (size_t col = 0; col < colCount; ++col) {
538 const auto scalar = dense[row][col];
539 if (col < leadCol) {
540 MATHICGB_ASSERT(scalar == 0);
541 } else if (col == leadCol) {
542 MATHICGB_ASSERT(scalar == 1);
543 } else {
544 MATHICGB_ASSERT(scalar == 0 || !columnHasPivot[col]);
545 }
546 }
547 }
548 }
549#endif
550
551 reduced.clear();
552 for (size_t row = 0; row < rowCount; ++row)
553 if (!dense[row].empty())
554 dense[row].appendTo(reduced);
555 return reduced;
556 }
557}

559void addRowMultipleInplace(
560 std::vector< std::vector<SparseMatrix::Scalar> >& matrix,
561 const SparseMatrix::RowIndex addRow,
562 const SparseMatrix::Scalar multiple,
563 const SparseMatrix::RowIndex row,
564 const SparseMatrix::ColIndex leadingCol,
565 const SparseMatrix::ColIndex colCount,
566 const SparseMatrix::Scalar modulus
567) {
568 assert(addRow < matrix.size());
569 assert(row < matrix.size());
570 assert(row != addRow);
571 assert(leadingCol < colCount);
572 assert(matrix[row].size() == colCount);
573 assert(matrix[addRow].size() == colCount);
574 for(auto col = leadingCol; col < colCount; ++col){
575 const auto product = modularProduct
576 (multiple, matrix[addRow][col], modulus);
577 matrix[row][col] = modularSum(matrix[row][col], product, modulus);
578 }
579}

581void makeRowUnitary(
582 std::vector< std::vector<SparseMatrix::Scalar>>& matrix,
583 const SparseMatrix::RowIndex row,
584 const SparseMatrix::ColIndex colCount,
585 const SparseMatrix::ColIndex leadingCol,
586 const SparseMatrix::Scalar modulus
587) {
588 assert(row<matrix.size());
589 assert(matrix[row].size() == colCount);
590 assert(leadingCol < colCount);
591 assert(modulus> 1);
592 const auto leadingScalar = matrix[row][leadingCol];
593 assert(leadingScalar != 0);
594 auto multiply = modularInverse(leadingScalar, modulus);
595 for(SparseMatrix::ColIndex col = leadingCol; col < colCount; ++col)
596 matrix[row][col] = modularProduct(matrix[row][col], multiply, modulus);
597}

614void rowReducedEchelonMatrix(
615 std::vector< std::vector<SparseMatrix::Scalar> >& matrix,
616 const SparseMatrix::ColIndex colCount,
617 const SparseMatrix::Scalar modulus
618) {
619 assert(matrix.empty() || matrix[0].size() == colCount);
620 const SparseMatrix::RowIndex rowCount =
621 static_cast<SparseMatrix::RowIndex>(matrix.size());
622 // pivotRowOfCol[i] is the pivot in column i or rowCount
623 // if we have not identified such a pivot so far.
624 std::vector<SparseMatrix::RowIndex> pivotRowOfCol(colCount, rowCount);
625
626 // row reduce to row echelon form
627 for(SparseMatrix::RowIndex row=0; row<rowCount;++row) {
628 SparseMatrix::ColIndex leadingCol = 0;
629 while (true) { // reduce row by previous pivots
630 leadingCol = leadingColumn(matrix, row, colCount, leadingCol);
631 if(leadingCol==colCount)
632 break; // row was zero
633 const auto pivotRow = pivotRowOfCol[leadingCol];
634 if(pivotRow == rowCount) {
635 makeRowUnitary(matrix, row, colCount, leadingCol, modulus);
636 pivotRowOfCol[leadingCol] = row;
637 break; // row is now a pivot
638 }
639 const auto multiple = modularNegative(matrix[row][leadingCol], modulus);
640 addRowMultipleInplace
641 (matrix, pivotRow, multiple, row, leadingCol, colCount, modulus);
642 }
643 }
644
645 // row reduce to reduced row echelon form
646 for (SparseMatrix::RowIndex row = 0; row < rowCount;++row) {
647 const auto lead = leadingColumn(matrix, row, colCount, 0);
648 if (lead == colCount)
649 continue; // row is zero
650 for (auto col = lead + 1; col < colCount; ++col) {
651 const auto pivotRow = pivotRowOfCol[col];
652 if(pivotRow == rowCount)
653 continue; // no pivot for this column
654 const auto multiple = modularNegative(matrix[row][col], modulus);
655 addRowMultipleInplace
656 (matrix, pivotRow, multiple, row, col, colCount, modulus);
657 }
658 }
659}

661SparseMatrix reduceToEchelonFormShrawan(
662 const SparseMatrix& toReduce,
663 SparseMatrix::Scalar modulus
664) {
665 const SparseMatrix::RowIndex rowCount = toReduce.rowCount();
666 const auto colCount = toReduce.computeColCount();
667
668 // Convert input matrix to dense format
669 std::vector< std::vector<SparseMatrix::Scalar>> matrix(rowCount);
670 for (SparseMatrix::RowIndex row = 0; row < rowCount; ++row) {
671 MATHICGB_ASSERT(!toReduce.emptyRow(row));
672 matrix[row].resize(colCount);
673 const auto end = toReduce.rowEnd(row);
674 for (auto it = toReduce.rowBegin(row); it != end; ++it) {
675 MATHICGB_ASSERT(it.index() < colCount);
