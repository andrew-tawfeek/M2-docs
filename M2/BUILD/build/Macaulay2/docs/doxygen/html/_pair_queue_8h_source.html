16 // A priority queue of integer pairs (col, row) with custom
17 // comparison function via template. Pairs must be added to the data
18 // structure one column index at a time, such as will be the case
19 // for S-pairs in Buchberger's algorithm.
20 //
21 // Think of the data structure as storing a subset of the following
22 // triangle
23 //
24 // row
25 // 3| x
26 // 2| x x
27 // 1| x x x
28 // 0| x x x x
29 // ---------
30 // 0 1 2 3 4 col
31 //
32 // As you can see in this triangle, any pair (col, row) must have
33 // col > row.
34 //
35 // The custom comparison is not directly between pairs but between
36 // triples (col, row, pd) where pd is extra information that is
37 // computed as a function pairData of the pair (col, row). The
38 // purpose of this is that pd=pairData(col, row) can contain
39 // information that is necessary to compare two pairs but that is
40 // expensive to compute. Storing pd then avoids the need to
41 // recompute pd on each pair when comparing two pairs. If you do not
42 // need pd then just let pairData compute an empty struct.
43 //
44 // The data structure stores only one pd value per column to cut
45 // down on the memory cost associated to storing many pd's. The pd
46 // value of a pair is computed exactly twice. This is a speed-memory
47 // trade-off between the extremes of storing no pd's (slow due to
48 // frequent recomputations) and storing a pd for every pair (can
49 // take a lot of memory).
50 //
51 // The data structure is the S-pair queue that is described from a
52 // high level in the paper "Practical Grobner Basis Computation"
53 // that is available at http://arxiv.org/abs/1206.6940
54 template<class Configuration>
55 class PairQueue;

385 typename PairQueue<C>::Column* PairQueue<C>::Column::create
386 (Index const col,
387 Iter const rowsBegin, Iter const rowsEnd,
388 C& conf,
389 memt::Arena& arena) {
390 Column* column = arena.allocObjectNoCon<Column>();
391 column->mColumnIndex = col;
392
393#ifdef MATHIC_DEBUG
394 // check that the passed in range is weakly descending according
395 // to the custom order.
396 if (rowsBegin != rowsEnd) {
397 Iter prevIt = rowsBegin;
398 Iter it = rowsBegin;
399 for (++it; it != rowsEnd; ++it, ++prevIt) {
400 memt::Arena::PtrNoConNoDecon<PairData> prevPd(arena);
401 memt::Arena::PtrNoConNoDecon<PairData> currentPd(arena);
402
403 PairQueueNamespace::constructPairData(prevPd.get(), col, *prevIt, conf);
404 try {
405 PairQueueNamespace::constructPairData
406 (currentPd.get(), col, *it, conf);
407 } catch (...) {
408 PairQueueNamespace::
409 destructPairData(prevPd.get(), col, *prevIt, conf);
410 throw;
411 }
412
413 // check prev >= current, which is equivalent to !(prev < current)
414 MATHIC_ASSERT
415 (!conf.cmpLessThan(conf.compare(col, *prevIt, *prevPd,
416 col, *it, *currentPd)));
417 PairQueueNamespace::
418 destructPairData(currentPd.get(), col, *prevIt, conf);
419 PairQueueNamespace::destructPairData(prevPd.get(), col, *prevIt, conf);
420 }
421 }
422#endif
423
424 size_t const entryCount = std::distance(rowsBegin, rowsEnd);
425 if (column->big()) {
426 std::pair<Index*, Index*> const range =
427 arena.allocArrayNoCon<Index>(entryCount);
428 column->bigBegin = range.first;
429 column->bigEnd = range.second;
430 Index* rangeIt = range.first;
431 Iter rowsIt = rowsBegin;
432 for (; rangeIt != range.second; ++rangeIt, ++rowsIt) {
433 MATHIC_ASSERT(rowsIt != rowsEnd);
434 MATHIC_ASSERT(*rowsIt < col);
435 MATHIC_ASSERT(*rowsIt < std::numeric_limits<Index>::max());
436 *rangeIt = static_cast<Index>(*rowsIt);
437 }
438 MATHIC_ASSERT(rowsIt == rowsEnd);
439 } else {
440 std::pair<SmallIndex*, SmallIndex*> range =
441 arena.allocArrayNoCon<SmallIndex>(entryCount);
442 column->smallBegin = range.first;
443 column->smallEnd = range.second;
444 SmallIndex* rangeIt = range.first;
445 Iter rowsIt = rowsBegin;
446 for (; rangeIt != range.second; ++rangeIt, ++rowsIt) {
447 MATHIC_ASSERT(rowsIt != rowsEnd);
448 MATHIC_ASSERT(*rowsIt < col);
449 MATHIC_ASSERT(*rowsIt < std::numeric_limits<SmallIndex>::max());
450 *rangeIt = static_cast<SmallIndex>(*rowsIt);
451 }
452 }
453 MATHIC_ASSERT(column->size() == entryCount);
454 MATHIC_ASSERT(column->empty() == (entryCount == 0));
455 PairQueueNamespace::constructPairData
456 (&column->mPairData, col, *rowsBegin, conf);
457 return column;
458 }

581 template<class C>
582 void PairQueue<C>::pop() {
583 MATHIC_ASSERT(!empty());
584 Column* topColumn = mColumnQueue.top();
585 do {
586 MATHIC_ASSERT(!empty());
587 MATHIC_ASSERT(topColumn == mColumnQueue.top());
588 MATHIC_ASSERT(topColumn != 0);
589 MATHIC_ASSERT(!topColumn->empty());
590
591 if (!SupportRetirement || !retired(topColumn->columnIndex())) {
592 // Note that all mathic queues allow doing this sequence of
593 // actions: top(), change top element in-place, do decreaseTop/pop.
594 topColumn->incrementRowIndex(*this);
595 if (!topColumn->empty()) {
596 MATHIC_ASSERT(!retired(topColumn->columnIndex()));
597 MATHIC_ASSERT(!retired(topColumn->rowIndex()));
598 mColumnQueue.decreaseTop(topColumn);
599 goto doNotDestroy;
600 }
601 }
602 topColumn->destruct(mConf);
603 mColumnQueue.pop();
604 doNotDestroy:;
605 if (!SupportRetirement || mColumnQueue.empty())
606 break;
607 topColumn = mColumnQueue.top();
608 } while (retired(topColumn->columnIndex()) ||
609 retired(topColumn->rowIndex()));
610 MATHIC_ASSERT(!SupportRetirement || empty() || !retired(topPair().first));
611 MATHIC_ASSERT(!SupportRetirement || empty() || !retired(topPair().second));
612 }
