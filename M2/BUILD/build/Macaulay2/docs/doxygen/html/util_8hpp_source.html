59inline M2_ArrayString stdvector_to_M2_ArrayString(
60 const std::vector<std::string>& strs)
61{
62 // needed since M2_ArrayString len field is int
63 auto len = static_cast<int>(strs.size());
64 M2_ArrayString a = getmemarraytype(M2_ArrayString, len);
65 for (auto i = 0; i < len; i++) a->array[i] = M2_tostring(strs[i].c_str());
66 a->len = len;
67 return a;
68}
