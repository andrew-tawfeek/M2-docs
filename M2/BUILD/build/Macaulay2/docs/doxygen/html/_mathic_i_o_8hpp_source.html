252template<class M, class BF>
253auto MathicIO<M, BF>::readOrder(
254 const VarIndex varCount,
255 const bool withComponent,
256 Scanner& in
257) -> Order {
258 const bool schreyering = in.match("schreyer");
259 bool lexBaseOrder = !in.match("revlex") && in.match("lex");
260
261 auto gradingCount = in.readInteger<VarIndex>();
262 bool componentsAscendingDesired = true;
263 auto componentCompareIndex = Order::ComponentAfterBaseOrder;
264 Gradings gradings;
265 gradings.reserve(static_cast<size_t>(varCount) * gradingCount);
266 for (VarIndex grading = 0; grading < gradingCount; ++grading) {
267 const bool com = in.match("component");
268 if (com || in.match("revcomponent")) {
269 if (!withComponent)
270 in.reportError("Cannot specify component comparison for non-modules.");
271 MATHICGB_ASSERT(Monoid::HasComponent);
272 if (componentCompareIndex != Order::ComponentAfterBaseOrder)
273 in.reportError("Component comparison must be specified at most once.");
274 componentsAscendingDesired = com;
275 componentCompareIndex = grading;
276 } else {
277 for (VarIndex i = 0; i < varCount; ++i)
278 gradings.emplace_back(in.readInteger<Exponent>());
279 }
280 }
281
282 const bool moreLex = in.match("_lex");
283 if (moreLex || in.match("_revlex")) {
284 lexBaseOrder = moreLex;
285 const bool moreCom = in.match("component");
286 if (moreCom || in.match("revcomponent")) {
287 if (!withComponent)
288 in.reportError("Cannot specify component comparison for non-modules.");
289 MATHICGB_ASSERT(Monoid::HasComponent);
290 componentsAscendingDesired = moreCom;
291 }
292 }
293
294 Order order(
295 varCount,
296 std::move(gradings),
297 lexBaseOrder ?
298 Order::LexBaseOrderFromRight : Order::RevLexBaseOrderFromRight,
299 componentCompareIndex,
300 componentsAscendingDesired,
301 schreyering
302 );
303 return order;
304}

306template<class M, class BF>
307void MathicIO<M, BF>::writeOrder(
308 const Order& order,
309 const bool withComponent,
310 std::ostream& out
311) {
312 MATHICGB_ASSERT(Monoid::HasComponent || !withComponent);
313
314 const auto baseOrder =
315 order.baseOrder() == Order::LexBaseOrderFromRight ? "lex" : "revlex";
316 const auto componentOrder =
317 order.componentsAscendingDesired() ? "component" : "revcomponent";
318
319 if (order.schreyering())
320 out << "schreyer ";
321 const bool componentLast =
322 order.componentBefore() == Order::ComponentAfterBaseOrder;
323 out << baseOrder << ' ' << order.gradingCount() + !componentLast << '\n';
324 for (VarIndex grading = 0; grading < order.gradingCount(); ++grading) {
325 if (withComponent && grading == order.componentBefore())
326 out << ' ' << componentOrder << '\n';
327 for (VarIndex var = 0; var < order.varCount(); ++var) {
328 const auto index = var + grading * order.varCount();
329 out << ' ' << unchar(order.gradings()[index]);
330 }
331 out << '\n';
332 }
333 if (withComponent && order.componentBefore() == order.gradingCount())
334 out << ' ' << componentOrder << '\n';
335
336 if (
337 withComponent &&
338 !order.componentsAscendingDesired() &&
339 order.componentBefore() == Order::ComponentAfterBaseOrder
340 ) {
341 out << " _" << baseOrder << "\n " << componentOrder << '\n';
342 }
343}

384template<class M, class BF>
385Poly MathicIO<M, BF>::readPolyDoNotOrder(
386 const PolyRing& ring,
387 const bool readComponent,
388 Scanner& in
389) {
390 Poly p(ring);
391
392 // also skips whitespace
393 if (in.match('0') || in.match("+0") || in.match("-0"))
394 return p;
395 MATHICGB_ASSERT(!in.peekWhite());
396
397 auto mono = ring.monoid().alloc();
398 auto coef = ring.field().zero();
399 do {
400 if (!p.isZero() && !in.peekSign() && (!readComponent || in.peek() != '<'))
401 in.expect('+', '-');
402 readTerm(ring, readComponent, coef, mono, in);
403 p.append(coef.value(), *mono);
404 } while (!in.peekWhite() && !in.matchEOF());
405 return p;
406}

432template<class M, class BF>
433void MathicIO<M, BF>::readTerm(
434 const PolyRing& ring,
435 const bool readComponent,
436 Coefficient& coef,
437 MonoRef mono,
438 Scanner& in
439) {
440 // ** Read coefficient, if any.
441 const auto& field = ring.field();
442 const auto& monoid = ring.monoid();
443 const bool negate = !in.match('+') && in.match('-');
444 if (in.peekDigit()) {
445 coef = in.readModular(field, negate);
446
447 if (!in.peekAlpha()) {
448 // Identify a number c on its own as the monomial 1 times c.
449 monoid.setIdentity(mono);
450 if (readComponent)
451 this->readComponent(monoid, mono, in);
452 return;
453 }
454 } else if (negate)
455 coef = field.minusOne();
456 else
457 coef = field.one();
458
459 readMonomial(monoid, readComponent, mono, in);
460}

463void MathicIO<M, BF>::writeTerm(
464 const PolyRing& ring,
465 const bool writeComponent,
466 Coefficient coef,
467 ConstMonoRef mono,
468 bool writeSignEvenIfPositive,
469 std::ostream& out
470) {
471 Coefficient coef1 = coef;
472 if (ring.field().isNegative(coef)) {
473 out << "-";
474 coef1 = ring.field().negativeNonZero(coef);
475 } else if (writeSignEvenIfPositive)
476 out << '+';
477 if (!ring.field().isOne(coef1)) {
478 out << unchar(coef1.value());
479 if (ring.monoid().isIdentity(mono)) {
480 if (writeComponent)
481 this->writeComponent(ring.monoid(), mono, out);
482 return;
483 }
484 }
485 writeMonomial(ring.monoid(), writeComponent, mono, out);

488template<class M, class BF>
489void MathicIO<M, BF>::readMonomial(
490 const Monoid& monoid,
491 const bool readComponent,
492 MonoRef mono,
493 Scanner& in
494) {
495 MATHICGB_ASSERT(!readComponent || Monoid::HasComponent);
496
497 monoid.setIdentity(mono);
498 if (in.peek() == '1') {
499 const auto e = in.readInteger<Exponent>();
500 if (e != 1) {
501 std::ostringstream err;
502 err << "Expected monomial, but got " << e << " (did you mean 1?).";
503 in.reportError(err.str());
504 }
505 } else if (!readComponent || in.peek() != '<') {
506 bool sawSome = false;
507 while (true) {
508 const auto letterCount = 'z' - 'a' + 1;
509 const auto letter = in.peek();
510
511 VarIndex var;
512 if ('a' <= letter && letter <= 'z')
513 var = letter - 'a';
514 else if ('A' <= letter && letter <= 'Z')
515 var = (letter - 'A') + letterCount;
516 else if (sawSome)
517 break;
518 else {
519 std::ostringstream err;
520 err << "Expected letter while reading monomial, but got '"
521 << static_cast<char>(letter) << "'.";
522 in.reportError(err.str());
523 return;
524 }
525 in.get(); // skip past letter
526
527 MATHICGB_ASSERT(var < 2 * letterCount);
528 if (var >= monoid.varCount()) {
529 std::ostringstream err;
530 err << "Saw the variable " << static_cast<char>(letter)
531 << ", but the monoid only has "
532 << monoid.varCount() << " variables.";
533 in.reportError(err.str());
534 return;
535 }
536 if (monoid.externalExponent(mono, var) > static_cast<Exponent>(0)) {
537 std::ostringstream err;
538 err << "Variable " << static_cast<char>(letter) <<
539 " must not be written twice in one monomial.";
540 in.reportError(err.str());
541 }
542
543 if (in.peekDigit())
544 monoid.setExternalExponent(var, in.readInteger<Exponent>(), mono);
545 else
546 monoid.setExternalExponent(var, static_cast<Exponent>(1), mono);
547 sawSome = true;
548 }
549 }
550
551 if (readComponent)
552 this->readComponent(monoid, mono, in);
553}

578template<class M, class BF>
579void MathicIO<M, BF>::writeMonomial(
580 const Monoid& monoid,
581 const bool writeComponent,
582 ConstMonoRef mono,
583 std::ostream& out
584) {
585 const auto letterCount = 'z' - 'a' + 1;
586
587 bool printedSome = false;
588 for (VarIndex var = 0; var < monoid.varCount(); ++var) {
589 const auto e = monoid.exponent(mono, var);
590 if (e == 0)
591 continue;
592 char letter;
593 if (var < letterCount)
594 letter = 'a' + static_cast<char>(var);
595 else if (var < 2 * letterCount)
596 letter = 'A' + (static_cast<char>(var) - letterCount);
597 else {
598 mathic::reportError("Too few letters in alphabet to print variable.");
599 return;
600 }
601 printedSome = true;
602 out << letter;
603 if (e != 1)
604 out << unchar(e);
605 }
606 if (!printedSome)
607 out << '1';
608 if (writeComponent)
609 this->writeComponent(monoid, mono, out);
610}
