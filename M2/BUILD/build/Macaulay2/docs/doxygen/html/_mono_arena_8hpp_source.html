200 void ensureSpace() {
201 MATHICGB_ASSERT(debugAssertValid());
202
203 if (!backVector().atCapacity())
204 return;
205
206 const auto oldSize = backVector().size();
207 if (oldSize > std::numeric_limits<decltype(oldSize)>::max() / 2)
208 throw std::bad_alloc();
209 const auto newSize = 2 * oldSize;

211 mVectors.emplace_back(monoid());
212 backVector().reserve(newSize);
