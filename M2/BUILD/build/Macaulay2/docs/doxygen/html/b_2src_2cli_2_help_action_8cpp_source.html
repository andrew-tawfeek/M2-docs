18 const char* header =
19 "MathicGB offers a set of logs that can be enabled or disabled "
20 "individually.\n"
21 "\n"
22 "A log can have a streaming component and a summary "
23 "component. The streaming component is printed as the logged event "
24 "occurs while the summary component is printed at the end. For example "
25 "a log for opening files would stream a message every time a file is "
26 "opened and it would report the total number of files opened when the "
27 "program has run to the end. The summary can also include how long "
28 "something took.\n"
29 "\n"
30 "An enabled log always prints its summary if it registered any events. "
31 "The streaming component can be turned off even for an enabled log, and "
32 "this is the default for some logs.\n"
33 "\n"
34 "You specify the log configuration using the option -log X, where X is "
35 "a comma-seperated list of log specifications. Here is an example:\n"
36 "\n"
37 " A,+B,-C,D+,E-\n"
38 "\n"
39 "This enables logs A, B, D and E and disables C. Furthermore, streaming "
40 "for D is turned on while it is turned off for E. The streaming setting "
41 "for A, B and C is the default for those logs.\n"
42 "\n"
43 "A prefix of - disables the log while no prefix or a prefix of + enables "
44 "it. A suffix of - turns off streaming while a suffix of + turns it on. "
45 "If there is no suffix then the setting for streaming is unchanged. A "
46 "prefix or suffix of 0 means do nothing.\n"
47 "\n"
48 "The following is a list of all compile-time enabled logs. The prefixes "
49 "and suffixes indicate the default state of the log.\n";
50 mathic::display(header);
51 auto& logs = LogDomainSet::singleton().logDomains();
52 for (auto it = logs.begin(); it != logs.end(); ++it) {
53 const auto toSign = [](const bool b) {return b ? '+' : '-';};
54 std::cerr
55 << "\n "
56 << toSign((*it)->enabled())
57 << (*it)->name()
58 << toSign((*it)->streamEnabledPure())
59 << '\n';
60 mathic::display((*it)->description(), " ");
61 }
62
63 const char* aliasDescription =
64 "\nA log alias is a short-hand where one name stands for several log "
65 "specifications. Prefixes and Suffixes also apply to aliases. If X "
66 "expands to A+,-B,C then +X- expands to +A-,+B-,+C-. 0all- turns off "
67 "all streaming output without enabling or disabling any logs.\n"
68 "\n"
69 "The following is a list of all aliases.\n";
70 mathic::display(aliasDescription);
71 auto& aliases = LogDomainSet::singleton().aliases();
72 for (auto it = aliases.begin(); it != aliases.end(); ++it) {
73 std::cerr << "\n " << it->first << " expands to\n";
74 std::string str = it->second;
75 std::replace(str.begin(), str.end(), ',', ' ');
76 mathic::display(str, " ");
77 }
78 std::cerr <<
79 "\n none expands to nothing\n"
80 "\n"
81 " all expands to all log names\n"
82 "\n";
83}
