41std::pair<size_t, size_t> SPairs::pop() {
42 MATHICGB_LOG_TIME(SPairLate);
43
44 // Must call addPairs for new elements before popping.
45 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
46
47 while (!mQueue.empty()) {
48 const auto p = mQueue.topPair();
49 if (mBasis.retired(p.first) || mBasis.retired(p.second)) {
50 mQueue.pop();
51 continue;
52 }
53 auto lcm = bareMonoid().alloc(); // todo: just keep one around instead
54 bareMonoid().copy(orderMonoid(), *mQueue.topPairData(), *lcm);
55 mQueue.pop();
56
57 MATHICGB_ASSERT(bareMonoid().isLcm(
58 monoid(), mBasis.leadMono(p.first),
59 monoid(), mBasis.leadMono(p.second),
60 *lcm
61 ));
62 if (!advancedBuchbergerLcmCriterion(p.first, p.second, *lcm)) {
63 mEliminated.setBit(p.first, p.second, true);
64 return p;
65 }
66 }
67 return std::make_pair(static_cast<size_t>(-1), static_cast<size_t>(-1));
68}

70std::pair<size_t, size_t> SPairs::pop(exponent& w) {
71 MATHICGB_LOG_TIME(SPairLate);
72
73 // Must call addPairs for new elements before popping.
74 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
75
76 for (; !mQueue.empty(); mQueue.pop()) {
77 const auto p = mQueue.topPair();
78 if (mBasis.retired(p.first) || mBasis.retired(p.second))
79 continue;
80 auto lcm = bareMonoid().alloc(); // todo: just keep one around instead
81 bareMonoid().copy(orderMonoid(), *mQueue.topPairData(), *lcm);
82
83 MATHICGB_ASSERT(bareMonoid().isLcm(
84 monoid(), mBasis.leadMono(p.first),
85 monoid(), mBasis.leadMono(p.second),
86 *lcm
87 ));
88 if (advancedBuchbergerLcmCriterion(p.first, p.second, *lcm))
89 continue;
90 if (w == 0)
91 w = bareMonoid().degree(*lcm);
92 else if (w != bareMonoid().degree(*lcm))
93 break;
94 mQueue.pop();
95 mEliminated.setBit(p.first, p.second, true);
96 MATHICGB_IF_STREAM_LOG(SPairLcm) {
97 stream << "Scheduling S-pair with lcm ";
98 MathicIO<BareMonoid>().writeMonomial
99 (bareMonoid(), BareMonoid::HasComponent, *lcm, stream);
100 stream << '.' << std::endl;
101 };
102 return p;
103 }
104 return std::make_pair(static_cast<size_t>(-1), static_cast<size_t>(-1));
105}

108 // Records multiples of a basis element.
109 // Used in addPairs().
110 class RecordIndexes : public MonoLookup::EntryOutput {
111 public:
112 RecordIndexes(
113 size_t newGen,
114 mathic::BitTriangle& eliminated,
115 std::vector<size_t>& indexes
116 ):
117 mNewGen(newGen),
118 mEliminated(eliminated),
119 mIndexes(indexes) {}
120
121 virtual bool proceed(size_t index) {
122 if (index == mNewGen)
123 return true;
124 mIndexes.push_back(index);
125
126 // The S-pair (newGen, *it) corresponds to reducing the non-minimal
127 // basis element *it. The S-polynomial corresponds to the first
128 // step of that reduction. We tell the caller to reduce *it, so we
129 // get to assume that this S-pair can be eliminated. This is important
130 // because it sometimes allows us to eliminate an S-pair (newGen, x)
131 // when (*it, x) has already been eliminated. We want to make use of
132 // this opportunity before removing all the information about *it.
133 mEliminated.setBit(mNewGen, index, true);
134 return true;
135 }
136 private:
137 size_t const mNewGen;
138 mathic::BitTriangle& mEliminated;
139 std::vector<size_t>& mIndexes;
140 };
141}

143void SPairs::addPairsAssumeAutoReduce(
144 size_t newGen,
145 std::vector<size_t>& toRetireAndReduce
146) {
147 {
148 MATHICGB_LOG_TIME(SPairEarly);
149
150 MATHICGB_ASSERT(mQueue.columnCount() == newGen);
151 MATHICGB_ASSERT(newGen < mBasis.size());

163 mBasis.monoLookup().multiples(mBasis.leadMono(newGen), indexes);
164 }
165 addPairs(newGen);
166}

200void SPairs::addPairs(size_t newGen) {
201 MATHICGB_LOG_TIME(SPairEarly);
202
203 // Must call addPairs with newGen parameter in the sequence 0, 1, ...
204 // newGen could be implicitly picked up from mQueue.columnCount(), but
205 // doing it this way ensures that what happens is what the client thinks
206 // is happening and offers an ASSERT to inform mistaken client code.
207 MATHICGB_ASSERT(mQueue.columnCount() == newGen);
208
209 MATHICGB_ASSERT(newGen < mBasis.size());
210 MATHICGB_ASSERT(!mBasis.retired(newGen));
211
212 while (mEliminated.columnCount() < mBasis.size()) {
213 if (mUseBuchbergerLcmHitCache) {
214 MATHICGB_ASSERT(mEliminated.columnCount() == mBuchbergerLcmHitCache.size());
215 mBuchbergerLcmHitCache.push_back(0);
216 }
217 mEliminated.addColumn();
218 }
219
220 if (newGen == std::numeric_limits<Queue::Index>::max())
221 throw std::overflow_error
222 ("Too large basis element index in constructing S-pairs.");
223
224 OrderMonoid::MonoVector prePairMonos(orderMonoid());
225 typedef std::pair<OrderMonoid::ConstMonoPtr, Queue::Index> PrePair;
226 std::vector<PrePair> prePairs;
227 prePairMonos.reserve(newGen);
228 prePairs.reserve(newGen);
229
230 auto newLead = mBasis.leadMono(newGen);
231 auto lcm = mBareMonoid.alloc();
232 for (size_t oldGen = 0; oldGen < newGen; ++oldGen) {
233 if (mBasis.retired(oldGen))
234 continue;
235 auto oldLead = mBasis.leadMono(oldGen);
236 if (monoid().component(newLead) != monoid().component(oldLead)) {
237 mEliminated.setBit(newGen, oldGen, true);
238 continue;
239 }
240 if (monoid().relativelyPrime(newLead, oldLead)) {
241 ++mStats.relativelyPrimeHits;
242 mEliminated.setBit(newGen, oldGen, true);
243 continue;
244 }
245 mBareMonoid.lcm(monoid(), newLead, monoid(), oldLead, *lcm);
246 if (simpleBuchbergerLcmCriterion(newGen, oldGen, *lcm)) {
247 mEliminated.setBit(newGen, oldGen, true);
248 continue;
249 }
250
251 prePairMonos.push_back(bareMonoid(), *lcm);
252 prePairs.emplace_back
253 (prePairMonos.back().ptr(), static_cast<Queue::Index>(oldGen));
254 }
255
256 std::sort(prePairs.begin(), prePairs.end(),
257 [&](const PrePair& a, const PrePair& b)
258 {
259 return mQueue.configuration().compare
260 (b.second, newGen, b.first, a.second, newGen, a.first);
261 });
262 mQueue.addColumnDescending
263 (makeSecondIterator(prePairs.begin()), makeSecondIterator(prePairs.end()));
264}

270bool SPairs::simpleBuchbergerLcmCriterion(
271 size_t a,
272 size_t b,
273 BareMonoid::ConstMonoRef lcmAB
274) const {
275 MATHICGB_ASSERT(
276 monoid().component(mBasis.leadMono(a)) ==
277 monoid().component(mBasis.leadMono(b))
278 );
279 MATHICGB_ASSERT(a < mBasis.size());
280 MATHICGB_ASSERT(b < mBasis.size());
281 MATHICGB_ASSERT(a != b);
282 MATHICGB_ASSERT(!mBasis.retired(a));
283 MATHICGB_ASSERT(!mBasis.retired(b));
284 MATHICGB_ASSERT(bareMonoid().isLcm
285 (monoid(), mBasis.leadMono(a), monoid(), mBasis.leadMono(b), lcmAB)
286 );
287 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
288
289 class Criterion : public MonoLookup::EntryOutput {
290 public:
291 Criterion(
292 const size_t a,
293 const size_t b,
294 BareMonoid::ConstMonoRef lcmAB,
295 const SPairs& sPairs
296 ):
297 mA(a), mB(b),
298 mLcmAB(lcmAB),
299 mSPairs(sPairs),
300 mMonoid(sPairs.monoid()),
301 mBareMonoid(sPairs.bareMonoid()),
302 mBasis(sPairs.basis()),
303 mHit(static_cast<size_t>(-1)),
304 mAlmostApplies(false) {}
305
306 virtual bool proceed(size_t index) {
307 MATHICGB_ASSERT(index < mBasis.size());
308 MATHICGB_ASSERT(!applies()); // should have stopped search in this case
309 MATHICGB_ASSERT
310 (mBareMonoid.divides(mMonoid, mBasis.leadMono(index), mLcmAB));
311 if (index == mA || index == mB)
312 return true;
313 mAlmostApplies = true;
314
315 // check lcm(a,index) != lcm(a,b) <=>
316 // exists i such that max(a[i], c[i]) != max(a[i],b[i]) <=>
317 // exists i such that b[i] > a[i] && b[i] > c[i] <=>
318 // exists i such that b[i] > max(a[i], c[i]) <=>
319 // b does not divide lcm(a[i], c[i])
320 auto leadA = mBasis.leadMono(mA);
321 auto leadB = mBasis.leadMono(mB);
322 auto leadC = mBasis.leadMono(index);
323
324 if (
325 !mSPairs.eliminated(index, mA) &&
326 mMonoid.dividesLcm(leadB, leadC, leadA)
327 )
328 return true; // we had lcm(a,index) == lcm(a,b)
329
330 // check lcm(b,index) != lcm(a,b)
331 if (
332 !mSPairs.eliminated(index, mB) &&
333 mMonoid.dividesLcm(leadA, leadC, leadB)
334 )
335 return true; // we had lcm(b,index) == lcm(a,b)
336
337 mHit = index;
338 return false; // stop search
339 }
340
341 BareMonoid::ConstMonoRef lcmAB() const {return mLcmAB;}
342 bool almostApplies() const {return mAlmostApplies;}
343 bool applies() const {return mHit != static_cast<size_t>(-1);}
344 size_t hit() const {return mHit;}
345
346 private:
347 const size_t mA;
348 const size_t mB;
349 BareMonoid::ConstMonoRef mLcmAB;
350 const SPairs& mSPairs;
351 const Monoid& mMonoid;
352 const BareMonoid& mBareMonoid;
353 const PolyBasis& mBasis;
354 size_t mHit; // the divisor that made the criterion apply
355 bool mAlmostApplies; // applies ignoring lcm(a,b)=lcm(a,c) complication
356 };
357
358 bool applies = false;
359 {
360 Criterion criterion(a, b, lcmAB, *this);
361 if (mUseBuchbergerLcmHitCache) {
362 // Check cacheB first since when I tried this there was a higher hit rate
363 // for cacheB than cacheA. Might not be a persistent phenomenon, but
364 // there's no downside to trying out cacheB first so I'm going for that.
365 //
366 // I update the cache if the second check is a hit but not if the first
367 // check is a hit. In the one test I did, the worst hit rate was from
368 // updating the cache every time, the second best hit rate was from
369 // not updating the cache (from cache hits) and the best hit rate was
370 // from doing this.
371 //
372 // The idea is that when the first cache check is a hit,
373 // the second cache member might have been a hit too, and updating it
374 // might replace a high hit rate element with a low hit rate element,
375 // which would be bad. When the second cache check is a hit, we know
376 // that the first one wasn't (or we would have taken an early exit),
377 // so we have reason to suspect that the first cache element is not
378 // a high hit rate element. So it should be better to replace it.
379 // That idea seems to be right since it worked better in the one
380 // test I did.
381 size_t cacheB = mBuchbergerLcmHitCache[b];
382 if (
383 !applies &&
384 !mBasis.retired(cacheB) &&
385 mBareMonoid.dividesWithComponent
386 (monoid(), mBasis.leadMono(cacheB), criterion.lcmAB())
387 )
388 applies = !criterion.Criterion::proceed(cacheB);
389
390 size_t cacheA = mBuchbergerLcmHitCache[a];
391 if (
392 !applies &&
393 !mBasis.retired(cacheA) &&
394 mBareMonoid.dividesWithComponent
395 (monoid(), mBasis.leadMono(cacheA), criterion.lcmAB())
396 ) {
397 applies = !criterion.Criterion::proceed(cacheA);
398 if (applies)
399 mBuchbergerLcmHitCache[b] = cacheA;
400 }
401 }
402 if (applies) {
403 if (mStats.late)
404 ++mStats.buchbergerLcmCacheHitsLate;
405 else
406 ++mStats.buchbergerLcmCacheHits;
407 } else {
408 MATHICGB_ASSERT(!criterion.applies());
409 // This will be a tough nut to crack in terms of getting the types to match.
410 mBasis.monoLookup().divisors
411 (Monoid::toRef(BareMonoid::toOld(criterion.lcmAB())), criterion);
412 applies = criterion.applies();
413
414 if (mUseBuchbergerLcmHitCache && applies) {
415 MATHICGB_ASSERT(criterion.hit() < mBasis.size());
416 mBuchbergerLcmHitCache[a] = criterion.hit();
417 mBuchbergerLcmHitCache[b] = criterion.hit();
418 }
419 }
420 }
421
422 if (applies)
423 {
424 if (mStats.late)
425 ++mStats.buchbergerLcmSimpleHitsLate;
426 else
427 ++mStats.buchbergerLcmSimpleHits;
428 }
429
430 MATHICGB_ASSERT(applies == simpleBuchbergerLcmCriterionSlow(a, b));
431 return applies;
432}

434bool SPairs::simpleBuchbergerLcmCriterionSlow(size_t a, size_t b) const {
435 MATHICGB_ASSERT(
436 monoid().component(mBasis.leadMono(a)) ==
437 monoid().component(mBasis.leadMono(b))
438 );
439 MATHICGB_ASSERT(a < mBasis.size());
440 MATHICGB_ASSERT(b < mBasis.size());
441 MATHICGB_ASSERT(a != b);
442 MATHICGB_ASSERT(!mBasis.retired(a));
443 MATHICGB_ASSERT(!mBasis.retired(b));
444 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
445
446 // todo: use iterators
447 auto lcmAB = bareMonoid().alloc();
448 auto lcm = bareMonoid().alloc();
449 bareMonoid().lcm(
450 monoid(), mBasis.leadMono(a),
451 monoid(), mBasis.leadMono(b),
452 *lcmAB
453 );
454 size_t stop = mBasis.size();
455 size_t i = 0;
456 for (; i < stop; ++i) {
457 if (mBasis.retired(i))
458 continue;
459 if (!bareMonoid().dividesWithComponent(monoid(), mBasis.leadMono(i), *lcmAB))
460 continue;
461 if (i == a || i == b)
462 continue;
463 if (!eliminated(i, a)) {
464 bareMonoid().lcm(
465 monoid(), mBasis.leadMono(a),
466 monoid(), mBasis.leadMono(i),
467 *lcm
468 );
469 if (bareMonoid().equal(*lcmAB, *lcm))
470 continue;
471 }
472
473 if (!eliminated(i, b)) {
474 bareMonoid().lcm(
475 monoid(), mBasis.leadMono(b),
476 monoid(), mBasis.leadMono(i),
477 *lcm
478 );
479 if (bareMonoid().equal(*lcmAB, *lcm))
480 continue;
481 }
482 break;
483 }
484 return i != stop;
485}

487bool SPairs::advancedBuchbergerLcmCriterion(
488 size_t a,
489 size_t b,
490 BareMonoid::ConstMonoRef lcmAB
491) const {
492 MATHICGB_ASSERT(a != b);
493 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
494 MATHICGB_ASSERT(bareMonoid().isLcm
495 (monoid(), mBasis.leadMono(a), monoid(), mBasis.leadMono(b), lcmAB)
496 );
497
498 mStats.late = true;
499 if (simpleBuchbergerLcmCriterion(a, b, lcmAB)) {
500 mStats.late = false;
501 MATHICGB_ASSERT(advancedBuchbergerLcmCriterionSlow(a, b));
502 return true;
503 }
504 mStats.late = false;
505
506 // *** Determine the graph vertices
507 // graph contains pairs (index, state). index is the index of a basis
508 // element that is a node in G. state indicates which of a and b that the
509 // node in question is so far known to be connected to, if any.
510
511 typedef std::vector<std::pair<size_t, Connection> > Graph;
512 class GraphBuilder : public MonoLookup::EntryOutput {
513 public:
514 GraphBuilder(Graph& graph): mGraph(graph) {graph.clear();}
515 virtual bool proceed(size_t index) {
516 mGraph.push_back(std::make_pair(index, NotConnected));
517 return true;
518 }
519 private:
520 Graph& mGraph;
521 };
522 Graph& graph = mAdvancedBuchbergerLcmCriterionGraph;
523 graph.clear();
524 GraphBuilder builder(graph);
525 // This will be a tough nut to crack in terms of getting the types to match.
526 mBasis.monoLookup().divisors
527 (Monoid::toRef(BareMonoid::toOld(lcmAB)), builder);
528
529 if (graph.size() <= 3) {
530 // For the graph approach to be better than the simpler approach of
531 // considering triples, there has to be more than 3 nodes in the graph.
532 MATHICGB_ASSERT(!advancedBuchbergerLcmCriterionSlow(a, b));
533 return false;
534 }
535
536 // *** Set up todo with a and b
537 // todo points to elements (nodes) of graph to process.
538 std::vector<Graph::iterator> todo;
539 Graph::iterator const graphEnd = graph.end();
540 for (Graph::iterator it = graph.begin(); it != graphEnd; ++it) {
541 if (it->first == a)
542 it->second = ConnectedA;
543 else if (it->first == b)
544 it->second = ConnectedB;
545 else
546 continue;
547 todo.push_back(it);
548 }
549
550 // *** Follow edges in the graph
551 // We stop as soon as we find a node that is connected to both a and b,
552 // since then a and b are connected so that the criterion applies.
553 bool applies = false;
554 while (!applies && !todo.empty()) {
555 size_t const currentIndex = todo.back()->first;
556 Connection const currentConnect = todo.back()->second;
557 MATHICGB_ASSERT(currentConnect != NotConnected);
558 todo.pop_back();
559
560 // loop through all potential edges (currentIndex, otherIndex)
561 auto currentLead = mBasis.leadMono(currentIndex);
562 for (Graph::iterator other = graph.begin(); other != graphEnd; ++other) {
563 Connection const otherConnect = other->second;
564 if (currentConnect == otherConnect)
565 continue;
566 size_t const otherIndex = other->first;
567 MATHICGB_ASSERT(otherIndex != currentIndex);
568
569 auto const otherLead = mBasis.leadMono(otherIndex);
570 // Note that
571 // lcm(c,d) != lcmAB <=>
572 // exists i such that max(c[i], d[i]) < lcmAB[i] <=>
573 // exists i such that lcmAB[i] > c[i] && lcmAB[i] > d[i] <=>
574 // exists i such that lcmAB[i] > max(c[i], d[i]) <=>
575 // lcmAB does not divide lcm(c[i], d[i])
576 if (
577 !eliminated(currentIndex, otherIndex) &&
578 monoid().dividesLcm
579 (bareMonoid(), lcmAB, monoid(), currentLead, otherLead)
580 )
581 continue; // not an edge in G
582
583 if (otherConnect == NotConnected) {
584 other->second = currentConnect;
585 todo.push_back(other);
586 } else {
587 // At this point we have found an edge between a node connected to
588 // a and a node connected to b. So a and b are connected.
589 MATHICGB_ASSERT(currentConnect != otherConnect);
590 applies = true;
591 break;
592 }
593 }
594 }
595
596 if (applies)
597 ++mStats.buchbergerLcmAdvancedHits;
598
599 // if (graph.size() >= 10)
600 // std::cout << "[adv size=" << graph.size() << " result= " << applies << std::endl;
601
602 MATHICGB_ASSERT(applies == advancedBuchbergerLcmCriterionSlow(a, b));
603 return applies;
604}

606bool SPairs::advancedBuchbergerLcmCriterionSlow(size_t a, size_t b) const {
607 MATHICGB_ASSERT(a != b);
608 MATHICGB_ASSERT(mEliminated.columnCount() == mBasis.size());
609
610 auto lcmAB = monoid().alloc();
611 auto lcm = monoid().alloc();
612 monoid().lcm(mBasis.leadMono(a), mBasis.leadMono(b), lcmAB);
613 size_t stop = mBasis.size();
614
615 // *** Build the graph vertices
616 // graph contains pairs (index, state). index is the index of a basis
617 // that is a node in G. state indicates which of a and b that the node
618 // in question is so far known to be connected to.
619 std::vector<std::pair<size_t, Connection> > graph;
620 std::vector<size_t> todo; // indexes into graph to process.
621 for (size_t i = 0; i != stop; ++i) {
622 if (mBasis.retired(i))
623 continue;
624 if (!monoid().dividesWithComponent(mBasis.leadMono(i), *lcmAB))
625 continue;
626 Connection con = NotConnected;
627 if (i == a) {
628 con = ConnectedA;
629 todo.push_back(graph.size());
630 } else if (i == b) {
631 con = ConnectedB;
632 todo.push_back(graph.size());
633 }
634 graph.push_back(std::make_pair(i, con));
635 }
636
637 // *** Follow edges in the graph
638 // We stop as soon as we find a node that is connected to both a and b,
639 // since then a and b are connected so that the criterion applies.
640 bool applies = false;
641 while (!applies && !todo.empty()) {
642 MATHICGB_ASSERT(todo.size() <= graph.size());
643 std::pair<size_t, Connection> const node = graph[todo.back()];
644 todo.pop_back();
645 MATHICGB_ASSERT(node.second != NotConnected);
646
647 // loop through all potential edges (node.first, i)
648 ConstMonoRef leadNode = mBasis.leadMono(node.first);
649 for (size_t i = 0; i < graph.size(); ++i) {
650 if (node.second == graph[i].second)
651 continue;
652 MATHICGB_ASSERT(graph[i].first != node.first);
653 const size_t other = graph[i].first;
654
655 auto const leadOther = mBasis.leadMono(other);
656 monoid().lcm(leadNode, leadOther, lcm);
657 if (!eliminated(node.first, other) && monoid().equal(*lcm, *lcmAB))
658 continue; // not an edge in G
659
660 if (graph[i].second == NotConnected) {
661 graph[i].second = node.second;
662 todo.push_back(i);
663 } else {
664 // At this point we have found an edge between something a node to
665 // a and a node connected to b. So a and b are connected.
666 MATHICGB_ASSERT(graph[i].second != node.second);
667 applies = true;
668 break;
669 }
670 }
671 }
672
673 MATHICGB_ASSERT(applies || !simpleBuchbergerLcmCriterionSlow(a, b));
674 return applies;
675}
