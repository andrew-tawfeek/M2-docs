27std::unique_ptr<Poly> TypicalReducer::regularReduce(
28 ConstMonoRef sig,
29 ConstMonoRef multiple,
30 size_t basisElement,
31 const SigPolyBasis& basis)
32{
33 const PolyRing& ring = basis.ring();
34 const auto& monoid = ring.monoid();
35
36 monomial tproduct = ring.allocMonomial(mArena);
37 monomial u = ring.allocMonomial(mArena);
38 monoid.multiply(multiple, basis.leadMono(basisElement), tproduct);
39
40 auto reducer = basis.regularReducer(sig, tproduct);
41 if (reducer == static_cast<size_t>(-1)) {
42 mArena.freeAllAllocs();
43 return nullptr; // singular reduction: no regular top reduction possible
44 }
45
46 monoid.divide(basis.leadMono(reducer), tproduct, u);
47
48 coefficient coef;
49 ring.coefficientSet(coef, 1);
50 insertTail(const_term(coef, Monoid::toOld(multiple)), &basis.poly(basisElement));
51
52 MATHICGB_ASSERT(ring.coefficientIsOne(basis.leadCoef(reducer)));
53 ring.coefficientFromInt(coef, -1);
54 insertTail(const_term(coef, u), &basis.poly(reducer));
55 basis.basis().usedAsReducer(reducer);
56
57 auto result = make_unique<Poly>(ring);
58
59 for (const_term v; leadTerm(v);) {
60 MATHICGB_ASSERT(v.coeff != 0);
61 reducer = basis.regularReducer(sig, v.monom);
62 if (reducer == static_cast<size_t>(-1)) { // no reducer found
63 result->append(v.coeff, v.monom);
64 removeLeadTerm();
65 } else { // reduce by reducer
66 basis.basis().usedAsReducer(reducer);
67 monomial mon = ring.allocMonomial(mArena);
68 monoid.divide(basis.leadMono(reducer), v.monom, mon);
69 ring.coefficientDivide(v.coeff, basis.leadCoef(reducer), coef);
70 ring.coefficientNegateTo(coef);
71 removeLeadTerm();
72 insertTail(const_term(coef, mon), &basis.poly(reducer));
73 }
74 }
75 if (!result->isZero())
76 result->makeMonic();
77
78 reset();
79 return result;
80}

90std::unique_ptr<Poly> TypicalReducer::classicTailReduce(const Poly& poly, const PolyBasis& basis) {
91 MATHICGB_ASSERT(&poly.ring() == &basis.ring());
92 MATHICGB_ASSERT(!poly.isZero());
93 term identity;
94 identity.monom = basis.ring().allocMonomial(mArena);
95 basis.ring().monomialSetIdentity(identity.monom);
96 basis.ring().coefficientSetOne(identity.coeff);
97 insertTail(identity, &poly);
98
99 std::unique_ptr<Poly> result(new Poly(basis.ring()));
100 result->append(poly.leadCoef(), poly.leadMono());
101
102 return classicReduce(std::move(result), basis);
103}

105std::unique_ptr<Poly> TypicalReducer::classicReduceSPoly(
106 const Poly& a,
107 const Poly& b,
108 const PolyBasis& basis
109) {
110 const auto& ring = basis.ring();
111 const auto& monoid = basis.ring().monoid();
112
113 monomial lcm = ring.allocMonomial();
114 monoid.lcm(a.leadMono(), b.leadMono(), lcm);
115
116 // insert tail of multiple of a
117 monomial multiple1 = ring.allocMonomial();
118 monoid.divide(a.leadMono(), lcm, multiple1);
119 coefficient plusOne;
120 ring.coefficientSet(plusOne, 1);
121 insertTail(const_term(plusOne, multiple1), &a);
122
123 // insert tail of multiple of b
124 monomial multiple2 = ring.allocMonomial();
125 monoid.divide(b.leadMono(), lcm, multiple2);
126 coefficient minusOne = plusOne;
127 ring.coefficientNegateTo(minusOne);
128 insertTail(const_term(minusOne, multiple2), &b);
129
130 std::unique_ptr<Poly> reduced = classicReduce(basis);
131 ring.freeMonomial(lcm);
132 ring.freeMonomial(multiple1);
133 ring.freeMonomial(multiple2);
134 return reduced;
135}

165std::unique_ptr<Poly> TypicalReducer::classicReduce
166 (std::unique_ptr<Poly> result, const PolyBasis& basis) {
167 const auto& ring = basis.ring();
168 const auto& monoid = ring.monoid();
169 MATHICGB_ASSERT(&result->ring() == &ring);
170
171 if (tracingLevel > 100)
172 std::cerr << "Classic reduction begun." << std::endl;
173
174 coefficient coef;
175 for (const_term v; leadTerm(v);) {
176 if (tracingLevel > 100) {
177 std::cerr << "from reducer queue: ";
178 basis.ring().monomialDisplay(std::cerr, v.monom);
179 std::cerr << std::endl;
180 }
181
182 size_t reducer = basis.classicReducer(v.monom);
183 if (reducer == static_cast<size_t>(-1)) { // no reducer found
184 MATHICGB_ASSERT(
185 result->isZero() ||
186 basis.monoid().lessThan(v.monom, result->backMono())
187 );
188 result->append(v.coeff, v.monom);
189 removeLeadTerm();
190 } else { // reduce by reducer
191 basis.usedAsReducer(reducer);
192 monomial mon = ring.allocMonomial(mArena);
193 monoid.divide(basis.leadMono(reducer), v.monom, mon);
194 ring.coefficientDivide(v.coeff, basis.leadCoef(reducer), coef);
195 ring.coefficientNegateTo(coef);
196 removeLeadTerm();
197 insertTail(const_term(coef, mon), &basis.poly(reducer));
198
199 if (tracingLevel > 100) {
200 std::cerr << "Reducing by basis element " << reducer << ": ";
201 MathicIO<>().writePoly(basis.poly(reducer), false, std::cerr);
202 std::cerr << std::endl;
203 std::cerr << "multiplied by: " << coef << " * ";
204 basis.ring().monomialDisplay(std::cerr, mon);
205 std::cerr << std::endl;
206 }
207 }
208 }
209 if (!result->isZero())
210 result->makeMonic();
211
212 if (tracingLevel > 100)
213 std::cerr << "Classic reduction done." << std::endl;
214
215 reset();
216 return result;
217}
