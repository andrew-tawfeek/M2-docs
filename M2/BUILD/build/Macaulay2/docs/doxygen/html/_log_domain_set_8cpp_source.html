59void LogDomainSet::performLogCommandInternal(
60 char prefix,
61 std::string cmd,
62 char suffix
63) {
64 const auto isSign =
65 [](const char c) {return c == '+' || c == '-' || c == '0';};
66 MATHICGB_ASSERT(prefix == ' ' || isSign(prefix));
67 MATHICGB_ASSERT(suffix == ' ' || isSign(suffix));
68
69 if (cmd.empty())
70 return;
71
72 // This could be more efficient, but this is not supposed to be a
73 // method that is called very often.
74
75 if (isSign(cmd[0])) {
76 if (prefix == ' ')
77 prefix = cmd[0];
78 cmd.erase(cmd.begin());
79 }
80
81 if (!cmd.empty() && isSign(*cmd.rbegin())) {
82 if (suffix == ' ')
83 suffix = *cmd.rbegin();
84 cmd.erase(cmd.end() - 1);
85 }
86
87 if (cmd == "none")
88 return;
89
90 if (cmd == "all") {
91 for (auto it = mLogDomains.begin(); it != mLogDomains.end(); ++it)
92 performLogCommandInternal(prefix, (*it)->name(), suffix);
93 return;
94 }
95
96 auto aliasOf = alias(cmd.c_str());
97 if (aliasOf != 0) {
98 performLogCommandsInternal(prefix, aliasOf, suffix);
99 return;
100 }
101
102 // The default modifiers are +X0.
103 if (prefix == ' ')
104 prefix = '+';
105 if (suffix == ' ')
106 suffix = '0';
107 auto log = logDomain(cmd.c_str());
108 if (log != 0) {
109 if (prefix != '0')
110 log->setEnabled(prefix != '-');
111 if (suffix != '0')
112 log->setStreamEnabled(suffix == '+');
113 return;
114 }
115
116 mathic::reportError("Unknown log \"" + cmd + "\".\n");
117}

124void LogDomainSet::printCountReport(std::ostream& out) const {
125 mathic::ColumnPrinter pr;
126 auto& names = pr.addColumn(true);
127 auto& counts = pr.addColumn(false);
128
129 names << "Log name \n";
130 counts << " Count\n";
131 pr.repeatToEndOfLine('-');
132
133 bool somethingToReport = false;
134 const auto end = logDomains().cend();
135 for (auto it = logDomains().cbegin(); it != end; ++it) {
136 const auto& log = **it;
137 if (!log.enabled() || !log.hasCount())
138 continue;
139 somethingToReport = true;
140
141 names << log.name() << " \n";
142 counts << " " << mathic::ColumnPrinter::commafy(log.count()) << '\n';
143 }
144 if (!somethingToReport)
145 return;
146
147 out << "***** Logging count report *****\n\n" << pr << '\n';
148}

150void LogDomainSet::printTimeReport(std::ostream& out) const {
151 const auto allTime = (mtbb::tick_count::now() - mStartTime).seconds();
152
153 mathic::ColumnPrinter pr;
154 auto& names = pr.addColumn(true);
155 auto& times = pr.addColumn(false);
156 auto& ratios = pr.addColumn(false);
157 times.precision(3);
158 times << std::fixed;
159 ratios.precision(3);
160 ratios << std::fixed;
161
162 names << "Log name \n";
163 times << " Time/s (real)\n";
164 ratios << " Ratio\n";
165 pr.repeatToEndOfLine('-');
166
167 double timeSum = 0;
168 bool somethingToReport = false;
169 const auto end = logDomains().cend();
170 for (auto it = logDomains().cbegin(); it != end; ++it) {
171 const auto& log = **it;
172 if (!log.enabled() || !log.hasTime())
173 continue;
174 somethingToReport = true;
175
176 const auto logTime = log.loggedSecondsReal();
177 timeSum += logTime;
178 names << log.name() << " \n";
179 times << logTime << '\n';
180 ratios << mathic::ColumnPrinter::percentDouble(logTime, allTime) << '\n';
181 }
182 if (!somethingToReport)
183 return;
184 pr.repeatToEndOfLine('-');
185 names << "sum\n";
186 times << timeSum;
187 ratios << mathic::ColumnPrinter::percentDouble(timeSum, allTime) << '\n';
188
189 const auto oldFlags = out.flags();
190 const auto oldPrecision = out.precision();
191 out << std::fixed;
192 out.precision(3);
193 out << "***** Logging time report *****\nTime elapsed: "
194 << allTime << "s\n\n" << pr << '\n';
195
196 // todo: restore the stream state using RAII, since the above code might
197 // throw an exception.
198 out.precision(oldPrecision);
199 out.flags(oldFlags);
200}
