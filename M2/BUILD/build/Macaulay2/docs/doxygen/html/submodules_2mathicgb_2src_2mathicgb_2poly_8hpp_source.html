64 Poly polyWithTermsDescending() {
65 // *** Sort terms in descending order of monomial.
66 // It would be possible but cumbersome to implement a sort directly
67 // on mMonos. That way no allocation would need to happen, however
68 // it is not clear that that would be any faster, since swapping around
69 // monomials in-place is slow. Swapping terms is faster, since terms
70 // just refer to the monomials. This way is also easier to implement.
71 //
75 auto greaterOrEqual = [&](const NewConstTerm& a, const NewConstTerm& b) {
76 return monoid().lessThan(*b.mono, *a.mono);
77 };
78 auto terms = rangeToVector(*this);
79 std::sort(std::begin(terms), std::end(terms), greaterOrEqual);
80
81 // *** Make a new polynomial with terms in that order
82 Poly poly(ring());
83 poly.reserve(termCount());
84 poly.append(terms);
85
86 MATHICGB_ASSERT(poly.termsAreInDescendingOrder());
87 MATHICGB_ASSERT(poly.termCount() == termCount());
88 return poly;
89 }

127 void makeMonic() {
