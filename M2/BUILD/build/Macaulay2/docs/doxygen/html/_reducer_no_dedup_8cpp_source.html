73template<template<typename> class Q>
74void ReducerNoDedup<Q>::insertTail(NewConstTerm multiple, const Poly& poly) {
75 if (poly.termCount() <= 1)
76 return;
77 mLeadTermKnown = false;
78
79 auto it = poly.begin();
80 const auto end = poly.end();
81 for (++it; it != end; ++it) {
82 NewTerm t;
83 t.mono = mRing.allocMonomial();
84 mRing.monoid().multiply(*multiple.mono, it.mono(), *t.mono);
85 mRing.coefficientMult(multiple.coef, it.coef(), t.coef);
86 mQueue.push(t);
87 }
88}

90template<template<typename> class Q>
91void ReducerNoDedup<Q>::insert(ConstMonoRef multiple, const Poly& poly) {
92 if (poly.isZero())
93 return;
94 mLeadTermKnown = false;
95
96 const auto end = poly.end();
97 for (auto it = poly.begin(); it != end; ++it) {
98 NewTerm t = {it.coef(), mRing.monoid().alloc().release()};
99 mRing.monoid().multiply(multiple, it.mono(), *t.mono);
100 mQueue.push(t);
101 }
102}

104template<template<typename> class Q>
105bool ReducerNoDedup<Q>::leadTerm(NewConstTerm& result) {
106 if (!mLeadTermKnown) {
107 do {
108 if (mQueue.empty())
109 return false;
110 mLeadTerm = mQueue.top();
111 mQueue.pop();
112
113 while (true) {
114 if (mQueue.empty())
115 break;
116
117 auto entry = mQueue.top();
118 if (!mRing.monoid().equal(*entry.mono, *mLeadTerm.mono))
119 break;
120 mRing.coefficientAddTo(mLeadTerm.coef, entry.coef);
121 mRing.monoid().freeRaw(*entry.mono);
122 mQueue.pop();
123 }
124 } while (mRing.coefficientIsZero(mLeadTerm.coef));
125 mLeadTermKnown = true;
126 }
127
128 result = mLeadTerm;
129 return true;
130}
