37bool almostEqual(const RingCCC *R, int nbits, ring_elem a, ring_elem b)
38{
39 mpfr_t epsilon;
40 mpfr_init2(epsilon, 100);
41 mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN);
42
43 ring_elem f = R->subtract(a, b);
44 auto f1 = BIGCC_RE(f);
45 bool re_is_zero = (mpfr_cmpabs(f1, epsilon) < 0);
46 auto f2 = BIGCC_IM(f);
47 bool im_is_zero = (mpfr_cmpabs(f2, epsilon) < 0);
48
49 bool ret = re_is_zero && im_is_zero;
50 mpfr_clear(epsilon);
51 return ret;
52}
