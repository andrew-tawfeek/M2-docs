46void SigGBAction::performAction() {
47 mParams.perform();
48 mGBParams.perform();
49
50 // read input file
51 const std::string inputBasisFile = mParams.inputFileNameStem(0) + ".ideal";
52 std::ifstream inputFile(inputBasisFile.c_str());
53 if (inputFile.fail())
54 mic::reportError("Could not read input file \"" + inputBasisFile + '\n');
55
56
57 Scanner in(inputFile);
58 auto p = MathicIO<>().readRing(true, in);
59 auto& ring = *p.first;
60 auto& processor = p.second;
61 auto basis = MathicIO<>().readBasis(ring, false, in);
62 if (processor.schreyering())
63 processor.setSchreyerMultipliers(basis);
64
65 SignatureGB alg(
66 std::move(basis),
67 std::move(processor),
68 Reducer::reducerType(mGBParams.mReducer.value()),
69 mGBParams.mMonoLookup.value(),
70 mGBParams.mMonomialTable.value(),
71 mPostponeKoszul.value(),
72 mUseBaseDivisors.value(),
73 mGBParams.mPreferSparseReducers.value(),
74 mUseSingularCriterionEarly.value(),
75 mGBParams.mSPairQueue.value());
76 alg.setBreakAfter(mGBParams.mBreakAfter.value());
77 alg.setPrintInterval(mGBParams.mPrintInterval.value());
78 alg.computeGrobnerBasis();
79
80 // print statistics
81 alg.displayStats(std::cout);
82 alg.displayPaperStats(std::cout);
83 {
84 std::ofstream statsOut(mParams.inputFileNameStem(0) + ".stats");
85 alg.displayStats(statsOut);
86 alg.displayPaperStats(statsOut);
87 }
88
89 if (mGBParams.mOutputResult.value()) {
90 // print basis
91 {
92 std::ofstream ogb(mParams.inputFileNameStem(0) + ".gb");
93 ogb << "-- gb: ----\n";
94 alg.getGB()->display(ogb);
95 }
96
97 // print syzygy basis
98 {
99 std::ofstream syzygyOut(mParams.inputFileNameStem(0) + ".syz");
100 syzygyOut << "-- syz: ----\n";
101 alg.getSyzTable()->display(syzygyOut);
102 syzygyOut << std::endl;
103 }
104 }
105}
