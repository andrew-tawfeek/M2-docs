344 template<class C>
345 void Geobucket<C>::push(Entry entry) {
346 ++_entryCount;
347 Bucket& bucket = *_bucketBegin;
348 if (bucket.size() == bucket.capacity()) {
349 const Entry* p = &entry;
350 insert(_bucketBegin, p, p + 1, 1, Premerge<C::premerge>());
351 return;
352 }
353 if (bucket.empty()) {
354 bucket.push_back(entry);
355 _front.insert(&bucket);
356 } else if (C::minBucketBinarySearch) {
357 size_t range = bucket.size();
358 Entry* begin = bucket.begin();
359 while (true) {
360 // invariant: the position is in the range [begin, begin + range].
361 if (range == 0) {
362 if (begin == bucket.end()) {
363 bucket.push_back(entry);
364 _front.keyIncreased(&bucket);
365 } else
366 bucket.insertAtNotEnd(begin, entry);
367 break;
368 }
369 Entry* mid = begin + range / 2;
370 typename C::CompareResult cmp = _conf.compare(entry, *mid);
371 if (_conf.cmpLessThan(cmp))
372 range /= 2;
373 else if (!C::supportDeduplication || !_conf.cmpEqual(cmp)) {
374 begin = mid;
375 ++begin;
376 range -= range / 2 + 1;
377 } else {
378 --_entryCount;
379 bucket.setEntry(mid, _conf.deduplicate(*mid, entry));
380 break;
381 }
382 }
383 } else {
384 Entry* end = bucket.end();
385 for (Entry* pos = bucket.begin(); pos != end; ++pos) {
386 typename C::CompareResult cmp = _conf.compare(entry, *pos);
387 if (_conf.cmpLessThan(cmp)) {
388 bucket.insertAtNotEnd(pos, entry);
389 return;
390 }
391 if (C::supportDeduplication && _conf.cmpEqual(cmp)) {
392 bucket.setEntry(pos, _conf.deduplicate(*pos, entry));
393 --_entryCount;
394 return;
395 }
396 }
397 bucket.push_back(entry);
398 _front.keyIncreased(&bucket);
399 }
400 }

403 template<class It>
404 void Geobucket<C>::push(It begin, It end) {
405#ifdef MATHIC_DEBUG
406 if (begin != end) {
407 for (It it = begin; it + 1 != end; ++it) {
408 // deduplication assumes that [begin, end) does not contain
409 // duplicates for efficiency, so support for deduplication implies
410 // that no element can be duplicated within a pushed range.
411 if (C::supportDeduplication) {
412 MATHIC_ASSERT(_conf.cmpLessThan(_conf.compare(*(it + 1), *it)));
413 } else {
414 MATHIC_ASSERT(!_conf.cmpLessThan(_conf.compare(*it, *(it + 1))));
415 }
416 }
417 }
418#endif
419 const size_t entryCount = end - begin;
420 const size_t adjCount = entryCount * C::insertFactor;
421 _entryCount += entryCount;
422 while (adjCount > (_bucketEnd - 1)->capacity())
423 addBucket();
424 for (Bucket* bucket = _bucketBegin; ; ++bucket) {
425 MATHIC_ASSERT(bucket != _bucketEnd);
426 if (adjCount <= bucket->capacity()) {
427 std::reverse_iterator<It> rBegin(end);
428 std::reverse_iterator<It> rEnd(begin);
429 insert(bucket, rBegin, rEnd, entryCount, Premerge<C::premerge>());
430 break;
431 }
432 }
433 MATHIC_SLOW_ASSERT(isValid());
434 }

504 template<class C>
505 void Geobucket<C>::addBucket() {
506 // calculate required amount of memory
507 const size_t newBucketSize = _bucketBegin == _bucketEnd ?
508 _conf.minBucketSize : (_bucketEnd - 1)->capacity() * _conf.geoBase;
509 size_t required = 0;
510 for (Bucket* it = _bucketBegin; it != _bucketEnd; ++it) {
511 required += C::bucketStorage == GeoStoreSameSizeBuffer ?
512 newBucketSize : it->capacity();
513 if (C::bucketStorage == GeoStoreDoubleBuffer)
514 required += it->capacity(); // for other buffer
515 }
516 required += newBucketSize; // for new bucket
517 if (C::bucketStorage == GeoStoreDoubleBuffer)
518 required += newBucketSize; // for other buffer
519
520 required += newBucketSize; // for _tmp
521 if (!C::premerge)
522 required += newBucketSize; // for _tmpForNoPremerge
523
524 // allocate new memory
525 std::vector<Entry> newMem(required);
526 std::copy(_mem.begin(), _mem.end(), newMem.begin());
527
528 // move to new memory buffer
529 size_t offset = 0;
530 for (Bucket* it = _bucketBegin; it != _bucketEnd; ++it) {
531 it->setBuffer(&(newMem[offset]));
532 offset += C::bucketStorage == GeoStoreSameSizeBuffer ?
533 newBucketSize : it->capacity();
534 if (C::bucketStorage == GeoStoreDoubleBuffer) {
535 it->setOtherBuffer(&(newMem[offset]));
536 offset += it->capacity();
537 }
538 }
539 Entry* buffer = &(newMem[offset]);
540 offset += newBucketSize;
541 Entry* otherBuffer = 0;
542 if (C::bucketStorage == GeoStoreDoubleBuffer) {
543 otherBuffer = &(newMem[offset]);
544 offset += newBucketSize;
545 }
546 // redirect pointers in _front to point to new memory for _buckets
547 Bucket* oldBucketBegin = _bucketBegin;
548 _buckets.push_back(Bucket(newBucketSize, buffer, otherBuffer));
549 _bucketBegin = &_buckets.front();
550 _bucketEnd = _bucketBegin + _buckets.size();
551 _front.bucketsInvalidated(oldBucketBegin, _bucketBegin);
552 _front.reserveCapacity(_bucketEnd - _bucketBegin);
553
554 _tmp = &(newMem[offset]);
555 offset += newBucketSize;
556 if (!C::premerge) {
557 _tmpForNoPremerge = &(newMem[offset]);
558 offset += newBucketSize;
559 }
560
561 MATHIC_ASSERT(offset == required);
562 _mem.swap(newMem);
563 }

634 template<class C>
635 template<class It>
636 void Geobucket<C>::insert
637 (Bucket* p, It begin, It end, size_t incomingSize, Premerge<false>) {
638 MATHIC_ASSERT(!C::premerge);
639 MATHIC_ASSERT(incomingSize == static_cast<size_t>(end - begin));
640 MATHIC_ASSERT(0 < incomingSize);
641 MATHIC_ASSERT(begin != end);
642
643 size_t b = p - _bucketBegin;
644 Bucket& bucket = _buckets[b];
645
646 if (bucket.empty()) {
647 bucket.assign(begin, end, incomingSize);
648 _front.insert(&bucket);
649 return;
650 }
651 if (bucket.size() + incomingSize <= bucket.capacity()) {
652 mergeToNonEmpty(bucket, begin, end);
653 return;
654 }
655
656 size_t accumulatedSize = bucket.size() + incomingSize;
657 for (size_t i = b + 1; ; ++i) {
658 if (i == _buckets.size())
659 addBucket(); // can invalidate bucket reference variable
660 accumulatedSize += _buckets[i].size();
661 if (accumulatedSize <= _buckets[i].capacity())
662 break;
663 }
664
665 // deal with first step separately as the iterator can have
666 // different type (grrrrr)
667 Bucket& current = _buckets[b];
668 MATHIC_ASSERT(b + 1 < _buckets.size());
669 Bucket& next = _buckets[b + 1];
670 if (next.empty()) {
671 mergeToEmpty(next, current, begin, end);
672 return;
673 }
674 Entry* tmpForNoPremergeEnd = merge(
675 begin, end, current.begin(), current.end(), _tmpForNoPremerge);
676 current.clear();
677 _front.remove(&current);
678 ++b;
679
680 while (true) {
681 Bucket& current = _buckets[b];
682 const size_t predictedSize =
683 (tmpForNoPremergeEnd - _tmpForNoPremerge) + current.size();
684 if (predictedSize <= current.capacity()) {
685 mergeToNonEmpty(current, _tmpForNoPremerge, tmpForNoPremergeEnd);
686 return;
687 }
688 MATHIC_ASSERT(b + 1 < _buckets.size());
689 Bucket& next = _buckets[b + 1];
690 if (next.empty()) {
691 mergeToEmpty(next, current, _tmpForNoPremerge, tmpForNoPremergeEnd);
692 return;
693 }
694 tmpForNoPremergeEnd = merge(_tmpForNoPremerge, tmpForNoPremergeEnd,
695 current.begin(), current.end(), _tmp);
696 std::swap(_tmp, _tmpForNoPremerge);
697 current.clear();
698 _front.remove(&current);
699 ++b;
700 }
701 }

703 template<class C>
704 template<class It>
705 void Geobucket<C>::insert
706 (Bucket* bucket, It begin, It end, size_t size, Premerge<true>) {
707 MATHIC_ASSERT(C::premerge);
708 // ** determine premerge range (bucket, rbegin]
709 Bucket* rbegin = bucket;
710 size_t incomingSize = size;
711 while (true) {
712 if (rbegin->size() + incomingSize <= rbegin->capacity())
713 break;
714 incomingSize = rbegin->size(); // this will be incoming for next bucket
715 ++rbegin;
716 if (rbegin == _bucketEnd) {
717 Bucket* oldBucketBegin = _bucketBegin;
718 addBucket();
719 bucket = (bucket - oldBucketBegin) + _bucketBegin;
720 rbegin = (rbegin - oldBucketBegin) + _bucketBegin;
721 break;
722 }
723 }
724 // ** perform premerges
725 if (bucket != rbegin) {
726 if (!rbegin->empty()) {
727 mergeToNonEmpty(*rbegin, *(rbegin - 1));
728 --rbegin;
729 }
730 for (Bucket* rit = rbegin; rit != bucket; --rit)
731 moveToEmpty(*rit, *(rit - 1));
732 }
733 // ** perform insertion
734 if (bucket->empty()) {
735 bucket->assign(begin, end, size);
736 _front.insert(bucket);
737 } else
738 mergeToNonEmpty(*bucket, begin, end);
739 }

741 template<class C>
742 template<class It1, class It2, class ResIt>
743 ResIt Geobucket<C>::merge
744 (It1 begin1, It1 end1, It2 begin2, It2 end2, ResIt res) {
745 if (begin1 == end1) goto range1Done;
746 if (begin2 == end2) goto range2Done;
747 while (true) {
748 typename C::CompareResult cmp = _conf.compare(*begin1, *begin2);
749 if (_conf.cmpLessThan(cmp)) {
750 *res = *begin1;
751 ++res;
752 ++begin1;
753 if (begin1 == end1) goto range1Done;
754 } else if (!C::supportDeduplication || !_conf.cmpEqual(cmp)) {
755 *res = *begin2;
756 ++res;
757 ++begin2;
758 if (begin2 == end2) goto range2Done;
759 } else {
760 *res = _conf.deduplicate(*begin1, *begin2);
761 ++res;
762 ++begin1;
763 ++begin2;
764 --_entryCount;
765 if (begin1 == end1) goto range1Done;
766 if (begin2 == end2) goto range2Done;
767 }
768 }
769 range1Done: return std::copy(begin2, end2, res);
770 range2Done: return std::copy(begin1, end1, res);
771 }

793 template<class C>
794 bool Geobucket<C>::isValid() const {
795#ifndef MATHIC_DEBUG
796 return true;
797#else
798 MATHIC_ASSERT(_conf.geoBase >= 2);
799 MATHIC_ASSERT(!_buckets.empty());
800 MATHIC_ASSERT(_bucketBegin == &_buckets.front());
801 MATHIC_ASSERT(_bucketEnd == _bucketBegin + _buckets.size());
802 MATHIC_ASSERT(_buckets.front().capacity() >= 1);
803
804 size_t entryCount = 0;
805 for (size_t b = 0; b < _buckets.size(); ++b) {
806 const Bucket& bucket = _buckets[b];
807 entryCount += bucket.size();
808 if (bucket.empty())
809 continue;
810 for (const Entry* it = bucket.begin(); it != bucket.end() - 1; ++it) {
811 MATHIC_ASSERT(!_conf.cmpLessThan(_conf.compare(*(it + 1), *it)));
