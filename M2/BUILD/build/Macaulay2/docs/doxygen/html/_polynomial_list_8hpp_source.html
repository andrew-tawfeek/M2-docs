294template<typename S>
295void toStream(const PolynomialList& Fs, S &str)
296{
297 str.idealBegin(Fs.size());
298 for (auto i=0; i<Fs.size(); ++i)
299 {
300 auto& F = Fs[i];
301 str.appendPolynomialBegin(F.mMonomials.size());
302 for (auto i=0; i<F.mMonomials.size(); ++i)
303 {
304 // get monomial
305 // write it out here using appendTermBegin, appendExponent, appendTermDone.
306 if (F.mComponents.empty())
307 str.appendTermBegin(0);
308 else
309 str.appendTermBegin(F.mComponents[i]);
310 MonomialView monom = Fs.monomialHashTable().monomialAt(F.mMonomials[i]);
311 for (auto ve = monom.begin(); ve != monom.end(); ++ve)
312 {
313 str.appendExponent(ve.var(), ve.power());
314 }
315 long val = Fs.vectorArithmetic().to_modp_long(F.mCoefficients, i);
316 str.appendTermDone(val);
317 }
318 str.appendPolynomialDone();
319 }
320 str.idealDone();
321}
