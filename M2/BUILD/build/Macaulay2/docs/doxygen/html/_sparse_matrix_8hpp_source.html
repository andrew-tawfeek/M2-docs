186 void rowDone() {
187 MATHICGB_ASSERT(mBlock.mColIndices.size() == mBlock.mScalars.size());
188 Row row;
189 row.mIndicesEnd = mBlock.mColIndices.end();
190 row.mScalarsEnd = mBlock.mScalars.end();
191 if (mBlock.mHasNoRows) {
192 row.mIndicesBegin = mBlock.mColIndices.begin();
193 row.mScalarsBegin = mBlock.mScalars.begin();
194 mBlock.mHasNoRows = false;
195 } else {
196 row.mIndicesBegin = mRows.back().mIndicesEnd;
197 row.mScalarsBegin = mRows.back().mScalarsEnd;

205 inline void appendEntry(ColIndex colIndex, Scalar scalar) {
206 MATHICGB_ASSERT(mBlock.mColIndices.size() == mBlock.mScalars.size());

208 MATHICGB_ASSERT(mBlock.mScalars.atCapacity() ==
209 mBlock.mColIndices.atCapacity());

211 growEntryCapacity();
212 MATHICGB_ASSERT(!mBlock.mScalars.atCapacity());
213 MATHICGB_ASSERT(!mBlock.mColIndices.atCapacity());
214
215 mBlock.mColIndices.rawPushBack(colIndex);
216 mBlock.mScalars.rawPushBack(scalar);

218 MATHICGB_ASSERT(mBlock.mColIndices.size() == mBlock.mScalars.size());

443void SparseMatrix::appendRow(
444 std::vector<T> const& v,
445 const ColIndex leadCol
446) {
447#ifdef MATHICGB_DEBUG
448 for (auto col = leadCol; col < leadCol; ++col) {
449 MATHICGB_ASSERT(v[col] == 0);
450 }
451#endif
452
453 const auto count = static_cast<ColIndex>(v.size());
454 for (ColIndex col = leadCol; col < count; ++col) {
455 MATHICGB_ASSERT(v[col] < std::numeric_limits<Scalar>::max());
456 if (v[col] != 0)
457 appendEntry(col, static_cast<Scalar>(v[col]));
458 }
459 rowDone();
460}
