46// This file contains classes for keeping sets of monomials.
47// There are two types of monomials:
48// a. fixed size (and that size is passed in as a parameter)
49// b. variable size, in which case the first int is the total length (so the range is [m, m + *m).
50//
51// Here are some assumptions about these monomials:
52// 1. each monomial is a contiguous sequence of int's.
53// 2. each monomial is hashed using all entries of the monomial. Equality is equality of all elements.
54// (and size, if variable length).

78class MonomialMemorySpace
79{
80public:
81 MonomialMemorySpace() : mCount(0) {}
82
83 // returns a range [begin, end), allocated in the block, such that end-begin == size.
84 std::pair<int*,int*> alloc(int size) { mCount++; return mArena.allocArrayNoCon<int>(size); }
85
86 int* allocate(int size) { auto result = alloc(size); return result.first; }
87
88 // if m is the begin part of the last range allocated here, then this pops that.
89 // Afterwards, one can pop the next, etc. If they are popped out of order though,
90 // undefined behavior results.
91 void popLastAlloc(int* m) { mCount--; mArena.freeTop(m); }
92
93 // If m is a pointer into the last monomial (i.e. >= a.first <= a.second), this resets the monomial
94 // to be [a.first, m). The public interface of memt::Arena should perhaps allow this use, but it does not.
95 // However, the code works fine, unless debugging is turned on. I might need to modify that code, and
96 // document that this can be done.
97 // Reasoning for wanting/using this: in the case when monomials are not fixed size, one can allocate
98 // the maximum size it can be, compute it in place, then shrink the monomial.
99 // UNSAFE. Use of this function requires taking care.
100 void shrinkLastAlloc(int* m) { mArena.freeTop(m); }
101
102 size_t size() const { return mCount; }
103
104 void freeAllAllocs() { mArena.freeAllAllocs(); }
105
106 void freeAllAllocsAndBackingMemory() { mArena.freeAllAllocsAndBackingMemory(); }
107private:
108 size_t mCount;
109 memt::Arena mArena;
110};

282 // There are two ways to insert monomials.
283 // 1. the monomial was just allocated on monomialMemorySpace().
284 // 2. the momomial exists outside of this memory block.
285 // In the latter case, the monomial will be copied to the block.
286 //
287 // Another way is to insert a monomial via a transformer function. TODO.
288 MonomialMemorySpace& monomialMemorySpace() { return mMonomialMemory; }
289
290 int elementSize() const { return mSet.elementSize(); }
291
292 // Number of (different) monomials in this set
293 std::size_t size() const { return mMonomialMemory.size(); }
294
295 // insert: two versions: depending on whether we need to copy the pointer.
296 // decide if monom is in the set: if so, (equiv monom, false) is returned.
297 // If it is not in the set, it is copied into the memory block, and (equiv monom, true) is returned.
298 // Note: the only pointers this function returns (as the first part of the pair) are pointers into
299 // the memory block.
300 std::pair<const int*, bool> findOrInsert(const int* monom)
301 {
302 std::pair<int*, int*> mon { mMonomialMemory.alloc(elementSize()) };
303 std::copy(monom, monom + elementSize(), mon.first);
304 assert(mon.second - mon.first >= elementSize());
305 return findOrInsertTopInternedMonomial(mon.first);
306 }
307
308 // assumption: 'monom' is currently the top (last) monomial on mMonomialMemory.
309 // If the monomial is found in the set, then monom is popped off the memory block.
310 std::pair<const int*, bool> findOrInsertTopInternedMonomial(int*& monom)
311 {
312 // if the monomial is found in the set, then monom is popped
313 auto result = mSet.findOrInsert(monom);
314 if (not result.second) // i.e. monomial was not needed.
315 mMonomialMemory.popLastAlloc(monom);
316 return result;
317 }
318
319 // find: this returns either: (nullptr,false), if not found, or the (canonicalized pointer, true).
320 const int* find(const int* monom) const
