46// This file contains classes for keeping sets of monomials.
47// There are two types of monomials:
48// a. fixed size (and that size is passed in as a parameter)
49// b. variable size, in which case the first int is the total length (so the range is [m, m + *m).
50//
51// Here are some assumptions about these monomials:
52// 1. each monomial is a contiguous sequence of int's.
53// 2. each monomial is hashed using all entries of the monomial. Equality is equality of all elements.
54// (and size, if variable length).

67class MonomialMemorySpace
68{
69public:
70 MonomialMemorySpace() : mCount(0) {}
71
72 // returns a range [begin, end), allocated in the block, such that end-begin == size.
73 std::pair<int*,int*> alloc(int size) { mCount++; return mArena.allocArrayNoCon<int>(size); }
74
75 int* allocate(int size) { auto result = alloc(size); return result.first; }
76
77 // if m is the begin part of the last range allocated here, then this pops that.
78 // Afterwards, one can pop the next, etc. If they are popped out of order though,
79 // undefined behavior results.
80 void popLastAlloc(int* m) { mCount--; mArena.freeTop(m); }
81
82 // If m is a pointer into the last monomial (i.e. >= a.first <= a.second), this resets the monomial
83 // to be [a.first, m). The public interface of memt::Arena should perhaps allow this use, but it does not.
84 // However, the code works fine, unless debugging is turned on. I might need to modify that code, and
85 // document that this can be done.
86 // Reasoning for wanting/using this: in the case when monomials are not fixed size, one can allocate
87 // the maximum size it can be, compute it in place, then shrink the monomial.
88 // UNSAFE. Use of this function requires taking care.
89 void shrinkLastAlloc(int* m) { mArena.freeTop(m); }
90
91 size_t size() const { return mCount; }
92
93 void freeAllAllocs() { mArena.freeAllAllocs(); }
94
95 void freeAllAllocsAndBackingMemory() { mArena.freeAllAllocsAndBackingMemory(); }
96private:
97 size_t mCount;
98 memt::Arena mArena;
99};

234 // There are two ways to insert monomials.
235 // 1. the monomial was just allocated on monomialMemorySpace().
236 // 2. the momomial exists outside of this memory block.
237 // In the latter case, the monomial will be copied to the block.
238 //
239 // Another way is to insert a monomial via a transformer function. TODO.
240 MonomialMemorySpace& monomialMemorySpace() { return mMonomialMemory; }
241
242 int elementSize() const { return mSet.elementSize(); }
243
244 // Number of (different) monomials in this set
245 std::size_t size() const { return mMonomialMemory.size(); }
246
247 // insert: two versions: depending on whether we need to copy the pointer.
248 // decide if monom is in the set: if so, (equiv monom, false) is returned.
249 // If it is not in the set, it is copied into the memory block, and (equiv monom, true) is returned.
250 // Note: the only pointers this function returns (as the first part of the pair) are pointers into
251 // the memory block.
252 std::pair<const int*, bool> findOrInsert(const int* monom)
253 {
254 std::pair<int*, int*> mon { mMonomialMemory.alloc(elementSize()) };
255 std::copy(monom, monom + elementSize(), mon.first);
256 assert(mon.second - mon.first >= elementSize());
257 return findOrInsertTopInternedMonomial(mon.first);
258 }
259
260 // assumption: 'monom' is currently the top (last) monomial on mMonomialMemory.
261 // If the monomial is found in the set, then monom is popped off the memory block.
262 std::pair<const int*, bool> findOrInsertTopInternedMonomial(int*& monom)
263 {
264 // if the monomial is found in the set, then monom is popped
265 auto result = mSet.findOrInsert(monom);
266 if (not result.second) // i.e. monomial was not needed.
267 mMonomialMemory.popLastAlloc(monom);
268 return result;
269 }
270
271 // find: this returns either: (nullptr,false), if not found, or the (canonicalized pointer, true).
272 const int* find(const int* monom) const
