6// This file contains classes for keeping sets of monomials.
7// There are two types of monomials:
8// a. fixed size (and that size is passed in as a parameter)
9// b. variable size, in which case the first int is the total length (so the range is [m, m + *m).
10//
11// Here are some assumptions about these monomials:
12// 1. each monomial is a contiguous sequence of int's.
13// 2. each monomial is hashed using all entries of the monomial. Equality is equality of all elements.
14// (and size, if variable length).

27class MonomialMemorySpace
28{
29public:
30 MonomialMemorySpace() : mCount(0) {}
31
32 // returns a range [begin, end), allocated in the block, such that end-begin == size.
33 std::pair<int*,int*> alloc(int size) { mCount++; return mArena.allocArrayNoCon<int>(size); }
34
35 int* allocate(int size) { auto result = alloc(size); return result.first; }
36
37 // if m is the begin part of the last range allocated here, then this pops that.
38 // Afterwards, one can pop the next, etc. If they are popped out of order though,
39 // undefined behavior results.
40 void popLastAlloc(int* m) { mCount--; mArena.freeTop(m); }
41
42 // If m is a pointer into the last monomial (i.e. >= a.first <= a.second), this resets the monomial
43 // to be [a.first, m). The public interface of memt::Arena should perhaps allow this use, but it does not.
44 // However, the code works fine, unless debugging is turned on. I might need to modify that code, and
45 // document that this can be done.
46 // Reasoning for wanting/using this: in the case when monomials are not fixed size, one can allocate
47 // the maximum size it can be, compute it in place, then shrink the monomial.
48 // UNSAFE. Use of this function requires taking care.
49 void shrinkLastAlloc(int* m) { mArena.freeTop(m); }
50
51 size_t size() const { return mCount; }
52
53 void freeAllAllocs() { mArena.freeAllAllocs(); }
54
55 void freeAllAllocsAndBackingMemory() { mArena.freeAllAllocsAndBackingMemory(); }
56private:
57 size_t mCount;
58 memt::Arena mArena;
59};

194 // There are two ways to insert monomials.
195 // 1. the monomial was just allocated on monomialMemorySpace().
196 // 2. the momomial exists outside of this memory block.
197 // In the latter case, the monomial will be copied to the block.
198 //
199 // Another way is to insert a monomial via a transformer function. TODO.
200 MonomialMemorySpace& monomialMemorySpace() { return mMonomialMemory; }
201
202 int elementSize() const { return mSet.elementSize(); }
203
204 // Number of (different) monomials in this set
205 std::size_t size() const { return mMonomialMemory.size(); }
206
207 // insert: two versions: depending on whether we need to copy the pointer.
208 // decide if monom is in the set: if so, (equiv monom, false) is returned.
209 // If it is not in the set, it is copied into the memory block, and (equiv monom, true) is returned.
210 // Note: the only pointers this function returns (as the first part of the pair) are pointers into
211 // the memory block.
212 std::pair<const int*, bool> findOrInsert(const int* monom)
213 {
214 std::pair<int*, int*> mon { mMonomialMemory.alloc(elementSize()) };
215 std::copy(monom, monom + elementSize(), mon.first);
216 assert(mon.second - mon.first >= elementSize());
217 return findOrInsertTopInternedMonomial(mon.first);
218 }
219
220 // assumption: 'monom' is currently the top (last) monomial on mMonomialMemory.
221 // If the monomial is found in the set, then monom is popped off the memory block.
222 std::pair<const int*, bool> findOrInsertTopInternedMonomial(int*& monom)
223 {
224 // if the monomial is found in the set, then monom is popped
225 auto result = mSet.findOrInsert(monom);
226 if (not result.second) // i.e. monomial was not needed.
227 mMonomialMemory.popLastAlloc(monom);
228 return result;
229 }
230
231 // find: this returns either: (nullptr,false), if not found, or the (canonicalized pointer, true).
232 const int* find(const int* monom) const

234 return mSet.find(monom);
