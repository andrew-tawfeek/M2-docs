29// This file contains classes for keeping sets of monomials.
30// There are two types of monomials:
31// a. fixed size (and that size is passed in as a parameter)
32// b. variable size, in which case the first int is the total length (so the range is [m, m + *m).
33//
34// Here are some assumptions about these monomials:
35// 1. each monomial is a contiguous sequence of int's.
36// 2. each monomial is hashed using all entries of the monomial. Equality is equality of all elements.
37// (and size, if variable length).

50class MonomialMemorySpace
51{
52public:
53 MonomialMemorySpace() : mCount(0) {}
54
55 // returns a range [begin, end), allocated in the block, such that end-begin == size.
56 std::pair<int*,int*> alloc(int size) { mCount++; return mArena.allocArrayNoCon<int>(size); }
57
58 int* allocate(int size) { auto result = alloc(size); return result.first; }
59
60 // if m is the begin part of the last range allocated here, then this pops that.
61 // Afterwards, one can pop the next, etc. If they are popped out of order though,
62 // undefined behavior results.
63 void popLastAlloc(int* m) { mCount--; mArena.freeTop(m); }
64
65 // If m is a pointer into the last monomial (i.e. >= a.first <= a.second), this resets the monomial
66 // to be [a.first, m). The public interface of memt::Arena should perhaps allow this use, but it does not.
67 // However, the code works fine, unless debugging is turned on. I might need to modify that code, and
68 // document that this can be done.
69 // Reasoning for wanting/using this: in the case when monomials are not fixed size, one can allocate
70 // the maximum size it can be, compute it in place, then shrink the monomial.
71 // UNSAFE. Use of this function requires taking care.
72 void shrinkLastAlloc(int* m) { mArena.freeTop(m); }
73
74 size_t size() const { return mCount; }
75
76 void freeAllAllocs() { mArena.freeAllAllocs(); }
77
78 void freeAllAllocsAndBackingMemory() { mArena.freeAllAllocsAndBackingMemory(); }
79private:
80 size_t mCount;
81 memt::Arena mArena;
82};

217 // There are two ways to insert monomials.
218 // 1. the monomial was just allocated on monomialMemorySpace().
219 // 2. the momomial exists outside of this memory block.
220 // In the latter case, the monomial will be copied to the block.
221 //
222 // Another way is to insert a monomial via a transformer function. TODO.
223 MonomialMemorySpace& monomialMemorySpace() { return mMonomialMemory; }
224
225 int elementSize() const { return mSet.elementSize(); }
226
227 // Number of (different) monomials in this set
228 std::size_t size() const { return mMonomialMemory.size(); }
229
230 // insert: two versions: depending on whether we need to copy the pointer.
231 // decide if monom is in the set: if so, (equiv monom, false) is returned.
232 // If it is not in the set, it is copied into the memory block, and (equiv monom, true) is returned.
233 // Note: the only pointers this function returns (as the first part of the pair) are pointers into
234 // the memory block.
235 std::pair<const int*, bool> findOrInsert(const int* monom)
236 {
237 std::pair<int*, int*> mon { mMonomialMemory.alloc(elementSize()) };
238 std::copy(monom, monom + elementSize(), mon.first);
239 assert(mon.second - mon.first >= elementSize());
240 return findOrInsertTopInternedMonomial(mon.first);
241 }
242
243 // assumption: 'monom' is currently the top (last) monomial on mMonomialMemory.
244 // If the monomial is found in the set, then monom is popped off the memory block.
245 std::pair<const int*, bool> findOrInsertTopInternedMonomial(int*& monom)
246 {
247 // if the monomial is found in the set, then monom is popped
248 auto result = mSet.findOrInsert(monom);
249 if (not result.second) // i.e. monomial was not needed.
250 mMonomialMemory.popLastAlloc(monom);
251 return result;
252 }
253
254 // find: this returns either: (nullptr,false), if not found, or the (canonicalized pointer, true).
255 const int* find(const int* monom) const

257 return mSet.find(monom);
