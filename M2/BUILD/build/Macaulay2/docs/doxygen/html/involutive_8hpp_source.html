140 template <typename MonomType>
141 const Matrix* BooleanInvolutiveBasis<MonomType>::ToMatrix() const
142 {
143 MatrixConstructor matrixConstructor(PRing->make_FreeModule(1), 0);
144 const Monoid* monoid = PRing->getMonoid();
145 const ring_elem coefficientUnit = PRing->getCoefficients()->one();
146 monomial tmpRingMonomial = monoid->make_one();
147
148 for (typename std::list<Polynom<MonomType>*>::const_iterator currentPolynom = GBasis.begin();
149 currentPolynom != GBasis.end();
150 ++currentPolynom)
151 {
152 if (!*currentPolynom)
153 {
154 continue;
155 }
156
157 ring_elem currentRingPolynomial;
158
159 for (const MonomType* currentMonom = &(**currentPolynom).Lm();
160 currentMonom;
161 currentMonom = currentMonom->Next)
162 {
163 exponents_t currentExponent = newarray_atomic_clear(int, MonomType::GetDimIndepend());
164 typename std::set<typename MonomType::Integer> variablesSet = currentMonom->GetVariablesSet();
165
166 for (typename std::set<typename MonomType::Integer>::const_iterator currentVariable = variablesSet.begin();
167 currentVariable != variablesSet.end();
168 ++currentVariable)
169 {
170 currentExponent[*currentVariable] = 1;
171 }
172
173 monoid->from_expvector(currentExponent, tmpRingMonomial);
174 freemem(currentExponent);
175
176 ring_elem tmpRingPolynomial = PRing->make_flat_term(coefficientUnit, tmpRingMonomial);
177 PRing->add_to(currentRingPolynomial, tmpRingPolynomial);
178 }
179
180 matrixConstructor.append(PRing->make_vec(0, currentRingPolynomial));
181 }
182
183 return matrixConstructor.to_matrix();
184 }

186 template <typename MonomType>
187 void BooleanInvolutiveBasis<MonomType>::FillInitialSet(const Matrix* matrix, std::list<Polynom<MonomType>*>& initialSet) const
188 {
189 const Monoid* monoid = PRing->getMonoid();
190 typename MonomType::Integer independ = MonomType::GetDimIndepend();
191
192 //construct Polynom for every column in matrix
193 for (int column = 0; column < matrix->n_cols(); ++column)
194 {
195 vec polynomVector = matrix->elem(column);
196 if (!polynomVector)
197 {
198 continue;
199 }
200
201 Polynom<MonomType>* currentPolynom = new Polynom<MonomType>();
202
203 for (Nterm& currentTerm : polynomVector->coeff)
204 {
205 exponents_t monomVector = newarray_atomic(int, independ);
206 monoid->to_expvector(currentTerm.monom, monomVector);
207
208 //construct Monom for every term
209 MonomType* currentMonom = new MonomType();
210 if (!currentMonom)
211 {
212 freemem(monomVector);
213 throw std::string("BIBasis::BooleanInvolutiveBasis::FillInitialSet(): got NULL instead of new monom.");
214 }
215
216 for (typename MonomType::Integer currentVariable = 0; currentVariable < independ; ++currentVariable)
217 {
218 if (monomVector[currentVariable])
219 {
220 *currentMonom *= currentVariable;
221 }
222 }
223
224 *currentPolynom += *currentMonom;
225 freemem(monomVector);
226 delete currentMonom;
227 }
228
229 initialSet.push_back(currentPolynom);
230 }

233 template <typename MonomType>
234 Polynom<MonomType>* BooleanInvolutiveBasis<MonomType>::NormalForm(const Triple<MonomType>* triple) const
235 {
236 /* As far as currentTriple can't be 0 (checked in QSET and TSET),
237 * no need to check for NULL pointer.
238 */
239
240 const Triple<MonomType>* involutiveDivisor = 0;
241 Polynom<MonomType>* originalForm = 0;
242 Polynom<MonomType>* normalForm = new Polynom<MonomType>();
243
244 if (triple->GetVariable() == -1)
245 {
246 originalForm = new Polynom<MonomType>(*triple->GetPolynom());
247 }
248 else
249 {
250 originalForm = new Polynom<MonomType>(*triple->GetWeakAncestor()->GetPolynom());
251 (*originalForm) *= triple->GetVariable();
252 }
253
254 while (!originalForm->IsZero())
255 {
256 involutiveDivisor = IntermediateBasis.Find(originalForm->Lm());
257 while (involutiveDivisor)
258 {
259 originalForm->HeadReduction(*involutiveDivisor->GetPolynom());
260 if (!originalForm->IsZero())
261 {
262 involutiveDivisor = IntermediateBasis.Find(originalForm->Lm());
263 }
264 else
265 {
266 involutiveDivisor = 0;
267 }
268 }
269
270 if (!originalForm->IsZero())
271 {
272 (*normalForm) += originalForm->Lm();
273 originalForm->RidOfLm();
274 }
275 }
276
277 delete originalForm;
278 return normalForm;
279 }

282 const Polynom<MonomType>* BooleanInvolutiveBasis<MonomType>::FindDivisor(const Polynom<MonomType>* polynom
283 , const std::list<Polynom<MonomType>*>& set
284 , bool toGroebner) const
285 {
286 if (!polynom || polynom->IsZero())
287 {
288 return 0;
289 }
290
291 typename std::list<Polynom<MonomType>*>::const_iterator it(set.begin()), setEnd(set.end());
292 const MonomType& plm = polynom->Lm();
293
294 while (it != setEnd)
295 {
296 if (toGroebner && plm.IsDivisibleBy((**it).Lm()))
297 {
298 return *it;
299 }
300 else if (!toGroebner && plm.IsPommaretDivisibleBy((**it).Lm()))
301 {
302 return *it;
303 }
304 ++it;
305 }
306
307 return 0;

310 template <typename MonomType>
311 Polynom<MonomType>* BooleanInvolutiveBasis<MonomType>::Reduce(Polynom<MonomType>* polynom
312 , const std::list<Polynom<MonomType>*>& set
313 , bool toGroebner) const
314 {
315 if (!polynom)
316 {
317 return 0;
318 }
319
320 Polynom<MonomType>* result = new Polynom<MonomType>();
321 const Polynom<MonomType>* currentReducer = 0;
322
323 while (!polynom->IsZero())
324 {
325 currentReducer = FindDivisor(polynom, set, toGroebner);
326 while (currentReducer)
327 {
328 polynom->Reduction(*currentReducer);
329 currentReducer = FindDivisor(polynom, set, toGroebner);
330 }
331 if (!polynom->IsZero())
332 {
333 (*result) += polynom->Lm();
334 polynom->RidOfLm();
335 }
336 }
337
338 polynom = result;
339 return result;
340 }

342 template <typename MonomType>
343 void BooleanInvolutiveBasis<MonomType>::ReduceSet(bool toGroebner)
344 {
345 std::list<Polynom<MonomType>*> tmpPolySet;
346 GBasis.sort(PointerMoreComparator<Polynom<MonomType> >());
347
348 while (!GBasis.empty())
349 {
350 Polynom<MonomType>* currentPolynom = GBasis.front();
351 GBasis.pop_front();
352 currentPolynom = Reduce(currentPolynom, tmpPolySet, toGroebner);
353
354 if (currentPolynom && !currentPolynom->IsZero())
355 {
356 const MonomType& hLm = currentPolynom->Lm();
357 typename std::list<Polynom<MonomType>*>::iterator iteratorTmpPolySet = tmpPolySet.begin();
358 while (iteratorTmpPolySet != tmpPolySet.end())
359 {
360 if ((**iteratorTmpPolySet).Lm().IsDivisibleBy(hLm))
361 {
362 GBasis.push_back(*iteratorTmpPolySet);
363 iteratorTmpPolySet = tmpPolySet.erase(iteratorTmpPolySet);
364 }
365 else
366 {
367 ++iteratorTmpPolySet;
368 }
369 }
370 tmpPolySet.push_back(currentPolynom);
371 }
372 }
373
374 unsigned tmpPolySetSize = static_cast<unsigned int>(tmpPolySet.size());
375 for (unsigned i = 0; i < tmpPolySetSize; ++i)
376 {
377 Polynom<MonomType>* currentPolynom = tmpPolySet.front();
378 tmpPolySet.pop_front();
379 currentPolynom = Reduce(currentPolynom, tmpPolySet, toGroebner);
380 if (!currentPolynom || currentPolynom->IsZero())
381 {
382 tmpPolySetSize--;
383 }
384 else
385 {
386 tmpPolySet.push_back(currentPolynom);
387 }
388 }
389
390 GBasis = tmpPolySet;
391 }

402 void BooleanInvolutiveBasis<MonomType>::Construct(const std::list<Polynom<MonomType>*>& set, bool toGroebner)
403 {
404 Reset();
405 GBasis = set;
406 ReduceSet(true);
407
408 ProlongationsSet.Insert(GBasis);
409 GBasis.clear();
410
411 ConstructInvolutiveBasis();
412 ProlongationsSet.Clear();
413
414 typename TSet<MonomType>::ConstIterator i2(IntermediateBasis.Begin());
415 while (i2 != IntermediateBasis.End())
416 {
417 GBasis.push_back(const_cast<Polynom<MonomType>*>((**i2).GetPolynom()));
418 ++i2;
419 }
420 ReduceSet(toGroebner);

423 template <typename MonomType>
424 void BooleanInvolutiveBasis<MonomType>::ConstructInvolutiveBasis()
425 {
426 typename TSet<MonomType>::Iterator tit(IntermediateBasis.Begin());
427 Polynom<MonomType>* newNormalForm = 0;
428 Triple<MonomType>* currentTriple = 0;
429
430 while (!ProlongationsSet.Empty())
431 {
432 currentTriple = ProlongationsSet.Get();
433 newNormalForm = NormalForm(currentTriple);
434 /* As far as currentTriple can't be 0 (checked in QSET and TSET),
435 * NormalForm can't return 0.
436 */
437
438 std::set<typename MonomType::Integer> currentNmpSet;
439 const Triple<MonomType>* currentAncestor = 0;
440 if (!newNormalForm->IsZero() && newNormalForm->Lm() == currentTriple->GetPolynomLm())
441 {
442 currentNmpSet = currentTriple->GetNmp();
443 currentAncestor = currentTriple->GetAncestor();
444 if (currentAncestor == currentTriple)
445 {
446 currentAncestor = 0;
447 }
448 }
449 delete currentTriple;
450
451 if (!newNormalForm->IsZero())
452 {
453 std::list<Triple<MonomType>*> newProlongations;
454 tit = IntermediateBasis.Begin();
455
456 while (tit != IntermediateBasis.End())
457 {
458 if ((**tit).GetPolynomLm().IsTrueDivisibleBy(newNormalForm->Lm()))
459 {
460 ProlongationsSet.DeleteDescendants(*tit);
461 newProlongations.push_back(*tit);
462 tit = IntermediateBasis.Erase(tit);
463 }
464 else
465 {
466 ++tit;
467 }
468 }
469
470 IntermediateBasis.PushBack(new Triple<MonomType>(newNormalForm, currentAncestor, currentNmpSet, 0, -1));
471 if (!newNormalForm->Degree())
472 {
473 return;
474 }
475
476 IntermediateBasis.CollectNonMultiProlongations(--IntermediateBasis.End(), newProlongations);
477 ProlongationsSet.Insert(newProlongations);
478 }
479 else
480 {
481 delete newNormalForm;
482 }
483 }
484 }
