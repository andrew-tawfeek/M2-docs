264Computation /* or null */ *rawStartComputation(Computation *C)
265/* start or continue the computation */
266{
267 try
268 {
269 clear_emit_size();
270 C->start_computation();
271
272 if (M2_gbTrace == 15)
273 {
274 ComputationStatusCode ret = C->status();
275 switch (ret)
276 {
277 case COMP_DONE_DEGREE_LIMIT:
278 emit_line("computation stopped at degree limit");
279 break;
280 case COMP_DONE:
281 emit_line("computation of GB completed");
282 break;
283 case COMP_DONE_PAIR_LIMIT:
284 emit_line("computation stopped at pair limit");
285 break;
286 case COMP_NEED_RESIZE:
287 case COMP_ERROR:
288 case COMP_INTERRUPTED:
289 case COMP_NOT_STARTED:
290 case COMP_INITIAL_STOP:
291 case COMP_DONE_LENGTH_LIMIT:
292 case COMP_DONE_SYZYGY_LIMIT:
293 case COMP_DONE_GB_LIMIT:
294 case COMP_DONE_SYZ_LIMIT:
295 case COMP_DONE_CODIM:
296 case COMP_DONE_MIN_GENS:
297 case COMP_DONE_STEPS:
298 case COMP_DONE_SUBRING_LIMIT:
299 case COMP_COMPUTING:
300 case COMP_OVERFLOWED:
301 emit_line("computation stopped for some good reason");
302 break;
303 default:
304 emit_line("incorrect status code encountered");
305 break;
306 }
307 }
308
309 return error() ? nullptr : C;
310 } catch (const exc::engine_error& e)
311 {
312 ERROR(e.what());
313 return nullptr;
314 }

698void rawDisplayMatrixStream(const Matrix *inputMatrix)
699{
700 const Ring *R = inputMatrix->get_ring();
701 const PolyRing *P = R->cast_to_PolyRing();
702 if (P == nullptr)
703 {
704 ERROR("expected a polynomial ring");
705 return;
706 }
707 if (P->characteristic() > std::numeric_limits<int>::max())
708 {
709 ERROR("characteristic is too large for mathic gb computation");
710 return;
711 }
712 int charac = static_cast<int>(P->characteristic());
713 int nvars = P->n_vars();
714
715 mgb::GroebnerConfiguration configuration(
716 charac, nvars, inputMatrix->n_rows());
717 mgb::GroebnerInputIdealStream input(configuration);
718
719 std::ostringstream computedStr;
720 mgb::IdealStreamLog<> computed(
721 computedStr, charac, nvars, inputMatrix->n_rows());
722 mgb::IdealStreamChecker<decltype(computed)> checked(computed);
723
724 matrixToStream(inputMatrix, checked);
725
726 std::cout << "result: " << std::endl;
727 std::cout << computedStr.str() << std::endl;
728}

765const Matrix *rawMGB(
766 const Matrix *inputMatrix,
767 int reducer,
768 int spairGroupSize, // a value of 0 means let the algorithm choose
769 int nthreads,
770 M2_string logging)
771{
772 try
773 {
774 const Ring *R = inputMatrix->get_ring();
775 const PolyRing *P = R->cast_to_PolyRing();
776 if (P == nullptr)
777 {
778 ERROR("expected a polynomial ring");
779 return nullptr;
780 }
781 if (nthreads < 0)
782 {
783 ERROR("mgb: expected a non-negative number of threads");
784 return nullptr;
785 }
786 if (P->characteristic() > std::numeric_limits<int>::max())
787 {
788 ERROR("characteristic is too large for mathic gb computation");
789 return nullptr;
790 }
791 if (P->characteristic() == 0)
792 {
793 ERROR(
794 "characteristic for mathic gb computation must be a prime "
795 "number");
796 return nullptr;
797 }
798 if (not P->getCoefficientRing()->isFinitePrimeField())
799 {
800 ERROR("coefficients for mathic gb computation must be a prime field");
801 return nullptr;
802 }
803 int charac = static_cast<int>(P->characteristic());
804 int nvars = P->n_vars();
805 MGBCallback callback;
806 mgb::GroebnerConfiguration configuration(
807 charac, nvars, inputMatrix->n_rows());
808
809 const auto reducerType = reducer == 0
810 ? mgb::GroebnerConfiguration::ClassicReducer
811 : mgb::GroebnerConfiguration::MatrixReducer;
812 configuration.setReducer(reducerType);
813 configuration.setMaxSPairGroupSize(spairGroupSize);
814 configuration.setMaxThreadCount(nthreads);
815 std::string log((char *)logging->array, logging->len);
816 configuration.setLogging(log.c_str());
817 configuration.setCallback(&callback);
818
819 // Now set the monomial ordering info
820 std::vector<int> mat;

896{

913template<typename PolyL>
914const Matrix* polyListToMatrix(const M2FreeAlgebraOrQuotient* A,
915 const PolyL& elems,
916 int numrows,
917 int numcols)
918{
919 if (elems.size() != numrows*numcols)
920 ERROR("Number of elements in list does not match matrix size.");
921 MatrixConstructor mat(A->make_FreeModule(numrows), numcols);
922 for (auto i = 0; i < elems.size(); ++i)
923 {
924 int curCol = i % numcols;
925 int curRow = (i - curCol) / numcols;
926 ring_elem a = const_cast<Nterm*>(reinterpret_cast<const Nterm*>(elems[i]));
927 mat.set_entry(curRow, curCol, a);
928 }
929 mat.compute_column_degrees();
930 return mat.to_matrix();

934{
935 const Ring* R = input->get_ring();

937 if (A != nullptr and input->n_rows() == 1)

947 G.compute(maxdeg); // this argument is actually the soft degree limit

954 G.compute(maxdeg); // this argument is actually the soft degree limit

957 }

959 }
960 ERROR("expected a one row matrix over a noncommutative algebra");
961 return nullptr;

965{
966 const Ring* R = toBeReduced->get_ring();
967 if (R != reducerMatrix->get_ring())
968 {
969 ERROR("expected matrices to be over the same ring");
970 return nullptr;
971 }
972 const M2FreeAlgebra* A = R->cast_to_M2FreeAlgebra();
973 if (A != nullptr and reducerMatrix->n_rows() == 1)
974 {
975 auto outRows = toBeReduced->n_rows();
976 auto outCols = toBeReduced->n_cols();
977 auto reducees = matrixToPolyList(A, toBeReduced);
978 auto reducers = matrixToPolyList(A, reducerMatrix);
979 NCGroebner G(A->freeAlgebra(),reducers, 0, 0);
980 G.initReductionOnly();
981 auto result = G.twoSidedReduction(reducees);
982 return polyListToMatrix(A, result, outRows, outCols); // consumes the Poly's in result.
983 }
984 ERROR("expected a matrix over a noncommutative algebra");
985 return nullptr;
