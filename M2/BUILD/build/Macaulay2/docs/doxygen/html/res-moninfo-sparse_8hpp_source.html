48// Format for monomials here:
49// a. length (in bytes) (int32)
50// b. hash value (int32)
51// c. component (int32)
52// d. v1 v2 ... vd
53// where d = length-2
54// v1 >= v2 >= ... >= vd >= 0 are indices of variables.
55// or, maybe also have degree before this, and other weight values...
56// SO, general form:
57// [length, hashval, component, w1, ..., wr, v, ..., vd]

132 void setWeightAndHash(res_packed_monomial result) const
133 {
134 // assumes the following is set already:
135 // result[0] -- length
136 // result[2] -- component
137 // result[mFirstVar..len-1] -- monomial part
138 // sets:
139 // result[1] -- hash code
140 // result[FirstWeight..mFirstVar-1] -- weight vector values
141 const int* wt = mWeightVectors.data();
142 for (int j = 0; j < mNumWeights; j++, wt += mNumVars)
143 {
144 res_monomial_word val = 0;
145 for (int* v = result + mFirstVar; v != result + *result; ++v)
146 val += wt[*v];
147 result[mFirstWeight + j] = val;
148 }
149 res_monomial_word val = 0;
150 for (int* v = result + mFirstVar; v != result + *result; ++v)
151 val += hashfcn[*v];
152 result[1] = val;
153 }
154 bool from_expvector(res_const_ntuple_monomial e,
155 component_index comp,
156 res_packed_monomial result) const
157 {
158 // Pack the vector e[0]..e[mNumVars-1],comp. Create the hash value at the
159 // same time.
160 ncalls_from_expvector++;
161 result[0] = 0; // length
162 result[1] = 0; // hash value, not set here (other than this...)
163 result[2] = comp; // component
164
165 int next_var_slot = mFirstVar;
166
167 for (int i = mNumVars - 1; i >= 0; --i)
168 {
169 if (e[i] > 0)
170 {
171 for (int j = 0; j < e[i]; j++)
172 {
173 result[1] += hashfcn[i];
174 result[next_var_slot] = i;
175 next_var_slot++;
176 }
177 }
178 }
179 result[0] = next_var_slot; // this is the length of this monomial
180 setWeightAndHash(result);
181 return true;
182 }

221 void to_varpower_monomial(res_const_packed_monomial m,
222 res_varpower_monomial result) const
223 {
224 // 'result' must have enough space allocated
225 ncalls_to_varpower++;
226 int len = 0;
227 res_varpower_word* r = result + 1;
228
229 if (mFirstVar != *m) // ie, m != 1.
230 {
231 int currentvar = m[mFirstVar];
232 int deg = 1;
233 const int* mend = m + *m;
234 for (auto p = m + mFirstVar + 1; p != mend; ++p)
235 {
236 if (*p == currentvar)
237 deg++;
238 else
239 {
240 *r++ = currentvar;
241 *r++ = deg;
242 len++;
243 currentvar = *p;
244 deg = 1;
245 }
246 }
247 }
248 result[0] = len;
249 }

251 void from_varpower_monomial(res_const_varpower_monomial m,
252 component_index comp,
253 res_packed_monomial result) const
254 {
255 // 'result' must have enough space allocated
256 ncalls_from_varpower++;
257 int len = 0;
258 int* r = result + mFirstVar;
259 for (index_res_varpower_monomial j = m; j.valid(); ++j)
260 {
261 res_varpower_word v = j.var();
262 res_varpower_word e = j.exponent();
263 for (int i = 0; i < e; i++) *r++ = v;
264 len += e;
265 }
266 result[0] = len;
267 result[2] = comp;
268 setWeightAndHash(result);
269 }

308 void unchecked_mult(res_const_packed_monomial a,
309 res_const_packed_monomial b,
310 res_packed_monomial result) const
311 {
312 // a: mFirstVar + degA
313 // b: mFirstVar + degB
314 // result size is mFirstVar + degA + degB = size(a) + size(b) - mFirstVar
315 ncalls_mult++;
316 result[0] = a[0] + b[0] - mFirstVar;
317 for (int i = 1; i < mFirstVar; ++i) result[i] = a[i] + b[i];
318 // Now we merge the rest
319 const int* v1 = a + mFirstVar;
320 const int* v2 = b + mFirstVar;
321 const int* end1 = a + *a;
322 const int* end2 = b + *b;
323 int* res = result + mFirstVar;
324 if (v1 == end1)
325 {
326 while (v2 != end2) *res++ = *v2++;
327 return;
328 }
329 if (v2 == end2)
330 {
331 while (v1 != end1) *res++ = *v1++;
332 return;
333 }
334 for (;;)
335 {
336 if (*v1 >= *v2)
337 {
338 *res++ = *v1;
339 v1++;
340 if (v1 == end1)
341 {
342 while (v2 != end2) *res++ = *v2++;
343 return;
344 }
345 }
346 else
347 {
348 *res++ = *v2;
349 v2++;
350 if (v2 == end2)
351 {
352 while (v1 != end1) *res++ = *v1++;
353 return;
354 }
355 }
356 }

359 bool divide(res_const_packed_monomial m,
360 res_const_packed_monomial n,
361 res_packed_monomial result) const
362 {
363 // computes m/n, or tries to. If n divides m, then return true and set
364 // 'result' to m/n
365 // otherwise return false. In the latter case, it is possible that result
366 // will have been written into partially,
367 // but the result is not a well-formed monomial.
368 ncalls_divide++;
369 if (*m < *n) return false;
370 const int* vm = m + mFirstVar;
371 const int* vn = n + mFirstVar;
372 const int* vend = n + *n;
373 int* vresult = result + mFirstVar;
374 for (; vn != vend; ++vn)
375 {
376 if (*vn > *vm) return false;
377 if (*vn == *vm) ++vm;
378 *vresult++ = *vm++;
379 }
380 for (int i = 1; i < mFirstVar; ++i) result[i] = m[i] - n[i];
381 result[0] = m[0] - n[0] + mFirstVar;
382 return true;
383 }

399 res_const_packed_monomial n) const
400 {
401 ncalls_compare_grevlex++;
402 for (int i = 3; i < mFirstVar; i++)
403 {
404 int cmp = m[i] - n[i];

406 if (cmp < 0) return -1;

409 const int* n1 = n + mFirstVar;
410 const int* mend = m + *m;

413 {
414 m1++;
415 n1++;
416 if (m1 == mend)
417 {
418 if (n1 == nend)
419 break;
420 else
421 {
422 return -1;
423 }
424 }
425 else
426 {
427 if (n1 == nend) return 1;

432 if (cmp < 0) return 1;
433 if (cmp > 0) return -1;
434 return 0;
