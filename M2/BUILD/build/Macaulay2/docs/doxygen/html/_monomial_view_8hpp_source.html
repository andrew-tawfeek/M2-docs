85 inline static bool monomialDivides(const MonomialView& divisor,
86 const MonomialView& divisee)
87 {
88 if (divisor.size() > divisee.size()) return false;
89 auto divisorIt = divisor.begin();
90 auto diviseeIt = divisee.begin();
91 auto divisorEnd = divisor.end();
92 auto diviseeEnd = divisee.end();
93 while (divisorIt != divisorEnd)
94 {
95 auto divisorVar = divisorIt.var();
96 while (true)
97 {
98 if (diviseeIt == diviseeEnd) return false;
99 if (diviseeIt.var() > divisorVar) return false;
100 if (diviseeIt.var() == divisorVar) break;
101 ++diviseeIt;
102 }
103 // at this point, both diviseeIt.var() and divisorIt.var() are equal
104 if (divisorIt.power() > diviseeIt.power()) return false;
105 ++divisorIt;
106 ++diviseeIt;
107 }
108 return true;
109 }
