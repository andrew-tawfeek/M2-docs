48 std::pair<QuadMatrixBuilder::LeftRightColIndex, ConstMonomial>
49 createCol(
50 typename Monoid::ConstMonoRef mono,
51 SparseMatrix& top,
52 SparseMatrix& bottom,
53 ToMono& toMonomial,
54 ToCol& toCol,
55 const Monoid& monoid,
56 const bool left
57 ) {
58 (void)top;
59 (void)bottom;
60 MATHICGB_ASSERT(typename ToCol::Reader(toCol).find(mono).first == 0);
61
62 const auto colCount =
63 static_cast<QuadMatrixBuilder::ColIndex>(toMonomial.size());
64 if (colCount == std::numeric_limits<QuadMatrixBuilder::ColIndex>::max())
65 throw std::overflow_error("Too many columns in QuadMatrixBuilder");
66
67 // allocate memory in toMonomial now to ensure no bad_alloc later
68 toMonomial.emplace_back(nullptr);
69 MATHICGB_SCOPE_EXIT(toMonomialGuard) {toMonomial.pop_back();};
70 auto copied = monoid.alloc();
71 monoid.copy(mono, copied);
72 auto inserted = toCol.insert(
73 std::make_pair(
74 copied.ptr(),
75 QuadMatrixBuilder::LeftRightColIndex(colCount, left)
76 )
77 );
78 MATHICGB_ASSERT(inserted.second);
79 MATHICGB_ASSERT(inserted.first.first != 0);
80 auto p(inserted.first);
81 toMonomial.back() = copied;
82
83 MATHICGB_ASSERT(monoid.equalHintTrue(*copied, *p.second));
84 MATHICGB_ASSERT(*p.first == QuadMatrixBuilder::LeftRightColIndex(colCount, left));
85
86 auto ptr = const_cast<exponent*>(Monoid::toOld(*p.second));
87 toMonomialGuard.dismiss();
88 copied.release();
89 return std::make_pair(*p.first, ptr);
90 }
