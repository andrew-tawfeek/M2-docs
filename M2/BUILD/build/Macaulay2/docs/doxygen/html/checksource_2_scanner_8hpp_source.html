209T Scanner::readInteger(const bool negate) {
210 static_assert(std::numeric_limits<T>::is_integer, "");
211
212 eatWhite();
213 const bool minus = !match('+') && match('-');
214 const bool positive = minus == negate;
215 if (!peekDigit())
216 reportErrorUnexpectedToken("an integer", "");
217 // Skip leading zeroes and return if the number is zero.
218 if (peek() == '0') {
219 while (peek() == '0')
220 get();
221 if (!peekDigit())
222 return static_cast<T>(0);
223 }
224
225 // Checking this here allows us to recognize -0 as non-negative.
226 if (!positive && !std::numeric_limits<T>::is_signed)
227 reportErrorUnexpectedToken("a positive integer", "");
228
229 const auto min = std::numeric_limits<T>::min();
230 const auto max = std::numeric_limits<T>::max();
231 auto t = static_cast<T>(0);
232 while (peekDigit()) {
233 const auto c = static_cast<char>(get());
234 const auto d = positive ? c - '0' : -(c - '0');
235 if (positive ? t > (max - d) / 10 : t < (min - d) / 10) {
236 std::ostringstream err;
237 err << "an integer in the range [" << min
238 << ", " << max << ']';
239 reportErrorUnexpectedToken(err.str(), "");
240 }
241 t = t * 10 + d;
242 }
