34 const Poly& addBasisElement(const std::string& str) {
35 std::istringstream in(str);
36 Scanner scanner(in);
37 auto p = make_unique<Poly>
38 (MathicIO<>().readPoly(*mRing, false, scanner));
39 mBasis.insert(std::move(p));
40 return mBasis.poly(mBasis.size() - 1);
41 }
42
43 F4MatrixBuilder& create() {
44 MATHICGB_ASSERT(mBuilder.get() == 0);
45 mBuilder.reset(new F4MatrixBuilder(mBasis));
46 return *mBuilder;
47 }
48
49 const PolyRing& ring() const {return *mRing;}
50
51 private:
52 std::unique_ptr<PolyRing> mRing;
53 Basis mIdeal;
54 PolyBasis mBasis;
55 std::unique_ptr<F4MatrixBuilder> mBuilder;
56 };

80TEST(F4MatrixBuilder, SPair) {

133TEST(F4MatrixBuilder, DirectReducers) {
134 for (int threadCount = 1; threadCount < 4; ++threadCount) {
135 BuilderMaker maker;
136 maker.addBasisElement("a6"); // reducer == to lead term
137 maker.addBasisElement("a3b2+a3c"); // reducer == to lower order term
138 maker.addBasisElement("c"); // reducer divides
139 maker.addBasisElement("d2"); // does not divide
140 F4MatrixBuilder& builder = maker.create();
141
142 Poly p1(builder.ring());
143 {
144 std::istringstream in("a3+b2+c+d");
145 Scanner scanner(in);
146 p1 = MathicIO<>().readPoly(builder.ring(), false, scanner);
147 builder.addPolynomialToMatrix(p1.leadMono(), p1);
148 }
149
150 Poly p2(builder.ring());
151 {
152 std::istringstream in("a3+2b2+3c+4d");
153 Scanner scanner(in);
154 p2 = MathicIO<>().readPoly(builder.ring(), false, scanner);
155 builder.addPolynomialToMatrix(p2.leadMono(), p2);
156 }
157
158 QuadMatrix qm(builder.ring());
159 builder.buildMatrixAndClear(qm);
160
161 const char* str =
162 "Left columns: a6 a3b2 a3c\n"
163 "Right columns: a3d\n"
164 "0: 2#1 | 0: \n"
165 "1: 1#1 2#1 | 1: \n"
166 "2: 0#1 | 2: \n"
167 " | \n"
168 "0: 0#1 1#1 2#1 | 0: 0#1\n"
169 "1: 0#1 1#2 2#3 | 1: 0#4\n";
170 qm = qm.toCanonical();
171 ASSERT_EQ(str, qm.toString()) << "** qm:\n" << qm;
172 }
173}
