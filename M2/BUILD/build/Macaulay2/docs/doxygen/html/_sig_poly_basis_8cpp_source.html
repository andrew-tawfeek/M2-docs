57void SigPolyBasis::insert(Mono ownedSig, std::unique_ptr<Poly> f) {
58 MATHICGB_ASSERT(f.get() != nullptr);
59 MATHICGB_ASSERT(!f->isZero());
60 MATHICGB_ASSERT(!field().isZero(f->leadCoef()));
61 MATHICGB_ASSERT(!ownedSig.isNull());
62 MATHICGB_ASSERT(monoid().fromPool(*ownedSig));
63
64 const auto index = mSignatures.size();
65 mSignatures.push_back(ownedSig.release());
66 auto sig = *mSignatures.back();
67
68 const auto component = monoid().component(sig);
69 MATHICGB_ASSERT(component < mSignatureLookup.size());
70 mSignatureLookup[component]->insert(sig, index);
71
72 auto ratio = ring().allocMonomial();
73 monoid().divideToNegative(f->leadMono(), sig, ratio);
74
75 mSigLeadRatio.push_back(ratio);
76
77 const auto lead = f->leadMono();
78 mBasis.insert(std::move(f));
79 if (mBasis.leadMinimal(mBasis.size() - 1)) {
80 mMinimalMonoLookup->removeMultiples(lead);
81 mMinimalMonoLookup->insert(lead, index);
82 }
83
84 MATHICGB_ASSERT(mMinimalMonoLookup->type() == 0 ||
85 mBasis.minimalLeadCount() == mMinimalMonoLookup->size());
86 MATHICGB_ASSERT(mSignatures.size() == index + 1);
87 MATHICGB_ASSERT(mBasis.size() == index + 1);
88 if (!mUseRatioRank)
89 return;
90
91 // compute rank of the ratio
92 auto pos = mRatioSorted.insert(index);
93again:
94 Rank prevRank;
95 if (pos == mRatioSorted.begin())
96 prevRank = 0;
97 else {
98 auto prev = pos;
99 --prev;
100 prevRank = mRatioRanks[*prev];
101 if (monoid().equal(ratio, *mSigLeadRatio[*prev])) {
102 mRatioRanks.push_back(prevRank);
103 return;
104 }
105 }
106
107 Rank nextRank;
108 auto next = pos;
109 ++next;
110 if (next == mRatioSorted.end())
111 nextRank = std::numeric_limits<Rank>::max();
112 else {
113 nextRank = mRatioRanks[*next];
114 if (monoid().equal(ratio, *mSigLeadRatio[*next])) {
115 mRatioRanks.push_back(nextRank);
116 return;
117 }
118 }
119 MATHICGB_ASSERT(prevRank < nextRank);
120
121 // this formula avoids the overflow inherent in prevRank + nextRank;
122 Rank rank = prevRank + (nextRank - prevRank) / 2;
123
124 // must have at least 1 space between ranks to support
125 // queries for non-basis element rank
126 if (rank == 0 || // must leave space for smaller ratio
127 rank == std::numeric_limits<Rank>::max() || // shouldn't happen
128 nextRank - prevRank < 4) { // 4 as require: prev, gap, new, gap, next
129 // size plus 1 to account for the gaps at the beginning and end.
130 size_t increment = std::numeric_limits<Rank>::max() / (mSignatures.size() + 1);
131 if (increment == 0)
132 increment = 2;
133 MATHICGB_ASSERT(!mRatioSorted.empty());
134 size_t rankSum = increment; // leave a gap at beginning
135 Rank prevRank = *mRatioRanks.begin();
136 auto end = mRatioSorted.end();
137 for (auto it = mRatioSorted.begin(); it != end; ++it) {
138 if (it == pos)
139 continue;
140 if (mRatioRanks[*it] != prevRank)
141 rankSum += increment;
142 prevRank = mRatioRanks[*it];
143 mRatioRanks[*it] = rankSum;
144 }
145 goto again;
146 }
147 MATHICGB_ASSERT(rank > 0);
148 MATHICGB_ASSERT(rank < std::numeric_limits<Rank>::max());
149 MATHICGB_ASSERT(prevRank + 1 < rank && rank < nextRank - 1);
150 mRatioRanks.push_back(rank);
151 MATHICGB_ASSERT(mRatioRanks.size() == index + 1);
152
153#ifdef MATHICGB_DEBUG
154 // Check that at least one space has been left between every rank
155 MATHICGB_ASSERT(mRatioRanks[*mRatioSorted.begin()] > 0);
156 MATHICGB_ASSERT(mRatioRanks[*mRatioSorted.rbegin()] <
157 std::numeric_limits<Rank>::max());
158 auto it2 = mRatioSorted.begin();
159 for (++it2; it2 != mRatioSorted.end(); ++it2) {
160 auto prev = it2;
161 --prev;
162 MATHICGB_ASSERT(mRatioRanks[*it2] == mRatioRanks[*prev] ||
163 mRatioRanks[*it2] - 1 > mRatioRanks[*prev]);
164 }
165#endif
166}

168size_t SigPolyBasis::regularReducer(
169 ConstMonoRef sig,
170 ConstMonoRef term
171) const {
172 size_t reducer = monoLookup().regularReducer(sig, term);
173#ifdef MATHICGB_SLOW_DEBUG
174 const size_t debugValue = regularReducerSlow(sig, term);
175 if (reducer == static_cast<size_t>(-1)) {
176 MATHICGB_SLOW_ASSERT(debugValue == static_cast<size_t>(-1));
177 } else {
178 MATHICGB_SLOW_ASSERT(debugValue != static_cast<size_t>(-1));
179 monomial m = ring().allocMonomial();
180 MATHICGB_SLOW_ASSERT
181 (ring().monomialIsDivisibleBy(term, leadMono(reducer)));
182 ring().monomialDivide(term, leadMono(reducer), m);
183 ring().monomialMultTo(m, signature(reducer));
184 MATHICGB_SLOW_ASSERT(monoid().lessThan(m, sig));
185 ring().freeMonomial(m);
186 }
187#endif
188 return reducer;
189}

191size_t SigPolyBasis::regularReducerSlow(
192 ConstMonoRef sig,
193 ConstMonoRef term
194) const {
195 monomial m = ring().allocMonomial();
196 const size_t stop = size();
197 for (size_t be = 0; be < stop; ++be) {
198 if (!monoid().divides(leadMono(be), term))
199 continue;
200 monoid().divide(leadMono(be), term, m);
201 monoid().multiplyInPlace(signature(be), m);
202 if (monoid().lessThan(m, sig)) {
203 ring().freeMonomial(m);
204 return be;
205 }
206 }
207 ring().freeMonomial(m);
208 return static_cast<size_t>(-1);
209}

232void SigPolyBasis::lowBaseDivisorsSlow(
233 std::vector<size_t>& divisors,
234 size_t maxDivisors,
235 size_t newGenerator) const
236{
237 MATHICGB_ASSERT(newGenerator < size());
238
239 divisors.clear();
240 divisors.reserve(maxDivisors + 1);
241
242 auto sigNew = signature(newGenerator);
243 for (size_t i = 0; i < newGenerator; ++i) {
244 auto sigi = signature(i);
245
246 if (monoid().component(sigi) != monoid().component(sigNew))
247 continue;
248 if (!monoid().divides(sigi, sigNew))
249 continue;
250 for (size_t j = 0; j <= divisors.size(); ++j) {
251 if (j == divisors.size()) {
252 divisors.push_back(i);
253 break;
254 }
255 if (ratioCompare(i, divisors[j]) == GT) {
256 divisors.insert(divisors.begin() + j, i);
257 break;
258 }
259 }
260 if (divisors.size() > maxDivisors)
261 divisors.pop_back();
262 MATHICGB_ASSERT(divisors.size() <= maxDivisors);
263 }
264 MATHICGB_ASSERT(divisors.size() <= maxDivisors);
265}

280size_t SigPolyBasis::highBaseDivisorSlow(size_t newGenerator) const {
281 MATHICGB_ASSERT(newGenerator < size());
282
283 auto highDivisor = static_cast<size_t>(-1);
284 auto leadNew = leadMono(newGenerator);
285 for (size_t i = 0; i < newGenerator; ++i) {
286 // continue if this generator would not be an improvement
287 // even if it does divide. This is a faster check than
288 // checking divisiblity, so do it first.
289 if (highDivisor != static_cast<size_t>(-1) &&
290 ratioCompare(highDivisor, i) == LT)
291 continue;
292 auto leadi = leadMono(i);
293 if (monoid().divides(leadi, leadNew))
294 highDivisor = i;
295 }
296 return highDivisor;
297}

306size_t SigPolyBasis::minimalLeadInSigSlow(ConstMonoRef sig) const {
307 monomial multiplier = ring().allocMonomial();
308 monomial minLead = ring().allocMonomial();
309
310 size_t minLeadGen = static_cast<size_t>(-1);
311 const auto sigComponent = monoid().component(sig);
312 const size_t genCount = size();
313 for (size_t gen = 0; gen < genCount; ++gen) {
314 if (monoid().component(signature(gen)) != sigComponent)
315 continue;
316 if (!monoid().divides(signature(gen), sig))
317 continue;
318 monoid().divide(signature(gen), sig, multiplier);
319 if (minLeadGen != static_cast<size_t>(-1)) {
320 auto genLead = leadMono(gen);
321 const auto leadCmp = monoid().compare(minLead, multiplier, genLead);
322 if (leadCmp == Monoid::LessThan)
323 continue;
324 if (leadCmp == Monoid::EqualTo) {
325 // If same lead monomial in signature, pick the one with fewer terms
326 // as that one might be less effort to reduce.
327 const size_t minTerms = poly(minLeadGen).termCount();
328 const size_t terms = poly(gen).termCount();
329 if (minTerms > terms)
330 continue;
331 if (minTerms == terms) {
332 // If same number of terms, pick the one with larger signature
333 // before being multiplied into the same signature. That one
334 // might be more reduced as the constraint on regular reduction
335 // is less.
336 const auto minSig = signature(minLeadGen);
337 const auto genSig = signature(gen);
338 const auto sigCmp = monoid().compare(minSig, genSig);
339
340 // no two generators have same signature
341 MATHICGB_ASSERT(sigCmp != Monoid::EqualTo);
342 if (sigCmp == GT)
343 continue;
344 }
345 }
346 }
347
348 minLeadGen = gen;
349 monoid().multiply(multiplier, leadMono(gen), minLead);
350 }
351 ring().freeMonomial(multiplier);
352 ring().freeMonomial(minLead);
353 return minLeadGen;
354}

356bool SigPolyBasis::isSingularTopReducibleSlow(
357 const Poly& poly,
358 ConstMonoRef sig
359) const {
360 if (poly.isZero())
361 return false;
362
363 monomial multiplier = ring().allocMonomial();
364 const size_t genCount = size();
365 const auto polyLead = poly.leadMono();
366 for (size_t i = 0; i < genCount; ++i) {
367 if (!monoid().divides(leadMono(i), polyLead))
368 continue;
369 monoid().divide(leadMono(i), polyLead, multiplier);
370 if (monoid().compare(sig, multiplier, signature(i)) == EQ)
371 return true;
372 }
373 ring().freeMonomial(multiplier);
374 return false;
375}

425size_t SigPolyBasis::getMemoryUse() const
426{
427 // Note: we do not count the signatures as they are counted elsewhere.
428 size_t total = 0;
429 total += mBasis.getMemoryUse();
430 total += mSignatures.capacity() * sizeof(mSignatures.front());
431 total += mSigLeadRatio.capacity() * sizeof(mSigLeadRatio.front());
432 total += mRatioRanks.capacity() * sizeof(mRatioRanks.front());
433 total += monoLookup().getMemoryUse();
434 total += mMinimalMonoLookup->getMemoryUse();
435
436 // This is an estimate of how much memory mRatioSorted uses per item.
437 // It is based on assuming a tree representation with a left pointer,
438 // a right pointer and a data member for each node. This is probably
439 // an underestimate.
440 const size_t perItemOverhead =
441 2 * sizeof(void*) + sizeof(*mRatioSorted.begin());
442 total += mRatioSorted.size() * perItemOverhead;
443
444 return total;
445}

447size_t SigPolyBasis::ratioRank(ConstMonoRef ratio) const {
448 MATHICGB_ASSERT(mUseRatioRank);
449 const size_t index = size();
450 if (index == 0)
451 return 0; // any value will do as there is nothing to compare to
452 auto& sigLeadRatioNonConst =
453 const_cast<std::vector<MonoPtr>&>(mSigLeadRatio);
454
455 sigLeadRatioNonConst.push_back(ratio.castAwayConst());
456 auto pos = mRatioSorted.lower_bound(index);
457 sigLeadRatioNonConst.pop_back();
458
459 if (pos == mRatioSorted.end()) {
460 MATHICGB_ASSERT
461 (ratioRank(*mRatioSorted.rbegin()) < std::numeric_limits<Rank>::max());
462 return std::numeric_limits<Rank>::max();
463 } else {
464 if (monoid().equal(ratio, sigLeadRatio(*pos)))
465 return ratioRank(*pos);
466 MATHICGB_ASSERT(ratioRank(*pos) > 0);
467#ifdef MATHICGB_DEBUG
468 if (pos != mRatioSorted.begin()) {
469 auto prev = pos;
470 --prev;
471 MATHICGB_ASSERT(ratioRank(*pos) - 1 > ratioRank(*prev));
472 }
473#endif
474 return ratioRank(*pos) - 1;
475 }
476}
