51void MatrixAction::performAction() {
52 mParams.perform();
53 for (size_t i = 0; i < mParams.inputFileCount(); ++i) {
54 const auto fileNameStem = mParams.inputFileNameStem(i);
55 const auto extension = mParams.inputFileNameExtension(i);
56 const auto quadFileName = fileNameStem + QuadMatrixExtension;
57 const auto lowerRightFileName = fileNameStem + LowerRightMatrixExtension;
58 const auto reducedLowerRightFileName =
59 fileNameStem + ReducedLowerRightMatrixExtension;
60 std::string inputFileName;
61
62 SparseMatrix lowerRightMatrix;
63 SparseMatrix::Scalar modulus = 0; // will be changed below to a (hopefully) prime number
64 if (
65 extension == QuadMatrixExtension ||
66 extension == "." ||
67 extension == ""
68 ) {
69 inputFileName = quadFileName;
70 CFile file(quadFileName, "rb");
71 QuadMatrix matrix;
72 modulus = matrix.read(file.handle());
73 file.close();
74 lowerRightMatrix = F4MatrixReducer(modulus).reduceToBottomRight(matrix);
75
76 if (!fileExists(lowerRightFileName)) {
77 CFile file(lowerRightFileName, "wb");
78 lowerRightMatrix.write(modulus, file.handle());
79 CFile pbmFile(lowerRightFileName + ".pbm", "wb");
80 lowerRightMatrix.writePBM(pbmFile.handle());
81 }
82 } else if (extension == LowerRightMatrixExtension) {
83 inputFileName = lowerRightFileName;
84 CFile file(lowerRightFileName, "rb");
85 modulus = lowerRightMatrix.read(file.handle());
86 } else {
87 mathic::reportError
88 ("Unknown input file extension of " + mParams.inputFileName(i));
89 }
90
91 lowerRightMatrix = F4MatrixReducer(modulus).
92 reducedRowEchelonForm(lowerRightMatrix);
93 lowerRightMatrix.sortRowsByIncreasingPivots();
94
95 if (!fileExists(reducedLowerRightFileName)) {
96 CFile file(reducedLowerRightFileName.c_str(), "wb");
97 lowerRightMatrix.write(modulus, file.handle());
98 CFile pbmFile(reducedLowerRightFileName + ".pbm", "wb");
99 lowerRightMatrix.writePBM(pbmFile.handle());
100 } else {
101 SparseMatrix referenceMatrix;
102 CFile file(reducedLowerRightFileName.c_str(), "rb");
103 referenceMatrix.read(file.handle());
104
105 if (lowerRightMatrix != referenceMatrix) {
106 const std::string wrongFile =
107 fileNameStem + ".out" + ReducedLowerRightMatrixExtension;
108 const std::string wrongFilePbm = fileNameStem + ".out.pbm";
109 std::cerr << "Reducing " << inputFileName
110 << " does not yield the matrix "
111 << reducedLowerRightFileName << ".\n"
112 << "Writing computed matrix to " << wrongFile << ".\n";
113 CFile file(wrongFile, "wb");
114 lowerRightMatrix.write(modulus, file.handle());
115 CFile filePbm(wrongFilePbm, "wb");
116 lowerRightMatrix.writePBM(filePbm.handle());
117 } else if (tracingLevel > 0) {
118 std::cerr << "Match for " << inputFileName
119 << " -> " << ReducedLowerRightMatrixExtension << ".\n";
120 }
121 }
122 }
123}
