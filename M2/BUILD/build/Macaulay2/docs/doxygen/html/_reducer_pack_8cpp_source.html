53 // Represents a term multiple of a polynomial,
54 // together with a current term of the multiple.
55 struct MultipleWithPos {
56 MultipleWithPos(const Poly& poly, NewConstTerm multiple);
57
58 Poly::ConstTermIterator pos;
59 const Poly::ConstTermIterator end;
60 NewTerm multiple;
61
62 // invariant: current is the monomial product of multiple.monom
63 // and pos.getMonomial().
64 MonoPtr current;
65
66 // Ensures the invariant, so sets current to the product of
67 // multiple.monom and pos.getMonomial().
68 void computeCurrent(const PolyRing& ring);
69 void currentCoefficient(const PolyRing& ring, coefficient& coeff);
70 void destroy(const PolyRing& ring);
71 };

90template<template<typename> class Q>
91void ReducerPack<Q>::insertTail(NewConstTerm multiple, const Poly& poly)
92{
93 if (poly.termCount() <= 1)
94 return;
95 mLeadTermKnown = false;
96
97 MultipleWithPos* entry =
98 new (mPool.alloc()) MultipleWithPos(poly, multiple);
99 ++entry->pos;
100 entry->computeCurrent(poly.ring());
101 mQueue.push(entry);
102}

154template<template<typename> class Q>
155bool ReducerPack<Q>::leadTerm(NewConstTerm& result)
156{
157 if (!mLeadTermKnown) {
158 do {
159 if (mQueue.empty())
160 return false;
161 MultipleWithPos* entry = mQueue.top();
162 std::swap(mLeadTerm.mono, entry->current);
163 entry->currentCoefficient(mRing, mLeadTerm.coef);
164
165 while (true) {
166 ++entry->pos;
167 if (entry->pos == entry->end) {
168 mQueue.pop();
169 entry->destroy(mRing);
170 mPool.free(entry);
171 } else {
172 entry->computeCurrent(mRing);
173 mQueue.decreaseTop(entry);
174 }
175
176 if (mQueue.empty())
177 break;
178
179 entry = mQueue.top();
180 if (!mRing.monoid().equal(*entry->current, *mLeadTerm.mono))
181 break;
182 coefficient coef;
183 entry->currentCoefficient(mRing, coef);
184 mRing.coefficientAddTo
185 (mLeadTerm.coef, const_cast<const coefficient&>(coef));
186 }
187 } while (mRing.coefficientIsZero(mLeadTerm.coef));
188 }
189
190 result = mLeadTerm;
191 mLeadTermKnown = true;
192 return true;
193}

205template<template<typename> class Q>
206void ReducerPack<Q>::resetReducer()
207{
208 class MonomialFree {
209 public:
210 MonomialFree(const PolyRing& ring): mRing(ring) {}
211
212 bool proceed(MultipleWithPos* entry) {
213 entry->destroy(mRing);
214 return true;
215 }
216
217 private:
218 const PolyRing& mRing;
219 };
220
221 MonomialFree freeer(mRing);
222 mQueue.forAll(freeer);
223 mQueue.clear();
224}
