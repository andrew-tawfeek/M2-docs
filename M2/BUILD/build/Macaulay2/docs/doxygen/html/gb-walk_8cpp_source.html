145void GBWalker::start_computation()
146{
147 if (stop_.always_stop) return; // don't change status
148
149 for (;;) switch (state)
150 {
151 case STATE_compute_w:
152 if (!compute_next_w())
153 {
154 // We are done!
155 state = STATE_done;
156 set_status(COMP_DONE);
157 return;
158 }
159 inwwG = G->get_parallel_lead_terms(ww);
160 gb_inwwG = make_gb(inwwG);
161 state = STATE_do_gb;
162 case STATE_do_gb:
163 // Now compute the GB object. If not interrupted, go on:
164 gb_inwwG->start_computation();
165 if (gb_inwwG->status() == COMP_INTERRUPTED)
166 {
167 set_status(COMP_INTERRUPTED);
168 return;
169 }
170 next_to_reduce = 0;
171 state = STATE_reduce;
172 [[fallthrough]];
173 case STATE_reduce:
174 while (next_to_reduce < 0) // TODO: consider the top of the loop
175 {
176 H = G->matrix_remainder(
177 gb_inwwG->get_gb()); // Not quite: need to subtract...
178 next_to_reduce++;
179 }
180 state = STATE_autoreduce;
181 break;
182 case STATE_autoreduce:
183 G->remove_gb();
184 delete G;
185 G1 = static_cast<MarkedGB *>(
186 GBDeclared::create(gb_inwwG->get_initial(-1), H, H, nullptr, nullptr));
187 state = STATE_compute_w;
188 case STATE_done:
189 set_status(COMP_DONE);
190 return;
191 }
192}
