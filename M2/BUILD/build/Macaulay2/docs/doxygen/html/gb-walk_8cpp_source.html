159void GBWalker::start_computation()
160{
161 if (stop_.always_stop) return; // don't change status
162
163 for (;;) switch (state)
164 {
165 case STATE_compute_w:
166 if (!compute_next_w())
167 {
168 // We are done!
169 state = STATE_done;
170 set_status(COMP_DONE);
171 return;
172 }
173 inwwG = G->get_parallel_lead_terms(ww);
174 gb_inwwG = make_gb(inwwG);
175 state = STATE_do_gb;
176 case STATE_do_gb:
177 // Now compute the GB object. If not interrupted, go on:
178 gb_inwwG->start_computation();
179 if (gb_inwwG->status() == COMP_INTERRUPTED)
180 {
181 set_status(COMP_INTERRUPTED);
182 return;
183 }
184 next_to_reduce = 0;
185 state = STATE_reduce;
186 [[fallthrough]];
187 case STATE_reduce:
188 while (next_to_reduce < 0) // TODO: consider the top of the loop
189 {
190 H = G->matrix_remainder(
191 gb_inwwG->get_gb()); // Not quite: need to subtract...
192 next_to_reduce++;
193 }
194 state = STATE_autoreduce;
195 break;
196 case STATE_autoreduce:
197 G->remove_gb();
198 delete G;
199 G1 = static_cast<MarkedGB *>(
200 GBDeclared::create(gb_inwwG->get_initial(-1), H, H, nullptr, nullptr));
201 state = STATE_compute_w;
202 case STATE_done:
203 set_status(COMP_DONE);
204 return;
205 }
206}
