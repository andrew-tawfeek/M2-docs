157template <typename CoeffRing>
158Matrix* toMatrix(const Ring* R, const DMat<CoeffRing>& A)
159{
160 int nrows = static_cast<int>(A.numRows());
161 int ncols = static_cast<int>(A.numColumns());
162 FreeModule* F = R->make_FreeModule(nrows);
163 MatrixConstructor result(F, ncols);
164 if (nrows == 0 || ncols == 0) return result.to_matrix();
165
166 for (int r = 0; r < nrows; ++r)
167 for (int c = 0; c < ncols; ++c)
168 {
169 const typename CoeffRing::ElementType & a = A.entry(r,c);
170 if (not A.ring().is_zero(a))
171 {
172 ring_elem ra;
173 A.ring().to_ring_elem(ra, a);
174 result.set_entry(r, c, ra);
175 }
176 }
177
178 result.compute_column_degrees();
179 return result.to_matrix();
180}
181template <typename CoeffRing>
182Matrix* toMatrix(const Ring* R, const SMat<CoeffRing>& A)
183{
184 int nrows = static_cast<int>(A.numRows());
185 int ncols = static_cast<int>(A.numColumns());
186 FreeModule* F = R->make_FreeModule(nrows);
187 MatrixConstructor result(F, ncols);
188 if (nrows == 0 || ncols == 0) return result.to_matrix();
189 ring_elem f;
190 auto i = A.begin();
191 for (int c = 0; c < ncols; c++)
192 {
193 ring_elem a;
194 for (i.set(c); i.valid(); i.next())
195 {
196 i.copy_elem(a);
197 int r = static_cast<int>(i.row());
198 result.set_entry(r, c, a);
199 }
200 }
201 result.compute_column_degrees();
202 return result.to_matrix();
203}

341 virtual bool get_entry(size_t r, size_t c, ring_elem& result) const
342 // Returns false if (r,c) is out of range or if result is 0. No error
343 // is returned. result <-- this(r,c), and is set to zero if false is returned.
344 {
345 if (r < n_rows() && c < n_cols())
346 {
347 Element a(mat.ring());
348
349 MatOps::getEntry(mat, r, c, a);
350 bool is_nonzero = not mat.ring().is_zero(a);
351 mat.ring().to_ring_elem(result, a);
352 return is_nonzero;
353 }
354
355 result = get_ring()->zero();
356 return false;
357 }

575 {
576 for (size_t r = 0; r < rows->len; r++)
577 if (rows->array[r] < 0 || rows->array[r] >= n_rows())
578 {
579 ERROR(
580 "row index %d out of bounds 0..%d", rows->array[r], n_rows() - 1);
581 return 0;
582 }
583 for (size_t c = 0; c < cols->len; c++)
584 if (cols->array[c] < 0 || cols->array[c] >= n_cols())

586 ERROR("column index %d out of bounds 0..%d",

588 n_cols() - 1);
589 return 0;
590 }

592 MatOps::setFromSubmatrix(getMat(), rows, cols, result->getMat());
