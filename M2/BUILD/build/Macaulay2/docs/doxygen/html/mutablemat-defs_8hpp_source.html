126template <typename CoeffRing>
127Matrix* toMatrix(const Ring* R, const DMat<CoeffRing>& A)
128{
129 int nrows = static_cast<int>(A.numRows());
130 int ncols = static_cast<int>(A.numColumns());
131 FreeModule* F = R->make_FreeModule(nrows);
132 MatrixConstructor result(F, ncols);
133 if (nrows == 0 || ncols == 0) return result.to_matrix();
134
135 for (int r = 0; r < nrows; ++r)
136 for (int c = 0; c < ncols; ++c)
137 {
138 const typename CoeffRing::ElementType & a = A.entry(r,c);
139 if (not A.ring().is_zero(a))
140 {
141 ring_elem ra;
142 A.ring().to_ring_elem(ra, a);
143 result.set_entry(r, c, ra);
144 }
145 }
146
147 result.compute_column_degrees();
148 return result.to_matrix();
149}
150template <typename CoeffRing>
151Matrix* toMatrix(const Ring* R, const SMat<CoeffRing>& A)
152{
153 int nrows = static_cast<int>(A.numRows());
154 int ncols = static_cast<int>(A.numColumns());
155 FreeModule* F = R->make_FreeModule(nrows);
156 MatrixConstructor result(F, ncols);
157 if (nrows == 0 || ncols == 0) return result.to_matrix();
158 ring_elem f;
159 auto i = A.begin();
160 for (int c = 0; c < ncols; c++)
161 {
162 ring_elem a;
163 for (i.set(c); i.valid(); i.next())
164 {
165 i.copy_elem(a);
166 int r = static_cast<int>(i.row());
167 result.set_entry(r, c, a);
168 }
169 }
170 result.compute_column_degrees();
171 return result.to_matrix();
172}

296 virtual size_t lead_row(size_t col, ring_elem& result) const
297 /* returns the largest index row which has a non-zero value in column 'col'.

310 virtual bool get_entry(size_t r, size_t c, ring_elem& result) const
311 // Returns false if (r,c) is out of range or if result is 0. No error
312 // is returned. result <-- this(r,c), and is set to zero if false is returned.
313 {
314 if (r < n_rows() && c < n_cols())
315 {
316 Element a(mat.ring());
317
318 MatOps::getEntry(mat, r, c, a);
319 bool is_nonzero = not mat.ring().is_zero(a);
320 mat.ring().to_ring_elem(result, a);
321 return is_nonzero;
322 }
323
324 result = get_ring()->zero();
325 return false;
326 }

545 for (size_t r = 0; r < rows->len; r++)

548 ERROR(
