303 void invert(ElementType &result, const ElementType &a) const
304 // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
305 {
306 mpfi_t norm, temp;
307 mpfi_init2(norm,get_precision());
308 mpfi_init2(temp,get_precision());
309 mpfi_set(norm,&a.re);
310 mpfi_mul(norm,norm,norm);
311 mpfi_set(temp,&a.im);
312 mpfi_mul(temp,temp,temp);
313 mpfi_add(norm,norm,temp);
314 mpfi_set(&result.re,&a.re);
315 mpfi_mul_si(&result.im,&a.im,-1);
316 mpfi_div(&result.re,&result.re,norm);
317 mpfi_div(&result.im,&result.im,norm);
318 }

360 void mult(ElementType &result,
361 const ElementType &a,
362 const ElementType &b) const
363 {
364 mpfi_t temp, retemp, imtemp;
365 mpfi_init2(temp,get_precision());
366 mpfi_init2(retemp,get_precision());
367 mpfi_init2(imtemp,get_precision());
368 mpfi_mul(retemp,&a.re,&b.re);
369 mpfi_mul(temp,&a.im,&b.im);
370 mpfi_sub(retemp,retemp,temp);
371 mpfi_mul(temp,&a.re,&b.im);
372 mpfi_mul(imtemp,&a.im,&b.re);
373 mpfi_add(&result.im,imtemp,temp);
374 mpfi_set(&result.re,retemp);
375 }

389 void power(ElementType &result, const ElementType &a, int n) const
390 {
391 if (n >= 2)
392 {
393 ElementType b;
394 init(b);
395 if (n%2 == 0)
396 {
397 power(b,a,n/2);
398 mult(result,b,b);
399 }
400 else
401 {
402 power(b,a,n-1);
403 mult(result,a,b);
404 }
405 }
406 else if (n == 1)
407 {
408 mpfi_set(&result.re,&a.re);
409 mpfi_set(&result.im,&a.im);
410 }
411 else if (n == 0)
412 {
413 mpfi_set_si(&result.re,1);
414 mpfi_set_si(&result.im,0);
415 }
416 else if (n<0)
417 throw 20;
418 }
