167 size_t regularReducer(
168 ConstMonoRef sig,
169 ConstMonoRef mono,
170 const SigPolyBasis& sigBasis,
171 const bool preferSparseReducers
172 ) const {
173 SigPolyBasis::StoredRatioCmp ratioCmp(sig, mono, sigBasis);
174 const auto& basis = sigBasis.basis();
175
176 auto reducer = size_t(-1);
177 auto proceed = [&](const Entry& e) {
178 if (ratioCmp.compare(e.data()) != GT)
179 return true;
180
181 if (reducer != size_t(-1)) {
182 if (preferSparseReducers) {
183 const auto newTermCount = basis.poly(e.data()).termCount();
184 const auto oldTermCount = basis.poly(reducer).termCount();
185 if (newTermCount > oldTermCount)
186 return true; // what we already have is sparser
187 // resolve ties by picking oldest
188 if (newTermCount == oldTermCount && e.data() > reducer)
189 return true;
190 } else { // pick oldest
191 if (e.data() > reducer)
192 return true; // what we already have is older
193 }
194 }
195 reducer = e.data();
196 return true;
197 };
198 auto wrap = lambdaWrap(proceed);
199 mLookup.findAllDivisors(mono, wrap);
200 return reducer;
201 }

203 void lowBaseDivisors(
204 std::vector<size_t>& divisors,
205 const size_t maxDivisors,
206 const size_t newGenerator,
207 const SigPolyBasis& basis
208 ) const {
209 MATHICGB_ASSERT(newGenerator < basis.size());
210 auto proceed = [&](const Entry& entry) {
211 if (entry.data() >= newGenerator)
212 return true;
213 for (size_t j = 0; j <= divisors.size(); ++j) {
214 if (j == divisors.size()) {
215 divisors.push_back(entry.data());
216 break;
217 }
218 int cmp = basis.ratioCompare(entry.data(), divisors[j]);
219 if (cmp == EQ && (entry.data() < divisors[j]))
220 cmp = GT; // prefer minimum index to ensure deterministic behavior
221 if (cmp == GT) {
222 divisors.insert(divisors.begin() + j, entry.data());
223 break;
224 }
225 }
226 if (divisors.size() > maxDivisors)
227 divisors.pop_back();
228 MATHICGB_ASSERT(divisors.size() <= maxDivisors);
229 return true;
230 };
231 divisors.clear();
232 divisors.reserve(maxDivisors + 1);
233 auto wrap = lambdaWrap(proceed);
234 mLookup.findAllDivisors(basis.signature(newGenerator), wrap);
235 }

237 size_t highBaseDivisor(
238 const size_t newGenerator,
239 const SigPolyBasis& basis
240 ) const {
241 MATHICGB_ASSERT(newGenerator < basis.size());
242 auto highDivisor = size_t(-1);
243 auto proceed = [&](const Entry& entry) {
244 if (entry.data() >= newGenerator)
245 return true;
246 if (highDivisor != size_t(-1)) {
247 int cmp = basis.ratioCompare(highDivisor, entry.data());
248 if (cmp == LT)
249 return true;
250 if (cmp == EQ && (entry.data() > highDivisor))
251 return true; // prefer minimum index to ensure deterministic behavior
252 }
253 highDivisor = entry.data();
254 return true;
255 };
256 auto wrap = lambdaWrap(proceed);
257 mLookup.findAllDivisors(basis.leadMono(newGenerator), wrap);
258 return highDivisor;
259 }

261 size_t minimalLeadInSig(
262 ConstMonoRef sig,
263 const SigPolyBasis& basis
264 ) const {
265 // Given signature sig, we want to minimize (S/G)g where
266 // g and G are the lead term and signature taken over basis elements
267 // whose signature G divide S. The code here instead maximizes G/g,
268 // which is equivalent and also faster since the basis has a data
269 // structure to accelerate comparisons between the ratio of
270 // signature to lead term.
271 //
272 // In case of ties, we select the sparser elements. If there is
273 // still a tie, we select the basis element with the largest
274 // signature. There can be no further ties since all basis
275 // elements have distinct signatures.
276 auto minLeadGen = size_t(-1);
277 auto proceed = [&](const Entry& entry) {
278 if (minLeadGen != size_t(-1)) {
279 const int ratioCmp = basis.ratioCompare(entry.data(), minLeadGen);
280 if (ratioCmp == LT)
281 return true;
282 if (ratioCmp == EQ) {
283 // If same lead monomial in signature, pick the one with fewer terms
284 // as that one might be less effort to reduce.
285 const size_t minTerms = basis.poly(minLeadGen).termCount();
286 const size_t terms = basis.poly(entry.data()).termCount();
287 if (minTerms > terms)
288 return true;
289 if (minTerms == terms) {
290 // If same number of terms, pick the one with larger signature
291 // before being multiplied into the same signature. That one
292 // might be more reduced as the constraint on regular reduction
293 // is less. Also, as no two generators have same signature, this
294 // ensures deterministic behavior.
295 const auto minSig = basis.signature(minLeadGen);
296 const auto genSig = basis.signature(entry.data());
297 if (basis.monoid().lessThan(genSig, minSig))
298 return true;
299 }
300 }
301 }
302 minLeadGen = entry.data();
303 return true;
304 };
305 auto wrap = lambdaWrap(proceed);
306 mLookup.findAllDivisors(sig, wrap);
307 return minLeadGen;
308 }

311
312 size_t classicReducer(
313 ConstMonoRef mono,
314 const PolyBasis& basis,
315 const bool preferSparseReducers
316 ) const {
317 auto reducer = size_t(-1);
318 auto proceed = [&](const Entry& entry) {
319 if (reducer == size_t(-1)) {
320 reducer = entry.data();
321 return true;
322 }
323 if (preferSparseReducers) {
324 const auto oldTermCount = basis.poly(reducer).termCount();
325 const auto newTermCount = basis.poly(entry.data()).termCount();
326 if (oldTermCount > newTermCount) {
327 reducer = entry.data(); // prefer sparser
328 return true;
329 }
330 if (oldTermCount < newTermCount)
331 return true;
332 } // break ties by age
333
334 if (reducer > entry.data())
335 reducer = entry.data(); // prefer older
336 return true;
337 };
338 auto wrap = lambdaWrap(proceed);
339 mLookup.findAllDivisors(mono, wrap);
340 return reducer;
341 }

395template<
396 template<bool, bool, bool> class Functor,
397 class ReturnType,
398 class Params
399>
400ReturnType staticMonoLookupMake(
401 int type,
402 bool allowRemovals,
403 Params&& params
404) {
405 switch (type) {
406 case 1:
407 if (allowRemovals)
408 return Functor<0, 1, 1>::make(std::forward<Params>(params));
409 else
410 return Functor<0, 0, 1>::make(std::forward<Params>(params));
411
412 case 2:
413 if (allowRemovals)
414 return Functor<1, 1, 1>::make(std::forward<Params>(params));
415 else
416 return Functor<1, 0, 1>::make(std::forward<Params>(params));
417
418 case 3:
419 if (allowRemovals)
420 return Functor<0, 1, 0>::make(std::forward<Params>(params));
421 else
422 return Functor<0, 0, 0>::make(std::forward<Params>(params));
423
424 case 4:
425 if (allowRemovals)
426 return Functor<1, 1, 0>::make(std::forward<Params>(params));
427 else
428 return Functor<1, 0, 0>::make(std::forward<Params>(params));
429
430 default:
431 MATHICGB_ASSERT_NO_ASSUME(false);
432 throw std::runtime_error("Unknown code for monomial data structure");
433 }
