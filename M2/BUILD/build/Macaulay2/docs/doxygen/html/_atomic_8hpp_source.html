190 MATHICGB_INLINE
191 T load(const std::memory_order order) const {
192 switch (order) {
193 case std::memory_order_relaxed:
194 // There are two constraints for memory_order_relaxed. The first
195 // constraint is that if you read *p, then you will never
196 // after that read a value of *p that was stored before the value
197 // you just read, where "before" is in terms of either the same thread
198 // that did the writing or external synchronization of another thread
199 // with this thread. This is automaticaly guaranteed on this platform
200 // and the compiler cannot break this guarantee.
201 //
202 // The second constraint is that writes performed by other threads will
203 // appear "in a timely manner" when reading the written-to memory
204 // location. It is not clear how much time is allowed to elapse before
205 // this constraint is broken, but clearly the written value must turn
206 // up eventually. This constraint could be broken in cases like this:
207 //
208 // while (x.load(std::memory_order_relaxed) == 0) {
209 // // do something that does not write to x
210 // }
211 //
212 // The problem is that the compiler might observe that x is not being
213 // written to inside the loop, so the expression x.load(...) should be
214 // the value for every iteration, so the code can safely be transformed
215 // to:
216 //
217 // if (x.load(std::memory_order_relaxed) == 0) {
218 // while (true) {
219 // // ...
220 // }
221 // }
222 //
223 // So we need to tell the compiler that the value can change due to
224 // code not in view of the current thread of execution. There are two
225 // solutions that I have considered:
226 //
227 // * A compiler read memory barrier. This is overkill because it stops
228 // the compiler from moving all reads across this line.
229 //
230 // * A volatile read. This is pretty much exactly what we need on gcc,
231 // but on MSVC it provides extra guarantees that we do not need here:
232 // http://msdn.microsoft.com/en-us/library/12a04hfd%28v=vs.80%29.aspx
233 //
234 // I do not think there is any best choice on MSVC, so I went for a
235 // volatile read since it is the best choice on GCC.
236 return const_cast<volatile T&>(mValue);
237
238 case std::memory_order_consume: {
239 // Loads in this thread that depend on the loaded value must not be
240 // reordered to before this load. So no DLL reorderings past this
241 // load from after to before (up). So we need a read barrier AFTER the
242 // load. It is a compiler only barrier since x86 and x64 CPUs do not do
243 // DLL reorderings.
244 const auto value = mValue;
245 compilerReadMemoryBarrier();
246 return value;
247 }
248
249 case std::memory_order_acquire: {
250 // Loads in this thread must not be reordered to before this load.
251 // So no LL reorderings past this load from after to before (up).
252 // So we need a read barrier AFTER the load. It is a compiler only
253 // barrier since x86 and x64 CPUs do not do LL reorderings.
254 const auto value = mValue;
255 compilerReadMemoryBarrier();
256 return value;
257 }
258
259 case std::memory_order_seq_cst: {
260 // There must be some global order in which all sequentially consistent
261 // atomic operations are considered to have happened in. On x86 and x64
262 // this is guaranteed by just a normal read as long as all writes use
263 // locked instructions like XHCG. See: http://goo.gl/U8xTK
264 //
265 // We still need to prevent the compiler from reordering the reads,
266 // which is the same constraint as for std::memory_order_acquire.
267 const auto value = mValue;
268 compilerReadMemoryBarrier();
269 return value;
270 }
271
272 case std::memory_order_release: // not available for load
273 case std::memory_order_acq_rel: // not available for load
274 default: // specified value is not a known std::memory_order
275 MATHICGB_UNREACHABLE;
276 }
277 }
278
279 MATHICGB_INLINE
280 void store(const T value, const std::memory_order order) {
281 switch (order) {
282 case std::memory_order_relaxed:
283 // There are no reordering constraints here but we need to tell the
284 // compiler that it must actually write out the value to memory in
285 // a scenario like this:
286 //
287 // x.store(1, std::memory_order_relaxed);
288 // while (true) {}
289 //
290 // So as for relaxed store we need either a volatile access or a memory
291 // barrier. I chose a volatile access for the same reason as for the
292 // store: MSVC has no best choice and for GCC the volatile is perfect.
293 const_cast<volatile T&>(mValue) = value;
294 break;
295
296 case std::memory_order_release:
297 // Stores in this thread must not be reordered to after this store.
298 // So no SS reorderings past this load from before to after (down).
299 // So we need a barrier BEFORE the load. It is a compiler only barrier
300 // since x86 and x64 CPUs do not do SS reorderings.
301 compilerWriteMemoryBarrier();
302 mValue = value;
303 break;
304
305 case std::memory_order_acq_rel:
306 // Combine the guarantees for std::memory_order_acquire and
307 // std::memory_order_release. So no loads moved up past here (SL) and
308 // no stores moved down past here (LL). We need a compiler barrier
309 // BEFORE the load to avoid LL and a CPU (+compiler) barrier AFTER the
310 // load to avoid SL, since x86 and x64 CPUs can in fact do SL
311 // reordering.
312 compilerWriteMemoryBarrier();
313 mValue = value;
314 cpuReadWriteMemoryBarrier();
315 break;
316
317 case std::memory_order_seq_cst:
318 // All operations happen in a globally consistent total order.
319 seqCstStore(value, mValue);
320
321 break;
322
323 case std::memory_order_consume: // not available for store
324 case std::memory_order_acquire: // not available for store
325 default: // specified value is not a known std::memory_order
326 MATHICGB_UNREACHABLE;
327 }
328 }
329
330 private:
331 T mValue;
332 };
