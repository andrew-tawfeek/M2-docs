115void F4MatrixBuilder::buildMatrixAndClear(QuadMatrix& matrix) {
116 MATHICGB_ASSERT(&matrix.ring() == &ring());
117 MATHICGB_LOG_TIME(F4MatrixBuild) <<
118 "\n***** Constructing matrix *****\n";
119
120 if (mTodo.empty()) {
121 matrix.clear();
122 return;
123 }
124
125 // todo: prefer sparse/old reducers among the inputs.
126
127 // Process pending rows until we are done. Note that the methods
128 // we are calling here can add more pending items.
129
130 struct ThreadData {
131 QuadMatrixBuilder builder;
132 MonoRef tmp1;
133 MonoRef tmp2;
134 };
135
136 mtbb::enumerable_thread_specific<ThreadData> threadData(
137 [&](){
138 const mtbb::lock_guard guard(mCreateColumnLock);
139 ThreadData data = {
140 QuadMatrixBuilder(
141 ring(),
142 mMap,
143 mMonomialsLeft,
144 mMonomialsRight,
145 mBuilder.memoryQuantum()
146 ),
147 *monoid().alloc().release(),
148 *monoid().alloc().release()
149 };
150 return data;
151 });
152
153 mtbb::parallel_for_each(mTodo.begin(), mTodo.end(),
154 [&](const RowTask& task, TaskFeeder& feeder)
155 {
156 auto& data = threadData.local();
157 QuadMatrixBuilder& builder = data.builder;
158 const Poly& poly = *task.poly;
159
160 if (task.sPairPoly != 0) {
161 MATHICGB_ASSERT(!task.addToTop);
162 monoid().colons(
163 poly.leadMono(),
164 task.sPairPoly->leadMono(),
165 data.tmp2,
166 data.tmp1
167 );
168 appendRowBottom
169 (poly, data.tmp1, *task.sPairPoly, data.tmp2, data.builder, feeder);
170 return;
171 }
172 if (task.desiredLead == nullptr)
173 monoid().setIdentity(data.tmp1);
174 else
175 monoid().divide(poly.leadMono(), *task.desiredLead, data.tmp1);
176 if (task.addToTop)
177 appendRowTop(data.tmp1, *task.poly, builder, feeder);
178 else
179 appendRowBottom
180 (data.tmp1, false, poly.begin(), poly.end(), builder, feeder);
181 });
182 MATHICGB_ASSERT(!threadData.empty()); // as mTodo empty causes early return
183 // Free the monomials from all the tasks
184 for (auto& mono : mTodo)
185 if (!mono.desiredLead.isNull())
186 monoid().freeRaw(*mono.desiredLead.castAwayConst());
187 mTodo.clear();
188
189 auto& builder = threadData.begin()->builder;
190 for (auto& data : threadData) {
191 if (&data.builder != &builder)
192 builder.takeRowsFrom(data.builder.buildMatrixAndClear());
193 monoid().freeRaw(data.tmp1);
194 monoid().freeRaw(data.tmp2);
195 }
196 matrix = builder.buildMatrixAndClear();
197 threadData.clear();
198
199 {
200 ColReader reader(mMap);
201 matrix.leftColumnMonomials.clear();
202 matrix.rightColumnMonomials.clear();
203 const auto end = reader.end();
204 for (auto it = reader.begin(); it != end; ++it) {
205 const auto p = *it;
206 auto copy = monoid().alloc();
207 monoid().copy(p.second, copy);
208 auto& monos = p.first.left() ?
209 matrix.leftColumnMonomials : matrix.rightColumnMonomials;
210 const auto index = p.first.index();
211 if (monos.size() <= index)
212 monos.resize(index + 1);
213 MATHICGB_ASSERT(monos[index].isNull());
214 monos[index] = copy.release();
215 }
216 }
217#ifdef MATHICGB_DEBUG
218 for (size_t side = 0; side < 2; ++side) {
219 auto& monos = side == 0 ?
220 matrix.leftColumnMonomials : matrix.rightColumnMonomials;
221 for (auto it = monos.begin(); it != monos.end(); ++it) {
222 MATHICGB_ASSERT(!it->isNull());
223 }
224 }
225#endif
226 matrix.sortColumnsLeftRightParallel();
227 mMap.clearNonConcurrent();
228}

230auto F4MatrixBuilder::createColumn(
231 ConstMonoRef monoA,
232 ConstMonoRef monoB,
233 TaskFeeder& feeder
234) -> std::pair<F4MatrixBuilder::LeftRightColIndex, ConstMonoRef> {
235 const mtbb::lock_guard guard(mCreateColumnLock);
236 // see if the column exists now after we have synchronized
237 {
238 const auto found(ColReader(mMap).findProduct(monoA, monoB));
239 if (found.first != 0)
240 return std::make_pair(*found.first, *found.second);
241 }
242
243 // The column really does not exist, so we need to create it
244 monoid().multiply(monoA, monoB, mTmp);
245 if (!monoid().hasAmpleCapacity(*mTmp))
246 mathic::reportError("Monomial exponent overflow in F4MatrixBuilder.");
247
248 // look for a reducer of mTmp
249 const size_t reducerIndex = mBasis.classicReducer(*mTmp);
250 const bool insertLeft = (reducerIndex != static_cast<size_t>(-1));
251
252 // Create the new left or right column
253 auto& colCount = insertLeft ? mLeftColCount : mRightColCount;
254 if (colCount == std::numeric_limits<ColIndex>::max())
255 throw std::overflow_error("Too many columns in QuadMatrix");
256 const auto inserted = mMap.insert
257 (std::make_pair(mTmp.ptr(), LeftRightColIndex(colCount, insertLeft)));
258 ++colCount;
259 MATHICGB_ASSERT(inserted.second);
260 MATHICGB_ASSERT(inserted.first.first != 0);
261
262 // schedule new task if we found a reducer
263 if (insertLeft) {
264 RowTask task = {};
265 task.addToTop = true;
266 task.poly = &mBasis.poly(reducerIndex);
267 task.desiredLead = inserted.first.second;
268 feeder.add(task);
269 }
270
271 return std::make_pair(*inserted.first.first, *inserted.first.second);
272}

274void F4MatrixBuilder::appendRowBottom(
275 ConstMonoRef multiple,
276 const bool negate,
277 const Poly::ConstTermIterator begin,
278 const Poly::ConstTermIterator end,
279 QuadMatrixBuilder& builder,
280 TaskFeeder& feeder
281) {
282 // todo: eliminate the code-duplication between here and appendRowTop.
283
284 auto it = begin;
285updateReader:
286 // Use an on-stack const reader to make it as obvious as possible to the
287 // optimizer's alias analysis that the pointer inside the reader never
288 // changes inside the loop.
289 const ColReader reader(mMap);
290 for (; it != end; ++it) {
291 const auto col = reader.findProduct(it.mono(), multiple);
292 if (col.first == 0) {
293 createColumn(it.mono(), multiple, feeder);
294 goto updateReader;
295 }
296
297 const auto origScalar = it.coef();
298 MATHICGB_ASSERT(!ring().field().isZero(origScalar));
299 const auto maybeNegated =
300 negate ? field().negativeNonZero(origScalar) : origScalar;
301 MATHICGB_ASSERT(maybeNegated < std::numeric_limits<Scalar>::max());
302 builder.appendEntryBottom(*col.first, static_cast<Scalar>(maybeNegated));
303 }
304 builder.rowDoneBottomLeftAndRight();
305}

307void F4MatrixBuilder::appendRowTop(
308 ConstMonoRef multiple,
309 const Poly& poly,
310 QuadMatrixBuilder& builder,
311 TaskFeeder& feeder
312) {
313 auto it = poly.begin();
314 const auto end = poly.end();
315 if ((poly.termCount() % 2) == 1) {
316 ColReader reader(mMap);
317 const auto col = findOrCreateColumn
318 (it.mono(), multiple, reader, feeder);
319 MATHICGB_ASSERT(it.coef() < std::numeric_limits<Scalar>::max());
320 MATHICGB_ASSERT(!field().isZero(it.coef()));
321 builder.appendEntryTop
322 (col.first, static_cast<Scalar>(it.coef()));
323 ++it;
324 }
325updateReader:
326 ColReader colMap(mMap);
327 while (it != end) {
328 MATHICGB_ASSERT(it.coef() < std::numeric_limits<Scalar>::max());
329 MATHICGB_ASSERT(!field().isZero(it.coef()));
330 const auto scalar1 = static_cast<Scalar>(it.coef());
331 const auto mono1 = it.mono();
332
333 auto it2 = it;
334 ++it2;
335 MATHICGB_ASSERT(it2.coef() < std::numeric_limits<Scalar>::max());
336 MATHICGB_ASSERT(!field().isZero(it2.coef()));
337 const auto scalar2 = static_cast<Scalar>(it2.coef());
338 const auto mono2 = it2.mono();
339
340 const auto colPair = colMap.findTwoProducts(mono1, mono2, multiple);
341 if (colPair.first == 0 || colPair.second == 0) {
342 createTwoColumns(mono1, mono2, multiple, feeder);
343 goto updateReader;
344 }
345
346 builder.appendEntryTop(*colPair.first, scalar1);
347 builder.appendEntryTop(*colPair.second, scalar2);
348 it = ++it2;
349 }
350 builder.rowDoneTopLeftAndRight();
351}

353void F4MatrixBuilder::appendRowBottom(
354 const Poly& poly,
355 ConstMonoRef multiply,
356 const Poly& sPairPoly,
357 ConstMonoRef sPairMultiply,
358 QuadMatrixBuilder& builder,
359 TaskFeeder& feeder
360) {
361 MATHICGB_ASSERT(!poly.isZero());
362 auto itA = poly.begin();
363 const auto endA = poly.end();
364
365 MATHICGB_ASSERT(!sPairPoly.isZero());
366 auto itB = sPairPoly.begin();
367 const auto endB = sPairPoly.end();
368
369 // skip leading terms since they cancel
370 MATHICGB_ASSERT(itA.coef() == itB.coef());
371 ++itA;
372 ++itB;
373
374 const ColReader colMap(mMap);
375
376 const auto mulA = multiply;
377 const auto mulB = sPairMultiply;
378 while (true) {
379 // Watch out: we are depending on appendRowBottom to finish the row, so
380 // if you decide not to call that function in case
381 // (itA == itA && itB == endB) then you need to do that yourself.
382 if (itB == endB) {
383 appendRowBottom(mulA, false, itA, endA, builder, feeder);
384 break;
385 }
386 if (itA == endA) {
387 appendRowBottom(mulB, true, itB, endB, builder, feeder);
388 break;
389 }
390
391 coefficient coeff = 0;
392 LeftRightColIndex col;
393 const auto colA = findOrCreateColumn
394 (itA.mono(), mulA, colMap, feeder);
395 const auto colB = findOrCreateColumn
396 (itB.mono(), mulB, colMap, feeder);
397 const auto cmp = monoid().compare(colA.second, colB.second);
398 if (cmp != LT) {
399 coeff = itA.coef();
400 col = colA.first;
401 ++itA;
402 }
403 if (cmp != GT) {
404 coeff = ring().coefficientSubtract(coeff, itB.coef());
405 col = colB.first;
406 ++itB;
407 }
408 MATHICGB_ASSERT(coeff < std::numeric_limits<Scalar>::max());
409 if (!field().isZero(coeff))
410 builder.appendEntryBottom(col, static_cast<Scalar>(coeff));
411 }
412}
