141template <class ARing>
142class SimpleARing : public RingInterface
143{
144 public:
148 class Element : public ElementImpl<typename ARing::ElementType>
149 {
150 typedef typename ARing::ElementType ElementType;
151 typedef ElementImpl<ElementType> Impl;
152
153 public:
154 explicit Element(const ARing &ring)
155 {
156 // without the Impl::, the compiler can't figure out where mValue comes
157 // from
158 ring.init(Impl::mValue);
159 }
160 Element(const ARing &ring, const ElementType& other)
161 {
162 ring.init_set(Impl::mValue,other);
163 }
164 ~Element() { ARing::clear(Impl::mValue); }
165 };

172 class ElementArray
173 {
174 typedef typename ARing::ElementType ElementType;
175 const size_t mSize;
176 std::unique_ptr<ElementType[]> mData;
177
178 public:
179 ElementArray(const ARing &ring, size_t size)
180 : mSize(size), mData(new ElementType[size])
181 {
182 for (size_t i = 0; i < size; i++) ring.init(mData[i]);
183 }
184 ~ElementArray()
185 {
186 for (size_t i = 0; i < mSize; i++) ARing::clear(mData[i]);
187 }
188 ElementType &operator[](size_t idx) { return mData[idx]; }
189 const ElementType &operator[](size_t idx) const { return mData[idx]; }
190 ElementType *data() { return mData.get(); }
191 const ElementType *data() const { return mData.get(); }
