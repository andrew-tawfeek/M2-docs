40bool testPolyParse(PolyRing* R, std::string s, bool withComponent)
41{
42 // parse poly, then see if it matches the orig string
43 std::istringstream i(s);
44 Scanner in(i);
45 auto f = MathicIO<>().readPolyDoNotOrder(*R, withComponent, in);
46 std::ostringstream o;
47 MathicIO<>().writePoly(f, withComponent, o);
48 // std::cout << "orig = " << s << std::endl;
49 // std::cout << "f = " << o.str() << std::endl;
50 return o.str() == s;
51}

206void testMonomialOps(const PolyRing* R, std::string s1, std::string s2)
207{
208 Monomial m1 = stringToMonomial(R, s1);
209 Monomial m2 = stringToMonomial(R, s2);
210 // Monomial m3 = stringToMonomial(R, "abcdef<0>");
211
212
213 Monomial m4 = R->allocMonomial();
214 Monomial lcm = R->allocMonomial();
215 Monomial m8 = R->allocMonomial();
216 Monomial m1a = R->allocMonomial();
217 Monomial m2a = R->allocMonomial();
218 Monomial m1b = R->allocMonomial();
219 Monomial m2b = R->allocMonomial();
220
221 R->monomialMult(m1,m2,m4);
222 R->monomialLeastCommonMultiple(m1,m2,lcm);
223
224 // lcm(m1,m2)/m1, lcm(m1,m2)/m2: relatively prime
225 EXPECT_TRUE(R->monomialIsDivisibleBy(lcm, m1));
226 EXPECT_TRUE(R->monomialIsDivisibleBy(lcm, m2));
227 R->monomialDivide(lcm, m1, m1a);
228 R->monomialDivide(lcm, m2, m2a);
229 EXPECT_TRUE(R->monomialRelativelyPrime(m1a,m2a));
230
231 EXPECT_TRUE(R->monomialIsDivisibleBy(lcm, m1a));
232 EXPECT_TRUE(R->monomialIsDivisibleBy(lcm, m2a));
233 R->monomialDivide(lcm, m1a, m1b);
234 R->monomialDivide(lcm, m2a, m2b);
235 EXPECT_TRUE(R->monomialEQ(m1, m1b));
236 EXPECT_TRUE(R->monomialEQ(m2, m2b));
237 R->monomialMult(m1a, m2a, m8);
238
239 size_t supp1 = R->monomialSizeOfSupport(m1a);
240 size_t supp2 = R->monomialSizeOfSupport(m2a);
241 size_t supp = R->monomialSizeOfSupport(m8);
242 EXPECT_EQ(supp1+supp2, supp)
243 << monomialToString(R,m1) << " " << monomialToString(R,m2) << "\n"
244 << monomialToString(R,m1a) << " " << monomialToString(R,m2a) << " "
245 << monomialToString(R,m8);
