62 DMat() : mRing(nullptr), mNumRows(0), mNumColumns(0), mArray(nullptr) {}
63 DMat(const ACoeffRing& R,
64 size_t nrows,
65 size_t ncols)
66 : mRing(&R),
67 mNumRows(nrows),
68 mNumColumns(ncols)
69 {
70 size_t len = mNumRows * mNumColumns;
71 if (len == 0)
72 {
73 mArray = nullptr;
74 }
75 else
76 {
77 mArray = newarray(ElementType,len);
78 for (size_t i = 0; i < len; i++)
79 {
80 ring().init(mArray[i]);
81 ring().set_zero(mArray[i]);
82 }
83 ElementType* next = mArray;
84 for (size_t r=0; r < mNumRows; ++r)
85 {
86 mRowPointers.push_back(next);
87 next += mNumColumns;
88 }
89 }
90 }
91 DMat(const DMat<ACoeffRing>& M)
92 : mRing(&M.ring()),
93 mNumRows(M.numRows()),
94 mNumColumns(M.numColumns())
95 {
96 size_t len = mNumRows * mNumColumns;
97 if (len == 0)
98 mArray = nullptr;
99 else
100 {
101 mArray = newarray(ElementType,len);
102 for (size_t i = 0; i < len; i++)
103 {
104 ring().init(mArray[i]);
105 ring().set_zero(mArray[i]);
106 }
107 ElementType* next = mArray;
108 for (size_t r=0; r < mNumRows; ++r)
109 {
110 mRowPointers.push_back(next);
111 next += mNumColumns;
112 }
113 for (size_t r = 0; r < mNumRows; ++r)
114 for (size_t c = 0; c < mNumColumns; ++c)
115 ring().init_set(entry(r,c), M.entry(r,c));
116 }
117 }

119 ~DMat()

121 size_t len = mNumRows * mNumColumns;

123 if (mArray != nullptr) freemem(mArray);
