52engine_RawRingElementArrayOrNull rawRoots(const RingElement *p,
53 long prec,
54 int unique)
55{
56 (void) unique;
57 const Ring *R = p->get_ring();
58 const PolynomialRing *P = R->cast_to_PolynomialRing();
59 const Monoid *M = P->getMonoid();
60 if (P == nullptr) {
61 ERROR("expected a polynomial ring");
62 return nullptr;
63 }
64 if (P->n_vars() != 1) {
65 ERROR("expected a univariate polynomial ring");
66 return nullptr;
67 }
68 if (p->is_zero()) {
69 ERROR("expected a nonzero polynomial");
70 return nullptr;
71 }
72 const Ring *K = P->getCoefficients();
73 int lodeg,hideg; // lowest,highest degree
74 P->degree_of_var(0,p->get_value(),lodeg,hideg);
75 if (prec == -1)
76 prec = (K->get_precision() == 0 ? 53 : K->get_precision());
77
78 mps_context *s = mps_context_new ();
79 mps_monomial_poly *mps_p = mps_monomial_poly_new (s, hideg);
80 mps_context_select_algorithm(s, MPS_ALGORITHM_SECULAR_GA);
81
82 const auto ID = K->ringID();
83 for (Nterm *t = p->get_value(); t != nullptr; t = t->next) {
84 int deg;
85 M->to_expvector(t->monom, &deg); // number of vars = 1
86 if (ID==M2::ring_RR or ID==M2::ring_CC) {
87 cc_doubles_struct cc;
88 if (ID==M2::ring_RR) {
89 cc.re = t->coeff.get_double();
90 cc.im = 0;
91 } else cc = *t->coeff.get_cc_doubles();
92 mps_monomial_poly_set_coefficient_d (s, mps_p, deg, cc.re, cc.im);
93 } else if (ID==M2::ring_RRR or ID==M2::ring_CCC) {
94 mpc_t mpc_cc;
95 mpc_init2(mpc_cc,K->get_precision());
96 if (ID==M2::ring_RRR) {
97 mpfr_get_f(mpc_cc->r,t->coeff.get_mpfr(),MPFR_RNDN);
98 } else {
99 mpfr_get_f(mpc_cc->r,&t->coeff.get_cc()->re,MPFR_RNDN);
100 mpfr_get_f(mpc_cc->i,&t->coeff.get_cc()->im,MPFR_RNDN);
101 }
102 mps_monomial_poly_set_coefficient_f (s, mps_p, deg, mpc_cc);
103 mpc_clear(mpc_cc);
104 } else if ((ID==M2::ring_old and K->is_ZZ()) or ID==M2::ring_QQ) {
105 mpq_t q, zero;
106 mpq_init(zero);
107 mpq_init(q);
108 if (ID==M2::ring_QQ)
109 mpq_set(q, t->coeff.get_mpq());
110 else mpq_set_z(q, t->coeff.get_mpz());
111 mps_monomial_poly_set_coefficient_q(s, mps_p, deg, q, zero);
112 mpq_clear(q);
113 mpq_clear(zero);
114 } else {
115 ERROR("'roots' expects coefficients in ZZ, QQ, RR or CC");
116 return nullptr;
117 }
118 }
119
120 /* Set the input polynomial */
121 mps_context_set_input_poly (s, MPS_POLYNOMIAL (mps_p));
122 mps_context_set_output_prec (s, prec);
123 mps_context_set_output_goal (s, MPS_OUTPUT_GOAL_APPROXIMATE);
124
125 /* Actually solve the polynomial */
126 mps_mpsolve (s);
127
128 // result to be returned
129 engine_RawRingElementArrayOrNull result = nullptr;
130 result = getmemarraytype(engine_RawRingElementArray, hideg);
131 result->len = static_cast<int>(hideg);
132
134 if (prec <= 53) {
135 /* Allocate space to hold the results. We check only floating point results in here */
136 cplx_t *result_mps = cplx_valloc (hideg);
137
138 /* Save roots computed in the vector results */
139 mps_context_get_roots_d (s, &result_mps, nullptr);
140
141 /* Print out roots */
142 // for (int i = 0; i < hideg; i++) {
143 // cplx_out_str (stdout, result_mps[i]);
144 // printf ("\n");
145 // }
146
148 // copy to mps_result to result
149
150 const Ring *C = IM2_Ring_CCC(prec);
151 for (int i = 0; i < hideg; i++) {
152 auto& mps_root = result_mps[i];
153 ring_elem m2_root;
154 C->from_complex_double(cplx_Re(mps_root), cplx_Im(mps_root),
155 m2_root);
156 result->array[i] = RingElement::make_raw(C, m2_root);
157 }
158
159 free (result_mps);
160 } else { // prec > 53
161 mpc_t *roots = nullptr;
162 rdpe_t *radii = nullptr;
163
164 mps_context_get_roots_m (s, &roots, &radii);
165 /* Sort roots in the order of increasing real part */
166 //sort_roots(hideg, roots, radii);

169 // printf ("\n");

172 const Ring *C = IM2_Ring_CCC(prec);
173 const M2::ARingCCC C0(prec);
174
175 M2::ARingCCC::Element cc (C0);
176 for (int i = 0; i < hideg; i++) {
177 auto& mps_root = roots[i];
178 mpfr_set_f(&cc.value().re,mpc_Re(mps_root),MPFR_RNDN);
179 mpfr_set_f(&cc.value().im,mpc_Im(mps_root),MPFR_RNDN);
180 ring_elem m2_root;
181 C0.to_ring_elem(m2_root, cc);
182 result->array[i] = RingElement::make_raw(C, m2_root);
183 }
184
185 free(roots);
186 free(radii);
187 }
188
189 mps_monomial_poly_free(s, (mps_polynomial *)mps_p);
190 mps_context_free(s);
191 return result;
192}
