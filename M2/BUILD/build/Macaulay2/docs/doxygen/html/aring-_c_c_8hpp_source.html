214 void invert(ElementType& res, const ElementType& a) const
215 // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
216 {
217 ElementType result;
218 if (fabs(a.re) >= fabs(a.im))
219 {
220 double p = a.im / a.re;
221 double denom = a.re + p * a.im;
222 result.re = 1.0 / denom;
223 result.im = -p / denom;
224 }
225 else
226 {
227 double p = a.re / a.im;
228 double denom = a.im + p * a.re;
229 result.re = p / denom;
230 result.im = -1.0 / denom;
231 }
232 set(res, result);
233 }

307 void divide(ElementType& res,
308 const ElementType& a,
309 const ElementType& b) const
310 {
311 RealElementType p, denom; // double
312
313 ElementType result;
314 if (fabs(b.re) >= fabs(b.im))
315 {
316 p = b.im / b.re;
317 denom = b.re + p * b.im;
318 result.re = (a.re + p * a.im) / denom;
319 result.im = (a.im - p * a.re) / denom;
320 }
321 else
322 {
323 p = b.re / b.im;
324 denom = b.im + p * b.re;
325 result.re = (a.im + p * a.re) / denom;
326 result.im = (p * a.im - a.re) / denom;
327 }
328 set(res, result);
329 }

341 void power(ElementType& result, const ElementType& a, int n) const
342 {
343 ElementType curr_pow;
344 init(curr_pow);
345 set_from_long(result, 1);
346 if (n == 0)
347 {
348 }
349 else if (n < 0)
350 {
351 n = -n;
352 invert(curr_pow, a);
353 }
354 else
355 {
356 set(curr_pow, a);
357 }
358 while (n > 0)
359 {
360 if (n % 2)
361 {
362 mult(result, result, curr_pow);
363 }
364 n = n / 2;
365 mult(curr_pow, curr_pow, curr_pow);
366 }
367 clear(curr_pow);
368 }
