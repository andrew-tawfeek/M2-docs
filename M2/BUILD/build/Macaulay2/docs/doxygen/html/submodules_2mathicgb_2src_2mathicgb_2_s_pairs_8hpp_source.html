119 // Returns true if Buchberger's second criterion for eliminating useless
120 // S-pairs applies to the pair (a,b). Define
121 // l(a,b) = lcm(lead(a), lead(b)).
122 // The criterion says that if there is some other basis element c such that
123 // lead(c)|l(a,b)
124 // and
125 // l(a,c) has a representation and
126 // l(b,c) has a representation
127 // then (a,b) has a representation too, so we do not need to reduce it.
128 //
129 // This criterion is easy to get wrong in cases where
130 // l(a,b) = l(a,c) = l(b,c)
131 // since then there is the potential to erroneously eliminate all the three
132 // pairs among a,b,c on the assumption that the other two pairs will reduce
133 // to zero. In fact only one of the pairs should be eliminated. We leave such
134 // cases to the advanced criterion, except if an S-pair has already been
135 // eliminated - in that case we do not check to see if the lcm's are the same
136 // as it is not necessary to do so.
137 bool simpleBuchbergerLcmCriterion(
138 size_t a,
139 size_t b,
140 BareMonoid::ConstMonoRef lcmAB
141 ) const;
142
143 // As the non-slow version, but uses simpler and slower code.
144 bool simpleBuchbergerLcmCriterionSlow(size_t a, size_t b) const;
145
146 // Improves on Buchberger's second criterion by using connection in a graph
147 // to determine if an S-pair can be eliminated. This can eliminate some pairs
148 // that cannot be eliminated by looking at any one triple of generators.
149 //
150 // The algorithm is based on considering an undirected graph G.
151 // Each vertex of G represents a basis element whose lead monomial divides
152 // lcmAB. There is an edge (c,d) if lcm(c,d) != lcm(a,b) or if (c,d) has
153 // been eliminated. It is a theorem that if there is a path from a to b
154 // in G then (a,b) is a useless S-pair that can be eliminated.
155 bool advancedBuchbergerLcmCriterion
156 (size_t a, size_t b, BareMonoid::ConstMonoRef lcmAB) const;
157
158 // As the non-slow version, but uses simpler and slower code.
159 bool advancedBuchbergerLcmCriterionSlow(size_t a, size_t b) const;

182 bool compare(
183 size_t colA, size_t rowA, OrderMonoid::ConstMonoPtr a,
184 size_t colB, size_t rowB, OrderMonoid::ConstMonoPtr b
185 ) const {
186 const auto cmp = orderMonoid().compare(*a, *b);
187 if (cmp == GT)
188 return true;
189 if (cmp == LT)
190 return false;
191
192 const bool aRetired = mBasis.retired(rowA) || mBasis.retired(colA);
193 const bool bRetired = mBasis.retired(rowB) || mBasis.retired(colB);
194 if (aRetired || bRetired)
195 return !bRetired;
196
197 if (mPreferSparseSPairs) {
198 const auto termCountA =
199 mBasis.basisElement(colA).termCount() +
200 mBasis.basisElement(rowA).termCount();
201 const auto termCountB =
202 mBasis.basisElement(colB).termCount() +
203 mBasis.basisElement(rowB).termCount();
204 if (termCountA > termCountB)
205 return true;
206 if (termCountA < termCountB)
207 return false;
208 }
209 return colA + rowA > colB + rowB;
210 }
