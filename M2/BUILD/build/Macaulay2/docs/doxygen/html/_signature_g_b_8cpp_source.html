28SignatureGB::SignatureGB(
29 Basis&& basis,
30 Processor&& processor,
31 Reducer::ReducerType reductiontyp,
32 int divlookup_type,
33 int montable_type,
34 bool postponeKoszul,
35 bool useBaseDivisors,
36 bool preferSparseReducers,
37 bool useSingularCriterionEarly,
38 size_t queueType
39):
40 mBreakAfter(0),
41 mPrintInterval(0),
42 R(basis.getPolyRing()),
43 mPostponeKoszul(postponeKoszul),
44 mUseBaseDivisors(useBaseDivisors),
45 stats_sPairSignaturesDone(0),
46 stats_sPairsDone(0),
47 stats_koszulEliminated(0),
48 stats_SignatureCriterionLate(0),
49 stats_relativelyPrimeEliminated(0),
50 stats_pairsReduced(0),
51 stats_nsecs(0.0),
52 GB(make_unique<SigPolyBasis>(*R, divlookup_type, montable_type, preferSparseReducers)),
53 mKoszuls(R->monoid()),
54 Hsyz(ModuleMonoSet::make(R->monoid(), montable_type, basis.size(), !mPostponeKoszul)),
55 Hsyz2(ModuleMonoSet::make(R->monoid(), montable_type, basis.size(), !mPostponeKoszul)),
56 reducer(Reducer::makeReducer(reductiontyp, *R)),
57 SP(make_unique<SigSPairs>(R, GB.get(), Hsyz.get(), reducer.get(), mPostponeKoszul, mUseBaseDivisors, useSingularCriterionEarly, queueType))
58{
59 mProcessor = make_unique<MonoProcessor<Monoid>>(std::move(processor));
60 if (basis.size() > std::numeric_limits<Component>::max())
61 mathic::reportError("Dimension of module too large.");
62 const auto componentCount = Component(basis.size());
63 mProcessor->setComponentCount(componentCount);
64
65 // Populate GB
66 for (Component j = 0; j < componentCount; j++)
67 GB->addComponent();
68
69 for (Component i = 0; i < componentCount; i++) {
70 auto g = make_unique<Poly>(*basis.getPoly(i));
71 g->makeMonic();
72
73 auto sig = monoid().alloc();
74 monoid().setIdentity(*sig);
75 monoid().setComponent(i, *sig);
76 mProcessor->preprocess(*sig);
77
78 GB->insert(std::move(sig), std::move(g));
79 }
80
81 // Populate SP
82 for (size_t i = 0; i < basis.size(); i++)
83 SP->newPairs(i);
84}

144bool SignatureGB::processSPair
145 (Mono sig, const SigSPairs::PairContainer& pairs)
146{
147 MATHICGB_ASSERT(!pairs.empty());
148 (void)pairs;
149
150 // the module term to reduce is multiple * GB->getSignature(gen)
151 size_t gen = GB->minimalLeadInSig(*sig);
152 MATHICGB_ASSERT(gen != static_cast<size_t>(-1));
153 monomial multiple = R->allocMonomial();
154 monoid().divide(GB->signature(gen), *sig, multiple);
155 GB->basis().usedAsStart(gen);
156
157 // reduce multiple * GB->getSignature(gen)
158 auto f = reducer->regularReduce(*sig, multiple, gen, *GB);
159
160 R->freeMonomial(multiple);
161
162 if (f == nullptr) { // singular reduction
163 MATHICGB_LOG(SigSPairFinal) << " eliminated by singular criterion.\n";
164 return true;
165 }
166
167 if (f->isZero()) { // reduction to zero
168 // todo: what are the correct ownership relations here?
169 auto ptr = sig.release();
170 Hsyz->insert(*ptr);
171 SP->setKnownSyzygies(mSpairTmp);
172 MATHICGB_LOG(SigSPairFinal) << " s-reduced to zero.\n";
173 return false;
174 }
175
176 // new basis element
177 MATHICGB_ASSERT(!GB->isSingularTopReducibleSlow(*f, *sig));
178
179 GB->insert(std::move(sig), std::move(f));
180
181 MATHICGB_LOG(SigSPairFinal) << " s-reduced to new basis element.\n";
182
183 MATHICGB_IF_STREAM_LOG(SigBasisChanged) {
184 stream << "New signature basis element. The basis is now:\n";
185 GB->displayFancy(stream, *mProcessor);
186 };
187
188 SP->newPairs(GB->size()-1);
189
190 return true;
191}

193bool SignatureGB::step() {
194 auto sig = SP->popSignature(mSpairTmp);
195 if (sig.isNull())
196 return false;
197 ++stats_sPairSignaturesDone;
198 stats_sPairsDone += mSpairTmp.size();
199
200 MATHICGB_IF_STREAM_LOG(SigSPairFinal) {
201 stream << "Final processing of signature ";
202 R->monomialDisplay(stream, Monoid::toOld(*sig));
203 stream << '\n';
204 };
205
206 if (Hsyz->member(*sig)) {
207 ++stats_SignatureCriterionLate;
208 SP->setKnownSyzygies(mSpairTmp);
209 MATHICGB_LOG(SigSPairFinal) << " eliminated by signature criterion.\n";
210 return true;
211 }
212
213 while (!mKoszuls.empty() && R->monoid().lessThan(mKoszuls.top(), *sig))
214 mKoszuls.pop();
215
216 if (!mKoszuls.empty() && R->monoid().equal(mKoszuls.top(), *sig)) {
217 ++stats_koszulEliminated;
218 // This signature is of a syzygy that is not in Hsyz, so add it
219 // todo: what are the correct ownership relations here?
220 auto ptr = sig.release();
221 Hsyz->insert(*ptr);
222 SP->setKnownSyzygies(mSpairTmp);
223 MATHICGB_LOG(SigSPairFinal) << " eliminated by Koszul criterion.\n";
224 return true;
225 }
226
227 if (mPostponeKoszul) {
228 // Relatively prime check
229 for (auto it = mSpairTmp.begin(); it != mSpairTmp.end(); ++it) {
230 auto a = GB->leadMono(it->first);
231 auto b = GB->leadMono(it->second);
232 if (monoid().relativelyPrime(a, b)) {
233 ++stats_relativelyPrimeEliminated;
234 // todo: what are the correct ownership relations here?
235 auto ptr = sig.release();
236 Hsyz->insert(*ptr);
237 SP->setKnownSyzygies(mSpairTmp);
238 MATHICGB_LOG(SigSPairFinal) <<
239 " eliminated by relatively prime criterion.\n";
240 return true;
241 }
242 }
243 }
244#ifdef MATHICGB_DEBUG
245 for (auto it = mSpairTmp.begin(); it != mSpairTmp.end(); ++it) {
246 auto a = GB->leadMono(it->first);
247 auto b = GB->leadMono(it->second);
248 MATHICGB_ASSERT(!monoid().relativelyPrime(a, b));
249 }
250#endif
251
252 // Reduce the pair
253 ++stats_pairsReduced;
254 if (!processSPair(std::move(sig), mSpairTmp) || !mPostponeKoszul)
255 return true;
256 for (auto it = mSpairTmp.begin(); it != mSpairTmp.end(); ++it) {
257 std::pair<size_t, size_t> p = *it;
258 if (GB->ratioCompare(p.first, p.second) == LT)
259 std::swap(p.first, p.second);
260
261 auto greaterSig = GB->signature(p.first);
262 auto smallerLead = GB->leadMono(p.second);
263 monomial koszul = R->allocMonomial();
264 monoid().multiply(greaterSig, smallerLead, koszul);
265 if (Hsyz->member(koszul))
266 R->freeMonomial(koszul);
267 else
268 mKoszuls.push(koszul);
269 }
270 return true;
271}

275 GB->getMemoryUse() +

280 SP->getMemoryUse() +
281 mKoszuls.getMemoryUse();
282}
