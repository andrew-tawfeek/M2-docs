113template <typename T, long int NSLAB>
114T *ResMemoryBlock<T, NSLAB>::reserve(int len)
115{
116 if (next_free + len > current_slab->block + NSLAB)
117 {
118 if (current_slab->next == nullptr)
119 {
120 last_slab->next = new_slab();
121 last_slab = last_slab->next;
122 current_slab = last_slab;
123 }
124 else
125 {
126 current_slab = current_slab->next;
127 }
128 next_free = current_slab->block;
129 }
130 return next_free;
131}
