33 void createColumns(const char* left, const char* right, QuadMatrixBuilder& b)
34 {
35 const PolyRing& ring = b.ring();
36 {
37 std::istringstream in(left);
38 Scanner scanner(in);
39 auto p = MathicIO<>().readPolyDoNotOrder(ring, true, scanner);
40 size_t colCount = 0;
41 for (auto it = p.begin(); it != p.end(); ++it) {
42 QuadMatrixBuilder::LeftRightColIndex lrCol =
43 b.createColumnLeft(it.mono()).first;
44 ASSERT_TRUE(lrCol.left());
45 ASSERT_FALSE(lrCol.right());
46 auto col = lrCol.leftIndex();
47 ASSERT_EQ(col, lrCol.index());
48 ASSERT_EQ(colCount, col);
49 ++colCount;
50 }
51 }
52 {
53 std::istringstream in(right);
54 Scanner scanner(in);
55 auto p = MathicIO<>().readPolyDoNotOrder(ring, true, scanner);
56 size_t colCount = 0;
57 for (auto it = p.begin(); it != p.end(); ++it) {
58 QuadMatrixBuilder::LeftRightColIndex lrCol =
59 b.createColumnRight(it.mono()).first;
60 ASSERT_TRUE(lrCol.right());
61 ASSERT_FALSE(lrCol.left());
62 auto col = lrCol.rightIndex();
63 ASSERT_EQ(col, lrCol.index());
64 ASSERT_EQ(colCount, col);
65 ++colCount;
66 }
67 }
68 }
69}

133TEST(QuadMatrixBuilder, ColumnQuery) {
134 std::unique_ptr<PolyRing> ring(ringFromString("32003 6 1\n1 1 1 1 1 1"));
135 QuadMatrixBuilder::Map map(*ring);
136 QuadMatrixBuilder::Monomials monoLeft;
137 QuadMatrixBuilder::Monomials monoRight;
138 QuadMatrixBuilder b(*ring, map, monoLeft, monoRight);
139 createColumns("a<1>+<0>", "b<0>+c<0>+bc<0>", b);
140
141 // coefficient 1X=left, 2X=right, 30=not there, % 10 = column index
142 std::istringstream in
143 ("10a<1>+11<0>+20b<0>+21c<0>+22bc<0>+30ab<0>+30e<0>+10a<1>");
144 Scanner scanner(in);
145 auto p = MathicIO<>().readPolyDoNotOrder(b.ring(), true, scanner);
146 for (auto it = p.begin(); it != p.end(); ++it) {
147 const QuadMatrixBuilder::LeftRightColIndex* col =
148 MonomialMap<QuadMatrixBuilder::LeftRightColIndex>::Reader(map).
149 find(it.mono()).first;
150 if (it.coef() / 10 == 3)
151 ASSERT_EQ(col, static_cast<void*>(0));
152 else {
153 ASSERT_TRUE(col != static_cast<void*>(0));
154 ASSERT_EQ(it.coef() % 10, col->index());
155 if (it.coef() / 10 == 2)
156 ASSERT_TRUE(col->right());
157 else
158 ASSERT_TRUE(col->left());
159 }
160 }
161}
