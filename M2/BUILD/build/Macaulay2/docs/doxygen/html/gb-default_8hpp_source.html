27 typedef int gbelem_type;
28 // ELEM_IN_RING: a bit set. If this is set, then the next two bits need to
29 // off.
30 // ELEM_MINGEN: a bit set: 0 means it is provably not needed to
31 // generated the ideal or submodule. 1 means it might be a minimal
32 // generator.
33 // NOTE: often the min gens obtained this way are not so good. Exceptions
34 // are
35 // for graded submodules. Then the meaning of 1 is: it IS a min gen.
36 // ELEM_MINGB: a bit set: 0 means is not part of the min gb (so far). 1 means
37 // it is.
38 // SO: possible values are 0, 1, 2, 4, 6.
39 // Test using code like this:
40 // if (g->minlevel & ELEM_MINGEN) { .. do this if it is a possible min gen
41 // .. } else { .. if not .. };
42
43 struct gbelem
44 {
45 POLY g;
46 int deg;
47 int gap; // the homogenizing degree, called "alpha", or the "ecart". It is
48 // deg g - deg lead term of g.
49 int size; // number of monomials, when the element is first inserted. After
50 // auto reduction, the number can change. It is not yet clear if we should
51 // change this value...
52 exponents_t lead; // -1..nvars-1, the -1 part is the component
53 gbelem_type minlevel;
54 };
