171void F4Reducer::classicReduceSPolySet(
172 std::vector<std::pair<size_t, size_t>>& spairs,
173 const PolyBasis& basis,
174 std::vector<std::unique_ptr<Poly>>& reducedOut
175) {
176 if (spairs.size() <= 1) {
177 if (tracingLevel >= 2)
178 std::cerr << "F4Reducer: Using fall-back reducer for "
179 << spairs.size() << " S-pairs.\n";
180 mFallback->classicReduceSPolySet(spairs, basis, reducedOut);
181 return;
182 }
183 reducedOut.clear();
184
185 MATHICGB_ASSERT(!spairs.empty());
186 if (tracingLevel >= 2)
187 std::cerr << "F4Reducer: Reducing " << spairs.size() << " S-polynomials.\n";
188
189 SparseMatrix reduced;
190 QuadMatrix::Monomials monomials;
191 {
192 QuadMatrix qm(basis.ring());
193 {
194 if (mType == OldType) {
195 F4MatrixBuilder builder(basis, mMemoryQuantum);
196 for (const auto& spair : spairs)
197 builder.addSPolynomialToMatrix
198 (basis.poly(spair.first), basis.poly(spair.second));
199 builder.buildMatrixAndClear(qm);
200 } else {
201 F4MatrixBuilder2 builder(basis, mMemoryQuantum);
202 for (const auto& spair : spairs)
203 builder.addSPolynomialToMatrix
204 (basis.poly(spair.first), basis.poly(spair.second));
205 builder.buildMatrixAndClear(qm);
206 }
207 }
208 MATHICGB_LOG_INCREMENT_BY(F4MatrixRows, qm.rowCount());
209 MATHICGB_LOG_INCREMENT_BY(F4MatrixTopRows, qm.topLeft.rowCount());
210 MATHICGB_LOG_INCREMENT_BY(F4MatrixBottomRows, qm.bottomLeft.rowCount());
211 MATHICGB_LOG_INCREMENT_BY(F4MatrixEntries, qm.entryCount());
212 saveMatrix(qm);
213 reduced = F4MatrixReducer(basis.ring().charac()).
214 reducedRowEchelonFormBottomRight(qm);
215 monomials = std::move(qm.rightColumnMonomials);
216 for (auto& mono : qm.leftColumnMonomials)
217 monoid().freeRaw(mono.castAwayConst());
218 }
219
220 for (SparseMatrix::RowIndex row = 0; row < reduced.rowCount(); ++row) {
221 auto p = make_unique<Poly>(basis.ring());
222 reduced.rowToPolynomial(row, monomials, *p);
223 reducedOut.push_back(std::move(p));
224 }
225 for (auto& mono : monomials)
226 monoid().freeRaw(mono.castAwayConst());
227}

229void F4Reducer::classicReducePolySet(
230 const std::vector< std::unique_ptr<Poly> >& polys,
231 const PolyBasis& basis,
232 std::vector< std::unique_ptr<Poly> >& reducedOut
233) {
234 if (polys.size() <= 1) {
235 if (tracingLevel >= 2)
236 std::cerr << "F4Reducer: Using fall-back reducer for "
237 << polys.size() << " polynomials.\n";
238 mFallback->classicReducePolySet(polys, basis, reducedOut);
239 return;
240 }
241
242 reducedOut.clear();
243
244 MATHICGB_ASSERT(!polys.empty());
245 if (tracingLevel >= 2)
246 std::cerr << "F4Reducer: Reducing " << polys.size() << " polynomials.\n";
247
248 SparseMatrix reduced;
249 QuadMatrix::Monomials monomials;
250 {
251 QuadMatrix qm(ring());
252 {
253 if (mType == OldType) {
254 F4MatrixBuilder builder(basis, mMemoryQuantum);
255 for (const auto& poly : polys)
256 builder.addPolynomialToMatrix(*poly);
257 builder.buildMatrixAndClear(qm);
258 } else {
259 F4MatrixBuilder2 builder(basis, mMemoryQuantum);
260 for (const auto& poly : polys)
261 builder.addPolynomialToMatrix(*poly);
262 builder.buildMatrixAndClear(qm);
263 }
264 }
265 MATHICGB_LOG_INCREMENT_BY(F4MatrixRows, qm.rowCount());
266 MATHICGB_LOG_INCREMENT_BY(F4MatrixTopRows, qm.topLeft.rowCount());
267 MATHICGB_LOG_INCREMENT_BY(F4MatrixBottomRows, qm.bottomLeft.rowCount());
268 MATHICGB_LOG_INCREMENT_BY(F4MatrixEntries, qm.entryCount());
269 saveMatrix(qm);
270 reduced = F4MatrixReducer(basis.ring().charac()).
271 reducedRowEchelonFormBottomRight(qm);
272 monomials = std::move(qm.rightColumnMonomials);
273 for (auto& mono : qm.leftColumnMonomials)
274 monoid().freeRaw(mono.castAwayConst());
275 }
276
277 if (tracingLevel >= 2 && false)
278 std::cerr << "F4Reducer: Extracted " << reduced.rowCount()
279 << " non-zero rows\n";
280
281 for (SparseMatrix::RowIndex row = 0; row < reduced.rowCount(); ++row) {
282 auto p = make_unique<Poly>(basis.ring());
283 reduced.rowToPolynomial(row, monomials, *p);
284 reducedOut.push_back(std::move(p));
285 }
286 for (auto& mono : monomials)
287 monoid().freeRaw(mono.castAwayConst());
288}

315void F4Reducer::saveMatrix(const QuadMatrix& matrix) {
316 if (mStoreToFile.empty())
317 return;
318 const auto entryCount = matrix.entryCount();
319 if (mMinEntryCountForStore > entryCount)
320 return;
321 ++mMatrixSaveCount;
322 std::ostringstream fileName;
323 fileName << mStoreToFile << '-' << mMatrixSaveCount << ".qmat";
324 if (tracingLevel > 2)
325 std::cerr << "F4Reducer: Saving matrix to " << fileName.str() << '\n';
326
327 CFile file(fileName.str(), "wb");
328 matrix.write
329 (static_cast<SparseMatrix::Scalar>(mRing.charac()), file.handle());
330}
