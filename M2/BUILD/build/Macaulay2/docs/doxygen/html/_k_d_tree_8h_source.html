329 void KDTree<C>::resetNumberOfChangesTillRebuild() {
330 const C& conf = getConfiguration();
331 if (conf.getUseDivisorCache())
332 _divisorCache = 0;

334 return;
335 MATHIC_ASSERT(conf.getRebuildRatio() > 0);
336 _changesTillRebuild = std::max

341 template<class C>
342 void KDTree<C>::reportChanges(size_t additions, size_t removals) {
343 if (getConfiguration().getUseDivisorCache() && (additions | removals) != 0)
344 _divisorCache = 0;
345 if (reportChangesRebuild(additions, removals))
346 rebuild();
347 }

349 template<class C>
350 bool KDTree<C>::reportChangesRebuild(size_t additions, size_t removals) {
351 // note how negative value/overflow of _changesTillRebuild cannot
352 // happen this way.
353 MATHIC_ASSERT(removals <= size() + additions);
354 _size = (size() + additions) - removals;
355 if (!getConfiguration().getDoAutomaticRebuilds())
356 return false;
357 const size_t changesMadeCount = additions + removals;
358 if (_changesTillRebuild > changesMadeCount) {
359 _changesTillRebuild -= changesMadeCount;
360 return false;
361 } else
362 return true;
363 }
