157 std::pair<std::pair<const mapped_type*, ConstMonoPtr>, bool>
158 insert(const value_type& value) {
159 const mtbb::lock_guard lockGuard(mInsertionMutex);
160 // const std::lock_guard<std::mutex> lockGuard(mInsertionMutex);
161 //const mtbb::mutex::scoped_lock lockGuard(mInsertionMutex);
162
163 // We can load mMap as std::memory_order_relaxed because we have already
164 // synchronized with all other mutators by locking mInsertionMutex;
165 auto map = mMap.load(std::memory_order_relaxed);
166
167 // this is a loop since it is possible to set the growth factor and
168 // the initial size so low that several rounds are required. This should
169 // only happen when debugging as otherwise such low parameters are
170 // not a good idea.
171 while (mCapacityUntilGrowth == 0) {
172 // Resize the table by making a bigger one and using that instead.
173 if (map->bucketCount() > // check overflow
174 std::numeric_limits<size_t>::max() / GrowthFactor)
175 {
176 throw std::bad_alloc();
177 }
178 const size_t newBucketCount = map->bucketCount() * GrowthFactor;
179 auto nextMap =
180 make_unique<FixedSizeMap>(newBucketCount, std::move(*map));
181 mOldMaps.emplace_back(map);
182 mCapacityUntilGrowth =
183 maxEntries(nextMap->bucketCount()) - maxEntries(map->bucketCount());
184
185 // Store with std::memory_order_seq_cst to force a memory flush so that
186 // readers see the new table as soon as possible.
187 map = nextMap.release();
188 mMap.store(map, std::memory_order_seq_cst);
189 }
190 MATHICGB_ASSERT(mCapacityUntilGrowth > 0);
191
192 auto p = map->insert(value);
193 if (p.second)
194 --mCapacityUntilGrowth;
195 return p;
196 }

200 size_t entryCount() const {
201 const mtbb::lock_guard lockGuard(mInsertionMutex);
202 // const std::lock_guard<std::mutex> lockGuard(mInsertionMutex);
203
204 // We can load with std::memory_order_relaxed because we are holding the
205 // lock.
206 auto& map = *mMap.load(std::memory_order_relaxed);
207 return maxEntries(map.bucketCount()) - mCapacityUntilGrowth;
208 }

210private:
211 static const size_t MinBucketsPerEntry = 3; // inverse of max load factor

213 static const size_t InitialBucketCount = 1 << 1;

215 static size_t maxEntries(const size_t bucketCount) {
216 return (bucketCount + (MinBucketsPerEntry - 1)) / MinBucketsPerEntry;
217 }
